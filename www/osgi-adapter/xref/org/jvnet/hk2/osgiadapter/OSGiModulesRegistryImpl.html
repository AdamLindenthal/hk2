85      /*package*/ OSGiModulesRegistryImpl(BundleContext bctx) {
86          super(null);
87          this.bctx = bctx;
88  
89          // Need to add a listener so that we get notification about
90          // bundles that get installed/uninstalled from now on...
91          // This must happen before we start iterating the existing bundles.
92          bctx.addBundleListener(this);
93  
94          try {
95              loadCachedData();
96          } catch (Exception e) {
97              Logger.logger.log(Level.WARNING, "Cannot load cached metadata, will recreate the cache", e);
98              cachedData.clear();
99          }
100 
101         // Populate registry with pre-installed bundles
102         for (final Bundle b : bctx.getBundles()) {
103             if (b.getLocation().equals (Constants.SYSTEM_BUNDLE_LOCATION)) {
104                 continue;
105             }
106             try {
107                 add(makeModule(b)); // call add as it processes provider names
108             } catch (Exception e) {
109                 logger.logp(Level.WARNING, "OSGiModulesRegistryImpl",
110                         "OSGiModulesRegistryImpl",
111                         "Not able convert bundle [{0}] having location [{1}] " +
112                                 "to module because of exception: {2}",
113                         new Object[]{b, b.getLocation(), e});
114                 continue;
115             }
116         }
117         ServiceReference ref = bctx.getServiceReference(PackageAdmin.class.getName());
118         pa = PackageAdmin.class.cast(bctx.getService(ref));
119     }

121     public void bundleChanged(BundleEvent event) {
122         // Extender implementation.
123         try {
124             final Bundle bundle = event.getBundle();
125             switch (event.getType()) {
126                 case BundleEvent.RESOLVED : {
127                     if (logger.isLoggable(Level.FINE)) {
128                         logger.fine("[" + bundle.getBundleId() + "] " + bundle.getSymbolicName() +  " resolved");
129                     }
130                     break;
131                 } 
132 
133                 case BundleEvent.INSTALLED :
134                 {
135                     // call add as it processes provider names
136                     OSGiModuleImpl m = makeModule(bundle);
137                     add(m);
138                     break;
139                 }
140                 case BundleEvent.UNINSTALLED :
141                 {
142                     final Module m = getModule(bundle);
143                     if (m!=null) {
144                         // getModule can return null if some bundle got uninstalled
145                         // before we have finished initialization. This can
146                         // happen if framework APIs are called in parallel
147                         // by some third party bundles.
148                         // We need to call remove as it processes provider names
149                         // and updates the cache.
150                         remove(m);
151                     }
152                     break;
153                 }
154                 case BundleEvent.UPDATED :
155                     final Module m = getModule(bundle);
156                     if (m!=null) {
157                         // getModule can return null if some bundle got uninstalled
158                         // before we have finished initialization. This can
159                         // happen if framework APIs are called in parallel
160                         // by some third party bundles.
161                         // We need to call remove as it processes provider names
162                         // and updates the cache.
163                         remove(m);
164                     }
165 
166                     // make a new module from the updated bundle data and add it
167                     add(makeModule(bundle));
168                     break;
169             }
170         } catch (Exception e) {
171             logger.logp(Level.WARNING, "OSGiModulesRegistryImpl", "bundleChanged",
172                     "e = {0}", new Object[]{e});
173         }
174     }

185     /***
186      * Loads the inhabitants metadata from the cache. metadata is saved in a file
187      * called inhabitants
188      *
189      * @throws Exception if the file cannot be read correctly
190      */
191     private void loadCachedData() throws Exception {
192         String cacheLocation = getProperty(HK2_CACHE_DIR);
193         if (cacheLocation == null) {
194             return;
195         }
196         File io = new File(cacheLocation, INHABITANTS_CACHE);
197         if (!io.exists()) return;
198         ObjectInputStream stream = new ObjectInputStream(new BufferedInputStream(new FileInputStream(io)));
199         cachedData = (Map<URI, ModuleDefinition>) stream.readObject();
200         stream.close();
201     }
202 
203     /***
204      * Saves the inhabitants metadata to the cache in a file called inhabitants
205      * @throws IOException if the file cannot be saved successfully
206      */
207     private void saveCache() throws IOException {
208         String cacheLocation = getProperty(HK2_CACHE_DIR);
209         if (cacheLocation == null) {
210             return;
211         }
212         File io = new File(cacheLocation, INHABITANTS_CACHE);
213         if (io.exists()) io.delete();
214         io.createNewFile();
215         Map<URI, ModuleDefinition> data = new HashMap<URI, ModuleDefinition>();
216         for (Module m : modules.values()) {
217             data.put(m.getModuleDefinition().getLocations()[0], m.getModuleDefinition());
218         }
219         ObjectOutputStream os = new ObjectOutputStream(new BufferedOutputStream(new FileOutputStream(io)));
220         os.writeObject(data);
221         os.close();
222     }
223 
224     private void deleteCache() {
225         String cacheLocation = getProperty(HK2_CACHE_DIR);
226         if (cacheLocation == null) {
227             return;
228         }
229         File io = new File(cacheLocation, INHABITANTS_CACHE);
230         if (io.exists()) {
231             if (io.delete()) {
232                 logger.logp(Level.FINE, "OSGiModulesRegistryImpl",
233                         "deleteCache", "deleted = {0}", new Object[]{io});
234             } else {
235                 logger.logp(Level.WARNING, "OSGiModulesRegistryImpl",
236                         "deleteCache", "failed to delete = {0}", new Object[]{io});
237             }
238         }
239     }
240     // Factory method
241     private OSGiModuleDefinition makeModuleDef(Bundle bundle)
242             throws IOException, URISyntaxException {
243         URI key = OSGiModuleDefinition.toURI(bundle);
244         if (cachedData.containsKey(key)) {
245             return OSGiModuleDefinition.class.cast(cachedData.get(key));
246         } else {
247             return new OSGiModuleDefinition(bundle);
248         }
249     }

271     // factory method
272     protected Module newModule(ModuleDefinition moduleDef) {
273         String location = moduleDef.getLocations()[0].toString();
274         try {
275             if (logger.isLoggable(Level.FINE)) {
276                 logger.logp(Level.FINE, "OSGiModulesRegistryImpl", "newModule",
277                     "location = {0}", location);
278             }
279             File l = new File(moduleDef.getLocations()[0]);
280             if (l.isDirectory()) {
281                 location = "reference:" + location;
282             }
283             Bundle bundle = bctx.installBundle(location);
284             // wrap Bundle by a Module object
285             return new OSGiModuleImpl(this, bundle, moduleDef);
286         } catch (BundleException e) {
287             logger.logp(Level.WARNING, "OSGiModulesRegistryImpl", "newModule",
288                     "Exception {0} while adding location = {1}", new Object[]{e, location});
289 //            throw new RuntimeException(e); // continue
290         }
291         return null;
292     }

310     public synchronized void shutdown() {
311         for (Module m : modules.values()) {
312             // Only stop modules that were started after ModulesRegistry
313             // came into existence.
314             if (OSGiModuleImpl.class.cast(m).isTransientlyActive()) {
315                  m.stop();
316             }
317         }
318         // Save the cache before clearing modules
319         try {
320             saveCache();
321         } catch (IOException e) {
322             Logger.logger.log(Level.WARNING, "Cannot save metadata to cache", e);
323         }
324         modules.clear();
325 
326         for (Repository repo : repositories.values()) {
327             try {
328                 repo.shutdown();
329             } catch(Exception e) {
330                 java.util.logging.Logger.getAnonymousLogger().log(Level.SEVERE, "Error while closing repository " + repo, e);
331                 // swallows
332             }
333         }
334         // don't try to stop the system bundle, as we may be embedded inside
335         // something like Eclipse.
336     }

357     /***
358      * Returns a ClassLoader capable of loading classes from a set of modules identified
359      * by their module definition and also load new urls.
360      *
361      * @param parent the parent class loader for the returned class loader instance
362      * @param mds module definitions for all modules this classloader should be
363      *        capable of loading
364      * @param urls urls to be added to the module classloader
365      * @return class loader instance
366      * @throws com.sun.enterprise.module.ResolveError if one of the provided module
367      *         definition cannot be resolved
368      */
369     public ClassLoader getModulesClassLoader(final ClassLoader parent,
370                                              Collection<ModuleDefinition> mds,
371                                              URL[] urls) throws ResolveError {
372         final List<ClassLoader> delegateCLs = new ArrayList<ClassLoader>();
373         final List<Module> delegateModules = new ArrayList<Module>();
374         for (ModuleDefinition md : mds) {
375             Module m = makeModuleFor(md.getName(), md.getVersion());
376             delegateModules.add(m);
377             delegateCLs.add(m.getClassLoader());
378         }
379         return new URLClassLoader(urls!=null?urls:new URL[0], parent) {
380             /*
381              * This is a delegating class loader.
382              * This extends URLClassLoader, because web layer (Jasper to be specific)
383              * expects it to be a URLClassLoader so that it can extract Classpath information
384              * used for javac.
385              * It always delegates to a chain of OSGi bundle's class loader.
386              * ClassLoader.defineClass() is never called in the context of this class.
387              * There will never be a class for which getClassLoader()
388              * would return this class loader.
389              */
390             @Override
391             public URL[] getURLs() {
392                 List<URL> result = new ArrayList<URL>();
393                 if (parent instanceof URLClassLoader) {
394                     URL[] parentURLs = URLClassLoader.class.cast(parent).getURLs();
395                     result.addAll(Arrays.asList(parentURLs));
396                 }
397                 for (Module m : delegateModules) {
398                     ModuleDefinition md = m.getModuleDefinition();
399                     URI[] uris = md.getLocations();
400                     URL[] urls = new URL[uris.length];
401                     for (int i = 0; i < uris.length; ++i) {
402                         try {
403                             urls[i] = uris[i].toURL();
404                         } catch (MalformedURLException e) {
405                             logger.warning("Exception " + e + " while converting " + uris[i] + " to URL");
406                         }
407                     }
408                     result.addAll(Arrays.asList(urls));
409                 }
410                 return result.toArray(new URL[0]);
411             }
412 
413             @Override
414             protected Class<?> findClass(String name) throws ClassNotFoundException {
415                 for (ClassLoader delegate : delegateCLs) {
416                     try {
417                         return delegate.loadClass(name);
418                     } catch(ClassNotFoundException e) {
419                         // This is expected, so ignore
420                     }
421                 }
422                 throw new ClassNotFoundException(name);
423             }
424 
425             @Override
426             public URL findResource(String name) {
427                 URL resource = null;
428                 for (ClassLoader delegate : delegateCLs) {
429                     resource = delegate.getResource(name);
430                     if (resource != null) {
431                         return resource;
432                     }
433                 }
434                 return resource;
435             }
436 
437             @Override
438             public Enumeration<URL> findResources(String name) throws IOException {
439                 List<Enumeration<URL>> enumerators = new ArrayList<Enumeration<URL>>();
440                 for (ClassLoader delegate : delegateCLs) {
441                     Enumeration<URL> enumerator = delegate.getResources(name);
442                     enumerators.add(enumerator);
443                 }
444                 return new CompositeEnumeration(enumerators);
445             }
446 
447         };
448     }

481     public void addModuleChangeListener(final ModuleChangeListener listener, final OSGiModuleImpl module) {
482         BundleListener bundleListener = new BundleListener() {
483             public void bundleChanged(BundleEvent event) {
484                 if ((event.getBundle() == module.getBundle()) &&
485                         ((event.getType() & BundleEvent.UPDATED) == BundleEvent.UPDATED)) {
486                     listener.changed(module);
487                 }
488             }
489         };
490         bctx.addBundleListener(bundleListener);
491         moduleChangeListeners.put(listener, bundleListener);
492     }

494     public boolean removeModuleChangeListener(ModuleChangeListener listener) {
495         BundleListener bundleListener = moduleChangeListeners.remove(listener);
496         if (bundleListener!= null) {
497             bctx.removeBundleListener(bundleListener);
498             return true;
499         }
500         return false;
501     }

503     public void register(final ModuleLifecycleListener listener) {
504         if (TracingUtilities.isEnabled()) {
505             bctx.addBundleListener(new SynchronousBundleListener() {
506             public void bundleChanged(BundleEvent event) {
507                 switch (event.getType()) {
508                     case BundleEvent.RESOLVED:
509                         TracingUtilities.traceResolution(OSGiModulesRegistryImpl.this,
510                                 event.getBundle().getBundleId(),
511                                 event.getBundle().getSymbolicName());
512                          break;
513                     case BundleEvent.STARTED:
514                         TracingUtilities.traceStarted(OSGiModulesRegistryImpl.this,
515                                 event.getBundle().getBundleId(),
516                                 event.getBundle().getSymbolicName());
517                         break;
518                  }
519             }
520             });
521         }
522         // This is purposefully made an asyncronous bundle listener
523         BundleListener bundleListener = new BundleListener() {
524             public void bundleChanged(BundleEvent event) {
525                 switch (event.getType()) {
526                     case BundleEvent.INSTALLED:
527                         listener.moduleInstalled(getModule(event.getBundle()));
528                         break;
529                     case BundleEvent.UPDATED:
530                         listener.moduleUpdated(getModule(event.getBundle()));
531                         break;
532                     case BundleEvent.RESOLVED:
533                         listener.moduleResolved(getModule(event.getBundle()));
534                         break;
535                     case BundleEvent.STARTED:
536                         listener.moduleStarted(getModule(event.getBundle()));
537                         break;
538                     case BundleEvent.STOPPED:
539                         listener.moduleStopped(getModule(event.getBundle()));
540                         break;
541                 }
542             }
543         };
544         bctx.addBundleListener(bundleListener);
545         moduleLifecycleListeners.put(listener,  bundleListener);
546     }

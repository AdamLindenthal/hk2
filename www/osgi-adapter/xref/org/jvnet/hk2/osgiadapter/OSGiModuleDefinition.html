84      public OSGiModuleDefinition(Jar jarFile, URI location) throws IOException {
85          /*
86          * When we support reading metadata from external manifest.mf file,
87          * we can create a custom URI and the URL handler can merge the
88          * manifest info. For now, just use the standard URI.
89          */
90          this.location = location;
91          manifest = new SerializableManifest(jarFile.getManifest());
92          Attributes mainAttr = manifest.getMainAttributes();
93          bundleName = manifest.getMainAttributes().getValue(Constants.BUNDLE_NAME);
94  
95          name = mainAttr.getValue(Constants.BUNDLE_SYMBOLICNAME);
96          // R3 bundles may not have any name, yet HK2 requires some name to be
97          // assigned. So, we use location in such cases. We encounter this
98          // problem when user has dropped some plain jars or R3 bundles
99          // in modules dir. If you choose to use a different name,
100         // please also change the code in OSGiModuleId class which makes
101         // similar assumption.
102         if (name == null) name = location.toString();
103         version = mainAttr.getValue(Constants.BUNDLE_VERSION);
104         lifecyclePolicyClassName = mainAttr.getValue(ManifestConstants.LIFECYLE_POLICY);
105         jarFile.loadMetadata(metadata);
106     }

112     static URI toURI(Bundle b) throws URISyntaxException {
113         try {
114             return new URI(b.getLocation());
115         } catch (URISyntaxException ue) {
116             // On Equinox, bundles started via autostart have a strange location.
117             // It is of the format: initial@reference:file:...
118             // It can't be turned into an URI, so we take out initial@.
119             if (b.getLocation().startsWith("initial@")) {
120                 return new URI(b.getLocation().substring("initial@".length()));
121             } else {
122                 throw ue;
123             }
124         }
125     }

137     /***
138      * @return List of bundles on which this bundle depends on using Require-Bundle
139      */
140     public ModuleDependency[] getDependencies() {
141         List<ModuleDependency> mds = new ArrayList<ModuleDependency>();
142         String requiredBundles =
143                 getManifest().getMainAttributes().getValue(Constants.REQUIRE_BUNDLE);
144         if (requiredBundles != null) {
145             Logger.logger.log(Level.INFO, name + " -> " + requiredBundles);
146             // The string looks like
147             // Require-Bundle: b1; version="[1.0, 2.0)", b2, b3;visbility:=reexport; version="1.0",...
148             // First remove the regions that appear between a pair of quotes (""), as that
149             // can confuse the tokenizer.
150             // Then, tokenize using comma(,) as that separates one bundle from another.
151             while (true) {
152                 int i1 = requiredBundles.indexOf('\"');
153                 if (i1 == -1) break;
154                 int i2 = requiredBundles.indexOf('\"', i1 + 1);
155                 StringBuilder sb = new StringBuilder();
156                 sb.append(requiredBundles.substring(0, i1));
157                 sb.append(requiredBundles.substring(i2 + 1));
158                 requiredBundles = sb.toString();
159             }
160             StringTokenizer st =
161                     new StringTokenizer(requiredBundles, ",", false);
162             while (st.hasMoreTokens()) {
163                 String requireBundle = st.nextToken();
164                 String requiredBundleName;
165                 int idx = requireBundle.indexOf(';');
166                 if (idx == -1) {
167                     requiredBundleName = requireBundle;
168                 } else {
169                     requiredBundleName = requireBundle.substring(0, idx);
170                     // TODO(Sahoo): parse version and other stuff
171                 }
172                 mds.add(new ModuleDependency(requiredBundleName, null));
173             }
174         }
175         return mds.toArray(new ModuleDependency[mds.size()]);
176     }

270         private void parseServiceDescriptors(ModuleMetadata result) {
271             /*
272              * No need for this optimisation, becasuse it fails when jar does not
273              * have directory entries.
274             if (b.getEntry(SERVICE_LOCATION) == null) return;
275              */
276             Enumeration<String> entries;
277             entries = b.getEntryPaths(SERVICE_LOCATION);
278             if (entries != null) {
279                 while (entries.hasMoreElements()) {
280                     String entry = entries.nextElement();
281                     String serviceName = entry.substring(SERVICE_LOCATION.length()+1);
282                     InputStream is = null;
283                     final URL url = b.getEntry(entry);
284                     try {
285                         is = url.openStream();
286                         result.load(url, serviceName, is);
287                     } catch (IOException e) {
288                         LogHelper.getDefaultLogger().log(Level.SEVERE,
289                                 "Error reading service provider in " + b.getLocation(), e);
290                     } finally {
291                         if (is != null) {
292                             try {
293                                 is.close();
294                             } catch (IOException e) {}
295                         }
296                     }
297                 }
298             }
299         }

301         private byte[] loadFully(URL url) throws IOException {
302             InputStream in = url.openStream();
303             byte[] buf = new byte[0];
304             try {
305                 int chunkSize = 512;
306                 byte[] chunk = new byte[chunkSize];
307                 while (true) {
308                     int count = in.read(chunk, 0, chunkSize);
309                     if (count == -1) break; // EOF
310                     final int curLength = buf.length;
311                     byte[] newbuf = new byte[curLength + count];
312                     System.arraycopy(buf, 0, newbuf, 0, curLength);
313                     System.arraycopy(chunk, 0, newbuf, curLength, count);
314                     buf = newbuf;
315                 }
316                 return buf;
317             } finally {
318                 in.close();
319             }
320         }

329         private SerializableManifest()

338         private Object writeReplace() throws ObjectStreamException {
339             return new ManifestData(this);
340         }
341 
342         /***
343          * a class which is used as a substiture for
344          * {@link SerializableManifest} during serialization.
345          * @see java.io.Serializable
346          * @see #readResolve
347          * @see org.jvnet.hk2.osgiadapter.OSGiModuleDefinition.SerializableManifest#writeReplace
348          */
349         private static class ManifestData implements Serializable {
350             private Map<String, String> mainAttributes;
351             private Map<String, Map<String, String>> entries;
352 
353             // called during serialization
354             private ManifestData(Manifest manifest)
355             {
356                 mainAttributes = new HashMap<String, String>();
357                 mainAttributes.putAll(toMap(manifest.getMainAttributes()));
358                 entries = new HashMap<String, Map<String, String>>();
359                 for (Map.Entry<String, Attributes> entry : manifest.getEntries().entrySet()) {
360                     entries.put(entry.getKey(), toMap(entry.getValue()));
361                 }
362             }
363 
364             // called during desrialization
365             private Object readResolve() throws ObjectStreamException {
366                 SerializableManifest m = new SerializableManifest();
367                 for (Map.Entry<String, String> entry : mainAttributes.entrySet()) {
368                     m.getMainAttributes().putValue(entry.getKey(), entry.getValue());
369                 }
370                 for (Map.Entry<String, Map<String, String>> entry : entries.entrySet()) {
371                     m.getEntries().put(entry.getKey(), toAttributes(entry.getValue()));
372                 }
373                 return m;
374             }
375 
376             private Attributes toAttributes(Map<String, String> map) {
377                 Attributes attrs = new Attributes();
378                 for (Map.Entry<String, String> entry : map.entrySet()) {
379                     attrs.putValue(entry.getKey(), entry.getValue());
380                 }
381                 return attrs;
382             }
383 
384             private Map<String, String> toMap(Attributes attrs) {
385                 Map<String, String> map = new HashMap<String, String>();
386                 for (Map.Entry<Object, Object> entry : attrs.entrySet()) {
387                     // the cast is OK, as Manifest only puts Name, String pairs
388                     // See javadocs of Manifest.
389                     map.put(Attributes.Name.class.cast(entry.getKey()).toString(),
390                             (String)entry.getValue());
391                 }
392                 return map;
393             }
394         }

84      public OSGiModuleDefinition(Jar jarFile, URI location) throws IOException {
85          /*
86          * When we support reading metadata from external manifest.mf file,
87          * we can create a custom URI and the URL handler can merge the
88          * manifest info. For now, just use the standard URI.
89          */
90          this.location = location;
91          final Manifest m = jarFile.getManifest();
92          manifest = m instanceof Serializable ? m : new SerializableManifest(m);
93          Attributes mainAttr = manifest.getMainAttributes();
94          bundleName = mainAttr.getValue(Constants.BUNDLE_NAME);
95  
96          name = mainAttr.getValue(Constants.BUNDLE_SYMBOLICNAME);
97          // R3 bundles may not have any name, yet HK2 requires some name to be
98          // assigned. So, we use location in such cases. We encounter this
99          // problem when user has dropped some plain jars or R3 bundles
100         // in modules dir. If you choose to use a different name,
101         // please also change the code in OSGiModuleId class which makes
102         // similar assumption.
103         if (name == null) name = location.toString();
104         version = mainAttr.getValue(Constants.BUNDLE_VERSION);
105         lifecyclePolicyClassName = mainAttr.getValue(ManifestConstants.LIFECYLE_POLICY);
106         jarFile.loadMetadata(metadata);
107     }

113     static URI toURI(Bundle b) throws URISyntaxException {
114         try {
115             return new URI(b.getLocation());
116         } catch (URISyntaxException ue) {
117             // On Equinox, bundles started via autostart have a strange location.
118             // It is of the format: initial@reference:file:...
119             // It can't be turned into an URI, so we take out initial@.
120             if (b.getLocation().startsWith("initial@")) {
121                 return new URI(b.getLocation().substring("initial@".length()));
122             } else {
123                 throw ue;
124             }
125         }
126     }

138     /**
139      * @return List of bundles on which this bundle depends on using Require-Bundle
140      */
141     public ModuleDependency[] getDependencies() {
142         List<ModuleDependency> mds = new ArrayList<ModuleDependency>();
143         String requiredBundles =
144                 getManifest().getMainAttributes().getValue(Constants.REQUIRE_BUNDLE);
145         if (requiredBundles != null) {
146             Logger.logger.log(Level.INFO, name + " -> " + requiredBundles);
147             // The string looks like
148             // Require-Bundle: b1; version="[1.0, 2.0)", b2, b3;visbility:=reexport; version="1.0",...
149             // First remove the regions that appear between a pair of quotes (""), as that
150             // can confuse the tokenizer.
151             // Then, tokenize using comma(,) as that separates one bundle from another.
152             while (true) {
153                 int i1 = requiredBundles.indexOf('\"');
154                 if (i1 == -1) break;
155                 int i2 = requiredBundles.indexOf('\"', i1 + 1);
156                 StringBuilder sb = new StringBuilder();
157                 sb.append(requiredBundles.substring(0, i1));
158                 sb.append(requiredBundles.substring(i2 + 1));
159                 requiredBundles = sb.toString();
160             }
161             StringTokenizer st =
162                     new StringTokenizer(requiredBundles, ",", false);
163             while (st.hasMoreTokens()) {
164                 String requireBundle = st.nextToken();
165                 String requiredBundleName;
166                 int idx = requireBundle.indexOf(';');
167                 if (idx == -1) {
168                     requiredBundleName = requireBundle;
169                 } else {
170                     requiredBundleName = requireBundle.substring(0, idx);
171                     // TODO(Sahoo): parse version and other stuff
172                 }
173                 mds.add(new ModuleDependency(requiredBundleName, null));
174             }
175         }
176         return mds.toArray(new ModuleDependency[mds.size()]);
177     }

266         private void parseServiceDescriptors(ModuleMetadata result) {
267             /*
268              * This optimisation was earlier not working because of FELIX-1210.
269              */
270             if (b.getEntry(SERVICE_LOCATION) == null) return;
271             Enumeration<String> entries;
272             entries = b.getEntryPaths(SERVICE_LOCATION);
273             if (entries != null) {
274                 while (entries.hasMoreElements()) {
275                     String entry = entries.nextElement();
276                     String serviceName = entry.substring(SERVICE_LOCATION.length()+1);
277                     InputStream is = null;
278                     final URL url = b.getEntry(entry);
279                     try {
280                         is = url.openStream();
281                         result.load(url, serviceName, is);
282                     } catch (IOException e) {
283                         LogHelper.getDefaultLogger().log(Level.SEVERE,
284                                 "Error reading service provider in " + b.getLocation(), e);
285                     } finally {
286                         if (is != null) {
287                             try {
288                                 is.close();
289                             } catch (IOException e) {}
290                         }
291                     }
292                 }
293             }
294         }

296         private byte[] loadFully(URL url) throws IOException {
297             InputStream in = url.openStream();
298             byte[] buf = new byte[0];
299             try {
300                 int chunkSize = 512;
301                 byte[] chunk = new byte[chunkSize];
302                 while (true) {
303                     int count = in.read(chunk, 0, chunkSize);
304                     if (count == -1) break; // EOF
305                     final int curLength = buf.length;
306                     byte[] newbuf = new byte[curLength + count];
307                     System.arraycopy(buf, 0, newbuf, 0, curLength);
308                     System.arraycopy(chunk, 0, newbuf, curLength, count);
309                     buf = newbuf;
310                 }
311                 return buf;
312             } finally {
313                 in.close();
314             }
315         }

321         /**
322          * Manifest of a bundle. It is optimized for serialization as it writes out only the bundle id to the stream.
323          * So, it can only be desrialized in the context of OSGi and that too only if the bundle id is same.
324          */
325         private static class BundleManifest extends Manifest implements Serializable {
326             private long bundleId;
327 
328             private BundleManifest() {
329             }
330 
331             private BundleManifest(Bundle b) {
332                 this.bundleId = b.getBundleId();
333                 init(b);
334             }
335 
336             private void init(Bundle b) {
337                 Attributes attrs = getMainAttributes();
338                 Dictionary headers = b.getHeaders();
339                 for (Object o : Collections.list(headers.keys())) {
340                     attrs.putValue((String)o, (String)headers.get(o));
341                 }
342             }
343 
344             // Serialization method
345             private void writeObject(ObjectOutputStream out) throws IOException {
346                 out.defaultWriteObject();
347             }
348 
349             private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
350                 in.defaultReadObject();
351                 ClassLoader cl;
352                 if (System.getSecurityManager() != null) {
353                     cl = AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
354                         public ClassLoader run() {
355                             return getClass().getClassLoader();
356                         }
357                     });
358                 } else {
359                     cl = getClass().getClassLoader();
360                 }
361 
362                 init(BundleReference.class.cast(cl).getBundle().getBundleContext().getBundle(bundleId));
363             }
364 
365         }
366     }

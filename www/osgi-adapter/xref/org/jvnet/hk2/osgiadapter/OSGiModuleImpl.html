100     public ModuleState getState() {
101         // We don't cache the module state locally. Instead we always map
102         // the underlying bundle's state to HK2 state. This avoids us
103         // from having to register a listener with OSGi to be updated with
104         // bundle state transitions.
105         return mapBundleStateToModuleState(bundle);
106     }
107 
108     /* package */ static ModuleState mapBundleStateToModuleState(Bundle bundle)
109     {
110         ModuleState state;
111         switch (bundle.getState())
112         {
113             case Bundle.INSTALLED:
114             case Bundle.UNINSTALLED:
115                 state = ModuleState.NEW;
116                 break;
117             case Bundle.RESOLVED:
118                 state = ModuleState.RESOLVED;
119                 break;
120             case Bundle.STARTING:
121                 state = ModuleState.PREPARING;
122                 break;
123             case Bundle.ACTIVE:
124                 state = ModuleState.READY;
125                 break;
126             default:
127                 throw new RuntimeException(
128                         "Does not know how to handle bundle with state [" +
129                                 bundle.getState() + "]");
130         }
131 
132         return state;
133     }

141     public synchronized void start() throws ResolveError {
142         int state = bundle.getState();
143         if (((Bundle.STARTING | Bundle.ACTIVE | Bundle.STOPPING) & state) != 0) {
144             if (logger.isLoggable(Level.FINER)) {
145                 logger.logp(Level.FINER, "OSGiModuleImpl", "start",
146                         "Ignoring start of bundle {0} as it is in {1} state",
147                         new Object[]{bundle, toString(bundle.getState())} );
148             }
149             return;
150         }
151         try {
152             SecurityManager sm = System.getSecurityManager();
153             if (sm != null) {
154                 try {
155                     AccessController.doPrivileged(new PrivilegedExceptionAction(){
156                         public Object run() throws BundleException
157                         {
158                             bundle.start(Bundle.START_TRANSIENT);
159                             return null;
160                         }
161                     });
162                 } catch (PrivilegedActionException e) {
163                     throw (BundleException)e.getException();
164                 }
165             } else {
166                 bundle.start(Bundle.START_TRANSIENT);
167             }
168             isTransientlyActive = true;
169             if (logger.isLoggable(Level.FINE)) {
170                 logger.logp(Level.FINE, "OSGiModuleImpl",
171                         "start", "Started bundle {0}", bundle);
172             }
173         } catch (BundleException e) {
174             throw new ResolveError("Failed to start "+this,e);
175         }
176 
177         // TODO(Sahoo): Remove this when hk2-apt generates equivalent BundleActivator
178         // if there is a LifecyclePolicy, then instantiate and invoke.
179         if (md.getLifecyclePolicyClassName()!=null) {
180             try {
181                 Class<LifecyclePolicy> lifecyclePolicyClass =
182                         (Class<LifecyclePolicy>) bundle.loadClass(md.getLifecyclePolicyClassName());
183                 lifecyclePolicy = lifecyclePolicyClass.newInstance();
184             } catch(ClassNotFoundException e) {
185                 throw new ResolveError("ClassNotFound : " + e.getMessage(), e);
186             } catch(java.lang.InstantiationException e) {
187                 throw new ResolveError(e);
188             } catch(IllegalAccessException e) {
189                 throw new ResolveError(e);
190             }
191         }
192         if (lifecyclePolicy!=null) {
193             lifecyclePolicy.start(this);
194         }
195     }

197     private String toString(int state)
198     {
199         String value;
200         switch (state) {
201             case Bundle.STARTING:
202                 value = "STARTING";
203                 break;
204             case Bundle.STOPPING:
205                 value = "STOPPING";
206                 break;
207             case Bundle.INSTALLED:
208                 value = "INSTALLED";
209                 break;
210             case Bundle.UNINSTALLED:
211                 value = "UNINSTALLED";
212                 break;
213             case Bundle.RESOLVED:
214                 value = "RESOLVED";
215                 break;
216             case Bundle.ACTIVE:
217                 value = "ACTIVE";
218                 break;
219             default:
220                 value = "UNKNOWN STATE [" + state + "]";
221                 logger.warning("No mapping exist for bundle state " + state);
222         }
223         return value;
224     }

233     public void detach() {
234         if (bundle.getState() != Bundle.ACTIVE) {
235             if (logger.isLoggable(Level.FINER)) {
236                 logger.logp(Level.FINER, "OSGiModuleImpl", "detach",
237                         "Ignoring stop of bundle {0} as it is in {1} state",
238                         new Object[]{bundle, toString(bundle.getState())} );
239             }
240             return;
241         }
242 
243         if (lifecyclePolicy!=null) {
244             lifecyclePolicy.stop(this);
245             lifecyclePolicy=null;
246         }
247 
248         try {
249             bundle.stop();
250             if (logger.isLoggable(Level.FINE))
251             {
252                 logger.logp(Level.FINE, "OSGiModuleImpl", "detach", "Stopped bundle = {0}", new Object[]{bundle});
253             }
254 //            bundle.uninstall();
255         } catch (BundleException e) {
256             throw new RuntimeException(e);
257         }
258     }

260     public void uninstall() {
261         // This method is called when the hk2-osgi-adapter module is stopped.
262         // During that time, we need to stop all the modules, hence no sticky check is
263         // performed in this method.
264         try {
265             bundle.uninstall();
266         } catch (BundleException e) {
267             throw new RuntimeException(e);
268         }
269         registry.remove(this);
270         this.registry = null;
271     }

273     public void refresh() {
274         URI location = md.getLocations()[0];
275         File f = new File(location);
276         if (f.lastModified() > bundle.getLastModified()) {
277             try {
278                 bundle.update();
279                 registry.getPackageAdmin().refreshPackages(new Bundle[]{bundle});
280             } catch (BundleException e) {
281                 throw new RuntimeException(e);
282             }
283         }
284     }

295     public Iterable<Class> getProvidersClass(String name) {
296         List<Class> r = new ArrayList<Class>();
297         for( String provider : getMetadata().getEntry(name).providerNames) {
298             try {
299                 r.add(getClassLoader().loadClass(provider));
300             } catch (ClassNotFoundException e) {
301                 logger.log(Level.SEVERE, "Failed to load "+provider+" from "+getName(),e);
302             }
303         }
304         return r;
305     }

327     /* package */ void parseInhabitants(String name, InhabitantsParser parser) throws IOException {
328         Holder<ClassLoader> holder = new Holder<ClassLoader>() {
329             public ClassLoader get() {
330                 //doprivileged needed for running with SecurityManager
331                 ClassLoader ret = (ClassLoader) AccessController.doPrivileged(new PrivilegedAction() {
332                     public java.lang.Object run() {
333                         return new ClassLoader() {
334                             @Override
335                             public synchronized Class<?> loadClass(
336                                     final String name) throws ClassNotFoundException {
337                                 if (logger.isLoggable(Level.FINE)) {
338                                     logger.logp(Level.FINE, "OSGiModuleImpl", "loadClass", "Loading {0} from bundle: {1}",
339                                             new Object[]{name, bundle});
340                                 }
341                                 start();
342                                 try {
343                                     final Class aClass = (Class)
344                                            AccessController.doPrivileged(new PrivilegedAction() {
345                                         public java.lang.Object run() {
346                                             try {
347                                                 return bundle.loadClass(name);
348                                             } catch (Throwable e) {
349                                                 logger.logp(Level.SEVERE, "OSGiModuleImpl", "loadClass", "Exception in module " + bundle.toString() + " : " + e.toString());
350                                                 throw new RuntimeException(e);
351                                             }
352                                         }
353                                     });
354                                     if (logger.isLoggable(Level.FINE)) {
355                                         logger.logp(Level.FINE, "OSGiModuleImpl", "loadClass",
356                                                 name + ".class.getClassLoader() = {0}",
357                                                 aClass.getClassLoader());
358                                     }
359                                     return aClass;
360                                 } catch (RuntimeException e) {
361                                     logger.logp(Level.SEVERE, "OSGiModuleImpl", "loadClass", "Exception in module " + bundle.toString() + " : " + e.toString());
362                                     throw new ClassNotFoundException(e.getCause().getMessage(), e.getCause());
363                                 }
364                             }
365                         };
366                     }
367                 });
368                 return ret;
369             }
370         };
371 
372         for (InhabitantsDescriptor d : md.getMetadata().getHabitats(name))
373             parser.parse(d.createScanner(),holder);
374     }

376     public ClassLoader getClassLoader() {
377         /*
378          * This is a delegating class loader.
379          * It always delegates to OSGi's bundle's class loader.
380          * ClassLoader.defineClass() is never called in the context of this class.
381          * There will never be a class for which getClassLoader()
382          * would return this class loader.
383          * It overrides loadClass(), getResource() and getResources() as opposed to
384          * their findXYZ() equivalents so that the OSGi export control mechanism
385          * is enforced even for classes and resources available in the system/boot
386          * class loader.
387          */
388         return new ClassLoader(Bundle.class.getClassLoader()) {
389             public static final String META_INF_SERVICES = "META-INF/services/";
390 
391             @Override
392             protected synchronized Class<?> loadClass(final String name, boolean resolve) throws ClassNotFoundException {
393                 try {
394                     //doprivileged needed for running with SecurityManager
395                     Class<?> ret =(Class<?>)
396                     AccessController.doPrivileged(new PrivilegedAction() {
397                         public java.lang.Object run() {
398                             try {
399                                 return bundle.loadClass(name);
400                             } catch (ClassNotFoundException e) {
401                                 // punch in. find the provider class, no matter where we are.
402                                 OSGiModuleImpl m =
403                                         (OSGiModuleImpl) registry.getProvidingModule(name);
404                                 if (m != null) {
405                                     try {
406                                         return m.bundle.loadClass(name);
407                                     } catch (ClassNotFoundException ex) {
408                                         throw new RuntimeException(ex);
409                                     }
410                                 }
411                                 throw new RuntimeException(e);
412                             }
413                         }
414                     });
415                     return ret;
416                     
417                 } catch (RuntimeException e) {
418                     if (e.getCause() instanceof ClassNotFoundException) {
419                         throw (ClassNotFoundException)e.getCause();
420                     }
421                     throw e;
422                 }
423 
424             }
425 
426             @Override
427             public URL getResource(String name) {
428                 URL result = bundle.getResource(name);
429                 if (result != null) return result;
430 
431                 // If this is a META-INF/services lookup, search in every
432                 // modules that we know of.
433                 if(name.startsWith(META_INF_SERVICES)) {
434                     // punch in. find the service loader from any module
435                     String serviceName = name.substring(
436                             META_INF_SERVICES.length());
437 
438                     for( Module m : registry.getModules() ) {
439                         List<URL> list = m.getMetadata().getDescriptors(
440                                 serviceName);
441                         if(!list.isEmpty())     return list.get(0);
442                     }
443                 }
444                 return null;
445             }
446 
447             @Override
448             public Enumeration<URL> getResources(String name) throws IOException {
449                 if(name.startsWith(META_INF_SERVICES)) {
450                     // search in the parent classloader
451                     Enumeration<URL> parentResources =
452                             Bundle.class.getClassLoader().getResources(name);
453 
454                     // punch in. find the service loader from any module
455                     String serviceName = name.substring(
456                             META_INF_SERVICES.length());
457 
458                     List<URL> punchedInURLs = new ArrayList<URL>();
459 
460                     for( Module m : registry.getModules() )
461                         punchedInURLs.addAll(m.getMetadata().getDescriptors(
462                                 serviceName));
463 
464                     List<Enumeration<URL>> enumerators = new ArrayList<Enumeration<URL>>();
465                     enumerators.add(parentResources);
466                     enumerators.add(Collections.enumeration(punchedInURLs));
467                     Enumeration<URL> result = new CompositeEnumeration(enumerators);
468                     return result;
469                 }
470                 Enumeration<URL> resources = bundle.getResources(name);
471                 if (resources==null) {
472                     // This check is needed, because ClassLoader.getResources()
473                     // expects us to return an empty enumeration.
474                     resources = new Enumeration<URL>(){
475 
476                         public boolean hasMoreElements() {
477                             return false;
478                         }
479 
480                         public URL nextElement() {
481                             throw new NoSuchElementException();
482                         }
483                     };
484                 }
485                 return resources;
486             }
487 
488             @Override
489             public String toString() {
490                 return "Class Loader for Bundle [" + bundle.toString() + " ]";
491             }
492         };
493     }

511     public List<Module> getImports() {
512         List<Module> result = new ArrayList<Module>();
513         RequiredBundle[] requiredBundles =
514                 registry.getPackageAdmin().getRequiredBundles(bundle.getSymbolicName());
515         if (requiredBundles!=null) {
516             for(RequiredBundle rb : requiredBundles) {
517                 Module m = registry.getModule(rb.getBundle());
518                 if (m!=null) {
519                     // module is known to the module system
520                     result.add(m);
521                 } else {
522                     // module is not known to us - may be the OSgi bundle depends on a native
523                     // OSGi bundle
524                 }
525             }
526 
527         }
528         return result;
529     }

90      public Habitat createHabitat(ModulesRegistry registry, StartupContext context) throws BootException {
91          HabitatInfo habitatInfo = new HabitatInfo();
92          habitatInfo.habitat = super.createHabitat(registry, context);
93          createHK2ServiceTracker(habitatInfo);
94          // register Habitat as an OSGi service
95          habitatInfo.habitatRegistration = ctx.registerService(Habitat.class.getName(), habitatInfo.habitat, context.getArguments());
96          habitatInfos.put(habitatInfo.habitat, habitatInfo);
97          return habitatInfo.habitat;
98      }

100     private void destroyHabitat(Habitat habitat) {
101         HabitatInfo habitatInfo = habitatInfos.get(habitat);
102         if (habitatInfo == null) {
103             return;
104         }
105 
106         // run code in the reverse order
107         habitatInfo.habitatRegistration.unregister();
108         stopHK2ServiceTracker(habitatInfo);
109         // AMX i shaving trouble if we release inhabitants. So temporarily disable this.
110         // habitat.release();
111         habitatInfo = null;
112         habitatInfos.remove(habitat);
113     }

115     private void createHK2ServiceTracker(HabitatInfo habitatInfo) {
116         habitatInfo.osgiServiceTracker = new ServiceTracker(
117                 ctx, new NonHK2ServiceFilter(), new HK2ServiceTrackerCustomizer(habitatInfo.habitat));
118         habitatInfo.osgiServiceTracker.open(true);
119     }

121     /***
122      * Stop service tracker associated with the given habitat
123      *

126     private void stopHK2ServiceTracker(HabitatInfo habitatInfo) {
127         if (habitatInfo.osgiServiceTracker != null) {
128             habitatInfo.osgiServiceTracker.close();
129             habitatInfo.osgiServiceTracker = null;
130         }

157     public void stop(BundleContext context) throws Exception {
158         // When OSGi framework shuts down, it shuts down all started bundles, but the order is unspecified.
159         // So, since we are going to shutdown the registry, it's better that we stop startup service just incase it is still running.
160         // Similarly, we can release the habitat.
161 
162         // Execute code in reverse order w.r.t. start()
163 
164         for (HabitatInfo habitatInfo : habitatInfos.values()) {
165             ModuleStartup startupService =
166                     habitatInfo.habitat.getComponent(ModuleStartup.class, habitatInfo.habitat.DEFAULT_NAME);
167             if (startupService != null) {
168                 try {
169                     logger.info("Stopping " + startupService);
170                     startupService.stop();
171                 } catch (Exception e) {
172                     logger.log(Level.WARNING, "HK2Main:stop():Exception while stopping ModuleStartup service.", e);
173                 }
174             }
175             destroyHabitat(habitatInfo.habitat);
176         }
177 
178         ModulesRegistry mr = (ModulesRegistry) ctx.getService(mrReg.getReference());
179         if (mr != null) {
180             mr.shutdown();
181             mr = null;
182         }
183 //        ctx.removeBundleListener(this); // used for debugging only. see start method
184 
185     }

211     private class HK2ServiceTrackerCustomizer implements ServiceTrackerCustomizer {
212         private final Habitat habitat;
213 
214         private HK2ServiceTrackerCustomizer(Habitat habitat) {
215             this.habitat = habitat;
216         }
217 
218         public Object addingService(final ServiceReference reference) {
219             final Object object = ctx.getService(reference);
220 
221             // let's get the list of implemented contracts
222             String[] contractNames = (String[]) reference.getProperty("objectclass");
223             if (contractNames != null && contractNames.length > 0) {
224                 // we will register this service under each contract it implements
225                 for (String contractName : contractNames) {
226                     String name = (String) reference.getProperty("component.name");
227                     if (name == null) {
228                         // let's get a name if possible, that will only work with Spring OSGi services
229                         // we may need to find a better way to get a potential name.
230                         name = (String) reference.getProperty("org.springframework.osgi.bean.name");
231                     }
232                     habitat.addIndex(new ExistingSingletonInhabitant(object), contractName, name);
233                     logger.logp(Level.FINE, "HK2Main$HK2ServiceTrackerCustomizer",
234                             "addingService", "registering service = {0}, contract = {1}, name = {2}", new Object[]{
235                                     object, contractName, name});
236                 }
237             } else {
238                 // this service does not implement a specific contract, let's register it by its type.
239                 habitat.add(new ExistingSingletonInhabitant(object));
240                 logger.logp(Level.FINE, "HK2Main$HK2ServiceTrackerCustomizer",
241                         "addingService", "registering service = {0}", object);
242             }
243             return object;
244         }
245 
246         public void modifiedService(ServiceReference reference, Object service) {
247         }
248 
249         public void removedService(ServiceReference reference, Object service) {
250             // we need to unregister the service for each contract it implements.
251             String[] contractNames = (String[]) reference.getProperty("objectclass");
252             if (contractNames != null && contractNames.length > 0) {
253                 for (String contractName : contractNames) {
254                     habitat.removeIndex(contractName, service);
255                     logger.logp(Level.FINE, "HK2Main$HK2ServiceTrackerCustomizer",
256                             "removingService", "removing service = {0}, contract = {1}",
257                             new Object[]{service, contractName});
258 
259                 }
260             } else {
261                 // it was registered by type
262                 Inhabitant<?> inhabitant = habitat.getInhabitantByType(service.getClass());
263                 if (inhabitant != null) {
264                     habitat.remove(inhabitant);
265                 } else {
266                     logger.logp(Level.WARNING, "HK2Main$HK2ServiceTrackerCustomizer",
267                             "removedService", "cannot removed singleton service = {0}", service);
268                 }
269             }
270         }
271 
272     }

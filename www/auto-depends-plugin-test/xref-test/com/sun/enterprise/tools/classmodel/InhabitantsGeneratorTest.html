361   static String callMain(File outputFile, boolean expectOutput, String inhabitantSources, String workingClassPath, Boolean sort) throws Exception {
362     outputFile.delete();
363     
364     inhabitantSources = (null == inhabitantSources) ? toString(getTestClassPathEntries(false)) : inhabitantSources;
365     workingClassPath = (null == workingClassPath) ? toString(getTestClassPathEntries(true)) : workingClassPath;
366     
367     System.setProperty(InhabitantsGenerator.PARAM_INHABITANT_TARGET_FILE, outputFile.getAbsolutePath());
368     System.setProperty(InhabitantsGenerator.PARAM_INHABITANTS_SOURCE_FILES, inhabitantSources);
369     System.setProperty(InhabitantsGenerator.PARAM_INHABITANTS_CLASSPATH, workingClassPath);
370     if (null != sort) {
371       System.setProperty(InhabitantsGenerator.PARAM_INHABITANTS_SORTED, sort.toString());
372     }
373     InhabitantsGenerator.main(null);
374 
375     assertEquals("expect output:" + outputFile, expectOutput, outputFile.exists());
376 
377     if (!outputFile.exists()) {
378       return null;
379     }
380     
381     FileInputStream fis = new FileInputStream(outputFile);
382     try {
383       String output = Utilities.sortInhabitantsDescriptor(toString(fis), false);
384       return output;
385     } finally {
386       fis.close();
387   
388       System.clearProperty(InhabitantsGenerator.PARAM_INHABITANT_TARGET_FILE);
389       System.clearProperty(InhabitantsGenerator.PARAM_INHABITANTS_SOURCE_FILES);
390       System.clearProperty(InhabitantsGenerator.PARAM_INHABITANTS_CLASSPATH);
391       System.clearProperty(InhabitantsGenerator.PARAM_INHABITANTS_SORTED);
392     }
393   }

395   static String toString(InputStream is) throws IOException {
396     StringBuilder sb = new StringBuilder();
397     BufferedReader reader = new BufferedReader(new InputStreamReader(is));
398     String line;
399     while (null != (line = reader.readLine())) {
400       sb.append(line).append("\n");
401     }
402     return sb.toString();
403   }

480   static ArrayList<File> getTestClassPathEntries(boolean worldView) {
481     ArrayList<File> entries = new ArrayList<File>();
482 
483       ClassPath classpath = ClassPath.create(null, false);
484       Set<String> cpSet = classpath.getEntries();
485       for (String entry : cpSet) {
486           if (worldView) {
487               entries.add(new File(entry));
488           } else {
489               if (entry.contains("test-inhabitant-generator") && !entry.contains("test-inhabitant-gen-ifaces")) {
490                   entries.add(new File(entry));
491               } else if (entry.contains("auto-depends-")
492                       && !entry.contains("auto-depends-plugin")) {
493                   if (worldView) {
494                       entries.add(new File(entry));
495                   } else if (!entry.contains("test-inhabitant-gen-ifaces")) {
496                       entries.add(new File(entry));
497                   }
498               } else if (entry.contains("test-classes")) {
499                   entries.add(new File(entry));
500               } else if (entry.contains("hk2-core") && worldView) {
501                   entries.add(new File(entry));
502               } else if (entry.contains("hk2-api")) {
503                   entries.add(new File(entry));
504               }
505           }
506       }
507 
508     if (entries.isEmpty()) {
509       throw new RuntimeException("can't find test-classes in " + cpSet);
510     }
511 
512     logger.info("test classpath for worldview=" + worldView + " is "
513         + ClassPath.create(null, entries));
514 
515     return entries;
516   }

126      * This is useful when the application that's hosting an HK2 environment

137     /**
138      * Parses the inhabitants file (which is represented by {@link InhabitantsScanner}).
139      *
140      * <p>
141      * All the earlier drop/replace commands will be honored during this process.
142      */
143     @SuppressWarnings("unchecked")
144     public void parse(Iterable<InhabitantParser> scanner, HK2Loader classLoader) throws IOException {
145         if (scanner==null)
146             return;
147         
148         Collection<InhabitantProviderInterceptor> interceptors = 
149             (null == serviceLocator) ? Collections.EMPTY_LIST :
150               serviceLocator.getAllServices(InhabitantProviderInterceptor.class);
151         
152         if (serviceLocator==null) {
153         	throw new RuntimeException("ServiceLocator not available");
154         }
155         
156 		DynamicConfigurationService dcs = serviceLocator
157 				.getService(DynamicConfigurationService.class);
158 		
159         for (InhabitantParser parser : scanner) {
160 
161     		DynamicConfiguration config = dcs.createDynamicConfiguration();
162 
163         	final String implName = parser.getImplName();
164 
165         	if (implName==null) 
166         		continue;
167         	
168 			final Iterable<String> indexes = parser.getIndexes();
169 
170 			final Set<String> contracts = new HashSet<String>();
171 
172 			Iterator<String> indexIterator = indexes.iterator();
173 
174 
175 
176 			final MultiMap<String, String> metadata = parser.getMetaData();
177 
178 		
179 			DescriptorImpl descriptor = new DescriptorImpl();
180 			
181 			descriptor.setImplementation(implName);
182 			descriptor.addAdvertisedContract(implName);
183 			descriptor.setScope(Singleton.class.getCanonicalName()); //TODO: scope support?
184 			descriptor.setLoader(classLoader);
185 			
186 			while (indexIterator.hasNext()) {
187 				String index = indexIterator.next();
188 
189 				int idx = index.indexOf(':');
190 				if (-1 != idx) {
191 					// v=contract:name
192 					index = index.substring(0, idx);
193 				}
194 
195 				descriptor.addAdvertisedContract(index);
196 
197 			}
198 			
199 		
200 					Iterator<String> cit = descriptor.getAdvertisedContracts()
201 							.iterator();
202 
203 					while (cit.hasNext()) {
204 						String contract = cit.next();
205 						List<String> name = metadata.get(contract);
206 
207 						if (name.size() != 0) {
208 							descriptor.setName(name.get(0));
209 							break;
210 						}
211 					}
212 			config.bind(descriptor);
213 		    config.commit();      
214         }
215     }

244     /**
245      * Adds the given inhabitant to the habitat, with all its indices.
246      */
247     protected void add(Inhabitant<?> i, InhabitantParser parser) {
248         add(i);
249 
250         for (String v : parser.getIndexes()) {
251             // register inhabitant to the index
252             int idx = v.indexOf(':');
253             if(idx==-1) {
254                 // no name
255                 addIndex(i,v,null);
256             } else {
257                 // v=contract:name
258                 String contract = v.substring(0, idx);
259                 String name = v.substring(idx + 1);
260                 addIndex(i, contract, name);
261             }
262         }
263     }

265     /**
266      * Returns the contract name given a raw index entry.
267      * 
268      * @param v
269      *    the raw index entry in the format of contract[:name]
270      * @param name 
271      *  a StringBuilder used to optionally store the service name
272      *  
273      * @return the contract name
274      */
275     public static String parseIndex(String v, StringBuilder name) {
276       int idx = v.indexOf(':');
277       if (-1 == idx) {
278         return v;
279       } else {
280           // v=contract:name
281           String contract = v.substring(0, idx);
282           if (null != name) {
283             name.append(v.substring(idx + 1));
284           }
285           return contract;
286       }
287     }

319       logger.log(Level.FINE, "removing inhabitant index: {0},{1} from habitat {2}",

321 		DynamicConfigurationService dcs = serviceLocator

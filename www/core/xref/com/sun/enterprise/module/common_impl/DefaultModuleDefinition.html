58  /**
59   * {@link ModuleDefinition} implementation that picks up most of the module

90      // Sahoo changed it to public so that it can be used by osgi-adapter
91      public DefaultModuleDefinition(File location, Attributes attr) throws IOException {
92  
93          classPath.add(location.toURI());
94          
95          Jar jarFile = Jar.create(location);
96          Manifest m = jarFile.getManifest();
97          if(m==null) m = EMPTY_MANIFEST;
98          manifest = m;
99  
100         if (attr==null)
101             attr = manifest.getMainAttributes();
102 
103         // no attributes whatsoever, I just use an empty collection to avoid 
104         // testing for null all the time.
105         if (attr==null) {
106             attr = new Attributes();
107         }
108         this.mainAttributes = attr;
109                 
110         // name
111         if (attr.getValue(ManifestConstants.BUNDLE_NAME)!=null) {
112             name = attr.getValue(ManifestConstants.BUNDLE_NAME);
113         } else {
114             name = location.getName();
115         }
116 
117         // classpath
118         parseClassPath(attr, location.toURI());
119         
120         // class exported...
121         String exported = attr.getValue(ManifestConstants.PKG_EXPORT_NAME);
122         ArrayList<String> tmpList = new ArrayList<String>();
123         for( String token : new Tokenizer(exported,",")) {
124             tmpList.add(token);
125         }
126         publicPkgs = tmpList.toArray(new String[tmpList.size()]);
127         
128         importPolicy = attr.getValue(ManifestConstants.IMPORT_POLICY);
129         lifecyclePolicy = attr.getValue(ManifestConstants.LIFECYLE_POLICY);
130         
131         version = "1.0"; // for now ;-)
132 
133         parseAttributes(attr);
134 
135         jarFile.loadMetadata(metadata);
136     }

138     /**
139      * Parses <tt>{@value ManifestConstants#CLASS_PATH}</tt> from manifest attributes
140      * and updates URI list.
141      */
142     protected void parseClassPath(Attributes attr, URI baseURI) throws IOException {
143         String classpath = attr.getValue(ManifestConstants.CLASS_PATH);
144         for( String classpathElement : new Tokenizer(classpath," ")) {
145             classpathElement = decorateClassPath(classpathElement);
146             URI result;
147             File ref = new File(classpathElement);
148             if (!ref.isAbsolute()) {
149                 try {
150                     result = baseURI.resolve(classpathElement);
151                 } catch (IllegalArgumentException e) {
152                     throw new IllegalArgumentException("Unable to parse Class-Path entry '"+classpath+"' in "+baseURI);
153                 }
154             } else
155                 result = ref.toURI();
156 
157             assert testClassPath(result);
158 
159             // look for /META-INF/services for classloader punch-in
160             File file = new File(result);
161             if (file.exists()) {
162                 Jar jar = Jar.create(file);
163                 Manifest m = jar.getManifest();
164                 if ((m != null) && (m.getMainAttributes().getValue("HK2-Bundle-Name")==null)) {
165                     // if Class-Path refers to other HK2 modules, ignore loading metadata
166                     // When someone makes HK2 module bootable from Java (that is, to have the main method),
167                     // it often needs to refer to other jars. The parent-first classloader delegation
168                     // helps us avoid loading dupliates, but metadata needs to be ignored here.
169                     jar.loadMetadata(metadata);
170                 }
171                 classPath.add(result);
172             } else {
173                 // even if the pointed resource doesn't exist, don't complain by default,
174                 // which is consistent with ClassLoader behavior of silently ignoring URLs
175                 // that doesn't resolve.
176                 // When the assertion is on, testClassPath above would do the warning.
177             }
178         }
179     }
180 
181     /**
182      * Optional error diagnostics performed during the development time
183      * to check if the URL pointed by the path actually exists.
184      */
185     private boolean testClassPath(URI uri) {
186         try {
187             if(uri.getScheme().equals("file")) {
188                 URLConnection c = uri.toURL().openConnection();
189                 if(c.getContentLength()==-1)
190                     LOGGER.warning(uri+" pointed from "+name+" in classpath doesn't exist");
191             }
192         } catch (IOException e) {
193             throw new AssertionError(e); // file is a valid URL
194         }
195         return true; // assertion should always succeed.
196     }

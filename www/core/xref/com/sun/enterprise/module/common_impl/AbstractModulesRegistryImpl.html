175     public ServiceLocator createServiceLocator(String name, InhabitantsParser parser) throws ComponentException {
176         try {
177             ServiceLocator serviceLocator = parser.serviceLocator;
178 
179             for (final Module module : getModules())
180                 parseInhabitants(module, name,parser);
181 
182             populateConfig(serviceLocator);
183 
184             // default modules registry is the one that created the habitat
185             DynamicConfigurationService dcs = serviceLocator.getService(DynamicConfigurationService.class);
186             DynamicConfiguration config = dcs.createDynamicConfiguration();
187             config.bind(BuilderHelper.createConstantDescriptor(this));
188             config.commit();
189             
190             habitats.put(name, serviceLocator);
191             
192             return serviceLocator;
193         } catch (Exception e) {
194             throw new ComponentException("Failed to create a habitat",e);
195         }
196     }

212     /**
213      * Add a new <code>Repository</code> to this registry. From now on
214      * the repository will be used to procure requested module not yet registered
215      * in this registry instance. Repository can be searched in a particular 
216      * order (to accomodate performance requirements like looking at local 
217      * repositories first), a search order (1 to 100) can be specified when 
218      * adding a repository to the registry (1 is highest priority). 
219      * @param repository new repository to attach to this registry
220      * @param weight int value from 1 to 100 to specify the search order
221      */
222     public synchronized void addRepository(Repository repository, int weight) {
223         // check that we don't already have this repository
224         for (Repository repo : repositories.values()) {
225             if (repo.getLocation().equals(repository.getLocation())) {
226                 throw new RuntimeException("repository at " + repository.getLocation() + " already registered");
227             }
228         }
229         while (repositories.containsKey(weight)) {
230             weight++;
231         }
232         repositories.put(weight, repository);
233     }

291     public Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError {
292         Module module;
293         Logger.getAnonymousLogger().fine("this.makeModuleFor("+name+ ", " +
294                 version + ", " + resolve + ") called.");
295         if(parent!=null) {
296             module = parent.makeModuleFor(name,version, resolve);
297             if(module!=null)        return module;
298         }
299         module = modules.get(AbstractFactory.getInstance().createModuleId(name, version));
300         if (module == null && version == null) {
301             Collection<Module> matchingModules = getModules(name);
302             if (!matchingModules.isEmpty()) {
303                 module = matchingModules.iterator().next();
304             }
305         }
306         if (module==null) {
307             module = loadFromRepository(name, version);
308             if (module!=null) {
309                 add(module);
310             }
311         }
312         if (module!=null && resolve) {
313             try {
314                 module.resolve();
315             } catch(Throwable e) {
316                 module.uninstall();
317                 throw new ResolveError(e);
318             }
319         }
320         Logger.getAnonymousLogger().info("this.makeModuleFor("+name+ ", " + version + ") returned " + module);
321         return module;
322     }

324     /**
325      * Find and return a loaded Module that has the package name in its list
326      * of exported interfaces.
327      *
328      * @param packageName the requested implementation package name.
329      * @return the <code>Module</code> instance implementing the package
330      * name or null if not found.
331      * @throws ResolveError if the module dependencies cannot be resolved
332      */
333     public Module makeModuleFor(String packageName) throws ResolveError {
334         if(parent!=null) {
335             Module m = parent.makeModuleFor(packageName);
336             if(m!=null)     return m;
337         }
338 
339         for (Module module : modules.values()) {
340             String[] exportedPkgs = module.getModuleDefinition().getPublicInterfaces();
341             for (String exportedPkg : exportedPkgs) {
342                 if (exportedPkg.equals(packageName)) {
343                     module.resolve();
344                     return module;
345                 }
346             }
347         }
348         return null;
349     }

351     protected Module loadFromRepository(String name, String version) {
352         Set<Integer> keys = repositories.keySet();
353         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
354         sortedKeys.addAll(keys);
355         for (Integer key : sortedKeys) {
356             Repository repo = repositories.get(key);
357             ModuleDefinition moduleDef = repo.find(name, version);
358             if (moduleDef!=null) {
359                 return newModule(moduleDef);
360             }
361         }
362         return null;
363     }

372     /**
373      * Add a new module to this registry. Once added, the module will be 
374      * available through one of the getServiceImplementor methods.
375      * @param newModule the new module
376      */
377     protected void add(Module newModule) {
378         //if (Utils.isLoggable(Level.INFO)) {
379         //    Utils.getDefaultLogger().info("New module " + newModule);
380         //}
381         assert newModule.getRegistry()==this;
382         // see if this module is already added. This can happen
383         // in case of OSGiModulesRegistryImpl which uses SynchronousBundleListener.
384         ModuleId id = AbstractFactory.getInstance().createModuleId(
385                 newModule.getModuleDefinition());
386         if (modules.get(id) != null) return;
387         modules.put(id, newModule);
388 
389         // pick up providers from this module
390         for( ModuleMetadata.Entry spi : newModule.getMetadata().getEntries() ) {
391             for( String name : spi.providerNames )
392                 providers.put(name,newModule);
393         }
394         for (Map.Entry<String, ServiceLocator> entry : habitats.entrySet()) {
395             String name = entry.getKey();
396             ServiceLocator h = entry.getValue();
397             try
398             {
399                 // TODO: should get the inhabitantsParser out of Main instead since
400                 // this could have been overridden
401                 parseInhabitants(newModule, name, createInhabitantsParser(h));
402             }
403             catch (IOException e)
404             {
405                 throw new RuntimeException("Not able to parse inhabitants information");
406             }
407         }
408     }

430     /** 
431      * Returns the list of shared Modules registered in this instance.
432      *
433      * <p>
434      * The returned list will not include the modules defined in the ancestor
435      * {@link AbstractModulesRegistryImpl}s.
436      *
437      * @return an umodifiable list of loaded modules
438      */
439     public Collection<Module> getModules() {
440 
441         // make a copy to avoid synchronizing since this API can be called while
442         // modules are added or removed by other threads.
443         Map<Integer, Repository> repos = new TreeMap<Integer, Repository>();
444         repos.putAll(repositories);
445 
446         // force repository extraction
447         Set<Integer> keys = repos.keySet();
448         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
449         sortedKeys.addAll(keys);
450         for (Integer key : sortedKeys) {
451             Repository repo = repos.get(key);
452             for (ModuleDefinition moduleDef : repo.findAll()) {
453                 if (modules.get(AbstractFactory.getInstance().createModuleId(moduleDef)) == null) {
454                     Module newModule = newModule(moduleDef);
455                     if (newModule != null) {
456                         // When some module can't get installed,
457                         // don't halt proceeding, instead continue
458                         add(newModule);
459                         // don't resolve such modules, we just want to know about them
460                     }
461                 }
462             }
463         }
464         return modules.values();
465     }

565     /**
566      * Registers a running service, this is useful when other components need
567      * to have access to a provider of a service without having to create
568      * a new instance and initialize it.
569      * @param serviceClass the service interface
570      * @param provider the provider of that service.
571      */
572     public <T> void registerRunningService(Class<T> serviceClass, T provider) {
573         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
574         if (rs==null) {
575             rs = new CopyOnWriteArrayList<T>();
576             CopyOnWriteArrayList existing = runningServices.putIfAbsent(serviceClass, rs);
577             if(existing!=null)
578                 rs = existing;
579         }
580         rs.add(provider);
581     }

583     /**
584      * Removes a running service, this is useful when a service instance is no longer
585      * available as a provider of a service.
586      */
587     public <T> boolean unregisterRunningService(Class<T> serviceClass, T provider) {
588         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
589         if (rs==null) {
590             return false;
591         }
592         return rs.remove(provider);
593     }

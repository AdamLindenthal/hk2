83  /**
84   * A module represents a set of resources accessible to third party modules. 
85   * Each module has a module definition which defines its name, its list of 
86   * exported resources and its dependencies to other modules. 
87   * A module instance stores references to the class loader instances giving
88   * access to the module's implementation. 
89   * A module instance belongs to a <code>ModuleRegistry</code> which can be used
90   * to get the list of available modules and/or get particular module
91   * implementation. 
92   * Modules can only satisfy their dependencies within the <code>ModuleRegistry
93   * </code> instance they are registered in.
94   *
95   * @author Jerome Dochez
96   */

123     /* package */ ModuleImpl(ModulesRegistryImpl registry, ModuleDefinition info) {
124         assert registry!=null && info!=null;
125         this.registry = registry;
126         moduleDef = info;
127         for (URI lib : info.getLocations()) {
128             File f = new File(lib);
129             if (f.exists()) {
130                 lastModifieds.put(f.getAbsolutePath(), f.lastModified());
131             }

134         state = ModuleState.NEW;

145         ClassLoaderFacade r=null;
146         if (publicCL!=null)
147             r = publicCL.get();
148         if (r!=null)
149             return r;
150 
151         ClassLoaderFacade facade = AccessController.doPrivileged(new PrivilegedAction<ClassLoaderFacade>() {
152             public ClassLoaderFacade run() {
153                 return new ClassLoaderFacade(getPrivateClassLoader());
154             }
155         });
156         facade.setPublicPkgs(moduleDef.getPublicInterfaces());
157         publicCL = new WeakReference<ClassLoaderFacade>(facade);
158         return facade;
159     }

161     /**
162      * Return the private class loader for this module. This class loader will
163      * be loading all the classes which are not explicitely exported in the 
164      * module definition
165      * @return the private <code>ClassLoader</code> instance
166      */
167     /*package*/ ModuleClassLoader getPrivateClassLoader() {
168         if (privateCL==null) {
169             synchronized(this) {
170                 if(privateCL==null) {
171                     URI[] locations = moduleDef.getLocations();
172                     URL[] urlLocations = new URL[locations.length];
173                     for (int i=0;i<locations.length;i++) {
174                         try {
175                             urlLocations[i] = locations[i].toURL();
176                         } catch(MalformedURLException e) {
177                             e.printStackTrace();
178                             return null;
179                         }
180                     }
181                     final URL[] urls = urlLocations.clone();
182                     privateCL = AccessController.doPrivileged(new PrivilegedAction<ModuleClassLoader>() {
183                         public ModuleClassLoader run() {
184                             return new ModuleClassLoader(
185                                 ModuleImpl.this,
186                                 urls,
187                                 registry.getParentClassLoader());
188                         }
189                     });
190                 }
191             }
192         }
193         return privateCL;
194     }

259     /**
260      * Trigger manual refresh mechanism, the module will check all its 
261      * URLs and generate change events if any of them has changed. This 
262      * will allow the owning registry to force a module upgrade at next 
263      * module request.
264      */
265     public void refresh() {
266         URI[] urls = moduleDef.getLocations();
267         boolean notify = false;
268         for (URI lib : urls) {
269             File f = new File(lib);
270             if (f.exists() && lastModifieds.containsKey(f.getAbsolutePath())) {
271                 if (lastModifieds.get(f.getAbsolutePath()) !=f.lastModified()) {
272                     //Utils.getDefaultLogger().info("Changed : " + this);
273                     notify = true;
274                     break;
275                 }
276             }
277         }
278         if(notify) {
279             fireChangeEvent();
280         }
281     }

294

312     /**
313      * Ensure that this module is {@link ModuleState#RESOLVED resolved}.
314      *
315      * <p>
316      * If the module is already resolved, this method does nothing.
317      * Otherwise, iterate over all declared ModuleDependency instances and use the
318      * associated <code>ModuleRegistry</code> to resolve it. After successful
319      * completion of this method, the module state is 
320      * {@link ModuleState#RESOLVED}.
321      *
322      * @throws com.sun.enterprise.module.ResolveError if any of the declared dependency of this module
323      * cannot be satisfied
324      */
325     public synchronized void resolve() throws ResolveError {
326         
327         // already resolved ?
328         if (state==ModuleState.ERROR)
329             throw new ResolveError("Module " + getName() + " is in ERROR state");
330         if (state.compareTo(ModuleState.RESOLVED)>=0)
331             return;
332 
333         if (state==ModuleState.PREPARING) {
334             Utils.identifyCyclicDependency(this, Logger.getAnonymousLogger());
335             throw new ResolveError("Cyclic dependency with " + getName());
336         }
337         state = ModuleState.PREPARING;
338         
339         if (moduleDef.getImportPolicyClassName()!=null) {
340             try {
341                 Class<ImportPolicy> importPolicyClass = (Class<ImportPolicy>) getPrivateClassLoader().loadClass(moduleDef.getImportPolicyClassName());
342                 ImportPolicy importPolicy = importPolicyClass.newInstance();
343                 importPolicy.prepare(this);
344             } catch(ClassNotFoundException e) {
345                 state = ModuleState.ERROR;
346                 throw new ResolveError(e);
347             } catch(java.lang.InstantiationException e) {
348                 state = ModuleState.ERROR;
349                 throw new ResolveError(e);
350             } catch(IllegalAccessException e) {
351                 state = ModuleState.ERROR;
352                 throw new ResolveError(e);
353             }
354         }
355         for (ModuleDependency dependency : moduleDef.getDependencies()) {
356             ModuleImpl depModule = (ModuleImpl)registry.makeModuleFor(dependency.getName(), dependency.getVersion());
357             if (depModule==null) {
358                 state = ModuleState.ERROR;                
359                 throw new ResolveError(dependency + " referenced from " 
360                         + moduleDef.getName() + " is not resolved");
361             }
362 
363             //if (Utils.isLoggable(Level.INFO)) {
364             //    Utils.getDefaultLogger().info("For module" + getName() + " adding new dependent " + module.getName());
365             //}
366             dependencies.add(depModule);
367         }
368 
369         // once we have proper import/export filtering for modules, we can
370         // build a look-up table to improve performance
371 
372         // build up the complete list of transitive dependency modules, without any duplication,
373         // in a breadth-first fashion. The reason we do this in breadth-first is to reduce
374         // the search time based on the assumption that classes tend to be discovered in close dependencies. 
375         List<ModuleImpl> transitiveDependencies = new ArrayList<ModuleImpl>();
376         Set<ModuleImpl> transitiveDependenciesSet = new HashSet<ModuleImpl>();
377         LinkedList<ModuleImpl> q = new LinkedList<ModuleImpl>();
378         q.addAll(dependencies);
379         while(!q.isEmpty()) {
380             ModuleImpl m = q.removeFirst();
381             if(transitiveDependenciesSet.add(m)) {
382                 // first time visited
383                 transitiveDependencies.add(m);
384                 m.resolve();
385                 q.addAll(m.dependencies);
386             }
387         }
388 
389         for (ModuleImpl m : transitiveDependencies) {
390             getPrivateClassLoader().addDelegate(m.getClassLoader());
391         }
392 
393         //Logger.global.info("Module " + getName() + " resolved");
394         state = ModuleState.RESOLVED;
395         for (ModuleLifecycleListener l : registry.getLifecycleListeners()) {
396             l.moduleResolved(this);
397         }
398     }

400     /**
401      * Forces module startup. In most cases, the runtime will take care 
402      * of starting modules when they are first used. There could be cases where
403      * code need to manually start a sub module. Invoking this method will 
404      * move the module to the {@link ModuleState#READY ModuleState.READY}, the 
405      * {@link LifecyclePolicy#start Lifecycle.start} method will be invoked.
406      */
407     public void start() throws ResolveError {
408         
409         if (state==ModuleState.READY)
410             return;
411         
412         // ensure RESOLVED state
413         resolve();      
414         
415         for (Module subModules : dependencies) {
416             subModules.start();
417         }
418 
419         // time to initialize the lifecycle instance
420         if (moduleDef.getLifecyclePolicyClassName()!=null) {
421             try {
422                 Class<LifecyclePolicy> lifecyclePolicyClass = (Class<LifecyclePolicy>) getPrivateClassLoader().loadClass(moduleDef.getLifecyclePolicyClassName());
423                 lifecyclePolicy = lifecyclePolicyClass.newInstance();
424             } catch(ClassNotFoundException e) {
425                 state = ModuleState.ERROR;
426                 throw new ResolveError("ClassNotFound : " + e.getMessage(), e);
427             } catch(java.lang.InstantiationException e) {
428                 state = ModuleState.ERROR;
429                 throw new ResolveError(e);
430             } catch(IllegalAccessException e) {
431                 state = ModuleState.ERROR;
432                 throw new ResolveError(e);
433             }
434         }
435         if (lifecyclePolicy!=null) {
436             lifecyclePolicy.start(this);
437         }
438         state = ModuleState.READY;
439 
440         //Logger.global.info("Module " + getName() + " started");
441 
442         // module started. notify listeners
443         for (ModuleLifecycleListener listener : registry.getLifecycleListeners())
444             listener.moduleStarted(this);
445     }

447     /** 
448      * Forces module stop. In most cases, the runtime will take care of stopping
449      * modules when the last module user released its interest. However, in 
450      * certain cases, it may be interesting to manually stop the module. 
451      * Stopping the module means that the module is removed from the registry, 
452      * the class loader references are released (note : the class loaders will
453      * only be released if all instances of any class loaded by them are gc'ed). 
454      * If a <code>LifecyclePolicy</code> for this module is defined, the 
455      * {@link LifecyclePolicy#stop(Module) Lifecycle.stop(Module)}
456      * method will be called and finally the module state will be 
457      * returned to {@link ModuleState#NEW ModuleState.NEW}.
458      *
459      * @return true if unloading was successful
460      */
461     public boolean stop() {
462 
463         if (sticky) {
464             return false;
465         } 
466         
467         if (lifecyclePolicy!=null) {
468             lifecyclePolicy.stop(this);
469             lifecyclePolicy=null;
470         }
471         
472         detach();
473         
474         // we do NOT stop our sub modules which are shared...
475         for (ModuleImpl subModule : dependencies) {
476             if (!subModule.isShared()) {
477                 subModule.stop();
478             }
479         }
480         // release all sub modules class loaders
481         privateCL = null;
482         publicCL = null;
483         dependencies.clear();
484         
485         state = ModuleState.NEW;
486 
487         // notify interested listeners that I have stopped.
488         for (ModuleLifecycleListener listener : registry.getLifecycleListeners()) {
489             listener.moduleStopped(this);
490         }
491         return true;
492     }

602     public Iterable<Class> getProvidersClass(String name) {
603         List<Class> r = serviceClasses.get(name);
604         if(r!=null) return r;
605 
606         // the worst case scenario in the race situation is we end up creating the same list twice,
607         // which is not a big deal.
608 
609         for( String provider : getMetadata().getEntry(name).providerNames) {
610             if(r==null)
611                 r = new ArrayList<Class>();
612             try {
613                 r.add(getPrivateClassLoader().loadClass(provider));
614             } catch (ClassNotFoundException e) {
615                 LogHelper.getDefaultLogger().log(Level.SEVERE, "Failed to load "+provider+" from "+getName(),e);
616             }
617         }
618 
619         if(r==null)
620             r = Collections.emptyList();
621 
622         serviceClasses.put(name, r);
623         return r;
624     }

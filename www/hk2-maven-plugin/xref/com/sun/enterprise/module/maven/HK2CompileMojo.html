70  /**
71   * Replacement of the default javac mojo that uses APT, so that we can not only
72   * compile source code but also generate HK2 related metadata at the same time.
73   *
74   * @goal hk2-compile
75   * @phase compile
76   * @author Kohsuke Kawaguchi
77   */
78  public class HK2CompileMojo extends CompilerMojo {
79      /**
80       * @parameter expression="${project}"
81       * @required
82       * @readonly
83       */
84      private MavenProject project;
85  
86      /**
87       * Additional jars that contain annotation processors.
88       *
89       * @parameter
90       */
91      public Processor[] processors;
92  
93      /**
94       * @component
95       */
96      private ArtifactFactory artifactFactory;
97  
98      /**
99       * @component
100      */
101     private ArtifactResolver artifactResolver;
102 
103     /**
104      * @parameter expression="${localRepository}"
105      * @required
106      */
107     private ArtifactRepository localRepository;
108 
109     /**
110      * @component
111      */
112     private ArtifactMetadataSource metadataSource;
113 
114     public void execute() throws MojoExecutionException, CompilationFailureException {
115         // overwrite the compilerId value. This seems to be the only way to
116         //do so without touching the copied files.
117         setField("compilerId", "hk2-apt");
118 
119         // invoke APT with the known set of annotation processors that we care about.
120         AptInvoker old = AptInvoker.replace(new AptInvoker() {
121             List<CompilerError> compileInProcess(String[] args) throws CompilerException {
122                 com.sun.tools.apt.Main aptTool = new com.sun.tools.apt.Main();
123                 int r = aptTool.process(
124                     new CompositeAnnotationProcessorFactory(
125                         new MetainfServiceGenerator(),
126                         new AnnotationProcessorFactoryImpl(),
127                         loadAdditionalAnnotationProcessors()
128                     ), new PrintWriter(System.out,true),args);
129                 if(r!=0)
130                     throw new CompilerException("APT failed: "+r);
131 
132                 // TODO: should I try to parse the output?
133                 return Collections.emptyList();
134             }
135         });
136         try {
137             super.execute();
138         } finally {
139             AptInvoker.replace(old);
140         }
141 
142         // TODO: ideally we should do this in the AptCompiler class, but I don't know
143         // how to get MavenProject injected there
144         project.getCompileSourceRoots().add(new File(project.getBasedir(),"target/apt-generated-sources").getAbsolutePath());
145     }
146 
147     /**
148      * Loads additional annotation processors specified in &lt;processor> configuration element.
149      *
150      * <p>
151      * We need to resolve their transitive dependencies as well, because otherwise they won't load.
152      */
153     private AnnotationProcessorFactory loadAdditionalAnnotationProcessors() throws CompilerException {
154         if(processors==null)
155             return new CompositeAnnotationProcessorFactory();   // none
156 
157         Set<Artifact> processorArtifacts = new HashSet<Artifact>();
158         for (Processor p : processors) {
159             processorArtifacts.add(p.createArtifact(artifactFactory));
160         }
161 
162         // TODO: perhaps we should create one URLClassLoader per each annotation processor to
163         // ensure sufficient isolation?
164         Set<URL> classpaths = new LinkedHashSet<URL>();
165         try {
166             ArtifactResolutionResult result = artifactResolver.resolveTransitively(
167                     processorArtifacts, project.getArtifact(),
168                     localRepository,
169                     this.project.getRemoteArtifactRepositories(),
170                     metadataSource, new ScopeArtifactFilter("runtime"));
171             for( Artifact a : (Set<Artifact>)result.getArtifacts()) {
172                 classpaths.add(a.getFile().toURL());
173             }
174         } catch (AbstractArtifactResolutionException e) {
175             throw new CompilerException("Failed to resolve annotation processors",e);
176         } catch (MalformedURLException e) {
177             throw new CompilerException("Failed to resolve annotation processors",e);
178         }
179 
180         // load them into the classloader
181         ClassLoader cl = new URLClassLoader(classpaths.toArray(new URL[classpaths.size()]),getClass().getClassLoader());
182 
183         List<AnnotationProcessorFactory> factories = new ArrayList<AnnotationProcessorFactory>();
184         for(AnnotationProcessorFactory apf : ServiceFinder.find(AnnotationProcessorFactory.class,cl)) {
185             factories.add(apf);
186             getLog().info("Picked up annotation processor "+apf.getClass().getName());
187         }
188 
189         return new CompositeAnnotationProcessorFactory(factories);
190     }

192     private void setField(String name, String value) {
193         try {
194             Field field = AbstractCompilerMojo.class.getDeclaredField(name);
195             field.setAccessible(true);
196             field.set(this, value);
197         } catch (NoSuchFieldException e) {
198             throw new AssertionError(e); // impossible
199         } catch (IllegalAccessException e) {
200             throw new AssertionError(e); // impossible
201         }
202     }

53  /** A Rudimentary Mojo to create scripts for Windows and non-Windows Platforms,
54   *  based on an intuitive file in a form similar to registry file.
55   * @goal createscripts
56   * @author Kedar Mhaswade (km@dev.java.net)
57   */
58  public final class ScriptCreatorMojo extends AbstractMojo {
59  
60      /**
61       * @parameter expression="${project}"
62       * @required
63       * @readonly
64       */
65      private MavenProject project;
66      
67      /* start: define the parameters */
68      /**
69       * @parameter
70       */
71      private String[] configFiles;
72      /**
73       * @parameter
74       */
75      private String[] destDirs;
76      /**
77       * @parameter
78       */
79      private String destDir;
80  
81      /**
82       * @parameter default-value=false
83       */
84      private boolean buildPlatformSpecific;
85      
86      /* end:   define the parameters */
87  
88      private final Log log = getLog();
89  
90      public void execute() throws MojoExecutionException {
91          //dumpProperties(project, log);
92          String basedir = project.getBasedir().getAbsolutePath();
93          if (configFiles == null || configFiles.length == 0) {
94              throw new MojoExecutionException("Required parameter <configFiles> not specified or no <configFile> specified, exiting ...");
95          }
96          if (destDir == null && destDirs == null) {
97              throw new MojoExecutionException("Either <destDir> or <destDirs> should be specified");
98          }
99          Map<String, String> sd = new LinkedHashMap<String, String>();
100         if (destDir != null) { // this has higher precedence
101             for (String file : configFiles) {
102                 sd.put(file, destDir);
103             }
104         } else {
105             if (destDirs.length != configFiles.length) 
106                 throw new MojoExecutionException("Number of <configFile>s and <destDir>s should be same");
107             for (int i = 0 ; i < configFiles.length ; i ++) {
108                 sd.put(configFiles[i], destDirs[i]);
109             } 
110         }
111         for (String file : configFiles) {
112             File cf = new File(basedir, file);
113             File dd = new File(basedir, sd.get(file));
114             try {
115                 Properties env;
116                 if (buildPlatformSpecific) {
117                     env = createBuildPlatformSpecificEnvironment();
118                     new ScriptCreator(env).create();
119                 } else {
120                     env = createEnvironment(cf, dd, WINDOWS);
121                     new ScriptCreator(env).create();
122                     env = createEnvironment(cf, dd, UNIX);
123                     new ScriptCreator(env).create();
124                 }
125             } catch (Exception e) {
126                 throw new MojoExecutionException(e.getMessage());
127             }
128         }
129     }
130 
131     private static void dumpProperties(MavenProject pro, Log log) {
132         Properties p = pro.getProperties();
133         log.debug("project: " + pro.getName());
134         log.debug(p.toString());
135     }
136     private Properties createBuildPlatformSpecificEnvironment() {
137         return (project.getProperties()); //for now ...
138     }
139     
140     private Properties createEnvironment(File src, File destDir, String id) {
141         Properties env = project.getProperties();
142         env.put(SRC, src.getAbsolutePath());
143         if (WINDOWS.equals(id)) {
144             env.put(OPERATING_SYSTEM, WINDOWS);
145             env.put(SCRIPT_HOME_TOKEN, WINDOWS_SCRIPT_HOME_VALUE);
146             String destFileName = src.getName() + WIN_SCRIPT_EXTENSION;
147             File destFilePath = new File(destDir, destFileName);
148             env.put(DEST, destFilePath.getAbsolutePath());
149         } else if (UNIX.equals(id)) {
150             env.put(OPERATING_SYSTEM, UNIX);
151             env.put(SCRIPT_HOME_TOKEN, UNIX_SCRIPT_HOME_VALUE);
152             String destFileName = src.getName();
153             File destFilePath = new File(destDir, destFileName);
154             env.put(DEST, destFilePath.getAbsolutePath());            
155         } else { //defaults to no-platform
156         }
157         return ( env );
158     }

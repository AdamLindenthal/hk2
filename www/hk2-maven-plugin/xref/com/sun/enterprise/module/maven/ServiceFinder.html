61  /**
62   * A simple service-provider lookup mechanism.  A <i>service</i> is a
63   * well-known set of interfaces and (usually abstract) classes.  A <i>service
64   * provider</i> is a specific implementation of a service.  The classes in a
65   * provider typically implement the interfaces and subclass the classes defined
66   * in the service itself.  Service providers may be installed in an
67   * implementation of the Java platform in the form of extensions, that is, jar
68   * files placed into any of the usual extension directories.  Providers may
69   * also be made available by adding them to the applet or application class
70   * path or by some other platform-specific means.
71   * <p/>
72   * <p> In this lookup mechanism a service is represented by an interface or an
73   * abstract class.  (A concrete class may be used, but this is not
74   * recommended.)  A provider of a given service contains one or more concrete
75   * classes that extend this <i>service class</i> with data and code specific to
76   * the provider.  This <i>provider class</i> will typically not be the entire
77   * provider itself but rather a proxy that contains enough information to
78   * decide whether the provider is able to satisfy a particular request together
79   * with code that can create the actual provider on demand.  The details of
80   * provider classes tend to be highly service-specific; no single class or
81   * interface could possibly unify them, so no such class has been defined.  The
82   * only requirement enforced here is that provider classes must have a
83   * zero-argument constructor so that they may be instantiated during lookup.
84   * <p/>
85   * <p> A service provider identifies itself by placing a provider-configuration
86   * file in the resource directory <tt>META-INF/services</tt>.  The file's name
87   * should consist of the fully-qualified name of the abstract service class.
88   * The file should contain a list of fully-qualified concrete provider-class
89   * names, one per line.  Space and tab characters surrounding each name, as
90   * well as blank lines, are ignored.  The comment character is <tt>'#'</tt>
91   * (<tt>0x23</tt>); on each line all characters following the first comment
92   * character are ignored.  The file must be encoded in UTF-8.
93   * <p/>
94   * <p> If a particular concrete provider class is named in more than one
95   * configuration file, or is named in the same configuration file more than
96   * once, then the duplicates will be ignored.  The configuration file naming a
97   * particular provider need not be in the same jar file or other distribution
98   * unit as the provider itself.  The provider must be accessible from the same
99   * class loader that was initially queried to locate the configuration file;
100  * note that this is not necessarily the class loader that found the file.
101  * <p/>
102  * <p> <b>Example:</b> Suppose we have a service class named
103  * <tt>java.io.spi.CharCodec</tt>.  It has two abstract methods:
104  * <p/>
105  * <pre>
106  *   public abstract CharEncoder getEncoder(String encodingName);
107  *   public abstract CharDecoder getDecoder(String encodingName);
108  * </pre>
109  * <p/>
110  * Each method returns an appropriate object or <tt>null</tt> if it cannot
111  * translate the given encoding.  Typical <tt>CharCodec</tt> providers will
112  * support more than one encoding.
113  * <p/>
114  * <p> If <tt>sun.io.StandardCodec</tt> is a provider of the <tt>CharCodec</tt>
115  * service then its jar file would contain the file
116  * <tt>META-INF/services/java.io.spi.CharCodec</tt>.  This file would contain
117  * the single line:
118  * <p/>
119  * <pre>
120  *   sun.io.StandardCodec    # Standard codecs for the platform
121  * </pre>
122  * <p/>
123  * To locate an codec for a given encoding name, the internal I/O code would
124  * do something like this:
125  * <p/>
126  * <pre>
127  *   CharEncoder getEncoder(String encodingName) {
128  *       for( CharCodec cc : ServiceFinder.find(CharCodec.class) ) {
129  *           CharEncoder ce = cc.getEncoder(encodingName);
130  *           if (ce != null)
131  *               return ce;
132  *       }
133  *       return null;
134  *   }
135  * </pre>
136  * <p/>
137  * The provider-lookup mechanism always executes in the security context of the
138  * caller.  Trusted system code should typically invoke the methods in this
139  * class from within a privileged security context.
140  *
141  * @author Mark Reinhold
142  * @version 1.11, 03/12/19
143  * @since 1.3
144  */
145 final class ServiceFinder<T> implements Iterable<T> {

147     private static final String prefix = "META-INF/services/";
148 
149     private final Class<T> serviceClass;
150     private final @Nullable
151     ClassLoader classLoader;

153     /**
154      * Locates and incrementally instantiates the available providers of a
155      * given service using the given class loader.
156      * <p/>
157      * <p> This method transforms the name of the given service class into a
158      * provider-configuration filename as described above and then uses the
159      * <tt>getResources</tt> method of the given class loader to find all
160      * available files with that name.  These files are then read and parsed to
161      * produce a list of provider-class names.  The iterator that is returned
162      * uses the given class loader to lookup and then instantiate each element
163      * of the list.
164      * <p/>
165      * <p> Because it is possible for extensions to be installed into a running
166      * Java virtual machine, this method may return different results each time
167      * it is invoked. <p>
168      *
169      * @param service The service's abstract service class
170      * @param loader  The class loader to be used to load provider-configuration files
171      *                and instantiate provider classes, or <tt>null</tt> if the system
172      *                class loader (or, failing that the bootstrap class loader) is to
173      *                be used
174      * @throws ServiceConfigurationError If a provider-configuration file violates the specified format
175      *                                   or names a provider class that cannot be found and instantiated
176      * @see #find(Class)
177      */
178     public static <T> ServiceFinder<T> find(@NotNull Class<T> service, @Nullable ClassLoader loader) {
179         return new ServiceFinder<T>(service,loader);
180     }

182     /**
183      * Locates and incrementally instantiates the available providers of a
184      * given service using the context class loader.  This convenience method
185      * is equivalent to
186      * <p/>
187      * <pre>
188      *   ClassLoader cl = Thread.currentThread().getContextClassLoader();
189      *   return Service.providers(service, cl);
190      * </pre>
191      *
192      * @param service The service's abstract service class
193      *
194      * @throws ServiceConfigurationError If a provider-configuration file violates the specified format
195      *                                   or names a provider class that cannot be found and instantiated
196      * @see #find(Class, ClassLoader)
197      */
198     public static <T> ServiceFinder<T> find(Class<T> service) {
199         return find(service, AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
200             @Override
201             public ClassLoader run() {
202                 return Thread.currentThread().getContextClassLoader();
203             }
204         }));
205     }

213      * Returns discovered objects incrementally.
214      *
215      * @return An <tt>Iterator</tt> that yields provider objects for the given

259     /**
260      * Parse a single line from the given configuration file, adding the name
261      * on the line to both the names list and the returned set iff the name is
262      * not already a member of the returned set.
263      */
264     private static int parseLine(Class service, URL u, BufferedReader r, int lc,
265                                  List<String> names, Set<String> returned)
266         throws IOException, ServiceConfigurationError {
267         String ln = r.readLine();
268         if (ln == null) {
269             return -1;
270         }
271         int ci = ln.indexOf('#');
272         if (ci >= 0) ln = ln.substring(0, ci);
273         ln = ln.trim();
274         int n = ln.length();
275         if (n != 0) {
276             if ((ln.indexOf(' ') >= 0) || (ln.indexOf('\t') >= 0))
277                 fail(service, u, lc, "Illegal configuration-file syntax");
278             int cp = ln.codePointAt(0);
279             if (!Character.isJavaIdentifierStart(cp))
280                 fail(service, u, lc, "Illegal provider-class name: " + ln);
281             for (int i = Character.charCount(cp); i < n; i += Character.charCount(cp)) {
282                 cp = ln.codePointAt(i);
283                 if (!Character.isJavaIdentifierPart(cp) && (cp != '.'))
284                     fail(service, u, lc, "Illegal provider-class name: " + ln);
285             }
286             if (!returned.contains(ln)) {
287                 names.add(ln);
288                 returned.add(ln);
289             }
290         }
291         return lc + 1;
292     }

294     /**
295      * Parse the content of the given URL as a provider-configuration file.
296      *
297      * @param service  The service class for which providers are being sought;
298      *                 used to construct error detail strings
299      * @param u        The URL naming the configuration file to be parsed
300      * @param returned A Set containing the names of provider classes that have already
301      *                 been returned.  This set will be updated to contain the names
302      *                 that will be yielded from the returned <tt>Iterator</tt>.
303      * @return A (possibly empty) <tt>Iterator</tt> that will yield the
304      *         provider-class names in the given configuration file that are
305      *         not yet members of the returned set
306      * @throws ServiceConfigurationError If an I/O error occurs while reading from the given URL, or
307      *                                   if a configuration-file format error is detected
308      */
309     @SuppressWarnings({"StatementWithEmptyBody"})
310     private static Iterator<String> parse(Class service, URL u, Set<String> returned)
311         throws ServiceConfigurationError {
312         InputStream in = null;
313         BufferedReader r = null;
314         ArrayList<String> names = new ArrayList<String>();
315         try {
316             in = u.openStream();
317             r = new BufferedReader(new InputStreamReader(in, "utf-8"));
318             int lc = 1;
319             while ((lc = parseLine(service, u, r, lc, names, returned)) >= 0) ;
320         } catch (IOException x) {
321             fail(service, ": " + x);
322         } finally {
323             try {
324                 if (r != null) r.close();
325                 if (in != null) in.close();
326             } catch (IOException y) {
327                 fail(service, ": " + y);
328             }
329         }
330         return names.iterator();
331     }

334     /**
335      * Private inner class implementing fully-lazy provider lookup
336      */
337     private static class LazyIterator<T> implements Iterator<T> {
338         Class<T> service;
339         @Nullable ClassLoader loader;
340         Enumeration<URL> configs = null;
341         Iterator<String> pending = null;
342         Set<String> returned = new TreeSet<String>();
343         String nextName = null;
344         URL currentConfig = null;
345 
346         private LazyIterator(Class<T> service, ClassLoader loader) {
347             this.service = service;
348             this.loader = loader;
349         }
350 
351         public boolean hasNext() throws ServiceConfigurationError {
352             if (nextName != null) {
353                 return true;
354             }
355             if (configs == null) {
356                 try {
357                     String fullName = prefix + service.getName();
358                     if (loader == null)
359                         configs = ClassLoader.getSystemResources(fullName);
360                     else
361                         configs = loader.getResources(fullName);
362                 } catch (IOException x) {
363                     fail(service, ": " + x);
364                 }
365             }
366             while ((pending == null) || !pending.hasNext()) {
367                 if (!configs.hasMoreElements()) {
368                     return false;
369                 }
370                 currentConfig = configs.nextElement();
371                 pending = parse(service, currentConfig, returned);
372             }
373             nextName = pending.next();
374             return true;
375         }
376 
377         public T next() throws ServiceConfigurationError {
378             if (!hasNext()) {
379                 throw new NoSuchElementException();
380             }
381             String cn = nextName;
382             nextName = null;
383             try {
384                 return service.cast(Class.forName(cn, true, loader).newInstance());
385             } catch (ClassNotFoundException x) {
386                 fail(service,
387                     "Provider " + cn + " is specified in "+currentConfig+" but not found");
388             } catch (Exception x) {
389                 fail(service,
390                     "Provider " + cn + " is specified in "+currentConfig+"but could not be instantiated: " + x, x);
391             }
392             return null;    /* This cannot happen */
393         }
394 
395         public void remove() {
396             throw new UnsupportedOperationException();
397         }
398     }

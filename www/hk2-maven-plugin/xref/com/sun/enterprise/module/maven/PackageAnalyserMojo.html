59  /***
60   * Analyses bundle manifest entries in a repository and
61   * generates various statistics like:
62   * no. of  bundles, no. of exported packages, package wiring details and
63   * split-packages.
64   *
65   * @author Sanjeeb.Sahoo@Sun.COM
66   * @goal analyse-packages
67   * @requiresDependencyResolution compile
68   * @requiresProject
69   */
70  public class PackageAnalyserMojo extends AbstractMojo {
71      private static Logger logger = Logger.getAnonymousLogger();
72  
73      /***
74       * The maven project.
75       *
76       * @parameter expression="${project}"
77       * @required
78       * @readonly
79       */
80      protected MavenProject project;
81  
82      /***
83       * The maven artifact.
84       *
85       * @parameter expression="${project.artifact}"
86       * @required
87       * @readonly
88       */
89      protected Artifact artifact;
90  
91      /***
92       * @component
93       */
94      private ArtifactResolver artifactResolver;
95  
96      /***
97       * @parameter expression="${localRepository}"
98       */
99      private ArtifactRepository localRepository;
100 
101     /***
102      * @component
103      */
104     private ArtifactFactory artifactFactory;
105 
106     public void execute() throws MojoExecutionException, MojoFailureException {
107         logger.logp(Level.INFO, "PackageAnalyserMojo", "execute", "Analysing modules");
108         try {
109             HK2Factory.initialize();
110             MavenProjectRepository repo = new MavenProjectRepository(
111                     project, artifactResolver, localRepository, artifactFactory);
112             repo.initialize();
113             PackageAnalyser analyser = new PackageAnalyser(repo, logger);
114             Collection<PackageAnalyser.Wire> wires = analyser.analyseWirings();
115             StringBuilder sb = new StringBuilder("Wiring details are given below:\n");
116             for (PackageAnalyser.Wire w : wires) {
117                 sb.append(w + "\n");
118             }
119             sb.append("Total number of wires = " + wires.size() + "\n");
120             sb.append("Split-Package details are given below:\n");
121             Collection<PackageAnalyser.SplitPackage> splitPkgs = analyser.findDuplicatePackages();
122             for (PackageAnalyser.SplitPackage p : splitPkgs) sb.append(p+"\n");
123             sb.append("Total number of Split Packages = " + splitPkgs.size() + "\n");
124 
125             int totalUnusedPkgs = 0;
126             for (PackageAnalyser.Bundle b : analyser.bundles) {
127                 Collection<PackageAnalyser.PackageCapability> unusedPackages = analyser.findUnusedExports(b);
128                 if (!unusedPackages.isEmpty()) {
129                     sb.append("<Bundle name=" + b.getName()+", totalUnusedPkgs = " + unusedPackages.size() + "> \n" );
130                     for (PackageAnalyser.PackageCapability p : unusedPackages) sb.append("\t" + p + "\n");
131                     sb.append("</Bundle>\n");
132                 }
133                 totalUnusedPkgs += unusedPackages.size();
134             }
135             sb.append("Total number of Unused Packages = " + totalUnusedPkgs);
136 
137             sb.append("******** GROSS STATISTICS *********\n");
138             sb.append("Total number of bundles in this repository: " + analyser.findAllBundles().size()+"\n");
139             sb.append("Total number of wires = " + wires.size() + "\n");
140             Collection<String> exportedPkgs = analyser.findAllExportedPackageNames();
141             sb.append("Total number of exported packages = " + exportedPkgs.size() + "\n");
142             sb.append("Total number of split-packages = " + splitPkgs.size()+"\n");
143             sb.append("Total number of unused-packages = " + totalUnusedPkgs +"\n");
144 
145             logger.logp(Level.INFO, "PackageAnalyserMojo", "execute", "{0}", new Object[]{sb});
146             String reportFilePath =
147                     System.getProperty("WiringReportPath",
148                             System.getProperty("java.io.tmpdir")+ File.separator + "wires.xml");
149             analyser.generateWiringReport(exportedPkgs, wires, new PrintStream(new FileOutputStream(new File(reportFilePath))));
150             System.out.println("Wiring reported can be found at " + reportFilePath);
151         } catch (IOException e) {
152             throw new MojoExecutionException("Unexpected exception", e);
153         }
154     }

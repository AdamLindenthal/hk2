53  /**
54   * Launches the module system by using all the dependencies as modules.
55   *
56   * <p>
57   * Not specifying "@execute phase=compile" so that it runs
58   * quickly even with a large project.
59   * </p>
60   * 
61   * @requiresProject
62   * @goal run
63   * @requiresDependencyResolution runtime
64   * @phase compile
65   * @execute phase=compile
66   * 
67   * @author Kohsuke Kawaguchi
68   */
69  public class RunMojo extends AbstractMojo {
70  
71      /**
72       * The maven project.
73       *
74       * @parameter expression="${project}"
75       * @required
76       * @readonly
77       */
78      private MavenProject project;
79  
80      /**
81       * @component
82       */
83      private ArtifactResolver artifactResolver;
84  
85      /**
86       * @parameter expression="${localRepository}"
87       */
88      private ArtifactRepository localRepository;
89  
90      /**
91       * @component
92       */
93      private ArtifactFactory artifactFactory;
94  
95      /**
96       * The root directory of the launched module system.
97       *
98       * @parameter expression="${rootDir}" default-value="${project.baseDir}"
99       */
100     private File rootDir;
101 
102     /**
103      * Command-line options to be passed to {@link StartupContext}.
104      *
105      * @parameter
106      */
107     private String[] args = new String[0];
108 
109     /**
110      * groupId:artifactId of the module that includes {@link ModuleStartup}.
111      * If omitted, the module list is searched to find one.
112      *
113      * @parameter expression="${mainModule}"
114      */
115     private String mainModule;
116 
117     public void execute() throws MojoExecutionException, MojoFailureException {
118         if(project.getPackaging().equals("pom")) {
119             // we want to run "mvn test" from parent module and run tests of all descendants,
120             // but there's nothing to test in POM module itself.
121             getLog().info("Skipping");
122             return;
123         }
124 
125         // sanity check
126         if(project.getPackaging().equals("hk2-jar") && (project.getArtifact()==null || project.getArtifact().getFile()==null)) {
127             getLog().warn("This project isn't compiled yet. Perhaps you meant 'mvn compile hk2:run'?");
128         }
129 
130         if(rootDir==null)
131             rootDir = project.getBasedir();
132     
133         try {
134             StartupContext context = new StartupContext(ArgumentManager.argsToMap(args));
135             if(mainModule==null)
136                 new Main().launch(createModuleRegistry(), context);
137             else
138                 new Main().launch(createModuleRegistry(),mainModule, context);
139         } catch (BootException e) {
140             throw new MojoExecutionException("Failed to boot up the module system",e);
141         } catch (IOException e) {
142             throw new MojoExecutionException("Failed to boot up the module system",e);
143         }
144     }
145 
146     /**
147      * Creates a fully configured module registry.
148      */
149     protected ModulesRegistry createModuleRegistry() throws IOException {
150         ModulesRegistry r = AbstractFactory.getInstance().createModulesRegistry();
151         // code in habitat needs to see HK2 classes, but we don't want them to see other classes and libraries
152         // that Maven loads into this maven plugin
153         r.setParentClassLoader(new MaskingClassLoader(this.getClass().getClassLoader(),"org.jvnet.hk2","com.sun.enterprise"));
154         MavenProjectRepository.prepareProject(project);
155         MavenProjectRepository lib = new MavenProjectRepository(project,artifactResolver,localRepository,artifactFactory);
156         r.addRepository(lib);
157         lib.initialize();
158         return r;
159     }
160 
161     public static String toModuleName(String groupId,String artifactId) {
162         return groupId+':'+artifactId;
163     }

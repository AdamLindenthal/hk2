56  /***
57   * Launches the module system by using all the dependencies as modules.
58   *
59   * <p>
60   * Not specifying "@execute phase=compile" so that it runs
61   * quickly even with a large project.
62   * </p>
63   * 
64   * @requiresProject
65   * @goal run
66   * @requiresDependencyResolution runtime
67   * @phase compile
68   * @execute phase=compile
69   * 
70   * @author Kohsuke Kawaguchi
71   */
72  public class RunMojo extends AbstractMojo {
73  
74      /***
75       * The maven project.
76       *
77       * @parameter expression="${project}"
78       * @required
79       * @readonly
80       */
81      private MavenProject project;
82  
83      /***
84       * @component
85       */
86      private ArtifactResolver artifactResolver;
87  
88      /***
89       * @parameter expression="${localRepository}"
90       */
91      private ArtifactRepository localRepository;
92  
93      /***
94       * @component
95       */
96      private ArtifactFactory artifactFactory;
97  
98      /***
99       * The root directory of the launched module system.
100      *
101      * @parameter expression="${rootDir}" default-value="${project.baseDir}"
102      */
103     private File rootDir;
104 
105     /***
106      * Command-line options to be passed to {@link StartupContext}.
107      *
108      * @parameter
109      */
110     private String[] args = new String[0];
111 
112     /***
113      * groupId:artifactId of the module that includes {@link ModuleStartup}.
114      * If omitted, the module list is searched to find one.
115      *
116      * @parameter expression="${mainModule}"
117      */
118     private String mainModule;
119 
120     public void execute() throws MojoExecutionException, MojoFailureException {
121         if(project.getPackaging().equals("pom")) {
122             // we want to run "mvn test" from parent module and run tests of all descendants,
123             // but there's nothing to test in POM module itself.
124             getLog().info("Skipping");
125             return;
126         }
127 
128         // sanity check
129         if(project.getPackaging().equals("hk2-jar") && (project.getArtifact()==null || project.getArtifact().getFile()==null)) {
130             getLog().warn("This project isn't compiled yet. Perhaps you meant 'mvn compile hk2:run'?");
131         }
132 
133         if(rootDir==null)
134             rootDir = project.getBasedir();
135     
136         try {
137             StartupContext context = new StartupContext(rootDir, args);
138             if(mainModule==null)
139                 new Main().launch(createModuleRegistry(), context);
140             else
141                 new Main().launch(createModuleRegistry(),mainModule, context);
142         } catch (BootException e) {
143             throw new MojoExecutionException("Failed to boot up the module system",e);
144         } catch (IOException e) {
145             throw new MojoExecutionException("Failed to boot up the module system",e);
146         }
147     }
148 
149     /***
150      * Creates a fully configured module registry.
151      */
152     protected ModulesRegistry createModuleRegistry() throws IOException {
153         ModulesRegistry r = AbstractFactory.getInstance().createModulesRegistry();
154         // code in habitat needs to see HK2 classes, but we don't want them to see other classes and libraries
155         // that Maven loads into this maven plugin
156         r.setParentClassLoader(new MaskingClassLoader(this.getClass().getClassLoader(),"org.jvnet.hk2","com.sun.enterprise"));
157         MavenProjectRepository.prepareProject(project);
158         MavenProjectRepository lib = new MavenProjectRepository(project,artifactResolver,localRepository,artifactFactory);
159         r.addRepository(lib);
160         lib.initialize();
161         return r;
162     }
163 
164     public static String toModuleName(String groupId,String artifactId) {
165         return groupId+':'+artifactId;
166     }

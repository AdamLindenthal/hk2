23      public List compile( CompilerConfiguration config ) throws CompilerException {
24          // force 1.5
25          config.setTargetVersion("1.5");
26          config.setSourceVersion("1.5");
27  
28  
29          File destinationDir = new File( config.getOutputLocation() );
30  
31          if ( !destinationDir.exists() )
32          {
33              destinationDir.mkdirs();
34          }
35  
36          String[] sourceFiles = getSourceFiles( config );
37  
38          if ( sourceFiles.length == 0 )
39          {
40              return Collections.EMPTY_LIST;
41          }
42  
43          getLogger().info( "Compiling " + sourceFiles.length + " " +
44                            "source file" + ( sourceFiles.length == 1 ? "" : "s" ) +
45                            " to " + destinationDir.getAbsolutePath() );
46  
47          // this is where the META-INF/services get generated.
48          config.addCompilerCustomArgument("-s",new File(config.getOutputLocation()).getAbsolutePath());
49          String[] args = buildCompilerArguments( config, sourceFiles );
50  
51          return compileInProcess( args );
52      }
53  
54      /***
55       * Compile the java sources in the current JVM, without calling an external executable,
56       * using <code>com.sun.tools.javac.Main</code> class
57       *
58       * @param args arguments for the compiler as they would be used in the command line javac
59       * @return List of CompilerError objects with the errors encountered.
60       * @throws CompilerException
61       */
62      protected List compileInProcess( String[] args ) throws CompilerException {
63          com.sun.tools.apt.Main aptTool = new com.sun.tools.apt.Main();
64          int r = aptTool.process(new MetainfServiceGenerator(),new PrintWriter(System.out,true),args);
65          if(r!=0)
66              throw new CompilerException("APT failed: "+r);
67  
68          // TODO: should I try to parse the output?
69          return Collections.emptyList();
70      }

24      public List compile( CompilerConfiguration config ) throws CompilerException {
25          // force 1.5
26          config.setTargetVersion("1.5");
27          config.setSourceVersion("1.5");
28  
29  
30          File destinationDir = new File( config.getOutputLocation() );
31  
32          if ( !destinationDir.exists() )
33          {
34              destinationDir.mkdirs();
35          }
36  
37          String[] sourceFiles = getSourceFiles( config );
38  
39          if ( sourceFiles.length == 0 )
40          {
41              return Collections.EMPTY_LIST;
42          }
43  
44          getLogger().info( "Compiling " + sourceFiles.length + " " +
45                            "source file" + ( sourceFiles.length == 1 ? "" : "s" ) +
46                            " to " + destinationDir.getAbsolutePath() );
47  
48          // this is where the META-INF/services get generated.
49          config.addCompilerCustomArgument("-s",new File(config.getOutputLocation()).getAbsolutePath());
50          String[] args = buildCompilerArguments( config, sourceFiles );
51  
52          return compileInProcess( args );
53      }
54  
55      /***
56       * Compile the java sources in the current JVM, without calling an external executable,
57       * using <code>com.sun.tools.javac.Main</code> class
58       *
59       * @param args arguments for the compiler as they would be used in the command line javac
60       * @return List of CompilerError objects with the errors encountered.
61       * @throws CompilerException
62       */
63      protected List compileInProcess( String[] args ) throws CompilerException {
64          com.sun.tools.apt.Main aptTool = new com.sun.tools.apt.Main();
65          int r = aptTool.process(
66              new CompositeAnnotationProcessorFactory(
67                  new MetainfServiceGenerator(),
68                  new AnnotationProcessorFactoryImpl()
69              ), new PrintWriter(System.out,true),args);
70          if(r!=0)
71              throw new CompilerException("APT failed: "+r);
72  
73          // TODO: should I try to parse the output?
74          return Collections.emptyList();
75      }

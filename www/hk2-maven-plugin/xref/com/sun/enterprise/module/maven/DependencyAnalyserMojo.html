46  import org.apache.maven.artifact.repository.ArtifactRepository;
47  import org.apache.maven.artifact.resolver.ArtifactResolver;
48  
49  import java.util.logging.Logger;
50  import java.util.logging.Level;
51  import java.util.HashSet;
52  import java.io.File;
53  import java.io.IOException;

58  /***
59   * Does static analysis of classes in a module to ensure that
60   * module dependency is correctly set up.
61   *
62   * @author Sanjeeb.Sahoo@Sun.COM
63   * @goal analyse-dependency
64   * @phase verify
65   * @requiresDependencyResolution compile
66   * @requiresProject
67   */
68  public class DependencyAnalyserMojo extends AbstractMojo {
69  
70      private static Logger logger = Logger.getAnonymousLogger();
71  
72      /***
73       * The maven project.
74       *
75       * @parameter expression="${project}"
76       * @required
77       * @readonly
78       */
79      protected MavenProject project;
80  
81      /***
82       * The maven artifact.
83       *
84       * @parameter expression="${project.artifact}"
85       * @required
86       * @readonly
87       */
88      protected Artifact artifact;
89  
90      /***
91       * @component
92       */
93      private ArtifactResolver artifactResolver;
94  
95      /***
96       * @parameter expression="${localRepository}"
97       */
98      private ArtifactRepository localRepository;
99  
100     /***
101      * @component
102      */
103     private ArtifactFactory artifactFactory;
104 
105     /***
106      * Directory containing the generated JAR.
107      *
108      * @parameter expression="${project.build.directory}"
109      * @required
110      */
111     protected File outputDirectory;
112 
113     /***
114      * Name of the generated JAR.
115      *
116      * @parameter alias="jarName" expression="${project.build.finalName}"
117      * @required
118      */
119     protected String finalName;
120 
121     /***
122      * By default, we just print a warning and proceed.
123      * @parameter default-value = false
124      */
125     private boolean failOnVerificationError;
126 
127     /***
128      * Flag that controls whether we analyse dependency or not.
129      * @parameter default-value = false
130      */
131     private boolean analyseDependency;
132 
133     /***
134      * Patterns excluded from dependency computation.
135      * @parameter
136      */
137     private HashSet<String> excludedPatterns;
138 
139     public void execute() throws MojoExecutionException, MojoFailureException {
140         // In addition to the configuration property called analyseDependency,
141         // we support a brute force way of skipping executio of this plugin.
142         if (!analyseDependency || Boolean.getBoolean("do-not-analyse-dependency")) {
143             logger.logp(Level.INFO, "DependencyAnalyserMojo", "execute", "Skipping analyse-dependency goal");
144             return;
145         }
146         logger.logp(Level.INFO, "DependencyAnalyserMojo", "execute", "Verifying module dependencies");
147 
148         if (!project.getPackaging().equals("hk2-jar")) {
149             System.out.println("Skipping this project as it does not use hk2-jar packaging");
150             return;
151         }
152         try {
153             MavenProjectRepository repo = new MavenProjectRepository(
154                     project, artifactResolver, localRepository, artifactFactory);
155             repo.initialize();
156             File location = artifact.getFile();
157             if (location == null) {
158                 // This can happen if the goal is executed directly outside of verify phase
159                 location = new File(outputDirectory, finalName + ".jar");
160                 if (!location.exists()) {
161                     throw new MojoExecutionException(location + " does not exist, so can't execute. Package the artifact first.");
162                 }
163             }
164             ModuleDefinition moduleDef = new MavenModuleDefinition(repo, location);
165             ModuleDependencyAnalyser analyser = new ModuleDependencyAnalyser(moduleDef, repo);
166             if (excludedPatterns!=null) {
167                 analyser.excludePatterns(excludedPatterns);
168             }
169             if (!analyser.analyse()) {
170                 String msg = "Missing dependency. See details below:\n" + analyser.getResultAsString();
171                 if (failOnVerificationError) {
172                     throw new MojoExecutionException(msg);
173                 } else {
174                     logger.logp(Level.WARNING, "DependencyAnalyserMojo", "execute", msg);
175                 }
176             } else {
177                 logger.logp(Level.INFO, "DependencyAnalyserMojo", "execute", "Dependency correctly set up");
178             }
179         } catch (IOException e) {
180             throw new MojoExecutionException("Unexpected exception", e);
181         }
182     }

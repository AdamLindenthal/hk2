248     private Set<DescriptorImpl> getAllDescriptorsFromInputStream(InputStream is) throws IOException {
249         BufferedReader pr = new BufferedReader(new InputStreamReader(is));
250         
251         Set<DescriptorImpl> retVal = new HashSet<DescriptorImpl>();
252         while (pr.ready()) {
253             DescriptorImpl di = new DescriptorImpl();
254             
255             if (!di.readObject(pr)) {
256                 continue;
257             }
258             
259             retVal.add(di);
260         }
261         
262         return retVal;
263     }
264     
265     private void checkDescriptors(Set<DescriptorImpl> dis) {
266         for (DescriptorImpl di : dis) {
267             Assert.assertTrue("Did not find " + di + " in the expected descriptors", EXPECTED_DESCRIPTORS.contains(di));
268         }
269         
270         
271         Assert.assertEquals("Expecting " + EXPECTED_DESCRIPTORS.size() + " descriptors, but only got " + dis.size(),
272                 EXPECTED_DESCRIPTORS.size(), dis.size());
273     }

275     /**
276      * Tests generating into a directory
277      * @throws IOException 
278      * @throws FileNotFoundException 
279      */
280     @Test
281     public void testDefaultDirectoryGeneration() throws IOException {
282         String argv[] = new String[2];
283         
284         argv[0] = FILE_ARGUMENT;
285         argv[1] = gendirDirectory.getAbsolutePath();
286         
287         File defaultOutput = new File(inhabitantsDirectory, DEFAULT);
288         if (defaultOutput.exists()) {
289             // Start with a clean plate
290             Assert.assertTrue(defaultOutput.delete());
291         }
292         
293         try {
294             int result = HabitatGenerator.embeddedMain(argv);
295             Assert.assertEquals("Got error code: " + result, 0, result);
296             
297             Assert.assertTrue("did not generate " + defaultOutput.getAbsolutePath(),
298                     defaultOutput.exists());
299             
300             Set<DescriptorImpl> generatedImpls = getAllDescriptorsFromInputStream(
301                     new FileInputStream(defaultOutput));
302             
303             checkDescriptors(generatedImpls);
304         }
305         finally {
306             // The test should be clean
307             defaultOutput.delete();
308         }
309     }

311     /**
312      * Tests generating into a directory
313      * @throws IOException On failure
314      */
315     @Test
316     public void testDefaultJarGeneration() throws IOException {
317         String argv[] = new String[4];
318         
319         argv[0] = FILE_ARGUMENT;
320         argv[1] = gendirJar.getAbsolutePath();
321         
322         argv[2] = OUTJAR_FILE_ARGUMENT;
323         argv[3] = OUTJAR_FILE.getAbsolutePath();
324         
325         Assert.assertTrue("Could not find file " + gendirJar.getAbsolutePath(),
326                 gendirJar.exists());
327         
328         if (OUTJAR_FILE.exists()) {
329             // Start with a clean plate
330             Assert.assertTrue(OUTJAR_FILE.delete());
331         }
332         
333         try {
334             int result = HabitatGenerator.embeddedMain(argv);
335             Assert.assertEquals("Got error code: " + result, 0, result);
336             
337             Assert.assertTrue("did not generate JAR " + OUTJAR_FILE.getAbsolutePath(),
338                     OUTJAR_FILE.exists());
339             
340             JarFile jar = new JarFile(OUTJAR_FILE);
341             ZipEntry entry = jar.getEntry(ZIP_FILE_INHABITANT_NAME);
342             Assert.assertNotNull(entry);
343             
344             InputStream is = jar.getInputStream(entry);
345             
346             Set<DescriptorImpl> generatedImpls = getAllDescriptorsFromInputStream(is);
347             
348             checkDescriptors(generatedImpls);
349         }
350         finally {
351             // The test should be clean
352             OUTJAR_FILE.delete();
353         }

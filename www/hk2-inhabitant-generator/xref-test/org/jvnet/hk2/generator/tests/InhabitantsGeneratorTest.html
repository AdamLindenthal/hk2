576     private Set<DescriptorImpl> getAllDescriptorsFromInputStream(InputStream is) throws IOException {
577         BufferedReader pr = new BufferedReader(new InputStreamReader(is));
578         
579         Set<DescriptorImpl> retVal = new HashSet<DescriptorImpl>();
580         while (pr.ready()) {
581             DescriptorImpl di = new DescriptorImpl();
582             
583             if (!di.readObject(pr)) {
584                 continue;
585             }
586             
587             retVal.add(di);
588         }
589         
590         return retVal;
591     }

594         for (DescriptorImpl di : dis) {
595             Assert.assertTrue("Did not find " + di + " in the expected descriptors <<<" +
596               EXPECTED_DESCRIPTORS + ">>>", EXPECTED_DESCRIPTORS.containsKey(di));
597             
598             // The rank is not part of the calculated equals or hash code (since it can change
599             // over the course of the lifeycle of the object) and hence must be checked

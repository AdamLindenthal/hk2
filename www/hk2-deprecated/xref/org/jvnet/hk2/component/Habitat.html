100     Habitat(Habitat parent, String name, Boolean concurrency_controls) {
101         if (parent != null || concurrency_controls != null) {
102             throw new UnsupportedOperationException("<clinit> (" + parent + "," + name + "," + concurrency_controls + ") in Habitat");
103         }
104         
105         if (name == null || name.length() <= 0) {
106             name = DEFAULT_NAME;
107         }
108         
109         delegate = ServiceLocatorFactory.getInstance().create(name);
110         
111         // Add this habitat in, so it can be looked up!
112         AbstractActiveDescriptor<Habitat> habitatDescriptor = BuilderHelper.createConstantDescriptor(this);
113         habitatDescriptor.removeContractType(ServiceLocator.class);
114         
115         DynamicConfigurationService dcs = delegate.getService(DynamicConfigurationService.class);
116         DynamicConfiguration config = dcs.createDynamicConfiguration();
117         
118         config.addActiveDescriptor(habitatDescriptor);
119         
120         config.commit();
121     }

360     /**
361      * Removes a Contracted service
362      *
363      * @param index               the contract name
364      * @param serviceOrInhabitant the service instance, or an Inhabitant instance
365      */
366     public boolean removeIndex(String index, Object serviceOrInhabitant) {
367         final Object o        = serviceOrInhabitant;
368         final String contract = index;
369         final Filter filter = new Filter() {
370             @Override
371             public boolean matches(Descriptor d) {
372                 if (d.getAdvertisedContracts().contains(contract)) {
373                     Descriptor baseDescriptor = d.getBaseDescriptor();
374                     if (baseDescriptor instanceof AliasDescriptor<?>) {
375                         if (((AliasDescriptor<?>)baseDescriptor).getDescriptor().getBaseDescriptor().equals(o)) {
376                             return true;
377                         }
378                     }
379                     ActiveDescriptor<?> activeDescriptor = (ActiveDescriptor<?>)d;
380 
381                     return (activeDescriptor.isReified() &&
382                             delegate.getServiceHandle(activeDescriptor).getService().equals(o));
383                 }
384                 return false;
385             }
386         };
387 
388         // TODO : notify not currently supported
389 
390         return Utilities.remove(delegate, filter);
391     }

533     /**
534      * Add an already instantiated component to this manager. The component has
535      * been instantiated by external code, however dependency injection,
536      * PostConstruct invocation and dependency extraction will be performed on
537      * this instance before it is store in the relevant scope's resource
538      * manager.
539      *
540      * @param component component instance
541      * @throws ComponentException if the passed object is not an HK2 component or
542      *                            injection/extraction failed.
543      */
544     public <T> void addComponent(T component) throws ComponentException {
545         if (component == null) return;
546         
547         DynamicConfigurationService dcs = delegate.getService(DynamicConfigurationService.class);
548         if (dcs == null) {
549             throw new ComponentException("No DynamicConfigurationService available");
550         }
551         
552         DynamicConfiguration config = dcs.createDynamicConfiguration();
553         
554         config.addActiveDescriptor(BuilderHelper.createConstantDescriptor(component));
555         
556         try {
557             config.commit();
558         }
559         catch (MultiException me) {
560             throw new ComponentException(me);
561         }
562     }

157     public Collection<Provider<T>> all(boolean stopAtFirstMatch) {
158 
159         Set<Provider<T>> providers = new HashSet<Provider<T>>();
160 
161         if (qualifiers.isEmpty()) {
162             return getNonQualifiedInhabitants(stopAtFirstMatch);
163         } else {
164             List<String> tmpQualifiers = new ArrayList<String>();
165             for (Class<? extends Annotation> annotation : qualifiers) {
166                 tmpQualifiers.add(annotation.getName());
167             }
168             for (Inhabitant<T> inh : inhabitants()) {
169                 List<String> declaredQualifiers = inh.metadata().get(InhabitantsFile.QUALIFIER_KEY);
170                 for (String declaredQualifier : declaredQualifiers) {
171                     // todo : we need to look at the instances of the Annotations
172                     // rather than the type on both the injection target and the
173                     // candidate so we can ensure that annotation attribute matching is performed
174                     tmpQualifiers.remove(declaredQualifier);
175                 }
176 
177                 // if the injection point qualifiers are all satisfied, stop the query
178                 if (tmpQualifiers.isEmpty()) {
179                     // check if the scope is fine.
180                     if (scope==null) {
181                         providers.add(inh);
182                     } else {
183                         if ((inh instanceof ScopedInhabitant) && ((ScopedInhabitant) inh).getScope().equals(scope)) {
184                             providers.add(inh);
185                         }
186                     }
187                     if (!providers.isEmpty() && stopAtFirstMatch) return providers;
188                 }
189             }
190         }
191         return providers;
192     }

194     private Inhabitant<T> getNonQualifiedInhabitant() {
195         if (name!=null && !name.isEmpty()) {
196             return provider();
197         }
198         Inhabitant<T> inh = provider();
199         if (inh==null) return null;
200         if (inh.metadata().get(InhabitantsFile.QUALIFIER_KEY).isEmpty()) {
201             return inh;
202         }
203         // we should find the inhabitant with no qualifier.
204         for (Inhabitant<T> inhabitant : inhabitants()) {
205             if (inhabitant.metadata().get(InhabitantsFile.QUALIFIER_KEY).isEmpty()) {
206                 return inhabitant;
207             }
208         }
209         return null;
210     }

212     private Collection<Provider<T>> getNonQualifiedInhabitants(boolean stopOnFirst) {
213         Set<Provider<T>> inhabitants = new HashSet<Provider<T>>();
214         if (name!=null && !name.isEmpty()) {
215             Inhabitant<T> provider = provider();
216             if (provider!=null) {
217                 inhabitants.add(provider);
218                 if (stopOnFirst) return inhabitants;
219             }
220         }
221         Inhabitant<T> inh = provider();
222         if (inh==null) return inhabitants;
223         if (inh.metadata().get(InhabitantsFile.QUALIFIER_KEY).isEmpty()) {
224             inhabitants.add(inh);
225             if (stopOnFirst) return inhabitants;
226         }
227         // we should find the inhabitant with no qualifier.
228         for (Inhabitant<T> inhabitant : inhabitants()) {
229             if (inhabitant.metadata().get(InhabitantsFile.QUALIFIER_KEY).isEmpty()) {
230                 //FIXME TODO if (inhabitant.getDescriptor().getNames().isEmpty()) {
231                 if (inhabitant.getDescriptor().getAdvertisedContracts().isEmpty()) {
232                     inhabitants.add(inhabitant);
233                     if (stopOnFirst) return inhabitants;
234                 }
235             }
236         }
237         return inhabitants;
238     }

628

630             Assert.assertEquals(VALUE2_1, values.get(0));
631             Assert.assertEquals(VALUE2_2, values.get(1));
632             Assert.assertEquals(VALUE2_3, values.get(2));
633             Assert.assertEquals(VALUE2_4, values.get(3));

43  package com.sun.enterprise.tools.verifier.apiscan.classfile;
44  
45  import java.util.Collection;
46  
47  /***
48   * This represents the information available in a Java .class file. This
49   * interface is used by {@link ClosureCompilerImpl} for closure computation. The
50   * interface only represents the information as needed by api scanning feature
51   * in verifier. I expect it to evolve over time to something very similar to
52   * BCEL's ClassFile. Now a note about different ways a Java class can be named
53   * and in different places different names are used for the same Java class. 1)
54   * In the format "p.a.b". This is what we use when declaring variables etc in a
55   * Java code. I do not use this because this is not an unambiguous
56   * representation of a Java class. By looking at p.a.b it is not possible to say
57   * if b is an inner class in class p.a or b is an outer class in package p.a. 2)
58   * In the format "p.outer$inner". It is what is used when we invoke java command
59   * or Class.forName(). It is same as what is returned by
60   * java.lang.Class.getName() method. It is an unambiguous representation of a
61   * class name, because by looking at it, we can tell "inner" is an inner class
62   * of a class "outer" which belongs to package p. By default our {@link
63   * #getName()} returns in this format. 3) In the format "p/outer$inner" This is
64   * the internal name of a class. It is called internal name because in byte code
65   * this is what is encoded. It is again an unambiguous representation as this a
66   * path expression. It is fairly simple to convert from 2 to 3 by a simple call
67   * to String.replace('.','/'). Similarly to convert from 3 to 2, call
68   * String.replace('/','.'); Here is a test of what you understood. What does
69   * this class name "a$b.c$d.Foo$Bar$Goo" mean? "Goo" is an inner class in a
70   * class "Foo$Bar" which is defined in a package "a$b.c$d".
71   *
72   * @author Sanjeeb.Sahoo@Sun.COM
73   */
74  public interface ClassFile {
75      /***
76       * @return names of the classes that are directly referenced by this class.
77       *         The returned class names are in external form.
78       */
79      Collection<String> getAllReferencedClassNames();
80  
81      /***
82       * @return names of the classes that are directly referenced by this class.
83       *         The returned class names are in internal form. This is done with
84       *         a purpose as it is easy to look up files with internal class
85       *         name.
86       */
87      Collection getAllReferencedClassNamesInInternalForm();
88  
89      /***
90       * @return the external name of this Java class. External name is of the
91       *         form java.util.Map$Entry. It is what is used when we invoke java
92       *         command or Class.forName(). It is same as what is returned by
93       *         java.lang.Class.getName() method. Pl note that a Java Class name
94       *         and package name can contain $, so when you see a$b, don't assume
95       *         it is an inner class.
96       * @see #getInternalName()
97       */
98      String getName();
99  
100     /***
101      * @return the internal name of the Java class. Internal name is what is
102      *         available in file system, e.g. java/util/Map$Entry Pl note that a
103      *         Java Class name and package name can contain $, so when you see
104      *         a$b, don't assume it is an inner class.
105      */
106     String getInternalName();
107 
108     /***
109      * @return internal package name of the Java class. Unlike class name,
110      *         package names do not have many forms. They are always specified
111      *         using dor notation (i.e. java.lang). See getName() method in
112      *         java.lang.Package class. Accordingly we have only one API for
113      *         package name. Returns "" for default package.
114      */
115     String getPackageName();
116 
117     /***
118      * @return all the methods that are present in this class. This includes
119      *         methods that are added by compiler as well, e.g. clinit and init
120      *         methods.
121      */
122     Collection<? extends Method> getMethods();
123 
124     /***
125      * @param methodRef is the reference of the method that is being looked for
126      * @return return the method object that matches the guven criteria. null,
127      *         otherwise.
128      */
129     Method getMethod(MethodRef methodRef);
130 
131     /***
132      * @return external name of super class
133      */
134     String getNameOfSuperClass();
135 
136     /***
137      * @return internal name of super class. Every class other than
138      *         java.lang.Object has a super class.
139      */
140     String getInternalNameOfSuperClass();
141 
142     /***
143      * @return external names of any interfaces implemented by this class.
144      */
145     String[] getNamesOfInterfaces();
146 
147     /***
148      * @return internal names of any interfaces implemented by this class.
149      */
150     String[] getInternalNamesOfInterfaces();
151 
152     /***
153      * @return true if this is an interface, else false
154      */
155     boolean isInterface();
156 
157     /***
158      * @return true if this is a public class, else false
159      */
160     boolean isPublic();
161 }

47  /***
48   * This is the base class for classes that actually implement ClosureCompiler
49   * interface, i.e. {@link BCELClosureCompilerImpl} and
50   * {@link ASMClosureCompilerImpl}.
51   * It contains common implementation for above classes.
52   *
53   * @author Sanjeeb.Sahoo@Sun.COM
54   */
55  public abstract class ClosureCompilerImplBase implements ClosureCompiler {
56  
57      protected ClassFileLoader loader;
58  
59      protected HashSet<String> excludedClasses = new HashSet<String>();
60  
61      protected HashSet<String> excludedPackages = new HashSet<String>();
62  
63      protected HashSet<String> excludedPatterns = new HashSet<String>();
64  
65      protected HashSet<String> visitedClasses = new HashSet<String>();

67      private static String resourceBundleName = "com.sun.enterprise.tools.verifier.apiscan.LocalStrings";
68      protected static Logger logger = Logger.getLogger("apiscan.classfile", resourceBundleName); // NOI18N

71      private static final String myClassName = "ClosureCompilerImplBase"; // NOI18N

73      /***
74       * @param loader the ClassFileLoader that is used to load the referenced
75       *               classes.
76       */
77      protected ClosureCompilerImplBase(ClassFileLoader loader) {
78          this.loader = loader;
79      }
80  
81      /***
82       * @param className the class name to be excluded from closure
83       *                     computation. It is in the external class name format
84       *                     (i.e. java.util.Map$Entry instead of java.util.Map.Entry).
85       *                     When the closure compiler sees a class matches this
86       *                     name, it does not try to compute its closure any
87       *                     more. It merely adds this name to the closure. So the
88       *                     final closure will contain this class name, but not
89       *                     its dependencies.
90       */
91      public void addExcludedClass(String className) {
92          excludedClasses.add(className);
93      }
94  
95      /***
96       * @param pkgName the package name of classes to be excluded from
97       *                     closure computation. It is in the external format
98       *                     (i.e. java.lang (See no trailing '.'). When the
99       *                     closure compiler sees a class whose package name
100      *                     matches this name, it does not try to compute the
101      *                     closure of that class any more. It merely adds that
102      *                     class name to the closure. So the final closure will
103      *                     contain that class name, but not its dependencies.
104      */
105     public void addExcludedPackage(String pkgName) {
106         excludedPackages.add(pkgName);
107     }
108 
109     /***
110      * @param pattern the pattern for the names of classes to be excluded from
111      *                closure computation. It is in the external format (i.e.
112      *                org.apache.). When the closure compiler sees a class whose
113      *                name begins with this pattern, it does not try to compute
114      *                the closure of that class any more. It merely adds that
115      *                class name to the closure. So the final closure will
116      *                contain that class name, but not its dependencies. Among
117      *                all the excluded list, it is given the lowest priority in
118      *                search order.
119      */
120     public void addExcludedPattern(String pattern) {
121         excludedPatterns.add(pattern);
122     }
123 
124     /***
125      * @param jar whose classes it will try to build closure of. This is a
126      *            convenience method which iterates over all the entries in a
127      *            jar file and computes their closure.
128      */
129     public boolean buildClosure(java.util.jar.JarFile jar) throws IOException {
130         boolean result = true;
131         for (java.util.Enumeration entries = jar.entries();
132              entries.hasMoreElements();) {
133             String clsName = ((java.util.jar.JarEntry) entries.nextElement()).getName();
134             if (clsName.endsWith(".class")) { // NOI18N
135                 String externalClsName = clsName.substring(0,
136                         clsName.lastIndexOf(".class")) // NOI18N
137                         .replace('/', '.');
138                 boolean newresult = this.buildClosure(externalClsName);
139                 result = newresult && result;
140             }
141         }//for all jar entries
142         return result;
143     }

145     public Collection<String> getNativeMethods() {
146         throw new UnsupportedOperationException();
147     }

149     /***
150      * @param className name of class in external format.
151      * @return
152      */
153     protected boolean needToBuildClosure(String className) {
154         boolean result = true;
155         if (visitedClasses.contains(className))
156             result = false;
157         else if (excludedClasses.contains(className)) {
158             result = false;
159         } else if (excludedPackages.contains(getPackageName(className))) {
160             result = false;
161         } else {
162             for (Iterator i = excludedPatterns.iterator(); i.hasNext();) {
163                 String pattern = (String) i.next();
164                 if (className.startsWith(pattern)) {
165                     result = false;
166                     break;
167                 }
168             }
169         }
170         logger.logp(Level.FINEST, myClassName, "needToBuildClosure", // NOI18N
171                 className + " " + result); // NOI18N
172         return result;
173     }
174 
175     /***
176      * @param className name of class in external format.
177      * @return package name in dotted format, e.g. java.lang for java.lang.void
178      */
179     protected static String getPackageName(String className) {
180         int idx = className.lastIndexOf('.');
181         if (idx != -1) {
182             return className.substring(0, idx);
183         } else
184             return "";
185     }

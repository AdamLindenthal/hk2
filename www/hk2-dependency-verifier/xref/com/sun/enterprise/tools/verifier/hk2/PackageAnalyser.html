93      /***
94       * Holds information about a range of Versions.
95       */
96      private static class VersionRange {
97          private final Version lowerVersion;
98          private final boolean lowerVersionInclussive; // is it [1.0 or (1.0
99          private final Version upperVersion;
100         private final boolean upperVersionInclussive; // is it 2.0] or 2.0)
101         private static final VersionRange DEFAULT_VERSION_RANGE = new VersionRange();
102         private static final String LSB = "[";
103         private static final String LP = "(";
104         private static final String RSB = "]";
105 
106         public VersionRange(Version lowerVersion, boolean lowerVersionInclussive, Version upperVersion, boolean upperVersionInclussive) {
107             this.lowerVersion = lowerVersion;
108             this.lowerVersionInclussive = lowerVersionInclussive;
109             this.upperVersion = upperVersion;
110             this.upperVersionInclussive = upperVersionInclussive;
111         }
112 
113         private VersionRange(Version lowerVersion) {
114             this(lowerVersion, true, null, false);
115         }
116 
117         public VersionRange() {
118             this(Version.emptyVersion, true, null, false);
119         }
120 
121         public static VersionRange valueOf(String s) {
122             if (s == null) s = "";
123             // The string can be any of the following:
124             // "v1" -> [v1, infinity)
125             // v1 -> [v1, inifinity)
126             // "[v1, v2)"
127             // "(v1, v2)"
128             // "(v1, v2]"
129             // "[v1, v2]"
130 
131             // Let's get rid of the sorrounding quotes.
132             if(s.startsWith("\"")) {
133                 assert(s.endsWith("\""));
134                 s = s.substring(1, s.length()-1);
135             }
136 
137             if (s.length() == 0) return DEFAULT_VERSION_RANGE;
138 
139             if (s.startsWith(LP) || s.startsWith(LSB)) {
140                 int comma = s.indexOf(COMMA, 1);
141                 String lvs = s.substring(1, comma).trim();
142                 Version lv = new Version(lvs);
143                 String uvs = s.substring(comma+1, s.length() -1).trim();
144                 Version uv = new Version(uvs);
145                 boolean lowerVersionInclussive = s.startsWith(LSB);
146                 boolean upperVersionInclussive = s.endsWith(RSB);
147                 return new VersionRange(lv, lowerVersionInclussive, uv, upperVersionInclussive);
148             } else {
149                 Version lv = new Version(s.trim());
150                 return new VersionRange(lv);
151             }
152         }
153 
154         /***
155          * Returns true if the given version falls in the version range.
156          * @param version
157          * @return
158          */
159         public boolean isInRange(Version version) {
160             final int distanceFromLowerVersion = version.compareTo(lowerVersion);
161             boolean lowerVersionMatches =
162                     lowerVersionInclussive ? (distanceFromLowerVersion >=0) : (distanceFromLowerVersion > 0);
163             if (lowerVersionMatches) {
164                 if (upperVersion != null) {
165                     final int distanceFromUpperVersion = version.compareTo(upperVersion);
166                     boolean upperVersionMatches =
167                             upperVersionInclussive ? (distanceFromUpperVersion <= 0) : (distanceFromUpperVersion < 0);
168                     return upperVersionMatches;
169                 } else {
170                     return true;
171                 }
172             } else {
173                 return false;
174             }
175         }
176 
177         @Override
178         public String toString() {
179             StringBuilder sb = new StringBuilder("\"");
180             sb.append(lowerVersionInclussive ? LSB : LP).append(lowerVersion).append(", ");
181             sb.append(upperVersion != null ? upperVersion.toString() : "infinity").append(upperVersionInclussive ? RSB : ")");
182             sb.append("\"");
183             return sb.toString();
184         }
185     }

516     private Set<PackageRequirement> computeImportedPackages(Bundle bundle) {
517         // The header looks like this:
518         // Import-Package: p1; p2; version=1.4; foo=bar, p3;version="[1, 5)", p4
519 
520         String importPkgHeader = bundle.getMd().getManifest().getMainAttributes().getValue(Constants.IMPORT_PACKAGE);
521         if (importPkgHeader == null) {
522             return Collections.EMPTY_SET;
523         }
524         Set<PackageRequirement> importedPkgs = new HashSet<PackageRequirement>();
525         List<Token> tokens = tokenize(importPkgHeader);
526         List<String> pkgNames = new ArrayList<String>();
527         VersionRange versionRange = VersionRange.DEFAULT_VERSION_RANGE;
528         boolean nextPkgGroup = false;
529         for (Token t : tokens) {
530             switch (t.type) {
531                 case PKG:
532                     if(nextPkgGroup) {
533                         // Add parsed package details and prepare for next iteration
534                         for (String name : pkgNames) {
535                             PackageRequirement pr = new PackageRequirement(name, versionRange);
536                             importedPkgs.add(pr);
537                         }
538                         pkgNames.clear();
539                         versionRange = VersionRange.DEFAULT_VERSION_RANGE;
540                         nextPkgGroup = false;
541                     }
542                     pkgNames.add(t.value);
543                     break;
544                 case DIRECTIVE:
545                     break;
546                 case ATTRIBUTE:
547                     int idx = t.value.indexOf(EQUALS);
548                     assert(idx != -1);
549                     String attrName = t.value.substring(0, idx);
550                     if (Constants.VERSION_ATTRIBUTE.equals(attrName)) {
551                         String versionRangeStr = t.value.substring(idx +1);
552                         versionRange = VersionRange.valueOf(versionRangeStr);
553                     }
554                     break;
555                 case PKG_GROUP_SEP:
556                     nextPkgGroup = true;
557                     break;
558                 default:
559                     throw new RuntimeException("Unknown token type. Fix the program.");
560             }
561         }
562         if (!pkgNames.isEmpty()) {
563             for (String name : pkgNames) {
564                 PackageRequirement pr = new PackageRequirement(name, versionRange);
565                 importedPkgs.add(pr);
566             }
567         }
568         return importedPkgs;
569     }

571     private Set<String> computeRequiredPackages(Bundle bundle) throws IOException {
572         Set<String> requiredPkgs = new HashSet<String>();
573         File moduleFile = new File(bundle.getMd().getLocations()[0]);
574         String classpath = moduleFile.getAbsolutePath();
575         JarFile moduleJar = new JarFile(moduleFile);
576         ClassFileLoader cfl = ClassFileLoaderFactory.newInstance(new Object[]{classpath});
577         final String classExt = ".class";
578         for (Enumeration<JarEntry> entries = moduleJar.entries(); entries.hasMoreElements();) {
579             JarEntry je = entries.nextElement();
580             if (je.getName().endsWith(classExt)) {
581                 String className = Util.convertToExternalClassName(je.getName().substring(0, je.getName().length() - classExt.length()));
582                 ClassFile cf = null;
583                 try {
584                     cf = cfl.load(className);
585                     for (String c : cf.getAllReferencedClassNames()) {
586                         requiredPkgs.add(Util.getPackageName(c));
587                     }
588                 } catch (IOException e) {
589                     logger.logp(Level.FINE, "PackageAnalyser", "computeRequiredPackages", "Skipping analysis of {0} as the following exception was thrown:\n {1}", new Object[]{className, e});
590                 }
591             }
592         }
593         return requiredPkgs;
594     }

596     static class Token {
597         enum TYPE {
598             PKG, ATTRIBUTE, DIRECTIVE, PKG_GROUP_SEP
599         }

602         final TYPE type;

606             this.type = type;

618     private List<Token> tokenize(String header) {
619         // The string looks like
620         // Export-Package: p1;p2;version=1.4;uses:="q1,q2...,qn",p3;uses:="q1,q2";p4;p5;version=...
621         List<Token> tokens = new ArrayList<Token>();
622         StringBuilder token = new StringBuilder();
623 
624         Token.TYPE type = PKG; // next token type
625         for (int i = 0; i < header.length(); i++) {
626             char c = header.charAt(i);
627             switch(c) {
628                 case QUOTE :
629                     int nextQuote = header.indexOf(QUOTE, i+1);
630                     token.append(header.substring(i, nextQuote+1));
631                     i = nextQuote;
632                     break;
633                 case COLON:
634                     token.append(c);
635                     char c2 = header.charAt(i+1);
636                     if (c2 == EQUALS) {
637                         type = DIRECTIVE;
638                         token.append(c2);
639                         i++;
640                     }
641                     break;
642                 case EQUALS:
643                     type = ATTRIBUTE;
644                     token.append(c);
645                     break;
646                 case COMMA :
647                     tokens.add(Token.createToken(token.toString(), type));
648                     // Add a special token as PKG GROUP SEPARATOR.
649                     tokens.add(Token.createToken(",", Token.TYPE.PKG_GROUP_SEP));
650                     token.delete(0, token.length());
651                     type = PKG;
652                     break;
653                 case SEMICOLON:
654                     tokens.add(Token.createToken(token.toString(), type));
655                     token.delete(0, token.length());
656                     type = PKG;
657                     break;
658                 default:
659                     token.append(c);
660             }
661         }
662         if (token.length() > 0) {
663             tokens.add(Token.createToken(token.toString(), type));
664         }
665         return tokens;
666     }

680     private List<PackageGroup> parseExportPackage(String header) {
681         // The string looks like
682         // Export-Package: p1;p2;version=1.4;uses:="q1,q2...,qn",p3;uses:="q1,q2";p4;p5;version=...
683 
684         List<PackageGroup> pkgGroups = new ArrayList<PackageGroup>();
685         List<Token> tokens = tokenize(header);
686         List<String> pkgNames = new ArrayList<String>();
687         List<String> pkgAttributes = new ArrayList<String>();
688         List<String> pkgDirectives = new ArrayList<String>();
689         boolean nextPkgGroup = false;
690         for (Token t : tokens) {
691             switch (t.type) {
692                 case PKG:
693                     if (nextPkgGroup) {
694                         // Add parsed package details and prepare for next iteration
695                         pkgGroups.add(new PackageGroup(pkgNames, pkgAttributes, pkgDirectives));
696                         pkgNames = new ArrayList<String>();
697                         pkgAttributes = new ArrayList<String>();
698                         pkgDirectives = new ArrayList<String>();
699                         nextPkgGroup = false;                   
700                     }
701                     pkgNames.add(t.value);
702                     break;
703                 case ATTRIBUTE:
704                     pkgAttributes.add(t.value);
705                     break;
706                 case DIRECTIVE:
707                     pkgDirectives.add(t.value);
708                     break;
709                 case PKG_GROUP_SEP:
710                     nextPkgGroup = true;
711                     break;
712                 default:
713                     throw new RuntimeException("Unknown token type. Fix the program");
714             }
715         }
716         if (!pkgNames.isEmpty()) {
717             // Add parsed package details
718             pkgGroups.add(new PackageGroup(pkgNames, pkgAttributes, pkgDirectives));
719         }
720         return pkgGroups;
721     }

723     private Set<PackageCapability> computeExportedPackages(Bundle bundle) {
724         Set<PackageCapability> exportedPkgs = new HashSet<PackageCapability>();
725         String exportedPkgsAttr = bundle.getMd().getManifest().getMainAttributes().getValue("Export-Package");
726         if (exportedPkgsAttr == null) return exportedPkgs;
727 
728         Collection<PackageGroup> pkgGroups = parseExportPackage(exportedPkgsAttr);
729         for (PackageGroup pg : pkgGroups) {
730             String version = null;
731             for (String attr : pg.pkgAttributes) {
732                 int idx = attr.indexOf(EQUALS);
733                 assert(idx!= -1);
734                 if (Constants.VERSION_ATTRIBUTE.equals(attr.substring(0, idx))) {
735                     version = attr.substring(idx+1);
736                     // if version contains quotes around it, remove them.
737                     if(version.startsWith("\"") && version.endsWith("\"")) {
738                         version = version.substring(1, version.length()-1);
739                     }
740                 }
741             }
742             for (String pkgName : pg.pkgNames) {
743                 PackageCapability p = new PackageCapability(pkgName, version);
744                 exportedPkgs.add(p);
745             }
746         }
747         return exportedPkgs;
748     }

763     public Collection<Wire> analyseWirings() throws IOException {
764         List<ModuleDefinition> moduleDefs =
765                 moduleDefs = moduleRepository.findAll();

767         for (ModuleDefinition moduleDef : moduleDefs) {

771         }
772         Set<Wire> wires = new HashSet<Wire>();
773         for (Bundle importer : bundles) {

775                 for (Bundle exporter : bundles) {
776                     if (exporter.provides(pkg)) {

780                 }

784         Collections.sort(sorted, new Comparator<Wire>() {
785             Collator collator = Collator.getInstance();
786 
787             public int compare(Wire o1, Wire o2) {

789             }
790         });
791         return sorted;

795      * Inspects bundles and reports duplicate packages.
796      * Before calling this method, you must call {@link this#analyseWirings()}

799      * @return set of split-packages, en empty set if none is found.

801     public Collection<SplitPackage> findDuplicatePackages() {
802         assert (bundles != null);
803         Map<String, Set<Bundle>> packages = new HashMap<String, Set<Bundle>>();
804         for (Bundle b : bundles) {
805             for (PackageCapability p : b.getExportedPkgs()) {
806                 Set<Bundle> exporters = packages.get(p.getName());
807                 if (exporters == null) {
808                     exporters = new HashSet<Bundle>();
809                     packages.put(p.getName(), exporters);
810                 }
811                 exporters.add(b);
812             }
813         }
814         Set<SplitPackage> duplicatePkgs = new HashSet<SplitPackage>();
815         for (Map.Entry<String, Set<Bundle>> entry : packages.entrySet()) {
816             if (entry.getValue().size() > 1) {
817                 duplicatePkgs.add(new SplitPackage(entry.getKey(), entry.getValue()));
818             }
819         }
820         List<SplitPackage> sortedDuplicatePkgs = new ArrayList<SplitPackage>(duplicatePkgs);
821         Collections.sort(sortedDuplicatePkgs, new Comparator<SplitPackage>() {
822             Collator collator = Collator.getInstance();
823 
824             public int compare(SplitPackage o1, SplitPackage o2) {
825                 return collator.compare(o1.name, o2.name);
826             }
827         });
828 
829         return sortedDuplicatePkgs;
830     }

875      * @return
876      */
877     public Collection<PackageCapability> findUnusedExports(Bundle exporter) {
878         List<PackageCapability> unusedPackages = new ArrayList<PackageCapability>();
879         for (PackageCapability p : exporter.getExportedPkgs()) {
880             boolean used = false;
881             for (Bundle importer : bundles) {
882                 if (importer != exporter && importer.requires(p)) {
883                     used = true;
884                     break;
885                 }
886             }
887             if (!used) unusedPackages.add(p);
888         }
889         Collections.sort(unusedPackages);
890         return unusedPackages;

79              // add all HK2 package patterns, because we don't handle reexport yet.
80              , "org.jvnet.hk2."
81              , "com.sun.hk2."
82              , "com.sun.enterprise.module."

86      /***
87       * Create a new analyser.
88       * @param moduleDef module whose dependency needs to be analysed
89       * @param moduleRepository repository used to satisfy dependencies
90       * @throws IOException
91       */
92      public ModuleDependencyAnalyser(ModuleDefinition moduleDef,
93                                      Repository moduleRepository) throws IOException {
94          this.moduleDef = moduleDef;
95          this.moduleRepository = moduleRepository;
96          moduleJar = new File(moduleDef.getLocations()[0]);
97          // Make a classpath consisting of only module jar file.
98          String classpath = moduleJar.getAbsolutePath();
99          ClassFileLoader cfl = ClassFileLoaderFactory.newInstance(new Object[]{classpath});
100         closure = new ClosureCompilerImpl(cfl);
101         for (String pattern : excludedPatterns) {
102             closure.addExcludedPattern(pattern);
103         }
104         excludeImportedPackages();
105         excludeExportedClasses();
106     }

108     /***
109      * Analyse dependency of a module. It uses the repository to look up
110      * modules that this module depends on.
111      * @return true if all the dependencies are OK, false if something is missing
112      * @throws IOException if there is any failure in reading module information
113      */
114     public synchronized boolean analyse()
115             throws IOException {
116         closure.buildClosure(new JarFile(moduleJar));
117         if (System.getProperty("debugOutput") != null) {
118             Logger.getLogger("apiscan.classfile").setLevel(Level.FINER);
119             PrintStream out = new PrintStream(new FileOutputStream(System.getProperty("debugOutput")));
120             out.println(closure);
121         }
122         return closure.getFailed().isEmpty();
123     }

152     /***
153      * This method adds packages imported by this bundle to
154      * to the list of excluded package names.
155      */
156     private void excludeImportedPackages() {
157         Attributes attributes = moduleDef.getManifest().getMainAttributes();
158         String exportedPkgsAttr = attributes.getValue("Import-Package");
159         if (exportedPkgsAttr==null) return;
160         StringTokenizer st = new StringTokenizer(exportedPkgsAttr, ",", false);
161         Set<String> importedPkgs = new HashSet<String>();
162         while (st.hasMoreTokens()) {
163             String token = st.nextToken().trim();
164             int idx = token.indexOf(';');
165             String pkg = (idx == -1) ? token : token.substring(0, idx);
166 //            System.out.println("pkg = " + pkg);
167             importedPkgs.add(pkg);
168         }
169         for (String pkg : importedPkgs) {
170             closure.addExcludedPackage(pkg);
171         }
172     }
173 
174     /***
175      * This method adds classes exported by all the bundles that this bundle imports
176      * to the list of excluded class names.
177      */
178     private void excludeExportedClasses() throws IOException {
179         ModuleDependency[] dependencies = moduleDef.getDependencies();
180         for (ModuleDependency dependency : dependencies) {
181             ModuleDefinition requiredModuleDef =
182                     moduleRepository.find(dependency.getName(), dependency.getVersion());
183             for (String cls : getExportedClasses(requiredModuleDef)) {
184                 closure.addExcludedClass(cls);
185             }
186         }
187     }

189     private Set<String> getExportedClasses(ModuleDefinition moduleDef) throws IOException {
190         Set<String> exportedPkgs = new HashSet<String>();
191         java.util.jar.Manifest m = moduleDef.getManifest();
192         if (m==null) return exportedPkgs;
193         Attributes attributes = m.getMainAttributes();
194         String exportedPkgsAttr = attributes.getValue("Export-Package");
195         StringTokenizer st = new StringTokenizer(exportedPkgsAttr, ",", false);
196         while (st.hasMoreTokens()) {
197             String token = st.nextToken().trim();
198             int idx = token.indexOf(';');
199             String pkg = (idx == -1) ? token : token.substring(0, idx);
200 //            System.out.println("pkg = " + pkg);
201             exportedPkgs.add(pkg);
202         }
203         Set<String> exportedClasses = new HashSet<String>();
204         JarFile moduleJar = new JarFile(new File(moduleDef.getLocations()[0]));
205         Enumeration<JarEntry> moduleEntries = moduleJar.entries();
206         while (moduleEntries.hasMoreElements()) {
207             String entry = moduleEntries.nextElement().getName();
208             if (entry.endsWith(".class")) {
209                 String clsName = Util.convertToExternalClassName(
210                         entry.substring(0, entry.length() - ".class".length()));
211                 String pkgName = Util.getPackageName(clsName);
212                 if (exportedPkgs.contains(pkgName)) {
213                     exportedClasses.add(clsName);
214                 }
215             }
216         }
217         return exportedClasses;
218     }

220     public String getResultAsString() {
221         StringBuilder sb = new StringBuilder();
222         Map<String, Collection<String>> failed = closure.getFailed();
223         if (failed.isEmpty()) return "";
224         for (String referencingPath : failed.keySet()) {
225             String referencingClass;
226             int idx = referencingPath.lastIndexOf('/');
227             if (idx == -1) {
228                 referencingClass = referencingPath;
229             } else {
230                 referencingClass = referencingPath.substring(idx + 1);
231             }
232             sb.append("\t").append(referencingClass);
233             sb.append(" -> {");
234             String[] notFoundClasses = failed.get(referencingPath).toArray(new String[0]);
235             for (int i = 0; i < notFoundClasses.length; ++i) {
236                 sb.append(notFoundClasses[i]);
237                 if (i != (notFoundClasses.length - 1)) { // skip appending ',' at the end
238                     sb.append(", ");
239                 }
240             }
241             sb.append("}\n");
242         }
243         return sb.toString();
244     }

250     public static void main(String[] args) throws Exception {
251         if (args.length == 0) {
252             System.out.println("Usage: java " + ModuleDependencyAnalyser.class.getName() +
253                     " <Repository Dir Path> [Path to module]...");
254 
255             System.out.println("Examples:\n" +
256                     "1. Following command verifies all modules in the specified repository:\n" +
257                     " java " + ModuleDependencyAnalyser.class.getName() +
258                     " /tmp/glassfish/modules/\n\n" +
259                     "2. Following command verifies only the specified modules in the specified repository:\n" +
260                     " java " + ModuleDependencyAnalyser.class.getName() +
261                     " /tmp/glassfish/modules/ kernel-10.0-SNAPSHOT.jar amx-impl-10.0SNAPSHOT.jar\n\n");
262             return;
263         }
264         String repoPath = args[0];
265         File f = new File(repoPath) {
266             @Override public File[] listFiles() {
267                 List<File> files = new ArrayList<File>();
268                 for (File f : super.listFiles()) {
269                     if (f.isDirectory()) {
270                         for (File f2 : f.listFiles()) {
271                             if (f2.isFile() && f2.getName().endsWith(".jar")) {
272                                 files.add(f2);
273                             }
274                         }
275                     } else if (f.isFile() && f.getName().endsWith(".jar")) {
276                         files.add(f);
277                     }
278                 }
279                 return files.toArray(new File[files.size()]);
280             }
281         };
282         Repository moduleRepository = new DirectoryBasedRepository("repo", f);
283         moduleRepository.initialize();
284         List<ModuleDefinition> moduleDefs = new ArrayList<ModuleDefinition>();
285         if (args.length > 1) {
286             for (int i = 1; i < args.length; ++i) {
287                 File moduleFile = new File(args[i]);
288                 if (!moduleFile.isAbsolute()) {
289                     moduleFile = new File(repoPath, args[i]);
290                 }
291                 ModuleDefinition moduleDef = new DefaultModuleDefinition(moduleFile);
292                 moduleDefs.add(moduleDef);
293             }
294         } else {
295             moduleDefs = moduleRepository.findAll();
296         }
297         List<URI> badModules = new ArrayList<URI>();
298         for (ModuleDefinition moduleDef : moduleDefs) {
299             ModuleDependencyAnalyser analyser =
300                     new ModuleDependencyAnalyser(moduleDef, moduleRepository);
301             if (System.getProperty("ExcludedPatterns")!=null) {
302                 StringTokenizer st = new StringTokenizer(
303                         System.getProperty("ExcludedPatterns"), ",", false);
304                 Set<String> patterns = new HashSet<String>();
305                 while (st.hasMoreTokens()) {
306                     patterns.add(st.nextToken());
307                 }
308                 analyser.excludePatterns(patterns);
309             }
310             if (!analyser.analyse()) {
311                 URI badModule = moduleRepository.getLocation().relativize(moduleDef.getLocations()[0]);
312                 badModules.add(badModule);
313                 System.out.println("<Module name = " + badModule + ">");
314                 System.out.println(analyser.getResultAsString());
315                 System.out.println("</Module>");
316             }
317         }
318         if (badModules.isEmpty()) {
319             System.out.println("All modules are OK");
320         } else {
321             System.out.println("Dependencies are not correctly set up for following modules:");
322             for (URI badModule : badModules) {
323                 System.out.print(badModule + " ");
324             }
325             System.out.println("");
326         }
327     }

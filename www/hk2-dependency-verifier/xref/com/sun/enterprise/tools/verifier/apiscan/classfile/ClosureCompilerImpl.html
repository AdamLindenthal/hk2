45  /**
46   * This is single most important class of the apiscan package. This class is
47   * used to compute the complete closure of a set of classes. It uses {@link
48   * ClassFile} to find out dependencies of a class. For each references class
49   * name, it loads the corresponding ClassFile using the ClassFileLoader passed
50   * to it in constructor. Then it recurssively computes the dependencies until it
51   * visits either Java primitive classes or the class name matches the exclude
52   * list. Example of using this class is given below...
53   * <blockquote><pre>
54   * String classpath="your own classpath";
55   * ClassFileLoader cfl=ClassFileLoaderFactory.newInstance(new
56   * Object[]{classpath});
57   * ClosureCompilerImpl cc=new ClosureCompilerImpl(cfl);
58   * cc.addExcludePattern("java.");//exclude all classes that start with java.
59   * Most of the J2SE classes will be excluded thus.
60    * cc.addExcludePattern("javax.");//Most of the J2EE classes can be excluded
61   * like this.
62   * cc.addExcludePackage("org.omg.CORBA");//Exclude classes whose package name is
63   * org.omg.CORBA
64   * cc.addExcludeClass("mypackage.Foo");//Exclude class whose name is
65   * myPackage.Foo
66   * boolean successful=cc.buildClosure("a.b.MyEjb");
67   * successful=cc.buildClosure("a.b.YourEjb") && successful; //note the order of
68   * &&.
69   * Collection closure=cc.getClosure();//now closure contains the union of
70   * closure for the two classes.
71   * Map failed=cc.getFailed();//now failure contains the union of failures for
72   * the two classes.
73   * cc.reset();//clear the results collected so far so that we can start afresh.
74   * //Now you can again start computing closure of another set of classes.
75   * //The excluded stuff are still valid.
76   * </pre></blockquote>
77   *
78   * @author Sanjeeb.Sahoo@Sun.COM
79   */
80  public class ClosureCompilerImpl implements ClosureCompiler {

97      /**
98       * @param loader the ClassFileLoader that is used to load the referenced
99       *               classes.
100      */
101     public ClosureCompilerImpl(ClassFileLoader loader) {
102         /*
103          * See how it binds to a runtime implementation
104          * TODO: Ideally we should introduce an AbstractFactory for
105          * both ClassFileLoader & ClassClosureCompiler product types.
106          */
107         if(loader instanceof BCELClassFileLoader ||
108                 loader instanceof BCELClassFileLoader1) {
109             imp = new BCELClosureCompilerImpl(loader);
110         } else {
111             throw new RuntimeException("Unknown loader type [" + loader + "]");
112         }
113     }
114 
115     /**
116      * I don't expect this constructor to be used. Only defined for
117      * testing purpose.
118      * @param imp the implementation in the bridge design pattern.
119      */
120     public ClosureCompilerImpl(ClosureCompilerImplBase imp) {
121         this.imp = imp;
122     }
123 
124     /**
125      * @param className the class name to be excluded from closure
126      *                     computation. It is in the external class name format
127      *                     (i.e. java.util.Map$Entry instead of java.util.Map.Entry).
128      *                     When the closure compiler sees a class matches this
129      *                     name, it does not try to compute its closure any
130      *                     more. It merely adds this name to the closure. So the
131      *                     final closure will contain this class name, but not
132      *                     its dependencies.
133      */
134     public void addExcludedClass(String className) {
135         imp.addExcludedClass(className);
136     }
137 
138     //takes in external format, i.e. java.util
139     /**
140      * @param pkgName the package name of classes to be excluded from
141      *                     closure computation. It is in the external format
142      *                     (i.e. java.lang (See no trailing '.'). When the
143      *                     closure compiler sees a class whose package name
144      *                     matches this name, it does not try to compute the
145      *                     closure of that class any more. It merely adds that
146      *                     class name to the closure. So the final closure will
147      *                     contain that class name, but not its dependencies.
148      */
149     public void addExcludedPackage(String pkgName) {
150         imp.addExcludedPackage(pkgName);
151     }
152 
153     /**
154      * @param pattern the pattern for the names of classes to be excluded from
155      *                closure computation. It is in the external format (i.e.
156      *                org.apache.). When the closure compiler sees a class whose
157      *                name begins with this pattern, it does not try to compute
158      *                the closure of that class any more. It merely adds that
159      *                class name to the closure. So the final closure will
160      *                contain that class name, but not its dependencies. Among
161      *                all the excluded list, it is given the lowest priority in
162      *                search order.
163      */
164     public void addExcludedPattern(String pattern) {
165         imp.addExcludedPattern(pattern);
166     }

168     //See corresponding method of ClosureCompiler for javadocs
169     public boolean buildClosure(String className) {
170         logger.entering(myClassName, "buildClosure", className); // NOI18N
171         return imp.buildClosure(className);
172     }
173 
174     /**
175      * @param jar whose classes it will try to build closure of. This is a
176      *            convenience method which iterates over all the entries in a
177      *            jar file and computes their closure.
178      */
179     public boolean buildClosure(java.util.jar.JarFile jar) throws IOException {
180         return imp.buildClosure(jar);
181     }

211     public static void main(String[] args) {
212         if (args.length < 2) {
213             System.out.println(
214                     "Usage : java " + com.sun.enterprise.tools.verifier.apiscan.classfile.ClosureCompilerImpl.class.getName() + // NOI18N
215                     " <classpath> <external class name(s)>"); // NOI18N
216             System.out.println("Example: to find the closure of " + // NOI18N
217                     "mypkg.MySessionBean which is packaged in myejb.jar run\n" + // NOI18N
218                     " java " + com.sun.enterprise.tools.verifier.apiscan.classfile.ClosureCompilerImpl.class.getName() + // NOI18N
219                     " path_to_j2ee.jar"+File.pathSeparator+"path_to_myejb.jar"+ // NOI18N
220                     " mypkg.MySessionBean"); // NOI18N
221             System.exit(1);
222         }
223 
224         String cp=args[0];
225         System.out.println("Using classpath " + cp); // NOI18N
226         ClassFileLoader cfl = ClassFileLoaderFactory.newInstance(
227                 new Object[]{cp});
228         ClosureCompilerImpl closure = new ClosureCompilerImpl(cfl);
229         closure.addExcludedPattern("java."); // NOI18N
230         for (int i = 1; i < args.length; i++) {
231             String clsName = args[i];
232             System.out.println("Building closure for " + clsName); // NOI18N
233             closure.reset();
234             closure.buildClosure(clsName);
235             System.out.println("The closure is [" + closure+"\n]"); // NOI18N
236         }
237     }

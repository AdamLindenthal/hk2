43  package com.sun.enterprise.tools.verifier.apiscan.classfile;
44  
45  import java.io.IOException;
46  import java.io.InputStream;
47  import java.util.logging.Logger;
48  
49  import com.sun.org.apache.bcel.internal.util.ClassPath;
50  
51  /***
52   * Yet another factory for {@link BCELClassFile}. This is not a public class, as
53   * I expect users to use {@link ClassFileLoaderFactory} interface. It differs
54   * from {@link BCELClassFileLoader} in the sense that it loads classfiles using
55   * bcel ClassPath class. Known Issues: Currently it ignores any INDEX
56   * information if available in a Jar file. This is because BCEL provided class
57   * ClassPath does not understand INDEX info. We should add this feature in
58   * future.
59   *
60   * @author Sanjeeb.Sahoo@Sun.COM
61   */
62  class BCELClassFileLoader1 implements ClassFileLoader {
63  
64      private static String resourceBundleName = "com.sun.enterprise.tools.verifier.apiscan.LocalStrings";
65      private static Logger logger = Logger.getLogger("apiscan.classfile", resourceBundleName); // NOI18N
66      private ClassPath cp;
67  
68      /***
69       * Creates a new instance of BCELClassFileLoader User should use {link
70       * ClassFileLoaderFactory} to create new instance of a loader.
71       *
72       * @param classPath represents the search path that is used by this loader.
73       *                  Please note that, it does not read the manifest entries
74       *                  for any jar file specified in the classpath, so if the
75       *                  jar files have optional package dependency, that must be
76       *                  taken care of in the classpath by ther caller.
77       */
78      public BCELClassFileLoader1(String classPath) {
79          logger.entering("BCELClassFileLoader1", "<init>(String)", classPath); // NOI18N
80          cp = new ClassPath(classPath);
81      }
82  
83      //See ClassFileLoader for description of this method.
84      public ClassFile load(String externalClassName) throws IOException {
85          logger.entering("BCELClassFileLoader1", "load", externalClassName); // NOI18N
86          //BCEL library expects me to pass in internal form.
87          String internalClassName = externalClassName.replace('.', '/');
88          //don't call getInputStream as it first tries to load using the
89          // getClass().getClassLoader().getResourceAsStream()
90          //return cp.getInputStream(extClassName);
91          InputStream is = cp.getClassFile(internalClassName, ".class") // NOI18N
92                  .getInputStream();
93          try {
94              ClassFile cf = new BCELClassFile(is, internalClassName + ".class"); // NOI18N
95              matchClassSignature(cf, externalClassName);
96              return cf;
97          } finally {
98              is.close();
99          }
100     }
101 
102     //This method is neede to be protected against users who are passing us
103     //internal class names instead of external class names or
104     //when the file actually represents some other class, but it isnot 
105     //available in in proper package hierarchy.
106     private void matchClassSignature(ClassFile cf, String externalClassName)
107             throws IOException {
108         String nameOfLoadedClass = cf.getName();
109         if (!nameOfLoadedClass.equals(externalClassName)) {
110             throw new IOException(
111                     externalClassName + ".class represents " +
112                     cf.getName() +
113                     ". Perhaps your package name is incorrect or you passed the" +
114                     " name using internal form instead of using external form.");
115         }
116     }
117 }

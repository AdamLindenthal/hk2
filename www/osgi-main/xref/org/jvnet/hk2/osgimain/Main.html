65  /***
66   * Goes through a directory structure recurssively and installs all the
67   * jar files as OSGi bundles. This bundle can be passed a list of bundle paths
68   * to be started automatically. The bundle path is treated relative to the
69   * directory from where it installs all the bundles. It does not stop those bundle
70   * when this bundle is stopped. It stops them in reverse order.
71   *
72   * This bundle is also responsible for updating or uninstalling bundle during
73   * subsequent restart if jars have been updated or deleted.
74   *
75   * It does not manage itself. So, we use reference: scheme to install this bundle
76   * which allows us to see changes to this jar file automatically.
77   *
78   * @author Sanjeeb.Sahoo@Sun.COM
79   */
80  public class Main implements BundleActivator

84       * This being a provisioning service itself can't expect too many other services to be available. So, it relies on
85       * core framework services only.
86       */

88      public static final String BUNDLES_DIR =
89              "org.jvnet.hk2.osgimain.bundlesDir";

91      // a comma separated list of dir names relative to bundles dir
92      // that need to be excluded. e.g., autostart
93      public static final String EXCLUDED_SUBDIRS = "org.jvnet.hk2.osgimain.excludedSubDirs";
94      public final static String HK2_CACHE_DIR = "com.sun.enterprise.hk2.cacheDir";
95      public final static String INHABITANTS_CACHE = "inhabitants";
96      public static final String AUTO_START_BUNDLES_PROP =
97              "org.jvnet.hk2.osgimain.autostartBundles";

106     // files under bundles dir structure that are excluded from processing
107     // if any of the excluded file is a directory, then entire content

111     private List<URI> autoStartBundleLocations = new ArrayList<URI>();
112     private Map<URI, Jar> currentManagedBundles = new HashMap<URI, Jar>();
113     private static final String THIS_JAR_NAME = "osgi-main.jar";

115     public void start(BundleContext context) throws Exception
116     {
117         this.context = context;
118         final String bundlesDirPath = getProperty(BUNDLES_DIR);
119         if (bundlesDirPath == null) {
120             // nothing to do, let's return
121             return;
122         }
123         bundlesDir = new File(bundlesDirPath);
124         String autostartBundlesProp = getProperty(AUTO_START_BUNDLES_PROP);
125         if (autostartBundlesProp != null) {
126             StringTokenizer st = new StringTokenizer(autostartBundlesProp, ",");
127             while (st.hasMoreTokens()) {
128                 String bundleRelPath = st.nextToken().trim();
129                 if (bundleRelPath.isEmpty()) break;
130                 URI bundleURI = new File(bundlesDir, bundleRelPath).toURI().normalize();
131                 autoStartBundleLocations.add(bundleURI);
132             }
133         }
134         String excludedFilesProp = getProperty(EXCLUDED_SUBDIRS);
135         if (excludedFilesProp != null) {
136             for (String s : excludedFilesProp.split(",")) {
137                 excludedSubDirs.add(new File(bundlesDir, s.trim()));
138             }
139         }
140 
141         traverse();
142 
143         for (URI location : autoStartBundleLocations) {
144             long id = currentManagedBundles.get(location).getBundleId();
145             if (id < 0) {
146                 logger.logp(Level.WARNING, "Main", "start", "Not able to locate autostart bundle for location = {0}", new Object[]{location});
147                 continue;
148             }
149             final Bundle bundle = context.getBundle(id);
150             // check is necessary as bundle could have been uninstalled
151             if (bundle != null) {
152                 try
153                 {
154                     bundle.start(Bundle.START_TRANSIENT);
155                 }
156                 catch (BundleException e)
157                 {
158                     logger.logp(Level.WARNING, "Main", "start", "Exception while starting bundle " + bundle, e);
159                 }
160             }
161         }
162     }

164     private String getProperty(String property) {
165         String value = context.getProperty(property);
166         // Check System properties to work around Equinox Bug:
167         // https://bugs.eclipse.org/bugs/show_bug.cgi?id=320459
168         if (value == null) value = System.getProperty(property);
169         return value;

172     /***
173      * Stops all the autostart bundles in reverse order
174      * @param context
175      * @throws Exception
176      */
177     public void stop(BundleContext context) throws Exception
178     {
179         List<URI> bundlesToStop = new ArrayList<URI>(autoStartBundleLocations);
180         Collections.reverse(bundlesToStop);
181         for (URI location : bundlesToStop) {
182             long id = currentManagedBundles.get(location).getBundleId();
183             if (id < 0) {
184                 logger.logp(Level.WARNING, "Main", "stop",
185                         "Not able to locate autostart bundle for location = {0}",
186                         new Object[]{location});
187                 continue;
188             }
189             final Bundle bundle = context.getBundle(id);
190             // check is necessary as bundle could have been uninstalled
191             if (bundle != null) bundle.stop();
192         }
193     }

195     private Set<Jar> discoverJars() {
196         final Set<Jar> jars = new HashSet<Jar>();
197         bundlesDir.listFiles(new FileFilter(){
198             final String JAR_EXT = ".jar";
199             public boolean accept(File file)
200             {
201                 if (file.isDirectory() && !excludedSubDirs.contains(file)) {
202                     file.listFiles(this);
203                 } else if (file.isFile()
204                         && file.getName().endsWith(JAR_EXT)
205                         && !file.getName().equals(THIS_JAR_NAME)) {
206                     jars.add(new Jar(file));
207                     return true;
208                 }
209                 return false;
210             }
211         });
212         return jars;
213     }

215     /***
216      * This method goes through all the currently installed bundles
217      * and returns information about those bundles whose location
218      * refers to a file in our {@link #bundlesDir}.
219      */
220     private void initCurrentManagedBundles()
221     {
222         Bundle[] bundles = this.context.getBundles();
223         String watchedDirPath = bundlesDir.toURI().normalize().getPath();
224         final long thisBundleId = context.getBundle().getBundleId();
225         for (Bundle bundle : bundles)
226         {
227             try
228             {
229                 final long id = bundle.getBundleId();
230                 if (id == 0 || id == thisBundleId) {
231                     // We can't manage system bundle or this bundle
232                     continue;
233                 }
234                 Jar jar = new Jar(bundle);
235                 String path = jar.getPath();
236                 if (path == null)
237                 {
238                     // jar.getPath is null means we could not parse the location
239                     // as a meaningful URI or file path. e.g., location
240                     // represented an Opaque URI.
241                     // We can't do any meaningful processing for this bundle.
242                     continue;
243                 }
244                 if (path.regionMatches(0, watchedDirPath, 0, watchedDirPath.length()))
245                 {
246                     currentManagedBundles.put(jar.getURI(), jar);
247                 }
248             }
249             catch (URISyntaxException e)
250             {
251                 // Ignore and continue.
252                 // This can never happen for bundles that have been installed
253                 // by FileInstall, as we always use proper filepath as location.
254             }
255         }
256     }

258     /***
259      * This method goes collects list of bundles that have been installed
260      * from the watched directory in previous run of the program,
261      * compares them with the current set of jar files,
262      * uninstalls old bundles, updates modified bundles, installs new bundles
263      * and refreshes the framework for the changes to take effect.
264      */
265     private void traverse() {
266         initCurrentManagedBundles();
267         final Collection<Jar> current = currentManagedBundles.values();
268         Set<Jar> discovered = discoverJars();
269 
270         // Find out all the new, deleted and common bundles.
271         // new = discovered - current
272         Set<Jar> newBundles = new HashSet<Jar>(discovered);
273         newBundles.removeAll(current);
274 
275         // deleted = current - discovered
276         Set<Jar> deletedBundles = new HashSet<Jar>(current);
277         deletedBundles.removeAll(discovered);
278 
279         // existing = intersection of current & discovered
280         Set<Jar> existingBundles = new HashSet<Jar>(discovered);
281         // We remove discovered ones from current, so that we are left
282         // with a collection of Jars made from files so that we can compare
283         // them with bundles.
284         existingBundles.retainAll(current);
285 
286         // We do the operations in the following order:
287         // uninstall, update, install, refresh & start.
288         int uninstalled = uninstall(deletedBundles);
289         int updated = update(existingBundles);
290         int installed = install(newBundles);
291         if ((installed + uninstalled + updated) > 0) {
292             refresh();
293         }
294     }

296     private int uninstall(Collection<Jar> bundles) {
297         int noOfBundlesUninstalled = 0;
298         for (Jar jar : bundles) {
299             Bundle bundle = context.getBundle(jar.getBundleId());
300             if (bundle == null) {
301                 // this is highly unlikely, but can't be ruled out.
302                 continue;
303             }
304             if (isExcludedFile(new File(jar.getPath()))) {
305                 // This is a bundle which is excluded from our processing.
306                 // The reason we have not discovered this jar is because
307                 // we have exclueded them in discoverJars(). So, don't uninstall
308                 // them.
309                 continue;
310             }
311             try {
312                 bundle.uninstall();
313                 noOfBundlesUninstalled++;
314                 logger.logp(Level.INFO, "Main", "uninstall",
315                         "Uninstalled bundle {0} installed from {1} ",
316                         new Object[]{bundle.getBundleId(), jar.getPath()});
317             } catch (Exception e) {
318                 logger.logp(Level.WARNING, "Main", "uninstall",
319                         "Failed to uninstall bundle " + jar.getPath(),
320                         e);
321             }
322         }
323         return noOfBundlesUninstalled;
324     }

326     private int update(Collection<Jar> jars) {
327         int updated = 0;
328         for (Jar jar : jars) {
329             final Jar existingJar= currentManagedBundles.get(jar.getURI());
330             if (jar.isNewer(existingJar)) {
331                 Bundle bundle = context.getBundle(existingJar.getBundleId());
332                 if (bundle == null) {
333                     // this is highly unlikely, but can't be ruled out.
334                     continue;
335                 }
336                 try {
337                     bundle.update();
338                     updated++;
339                     logger.logp(Level.INFO, "Main", "update",
340                             "Updated bundle {0} from {1} ",
341                             new Object[]{bundle.getBundleId(), jar.getPath()});
342                 } catch (Exception e) {
343                     logger.logp(Level.WARNING, "Main", "update",
344                             "Failed to update " + jar.getPath(),
345                             e);
346                 }
347             }
348         }
349         return updated;
350     }

352     private int install(Collection<Jar> jars) {
353         int installed = 0;
354         for (Jar jar : jars) {
355             try {
356                 final String path = jar.getPath();
357                 File file = new File(path);
358                 final FileInputStream is = new FileInputStream(file);
359                 try {
360                     Bundle b = context.installBundle(jar.getURI().toString(), is);
361                     installed++;
362                     currentManagedBundles.put(jar.getURI(), new Jar(b));
363                     logger.logp(Level.FINE, "Main", "install",
364                             "Installed bundle {0} from {1} ",
365                             new Object[]{b.getBundleId(), jar.getURI()});
366                 } finally {
367                     try {
368                         is.close();
369                     } catch (IOException e) {
370                     }
371                 }
372             } catch (Exception e) {
373                 logger.logp(Level.WARNING, "Main", "install",
374                         "Failed to install " + jar.getURI(),
375                         e);
376             }
377         }
378         return installed;
379     }

381     private void refresh() {
382         final ServiceReference reference =
383                 context.getServiceReference(PackageAdmin.class.getName());
384         PackageAdmin pa = PackageAdmin.class.cast(
385                 context.getService(reference));
386         pa.refreshPackages(null); // null to refresh any bundle that's obsolete
387         context.ungetService(reference);
388 
389         // This is a HACK - thanks to some weired optimization trick
390         // done for GlassFish. HK2 maintains a cache of inhabitants and
391         // that needs  to be recreated when there is a change in modules dir.
392         final String cacheDir = getProperty(HK2_CACHE_DIR);
393         if (cacheDir != null) {
394             File inhabitantsCache = new File(cacheDir, INHABITANTS_CACHE);
395             if (inhabitantsCache.exists()) inhabitantsCache.delete();
396         }
397     }

399     private boolean isExcludedFile(File f) {
400         String path = f.getPath();
401         for (File excludedSubDir : excludedSubDirs) {
402             String excludedSubDirPath = excludedSubDir.getPath();
403             if (path.regionMatches(0, excludedSubDirPath, 0, excludedSubDirPath.length())) {
404                 return true;
405             }
406         }
407         return false;
408     }

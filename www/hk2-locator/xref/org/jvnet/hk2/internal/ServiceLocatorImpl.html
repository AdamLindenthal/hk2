130     /**
131      * Must be called under lock
132      * 
133      * @param descriptor The descriptor to validate
134      * @param onBehalfOf The fella who is being validated (or null)
135      * @return true if every validator returned true
136      */
137     private boolean validate(SystemDescriptor<?> descriptor, Injectee onBehalfOf, Filter filter) {
138         for (ValidationService vs : allValidators) {
139             if (!descriptor.isValidating(vs)) continue;
140             
141             if (!vs.getValidator().validate(new ValidationInformationImpl(
142                     Operation.LOOKUP, descriptor, onBehalfOf, filter))) {
143                 return false;
144             }
145         }
146         
147         return true;
148     }

150     private List<ActiveDescriptor<?>> getDescriptors(Filter filter, Injectee onBehalfOf, boolean getParents) {
151         if (filter == null) throw new IllegalArgumentException("filter is null");
152         
153         synchronized (lock) {
154             List<SystemDescriptor<?>> sortMeOut;
155             if (filter instanceof IndexedFilter) {
156                 IndexedFilter df = (IndexedFilter) filter;
157                 
158                 if (df.getName() != null) {
159                     List<SystemDescriptor<?>> scopedByName;
160                     
161                     String name = df.getName();
162                     
163                     IndexedListData ild = descriptorsByName.get(name);
164                     scopedByName = (ild == null) ? null : ild.getSortedList();
165                     if (scopedByName == null) {
166                         scopedByName = Collections.emptyList();
167                     }
168                     
169                     if (df.getAdvertisedContract() != null) {
170                         sortMeOut = new LinkedList<SystemDescriptor<?>>();
171                         
172                         for (SystemDescriptor<?> candidate : scopedByName) {
173                             if (candidate.getAdvertisedContracts().contains(df.getAdvertisedContract())) {
174                                 sortMeOut.add(candidate);
175                             }
176                         }
177                     }
178                     else {
179                         sortMeOut = scopedByName;
180                     }
181                 }
182                 else if (df.getAdvertisedContract() != null) {
183                     String advertisedContract = df.getAdvertisedContract();
184                     
185                     IndexedListData ild = descriptorsByAdvertisedContract.get(advertisedContract);
186                     sortMeOut = (ild == null) ? null : ild.getSortedList();
187                     if (sortMeOut == null) {
188                         sortMeOut = Collections.emptyList();
189                         
190                     }
191                 }
192                 else {
193                     sortMeOut = allDescriptors.getSortedList();
194                 }
195             }
196             else {
197                 sortMeOut = allDescriptors.getSortedList();
198             }
199             
200             LinkedList<ActiveDescriptor<?>> retVal = new LinkedList<ActiveDescriptor<?>>();
201             
202             for (SystemDescriptor<?> candidate : sortMeOut) {
203                 if (!validate(candidate, onBehalfOf, filter)) continue;
204                 
205                 if (filter.matches(candidate)) {
206                     retVal.add(candidate);
207                 }
208             }
209             
210             if (getParents && parent != null) {
211                 TreeSet<ActiveDescriptor<?>> sorter = new TreeSet<ActiveDescriptor<?>>(DESCRIPTOR_COMPARATOR);
212                 
213                 sorter.addAll(retVal);
214                 sorter.addAll(parent.getDescriptors(filter));
215                 
216                 retVal.clear();
217                 
218                 retVal.addAll(sorter);
219             }
220             
221             return retVal;
222         }
223         
224     }

257     /* (non-Javadoc)
258      * @see org.glassfish.hk2.api.ServiceLocator#reifyDescriptor(org.glassfish.hk2.api.Descriptor)
259      */
260     @Override
261     public ActiveDescriptor<?> reifyDescriptor(Descriptor descriptor, Injectee injectee)
262             throws MultiException {
263         checkState();
264         if (descriptor == null) throw new IllegalArgumentException();
265         
266         if (!(descriptor instanceof ActiveDescriptor)) {
267             SystemDescriptor<?> sd = new SystemDescriptor<Object>(descriptor, new Long(id), new Long(getNextServiceId()));
268             
269             Class<?> implClass = loadClass(descriptor, injectee);
270             
271             Collector collector = new Collector();
272             sd.reify(implClass, this, collector);
273             
274             collector.throwIfErrors();
275             
276             return sd;
277         }
278         
279         // Descriptor is an active descriptor
280         ActiveDescriptor<?> active = (ActiveDescriptor<?>) descriptor;
281         if (active.isReified()) return active; 
282         
283         SystemDescriptor<?> sd;
284         if (active instanceof SystemDescriptor) {
285             sd = (SystemDescriptor<?>) active;
286         }
287         else {
288             sd = new SystemDescriptor<Object>(descriptor, new Long(id), new Long(getNextServiceId()));
289         }
290         
291         Class<?> implClass = sd.getPreAnalyzedClass();
292         if (implClass == null) {
293             implClass = loadClass(descriptor, injectee);
294         }
295         
296         Collector collector = new Collector();
297         synchronized(lock) {
298             sd.reify(implClass, this, collector);
299         }
300         
301         collector.throwIfErrors();
302         
303         return sd;
304     }

316     /* (non-Javadoc)
317      * @see org.glassfish.hk2.api.ServiceLocator#getInjecteeDescriptor(org.glassfish.hk2.api.Injectee)
318      */
319     @Override
320     public ActiveDescriptor<?> getInjecteeDescriptor(Injectee injectee)
321             throws MultiException {
322         if (injectee == null) throw new IllegalArgumentException();
323         checkState();
324         
325         Type requiredType = injectee.getRequiredType();
326         Class<?> rawType = ReflectionHelper.getRawClass(requiredType);
327         if (rawType == null) {
328             throw new MultiException(new IllegalArgumentException(
329                     "Invalid injectee with required type of " + injectee.getRequiredType() + " passed to getInjecteeDescriptor"));
330         }
331         
332         if (Provider.class.equals(rawType) || IterableProvider.class.equals(rawType) ) {
333             IterableProviderImpl<?> value = new IterableProviderImpl<Object>(this,
334                     (Utilities.getFirstTypeArgument(requiredType)),
335                     injectee.getRequiredQualifiers());
336             
337             return new ConstantActiveDescriptor<Object>(value, id);
338         }
339         
340         Set<Annotation> qualifiersAsSet = injectee.getRequiredQualifiers();
341         String name = Utilities.getNameFromAllQualifiers(qualifiersAsSet, injectee.getParent());
342         
343         Annotation qualifiers[] = qualifiersAsSet.toArray(new Annotation[qualifiersAsSet.size()]);
344         
345         ServiceHandle<?> handle = internalGetServiceHandle(injectee, requiredType, name, qualifiers);
346         if (handle == null) return null;
347         
348         return handle.getActiveDescriptor();
349     }

493     /* (non-Javadoc)
494      * @see org.glassfish.hk2.api.ServiceLocator#shutdown()
495      */
496     @Override
497     public void shutdown() {
498         synchronized (lock) {
499             if (shutdown) return;
500             
501 
502             List<ServiceHandle<?>> handles = getAllServiceHandles(BuilderHelper.createContractFilter(Context.class.getName()));
503 
504             for (ServiceHandle<?> handle : handles){
505                 if (handle.isActive()) {
506                     Context context = (Context) handle.getService();
507                     context.shutdown();
508                 }
509             }
510 
511             shutdown = true;
512 
513             allDescriptors.clear();
514             descriptorsByAdvertisedContract.clear();
515             descriptorsByName.clear();
516             allResolvers.clear();
517             allValidators.clear();
518             errorHandlers.clear();
519         }
520 
521     }

592     @SuppressWarnings("unchecked")
593     private <T> ServiceHandle<T> internalGetServiceHandle(Injectee onBehalfOf, Type contractOrImpl,
594             String name,
595             Annotation... qualifiers) throws MultiException {
596         if (contractOrImpl == null) throw new IllegalArgumentException();
597         
598         Class<?> rawClass = ReflectionHelper.getRawClass(contractOrImpl);
599         if (rawClass == null) return null;  // Can't be a TypeVariable or Wildcard
600         Utilities.checkLookupType(rawClass);
601         
602         rawClass = Utilities.translatePrimitiveType(rawClass);
603         
604         Filter filter = BuilderHelper.createNameAndContractFilter(rawClass.getName(), name);
605         NarrowResults results;
606         LinkedList<ErrorService> currentErrorHandlers = null;
607         synchronized (lock) {
608             List<ActiveDescriptor<?>> candidates = getDescriptors(filter, onBehalfOf, true);
609             results = narrow(candidates, contractOrImpl, name, false, onBehalfOf, qualifiers);
610             if (!results.getErrors().isEmpty()) {
611                 currentErrorHandlers = new LinkedList<ErrorService>(errorHandlers);
612             }
613         }
614         
615         if (currentErrorHandlers != null) {
616             // Do this next call OUTSIDE of the lock
617             Utilities.handleErrors(results, currentErrorHandlers);
618         }
619         
620         ActiveDescriptor<?> topDog = Utilities.getFirstThingInList(results.getResults());
621         if (topDog == null) return null;
622         
623         return getServiceHandle((ActiveDescriptor<T>) topDog);
624     }

649     /* (non-Javadoc)
650      * @see org.glassfish.hk2.api.ServiceLocator#getAllServiceHandles(java.lang.reflect.Type, java.lang.annotation.Annotation[])
651      */
652     @Override
653     public List<ServiceHandle<?>> getAllServiceHandles(
654             Type contractOrImpl, Annotation... qualifiers)
655             throws MultiException {
656         if (contractOrImpl == null) throw new IllegalArgumentException();
657         checkState();
658         
659         Class<?> rawClass = ReflectionHelper.getRawClass(contractOrImpl);
660         if (rawClass == null) {
661             throw new MultiException(new IllegalArgumentException("Type must be a class or parameterized type, it was " + contractOrImpl));
662         }
663         
664         Filter filter = BuilderHelper.createContractFilter(rawClass.getName());
665         NarrowResults results;
666         LinkedList<ErrorService> currentErrorHandlers = null;
667         synchronized (lock) {
668             List<ActiveDescriptor<?>> candidates = getDescriptors(filter);
669             results = narrow(candidates, contractOrImpl, null, true, null, qualifiers);
670             if (!results.getErrors().isEmpty()) {
671                 currentErrorHandlers = new LinkedList<ErrorService>(errorHandlers);
672             }
673         }
674         
675         if (currentErrorHandlers != null) {
676             // Do this next call OUTSIDE of the lock
677             Utilities.handleErrors(results, currentErrorHandlers);
678         }
679         
680         LinkedList<ServiceHandle<?>> retVal = new LinkedList<ServiceHandle<?>>();
681         for (ActiveDescriptor<?> candidate : results.getResults()) {
682             retVal.add(getServiceHandle(candidate));
683         }
684         
685         return retVal;
686     }

762     /**
763      * Checks the configuration operation before anything happens to the internal data structures.
764      * 
765      * @param dci The configuration that contains the proposed modifications
766      * @return A set of descriptors that is being removed fromthe configuration
767      */
768     private CheckConfigurationData checkConfiguration(DynamicConfigurationImpl dci) {
769         TreeSet<SystemDescriptor<?>> retVal = new TreeSet<SystemDescriptor<?>>(DESCRIPTOR_COMPARATOR);
770         boolean addOrRemoveOfInstanceListener = false;
771         boolean addOrRemoveOfInjectionResolver = false;
772         boolean addOrRemoveOfErrorHandler = false;
773         
774         for (Filter unbindFilter : dci.getUnbindFilters()) {
775             List<ActiveDescriptor<?>> results = getDescriptors(unbindFilter, null, false);
776             
777             for (ActiveDescriptor<?> result : results) {
778                 SystemDescriptor<?> candidate = (SystemDescriptor<?>) result;
779                 
780                 if (retVal.contains(candidate)) continue;
781                 
782                 for (ValidationService vs : allValidators) {
783                     if (!vs.getValidator().validate(new ValidationInformationImpl(
784                             Operation.UNBIND, candidate))) {
785                         throw new MultiException(new IllegalArgumentException("Descriptor " +
786                             candidate + " did not pass the UNBIND validation"));
787                     }
788                 }
789                 
790                 if (candidate.getAdvertisedContracts().contains(InstanceLifecycleListener.class.getName())) {
791                     addOrRemoveOfInstanceListener = true;
792                 }
793                 if (candidate.getAdvertisedContracts().contains(InjectionResolver.class.getName())) {
794                     addOrRemoveOfInjectionResolver = true;
795                 }
796                 if (candidate.getAdvertisedContracts().contains(ErrorService.class.getName())) {
797                     addOrRemoveOfErrorHandler = true;
798                 }
799                 
800                 retVal.add(candidate);
801             }
802         }
803         
804         for (SystemDescriptor<?> sd : dci.getAllDescriptors()) {
805             boolean checkScope = false;
806             if (sd.getAdvertisedContracts().contains(ValidationService.class.getName()) ||
807                 sd.getAdvertisedContracts().contains(ErrorService.class.getName()) ||
808                 sd.getAdvertisedContracts().contains(InstanceLifecycleListener.class.getName())) {
809                 // These gets reified right away
810                 reifyDescriptor(sd);
811                 
812                 checkScope = true;
813                 
814                 if (sd.getAdvertisedContracts().contains(ErrorService.class.getName())) {
815                     addOrRemoveOfErrorHandler = true;
816                 }
817                 if (sd.getAdvertisedContracts().contains(InstanceLifecycleListener.class.getName())) {
818                     addOrRemoveOfInstanceListener = true;
819                 }
820             }
821             
822             if (sd.getAdvertisedContracts().contains(InjectionResolver.class.getName())) {
823                 // This gets reified right away
824                 reifyDescriptor(sd);
825                 
826                 checkScope = true;
827                 
828                 if (Utilities.getInjectionResolverType(sd) == null) {
829                     throw new MultiException(new IllegalArgumentException(
830                             "An implementation of InjectionResolver must be a parameterized type and the actual type" +
831                             " must be an annotation"));
832                 }
833                 
834                 addOrRemoveOfInjectionResolver = true;
835             }
836             
837             if (sd.getAdvertisedContracts().contains(Context.class.getName())) {
838                 // This one need not be reified, it will get checked later
839                 checkScope = true;
840             }
841             
842             if (checkScope) {
843                 String scope = (sd.getScope() == null) ? PerLookup.class.getName() : sd.getScope() ;
844                 if (!scope.equals(Singleton.class.getName())) {
845                     throw new MultiException(new IllegalArgumentException(
846                             "The implementation class " +  sd.getImplementation() + " must be in the Singleton scope"));
847                 }
848             }
849             
850             for (ValidationService vs : allValidators) {
851                 Validator validator = vs.getValidator();
852                 if (validator == null) {
853                     throw new MultiException(new IllegalArgumentException("Validator was null from validation service" + vs));
854                 }
855                 
856                 if (!vs.getValidator().validate(new ValidationInformationImpl(
857                         Operation.BIND, sd))) {
858                     throw new MultiException(new IllegalArgumentException("Descriptor " + sd + " did not pass the BIND validation"));
859                 }
860             }
861         }
862         
863         return new CheckConfigurationData(retVal,
864                 addOrRemoveOfInstanceListener,
865                 addOrRemoveOfInjectionResolver,
866                 addOrRemoveOfErrorHandler);
867     }

1109     private NarrowResults narrow(List<ActiveDescriptor<?>> candidates,
1110             Type requiredType, String name, boolean getAll, Injectee injectee, Annotation... qualifiers) {
1111         NarrowResults retVal = new NarrowResults();
1112         
1113         Set<Annotation> requiredAnnotations = Utilities.fixAndCheckQualifiers(qualifiers, name);
1114         
1115         for (ActiveDescriptor<?> candidate : candidates) {
1116             // We will not reify them all, we will only reify until we match
1117             if (!candidate.isReified()) {
1118                 try {
1119                     candidate = reifyDescriptor(candidate, injectee);
1120                 }
1121                 catch (MultiException me) {
1122                     retVal.addError(candidate, injectee, me);
1123                     continue;
1124                 }
1125                 catch (Throwable th) {
1126                     retVal.addError(candidate, injectee, new MultiException(th));
1127                     continue;
1128                 }
1129             }
1130             
1131             // Now match it
1132             if (requiredType != null) {
1133                 boolean safe = false;
1134                 for (Type candidateType : candidate.getContractTypes()) {
1135                     if (TypeChecker.isTypeSafe(requiredType, candidateType)) {
1136                         safe = true;
1137                         break;
1138                     }
1139                 }
1140             
1141                 if (!safe) {
1142                     // Sorry, not type safe
1143                     continue;
1144                 }
1145             }
1146             
1147             // Now match the qualifiers
1148             Set<Annotation> candidateAnnotations = candidate.getQualifierAnnotations();
1149             
1150             // Checking requiredAnnotations isEmpty is a performance optimization which avoids
1151             // a potentially expensive doPriv call in the second part of the AND statement
1152             if (!requiredAnnotations.isEmpty() &&
1153                     !Utilities.annotationContainsAll(candidateAnnotations, requiredAnnotations)) {
1154                 // The qualifiers do not match
1155                 continue;
1156             }
1157             
1158             // If we are here, then this one matches
1159             retVal.addGoodResult(candidate);
1160             
1161             if (!getAll) {
1162                 // We found one, we don't want to reify any more than we have to
1163                 break;
1164             }
1165         }
1166         
1167         return retVal;
1168     }

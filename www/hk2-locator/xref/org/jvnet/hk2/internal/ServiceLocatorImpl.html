126     /**
127      * Must be called under lock
128      * 
129      * @param descriptor The descriptor to validate
130      * @param onBehalfOf The fella who is being validated (or null)
131      * @return true if every validator returned true
132      */
133     private boolean validate(SystemDescriptor<?> descriptor, Injectee onBehalfOf, Filter filter) {
134         for (ValidationService vs : allValidators) {
135             if (!descriptor.isValidating(vs)) continue;
136             
137             if (!vs.getValidator().validate(new ValidationInformationImpl(
138                     Operation.LOOKUP, descriptor, onBehalfOf, filter))) {
139                 return false;
140             }
141         }
142         
143         return true;
144     }

146     private List<ActiveDescriptor<?>> getDescriptors(Filter filter, Injectee onBehalfOf, boolean getParents) {
147         if (filter == null) throw new IllegalArgumentException("filter is null");
148         
149         synchronized (lock) {
150             List<SystemDescriptor<?>> sortMeOut;
151             if (filter instanceof IndexedFilter) {
152                 IndexedFilter df = (IndexedFilter) filter;
153                 
154                 if (df.getName() != null) {
155                     List<SystemDescriptor<?>> scopedByName;
156                     
157                     String name = df.getName();
158                     
159                     scopedByName = descriptorsByName.get(name);
160                     if (scopedByName == null) {
161                         scopedByName = Collections.emptyList();
162                     }
163                     
164                     if (df.getAdvertisedContract() != null) {
165                         sortMeOut = new LinkedList<SystemDescriptor<?>>();
166                         
167                         for (SystemDescriptor<?> candidate : scopedByName) {
168                             if (candidate.getAdvertisedContracts().contains(df.getAdvertisedContract())) {
169                                 sortMeOut.add(candidate);
170                             }
171                         }
172                     }
173                     else {
174                         sortMeOut = scopedByName;
175                     }
176                 }
177                 else if (df.getAdvertisedContract() != null) {
178                     String advertisedContract = df.getAdvertisedContract();
179                     
180                     sortMeOut = descriptorsByAdvertisedContract.get(advertisedContract);
181                     if (sortMeOut == null) {
182                         sortMeOut = Collections.emptyList();
183                         
184                     }
185                 }
186                 else {
187                     sortMeOut = allDescriptors;
188                 }
189             }
190             else {
191                 sortMeOut = allDescriptors;
192             }
193             
194             TreeSet<ActiveDescriptor<?>> sorter = new TreeSet<ActiveDescriptor<?>>(DESCRIPTOR_COMPARATOR);
195             
196             for (SystemDescriptor<?> candidate : sortMeOut) {
197                 if (!validate(candidate, onBehalfOf, filter)) continue;
198                 
199                 if (filter.matches(candidate)) {
200                     sorter.add(candidate);
201                 }
202             }
203             
204             if (getParents && parent != null) {
205                 sorter.addAll(parent.getDescriptors(filter));
206             }
207             
208             return new LinkedList<ActiveDescriptor<?>>(sorter);
209         }
210         
211     }

244     /* (non-Javadoc)
245      * @see org.glassfish.hk2.api.ServiceLocator#reifyDescriptor(org.glassfish.hk2.api.Descriptor)
246      */
247     @Override
248     public ActiveDescriptor<?> reifyDescriptor(Descriptor descriptor, Injectee injectee)
249             throws MultiException {
250         checkState();
251         if (descriptor == null) throw new IllegalArgumentException();
252         
253         if (!(descriptor instanceof ActiveDescriptor)) {
254             SystemDescriptor<?> sd = new SystemDescriptor<Object>(descriptor, new Long(id), new Long(getNextServiceId()));
255             
256             Class<?> implClass = loadClass(descriptor, injectee);
257             
258             Collector collector = new Collector();
259             sd.reify(implClass, this, collector);
260             
261             collector.throwIfErrors();
262             
263             return sd;
264         }
265         
266         // Descriptor is an active descriptor
267         ActiveDescriptor<?> active = (ActiveDescriptor<?>) descriptor;
268         if (active.isReified()) return active; 
269         
270         SystemDescriptor<?> sd;
271         if (active instanceof SystemDescriptor) {
272             sd = (SystemDescriptor<?>) active;
273         }
274         else {
275             sd = new SystemDescriptor<Object>(descriptor, new Long(id), new Long(getNextServiceId()));
276         }
277         
278         Class<?> implClass = loadClass(descriptor, injectee);
279         Collector collector = new Collector();
280         synchronized(lock) {
281             sd.reify(implClass, this, collector);
282         }
283         
284         collector.throwIfErrors();
285         
286         return sd;
287     }

299     /* (non-Javadoc)
300      * @see org.glassfish.hk2.api.ServiceLocator#getInjecteeDescriptor(org.glassfish.hk2.api.Injectee)
301      */
302     @Override
303     public ActiveDescriptor<?> getInjecteeDescriptor(Injectee injectee)
304             throws MultiException {
305         if (injectee == null) throw new IllegalArgumentException();
306         checkState();
307         
308         Type requiredType = injectee.getRequiredType();
309         Class<?> rawType = ReflectionHelper.getRawClass(requiredType);
310         if (rawType == null) {
311             throw new MultiException(new IllegalArgumentException(
312                     "Invalid injectee with required type of " + injectee.getRequiredType() + " passed to getInjecteeDescriptor"));
313         }
314         
315         if (Provider.class.equals(rawType) || IterableProvider.class.equals(rawType) ) {
316             IterableProviderImpl<?> value = new IterableProviderImpl<Object>(this,
317                     (Utilities.getFirstTypeArgument(requiredType)),
318                     injectee.getRequiredQualifiers());
319             
320             return new ConstantActiveDescriptor<Object>(value, id);
321         }
322         
323         Set<Annotation> qualifiersAsSet = injectee.getRequiredQualifiers();
324         String name = Utilities.getNameFromAllQualifiers(qualifiersAsSet, injectee.getParent());
325         
326         Annotation qualifiers[] = qualifiersAsSet.toArray(new Annotation[qualifiersAsSet.size()]);
327         
328         ServiceHandle<?> handle = internalGetServiceHandle(injectee, requiredType, name, qualifiers);
329         if (handle == null) return null;
330         
331         return handle.getActiveDescriptor();
332     }

538     @SuppressWarnings("unchecked")
539     private <T> ServiceHandle<T> internalGetServiceHandle(Injectee onBehalfOf, Type contractOrImpl,
540             String name,
541             Annotation... qualifiers) throws MultiException {
542         if (contractOrImpl == null) throw new IllegalArgumentException();
543         
544         Class<?> rawClass = ReflectionHelper.getRawClass(contractOrImpl);
545         if (rawClass == null) return null;  // Can't be a TypeVariable or Wildcard
546         rawClass = Utilities.translatePrimitiveType(rawClass);
547         
548         Filter filter = BuilderHelper.createNameAndContractFilter(rawClass.getName(), name);
549         NarrowResults results;
550         LinkedList<ErrorService> currentErrorHandlers = null;
551         synchronized (lock) {
552             List<ActiveDescriptor<?>> candidates = getDescriptors(filter, onBehalfOf, true);
553             results = narrow(candidates, contractOrImpl, name, false, onBehalfOf, qualifiers);
554             if (!results.getErrors().isEmpty()) {
555                 currentErrorHandlers = new LinkedList<ErrorService>(errorHandlers);
556             }
557         }
558         
559         if (currentErrorHandlers != null) {
560             // Do this next call OUTSIDE of the lock
561             Utilities.handleErrors(results, currentErrorHandlers);
562         }
563         
564         ActiveDescriptor<?> topDog = Utilities.getFirstThingInList(results.getResults());
565         if (topDog == null) return null;
566         
567         return getServiceHandle((ActiveDescriptor<T>) topDog);
568     }

594      * @see org.glassfish.hk2.api.ServiceLocator#getAllServiceHandles(java.lang.reflect.Type, java.lang.annotation.Annotation[])
595      */
596     @Override
597     public List<ServiceHandle<?>> getAllServiceHandles(
598             Type contractOrImpl, Annotation... qualifiers)
599             throws MultiException {
600         if (contractOrImpl == null) throw new IllegalArgumentException();
601         checkState();
602         
603         Class<?> rawClass = ReflectionHelper.getRawClass(contractOrImpl);
604         if (rawClass == null) {
605             throw new MultiException(new IllegalArgumentException("Type must be a class or parameterized type, it was " + contractOrImpl));
606         }
607         
608         Filter filter = BuilderHelper.createContractFilter(rawClass.getName());
609         NarrowResults results;
610         LinkedList<ErrorService> currentErrorHandlers = null;
611         synchronized (lock) {
612             List<ActiveDescriptor<?>> candidates = getDescriptors(filter);
613             results = narrow(candidates, contractOrImpl, null, true, null, qualifiers);
614             if (!results.getErrors().isEmpty()) {
615                 currentErrorHandlers = new LinkedList<ErrorService>(errorHandlers);
616             }
617         }
618         
619         if (currentErrorHandlers != null) {
620             // Do this next call OUTSIDE of the lock
621             Utilities.handleErrors(results, currentErrorHandlers);
622         }
623         
624         LinkedList<ServiceHandle<?>> retVal = new LinkedList<ServiceHandle<?>>();
625         for (ActiveDescriptor<?> candidate : results.getResults()) {
626             retVal.add(getServiceHandle(candidate));
627         }
628         
629         return retVal;
630     }

663             Utilities.handleErrors(results, currentErrorHandlers);
664         }
665         
666         SortedSet<ServiceHandle<?>> retVal = new TreeSet<ServiceHandle<?>>(HANDLE_COMPARATOR);
667         for (ActiveDescriptor<?> candidate : results.getResults()) {
668             retVal.add(getServiceHandle(candidate));
669         }

671         return new LinkedList<ServiceHandle<?>>(retVal);

682         if (qualifier == null) throw new IllegalArgumentException("qualifier is null");
683         
684         final Set<String> allQualifiers = new LinkedHashSet<String>();
685         allQualifiers.add(qualifier.annotationType().getName());
686         
687         for (Annotation anno : qualifiers) {
688             String addMe = anno.annotationType().getName();
689             if (allQualifiers.contains(addMe)) {
690                 throw new IllegalArgumentException("Multiple qualifiers with name " + addMe);
691             }
692             
693             allQualifiers.add(addMe);
694         }
695         
696         return getAllServiceHandles(new Filter() {
697 
698             @Override
699             public boolean matches(Descriptor d) {
700                 return d.getQualifiers().containsAll(allQualifiers);
701             }
702             
703         });

707      * Checks the configuration operation before anything happens to the internal data structures.

709      * @param dci The configuration that contains the proposed modifications
710      * @return A set of descriptors that is being removed fromthe configuration
711      */
712     private CheckConfigurationData checkConfiguration(DynamicConfigurationImpl dci) {
713         TreeSet<SystemDescriptor<?>> retVal = new TreeSet<SystemDescriptor<?>>(DESCRIPTOR_COMPARATOR);
714         boolean addOrRemoveOfInstanceListener = false;
715         boolean addOrRemoveOfInjectionResolver = false;
716         boolean addOrRemoveOfErrorHandler = false;
717         
718         for (Filter unbindFilter : dci.getUnbindFilters()) {
719             List<ActiveDescriptor<?>> results = getDescriptors(unbindFilter, null, false);
720             
721             for (ActiveDescriptor<?> result : results) {
722                 SystemDescriptor<?> candidate = (SystemDescriptor<?>) result;
723                 
724                 if (retVal.contains(candidate)) continue;
725                 
726                 for (ValidationService vs : allValidators) {
727                     if (!vs.getValidator().validate(new ValidationInformationImpl(
728                             Operation.UNBIND, candidate))) {
729                         throw new MultiException(new IllegalArgumentException("Descriptor " +
730                             candidate + " did not pass the UNBIND validation"));
731                     }
732                 }
733                 
734                 if (candidate.getAdvertisedContracts().contains(InstanceLifecycleListener.class.getName())) {
735                     addOrRemoveOfInstanceListener = true;
736                 }
737                 if (candidate.getAdvertisedContracts().contains(InjectionResolver.class.getName())) {
738                     addOrRemoveOfInjectionResolver = true;
739                 }
740                 if (candidate.getAdvertisedContracts().contains(ErrorService.class.getName())) {
741                     addOrRemoveOfErrorHandler = true;
742                 }
743                 
744                 retVal.add(candidate);
745             }
746         }
747         
748         for (SystemDescriptor<?> sd : dci.getAllDescriptors()) {
749             boolean checkScope = false;
750             if (sd.getAdvertisedContracts().contains(ValidationService.class.getName()) ||
751                 sd.getAdvertisedContracts().contains(ErrorService.class.getName()) ||
752                 sd.getAdvertisedContracts().contains(InstanceLifecycleListener.class.getName())) {
753                 // These gets reified right away
754                 reifyDescriptor(sd);
755                 
756                 checkScope = true;
757                 
758                 if (sd.getAdvertisedContracts().contains(ErrorService.class.getName())) {
759                     addOrRemoveOfErrorHandler = true;
760                 }
761                 if (sd.getAdvertisedContracts().contains(InstanceLifecycleListener.class.getName())) {
762                     addOrRemoveOfInstanceListener = true;
763                 }
764             }
765             
766             if (sd.getAdvertisedContracts().contains(InjectionResolver.class.getName())) {
767                 // This gets reified right away
768                 reifyDescriptor(sd);
769                 
770                 checkScope = true;
771                 
772                 if (Utilities.getInjectionResolverType(sd) == null) {
773                     throw new MultiException(new IllegalArgumentException(
774                             "An implementation of InjectionResolver must be a parameterized type and the actual type" +
775                             " must be an annotation"));
776                 }
777                 
778                 addOrRemoveOfInjectionResolver = true;
779             }
780             
781             if (sd.getAdvertisedContracts().contains(Context.class.getName())) {
782                 // This one need not be reified, it will get checked later
783                 checkScope = true;
784             }
785             
786             if (checkScope) {
787                 String scope = (sd.getScope() == null) ? PerLookup.class.getName() : sd.getScope() ;
788                 if (!scope.equals(Singleton.class.getName())) {
789                     throw new MultiException(new IllegalArgumentException(
790                             "The implementation class " +  sd.getImplementation() + " must be in the Singleton scope"));
791                 }
792             }
793             
794             for (ValidationService vs : allValidators) {
795                 if (!vs.getValidator().validate(new ValidationInformationImpl(
796                         Operation.BIND, sd))) {
797                     throw new MultiException(new IllegalArgumentException("Descriptor " + sd + " did not pass the BIND validation"));
798                 }
799             }
800         }
801         
802         return new CheckConfigurationData(retVal,
803                 addOrRemoveOfInstanceListener,
804                 addOrRemoveOfInjectionResolver,
805                 addOrRemoveOfErrorHandler);
806     }

972         if (scope.equals(Singleton.class)) return singletonContext;
973         if (scope.equals(PerLookup.class)) return perLookupContext;

975         Type actuals[] = new Type[1];
976         actuals[0] = scope;

980                 protectedGetAllServiceHandles(findContext));

982         try {
983             Context<?> retVal = null;
984             for (ServiceHandle<Context<?>> contextHandle : contextHandles) {
985                 Context<?> context = contextHandle.getService();
986                 
987                 if (!context.isActive()) continue;

1016     private NarrowResults narrow(List<ActiveDescriptor<?>> candidates,
1017             Type requiredType, String name, boolean getAll, Injectee injectee, Annotation... qualifiers) {
1018         NarrowResults retVal = new NarrowResults();
1019         
1020         Set<Annotation> requiredAnnotations = Utilities.fixAndCheckQualifiers(qualifiers, name);
1021         
1022         for (ActiveDescriptor<?> candidate : candidates) {
1023             // We will not reify them all, we will only reify until we match
1024             if (!candidate.isReified()) {
1025                 try {
1026                     candidate = reifyDescriptor(candidate, injectee);
1027                 }
1028                 catch (MultiException me) {
1029                     retVal.addError(candidate, injectee, me);
1030                     continue;
1031                 }
1032                 catch (Throwable th) {
1033                     retVal.addError(candidate, injectee, new MultiException(th));
1034                     continue;
1035                 }
1036             }
1037             
1038             // Now match it
1039             if (requiredType != null) {
1040                 boolean safe = false;
1041                 for (Type candidateType : candidate.getContractTypes()) {
1042                     if (TypeChecker.isTypeSafe(requiredType, candidateType)) {
1043                         safe = true;
1044                         break;
1045                     }
1046                 }
1047             
1048                 if (!safe) {
1049                     // Sorry, not type safe
1050                     continue;
1051                 }
1052             }
1053             
1054             // Now match the qualifiers
1055             Set<Annotation> candidateAnnotations = candidate.getQualifierAnnotations();
1056             
1057             // Checking requiredAnnotations isEmpty is a performance optimization which avoids
1058             // a potentially expensive doPriv call in the second part of the AND statement
1059             if (!requiredAnnotations.isEmpty() &&
1060                     !Utilities.annotationContainsAll(candidateAnnotations, requiredAnnotations)) {
1061                 // The qualifiers do not match
1062                 continue;
1063             }
1064             
1065             // If we are here, then this one matches
1066             retVal.addGoodResult(candidate);
1067             
1068             if (!getAll) {
1069                 // We found one, we don't want to reify any more than we have to
1070                 break;
1071             }
1072         }
1073         
1074         return retVal;
1075     }

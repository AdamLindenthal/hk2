52  /**

56  public class DynamicConfigurationImpl implements DynamicConfiguration {
57      private final ServiceLocatorImpl locator;
58      private final LinkedList<SystemDescriptor<?>> allDescriptors = new LinkedList<SystemDescriptor<?>>();
59      private final LinkedList<Filter> allUnbindFilters = new LinkedList<Filter>();
60      
61      private final Object lock = new Object();
62      private boolean committed = false;
63  
64      /* package */ DynamicConfigurationImpl(ServiceLocatorImpl locator) {
65          this.locator = locator;
66      }
67  
68      /* (non-Javadoc)
69       * @see org.glassfish.hk2.api.Configuration#bind(org.glassfish.hk2.api.Descriptor)
70       */
71      @Override
72      public ActiveDescriptor<?> bind(Descriptor key) {
73          checkState();
74          checkDescriptor(key);
75          
76          SystemDescriptor<?> sd = new SystemDescriptor<Object>(key,
77                  new Long(locator.getLocatorId()),
78                  new Long(locator.getNextServiceId()));
79          
80          allDescriptors.add(sd);
81          
82          return sd;
83      }
84      
85      /* (non-Javadoc)
86       * @see org.glassfish.hk2.api.Configuration#bind(org.glassfish.hk2.api.FactoryDescriptors)
87       */
88      @Override
89      public FactoryDescriptors bind(FactoryDescriptors factoryDescriptors) {
90          if (factoryDescriptors == null) throw new IllegalArgumentException("factoryDescriptors is null");
91          
92          // Now a bunch of validations
93          Descriptor asService = factoryDescriptors.getFactoryAsAService();
94          Descriptor asFactory = factoryDescriptors.getFactoryAsAFactory();
95          
96          checkDescriptor(asService);
97          checkDescriptor(asFactory);
98          
99          String implClassService = asService.getImplementation();
100         String implClassFactory = asFactory.getImplementation();
101         
102         if (!implClassService.equals(implClassFactory)) {
103             throw new IllegalArgumentException("The implementation classes must match (" +
104                 implClassService + "/" + implClassFactory + ")");
105         }
106         
107         if (!asService.getDescriptorType().equals(DescriptorType.CLASS)) {
108             throw new IllegalArgumentException("The getFactoryAsService descriptor must be of type CLASS");
109         }
110         if (!asFactory.getDescriptorType().equals(DescriptorType.FACTORY)) {
111             throw new IllegalArgumentException("The getFactoryAsFactory descriptor must be of type FACTORY");
112         }
113         
114         // Bind the factory first, so normally people get the factory, not the service
115         final ActiveDescriptor<?> boundAsFactory = bind(asFactory);
116         final ActiveDescriptor<?> boundAsService = bind(asService);
117         
118         return new FactoryDescriptors() {
119 
120             @Override
121             public Descriptor getFactoryAsAService() {
122                 return boundAsService;
123             }
124 
125             @Override
126             public Descriptor getFactoryAsAFactory() {
127                 return boundAsFactory;
128             }
129             
130         };
131     }

133     /* (non-Javadoc)
134      * @see org.glassfish.hk2.api.Configuration#addActiveDescriptor(org.glassfish.hk2.api.ActiveDescriptor)
135      */
136     @Override
137     public <T> ActiveDescriptor<T> addActiveDescriptor(ActiveDescriptor<T> activeDescriptor)
138             throws IllegalArgumentException {
139         checkState();
140         checkDescriptor(activeDescriptor);
141         
142         if (!activeDescriptor.isReified()) {
143             throw new IllegalArgumentException();
144         }
145         
146         SystemDescriptor<T> retVal = new SystemDescriptor<T>(activeDescriptor,
147                 new Long(locator.getLocatorId()),
148                 new Long(locator.getNextServiceId()));
149         
150         allDescriptors.add(retVal);
151         
152         return retVal;
153     }
154     
155     /* (non-Javadoc)
156      * @see org.glassfish.hk2.api.Configuration#addActiveDescriptor(java.lang.Class)
157      */
158     @Override
159     public <T> ActiveDescriptor<T> addActiveDescriptor(Class<T> rawClass)
160             throws IllegalArgumentException {
161         ActiveDescriptor<T> ad = Utilities.createAutoDescriptor(rawClass, locator);
162         
163         return addActiveDescriptor(ad);
164     }
165     
166     /* (non-Javadoc)
167      * @see org.glassfish.hk2.api.Configuration#addUnbindFilter(org.glassfish.hk2.api.Filter)
168      */
169     @Override
170     public void addUnbindFilter(Filter unbindFilter)
171             throws IllegalArgumentException {
172         if (unbindFilter == null) throw new IllegalArgumentException();
173         
174         allUnbindFilters.add(unbindFilter);
175     }
176 
177     /* (non-Javadoc)
178      * @see org.glassfish.hk2.api.DynamicConfiguration#commit()
179      */
180     @Override
181     public void commit() throws MultiException {
182         synchronized (lock) {
183             checkState();
184             
185             committed = true;
186         }
187         
188         locator.addConfiguration(this);
189     }
190     
191     private void checkState() {
192         synchronized (lock) {
193             if (committed) throw new IllegalStateException();
194         }
195     }
196     
197     private static void checkDescriptor(Descriptor d) {
198         if (d == null) throw new IllegalArgumentException();
199         if (d.getImplementation() == null) throw new IllegalArgumentException();
200         if (d.getAdvertisedContracts() == null) throw new IllegalArgumentException();
201         if (d.getDescriptorType() == null) throw new IllegalArgumentException();
202         if (d.getMetadata() == null) throw new IllegalArgumentException();
203         if (d.getQualifiers() == null) throw new IllegalArgumentException();
204     }

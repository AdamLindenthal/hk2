50  /**
51   * Tests for the &#064;Unqualified directive
52   * 
53   * @author jwells
54   *
55   */
56  public class UnqualifiedTest {
57      private final static String TEST_NAME = "UnqualifiedTest";
58      private final static ServiceLocator locator = LocatorHelper.create(TEST_NAME, new UnqualifiedModule());
59      
60      public final static String SERVER_HEALTH_COMMAND = "ServerHealthCommand";
61      public final static String SERVER_DATA_COMMAND = "ServerDataCommand";
62      public final static String CLIENT_HEALTH_COMMAND = "ClientHealthCommand";
63      public final static String CLIENT_DATA_COMMAND = "ClientDataCommand";
64      public final static String LIST_COMMAND_COMMAND = "ListCommandCommand";
65      
66      public final static String TRACTOR_ELEPHANT_SHOE_TOY = "TractorElephantShoeToy";
67      public final static String SHOE_TOY = "ShoeToy";
68      public final static String UNKNOWN_TOY = "UnknownToy";
69      
70      @Test
71      public void testUnqualifiedOnlyGetsUnqualified() {
72          ListCommandCommand lcc = locator.getService(ListCommandCommand.class);
73          Assert.assertNotNull(lcc);
74          
75          {
76              List<Command> remoteCommands = lcc.getRemoteCommands();
77              Assert.assertSame(2, remoteCommands.size());
78          
79              Assert.assertSame(SERVER_DATA_COMMAND, remoteCommands.get(0).getName());
80              Assert.assertSame(SERVER_HEALTH_COMMAND, remoteCommands.get(1).getName());
81          }
82          
83          {
84              List<Command> allCommands = lcc.getAllCommands();
85              Assert.assertSame(5, allCommands.size());
86          
87              Assert.assertSame(SERVER_DATA_COMMAND, allCommands.get(0).getName());
88              Assert.assertSame(CLIENT_DATA_COMMAND, allCommands.get(1).getName());
89              Assert.assertSame(CLIENT_HEALTH_COMMAND, allCommands.get(2).getName());
90              Assert.assertSame(SERVER_HEALTH_COMMAND, allCommands.get(3).getName());
91              Assert.assertSame(LIST_COMMAND_COMMAND, allCommands.get(4).getName());
92          }
93          
94          // And  now the real test
95          {
96              List<Command> localCommands = lcc.getLocalCommands();  // Local commands are unqualified
97              Assert.assertSame(2, localCommands.size());
98          
99              Assert.assertSame(CLIENT_DATA_COMMAND, localCommands.get(0).getName());
100             Assert.assertSame(CLIENT_HEALTH_COMMAND, localCommands.get(1).getName());
101         }
102         
103         // Lets make sure it also got the correct "first" unqualified service
104         Assert.assertSame(CLIENT_DATA_COMMAND, lcc.getFirstUnqualifiedCommand().getName());
105         
106         // Now not just unqualified, but those not qualified with @Remote
107         {
108             List<Command> notRemoteCommands = lcc.getNotRemoteCommands();  // Local commands are unqualified
109             Assert.assertSame(3, notRemoteCommands.size());
110         
111             Assert.assertSame(CLIENT_DATA_COMMAND, notRemoteCommands.get(0).getName());
112             Assert.assertSame(CLIENT_HEALTH_COMMAND, notRemoteCommands.get(1).getName());
113             Assert.assertSame(LIST_COMMAND_COMMAND, notRemoteCommands.get(2).getName());
114         }
115         
116     }

242     @Test
243     public void testFindOneDescriptor() {
244         Descriptor addMe = BuilderHelper.link(SimpleService3.class).
245                 to(SimpleContract.class).build();
246         
247         ActiveDescriptor<?> ss3Desc1 = ServiceLocatorUtilities.addOneDescriptor(locator, addMe);
248         ActiveDescriptor<?> ss3Desc2 = ServiceLocatorUtilities.addOneDescriptor(locator, addMe);
249         
250         // Look up two ways, with the specific and the non-specific descriptor
251         ActiveDescriptor<?> found1 = ServiceLocatorUtilities.findOneDescriptor(locator, addMe);
252         Assert.assertSame(ss3Desc1, found1);
253         
254         ActiveDescriptor<?> found2 = ServiceLocatorUtilities.findOneDescriptor(locator, ss3Desc1);
255         Assert.assertSame(ss3Desc1, found2);
256         
257         ActiveDescriptor<?> found3 = ServiceLocatorUtilities.findOneDescriptor(locator, ss3Desc2);
258         Assert.assertSame(ss3Desc2, found3);
259         
260         // Now remove first one, ensure I can still get with the not-found specific descriptor
261         ServiceLocatorUtilities.removeOneDescriptor(locator, ss3Desc1);
262         
263         ActiveDescriptor<?> found4 = ServiceLocatorUtilities.findOneDescriptor(locator, ss3Desc1);
264         Assert.assertSame(ss3Desc2, found4);
265         
266         ServiceLocatorUtilities.removeOneDescriptor(locator, ss3Desc2);
267         
268         Assert.assertNull(ServiceLocatorUtilities.findOneDescriptor(locator, addMe));
269         Assert.assertNull(ServiceLocatorUtilities.findOneDescriptor(locator, ss3Desc1));
270         Assert.assertNull(ServiceLocatorUtilities.findOneDescriptor(locator, ss3Desc2));

401     public static class NonReifiedActiveDescriptor<T> extends AbstractActiveDescriptor<T> implements ActiveDescriptor<T> {
402         /**
403          * 
404          */
405         private static final long serialVersionUID = 8750311164952618038L;
406         
407         private final AbstractActiveDescriptor<T> delegate;
408         
409         private NonReifiedActiveDescriptor(AbstractActiveDescriptor<T> delegate) {
410             super(delegate.getContractTypes(),
411                     delegate.getScopeAnnotation(),
412                     delegate.getName(),
413                     delegate.getQualifierAnnotations(),
414                     delegate.getDescriptorType(),
415                     delegate.getDescriptorVisibility(),
416                     delegate.getRanking(),
417                     delegate.isProxiable(),
418                     delegate.getMetadata());
419             
420             this.delegate = delegate;
421         }
422         
423         /**
424          * This method is the point of this class.  Since this ActiveDescriptor is not reified the
425          * method should treat it as a descriptor, not an active descriptor.  And hence the constant
426          * should NOT be honored.
427          */
428         @Override
429         public boolean isReified() {
430             return false;
431         }
432         
433         public String getImplementation() {
434             return delegate.getImplementation();
435         }
436 
437         /* (non-Javadoc)
438          * @see org.glassfish.hk2.api.ActiveDescriptor#getImplementationClass()
439          */
440         @Override
441         public Class<?> getImplementationClass() {
442             return delegate.getImplementationClass();
443         }
444 
445         /* (non-Javadoc)
446          * @see org.glassfish.hk2.api.ActiveDescriptor#create(org.glassfish.hk2.api.ServiceHandle)
447          */
448         @Override
449         public T create(ServiceHandle<?> root) {
450             return delegate.create(root);
451         }
452         
453     }

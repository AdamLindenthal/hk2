19      /***
20       * Returns instances of all the classes implementing the
21       * service interface passed as a parameter.
22       * 
23       * @param serviceClass the service interface
24       * @return an iterator of instances of classes implementing the passed
25       * interface
26       */
27      public <T> Iterable<T> getProviders(final Class<T> serviceClass) {
28          return new Iterable<T>() {
29              final Iterable<Class<? extends T>> classes = getProvidersClass(serviceClass);
30  
31              public Iterator<T> iterator() {
32                  return new AdapterIterator<T,Class<? extends T>>(classes.iterator()) {
33                      protected T adapt(Class<? extends T> providerClass) {
34                          try {
35                              return providerClass.newInstance();
36                          } catch(java.lang.InstantiationException e) {
37                              Utils.getDefaultLogger().log(Level.SEVERE, "Cannot instantiate " +serviceClass + " service "
38                                      + serviceClass.getName() + " implementation : ", e);
39                          } catch(IllegalAccessException e) {
40                              Utils.getDefaultLogger().log(Level.SEVERE, "Cannot instantiate " + serviceClass + " service "
41                                      + serviceClass.getName() + " implementation : ", e);
42                          } catch(RuntimeException e) {
43                              Utils.getDefaultLogger().log(Level.SEVERE, "Runtime exception while instantiating " +serviceClass + " service "
44                                      + serviceClass.getName() + " implementation : ", e);
45                          }
46                          return null;
47                      }

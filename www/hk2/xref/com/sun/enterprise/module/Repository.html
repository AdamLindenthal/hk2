24  package com.sun.enterprise.module;
25  
26  import java.io.IOException;
27  import java.net.URI;
28  import java.util.List;
29  
30  /***
31   * A Repository instance is an abstraction of a set of accessible
32   * modules. Repository can be local or remote and are used to
33   * procure modules implementation based on constraints like name or
34   * version.
35   *
36   * @author Jerome Dochez
37   */
38  public interface Repository {
39      
40      /***
41       * Returns the repository name
42       * @return repository name
43       */
44      public String getName();
45      
46      /***
47       * Returns the repository location
48       * @return the URI for the repository location
49       */
50      public URI getLocation();
51      
52      /***
53       * Finds and returns a <code>DefaultModuleDefinition</code> instance
54       * for a module given the name and version constraints.
55       * @param name the requested module name
56       * @param version
57       *      the module version. Can be null if the caller doesn't care about the version.
58       * @return a <code>DefaultModuleDefinition</code> or null if not found
59       * in this repository.
60       */
61      public ModuleDefinition find(String name, String version);
62      
63      /***
64       * Returns a list of all modules available in this repository 
65       * @return a list of available modules
66       */
67      public List<ModuleDefinition> findAll();
68      
69      /***
70       * Finds and returns a list of all the available versions of a 
71       * module given its name.
72       * @param name the requested module name
73       */
74      public List<ModuleDefinition> findAll(String name);
75      
76      /***
77       * Initialize the repository for use. This need to be called at least
78       * once before any find methods is invoked.  
79       * @throws IOException if an error occur accessing the repository
80       */
81      public void initialize() throws IOException;
82      
83      /***
84       * Shutdown the repository. After this call return, the find methods cannot 
85       * be used until initialize() is called again.
86       * @throws IOException if an error occur accessing the repository
87       */
88      public void shutdown() throws IOException;
89  
90      /***
91       * Creates a module instance from a Module definition.
92       * Different implementation of repositories may need to use different means
93       * of initializing a Module instance depending on the module's definition nature.
94       * For instance, an Netbeans module or a JSR277/Glassfish module could use different
95       * implementation of Module.
96       *
97       * @param registry is the registry that will own the module
98       * @param moduleDef the requested module definition
99       * @return the uninitialized Module instance
100      */
101     public Module newModule(ModulesRegistry registry, ModuleDefinition moduleDef);
102

104      * This module adds a new Module Definition to the repository.
105      *
106      * @param definition is the module definition
107      * @return true if the addition was successful
108      */

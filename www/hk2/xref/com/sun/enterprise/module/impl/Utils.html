52      public static void identifyCyclicDependency(Module m, Logger logger) {
53  
54          StringBuffer tree = new StringBuffer();
55          tree.append(m.getName());
56          Vector<Module> traversed = new Vector<Module>();
57          boolean success = traverseAndFind(m, m, traversed);
58          if (success) {
59              traversed.remove(0);
60              for (Module mod : traversed) {
61                  tree.append("-->" + mod.getName());
62              }
63              tree.append("-->" + m.getName());
64          logger.log(Level.SEVERE, "Cyclic dependency : " + tree.toString());
65          }
66      }
67  
68      static private boolean traverseAndFind(Module toTraverse, Module toFind, Vector<Module> traversed) {
69  
70          traversed.add(toTraverse);
71          for (ModuleDependency md : toTraverse.getModuleDefinition().getDependencies())  {
72              ModulesRegistry registry = toTraverse.getRegistry();
73              for (Module mod : registry.getModules()) {
74                  if (mod.getName().equals(md.getName())) {
75                      if (mod!=null) {
76                          if (mod.getName().equals(toFind.getName())) {
77                              return true;
78                          }
79                          if (traverseAndFind(mod, toFind, traversed)) {
80                              return true;
81                          }
82                      }
83  
84                  }
85              }
86          }
87          traversed.remove(toTraverse);
88          return false;
89      }

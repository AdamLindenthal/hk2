40  /***
41   * {@link ModuleDefinition} implementation that picks up most of the module
42   * properties from the manifest file of the jar, as baked in by
43   * the hk2-maven-plugin.
44   *
45   * @author Jerome Dochez
46   */
47  public class DefaultModuleDefinition implements ModuleDefinition {
48      
49      private final String name;
50      private final String version;
51      private final String[] publicPkgs;
52      protected final List<ModuleDependency> dependencies = new ArrayList<ModuleDependency>();
53      protected final List<URI> classPath = new ArrayList<URI>();
54      private final String importPolicy;
55      private final String lifecyclePolicy;
56      private final Manifest manifest;
57      /***
58       * Main attributes section of the manifest.
59       * Always non-null.
60       */
61      protected final Attributes mainAttributes;
62      /***
63       * Service entries exposed by this module.
64       */
65      private final List<ServiceProviderInfo> services = new ArrayList<ServiceProviderInfo>();
66  
67      /*** TO DO need to support a URI constructor */
68      public DefaultModuleDefinition(File location) throws IOException {
69          this(location, null);
70      }
71      
72      protected DefaultModuleDefinition(File location, Attributes attr) throws IOException {
73  
74          classPath.add(location.toURI());
75          
76          Jar jarFile = Jar.create(location);
77          manifest = jarFile.getManifest();
78          if (attr==null && manifest!=null) {
79              attr = manifest.getMainAttributes();
80          }
81  
82          // no attributes whatsoever, I just use an empty collection to avoid 
83          // testing for null all the time.
84          if (attr==null) {
85              attr = new Attributes();
86          }
87          this.mainAttributes = attr;
88                  
89          // name
90          if (attr.getValue(ManifestConstants.BUNDLE_NAME)!=null) {
91              name = attr.getValue(ManifestConstants.BUNDLE_NAME);
92          } else {
93              name = location.getName();
94          }
95  
96          // classpath
97          parseClassPath(attr, location.toURI());
98          
99          // class exported...
100         String exported = attr.getValue(ManifestConstants.PKG_EXPORT_NAME);
101         ArrayList<String> tmpList = new ArrayList<String>();
102         for( String token : new Tokenizer(exported,",")) {
103             tmpList.add(token);
104         }
105         publicPkgs = tmpList.toArray(new String[tmpList.size()]);
106         
107         // class imported
108         String imported = attr.getValue(ManifestConstants.BUNDLE_IMPORT_NAME);
109         for( String token : new Tokenizer(imported,",")) {
110             // no versioning so far...
111             dependencies.add(new ModuleDependency(token,""));
112         }
113 
114         importPolicy = attr.getValue(ManifestConstants.IMPORT_POLICY);
115         lifecyclePolicy = attr.getValue(ManifestConstants.LIFECYLE_POLICY);
116         
117         version = "1.0"; // for now ;-)
118 
119         parseAttributes(attr);
120 
121         jarFile.getServiceProviders(services);
122     }
123 
124     /***
125      * Parses <tt>{@value ManifestConstants#CLASS_PATH}</tt> from manifest attributes
126      * and updates URI list.
127      */
128     protected void parseClassPath(Attributes attr, URI baseURI) {
129         String classpath = attr.getValue(ManifestConstants.CLASS_PATH);
130         for( String classpathElement : new Tokenizer(classpath," ")) {
131             classpathElement = decorateClassPath(classpathElement);
132             URI result;
133             File ref = new File(classpathElement);
134             if (!ref.isAbsolute()) {
135                 try {
136                     result = baseURI.resolve(classpathElement);
137                 } catch (IllegalArgumentException e) {
138                     throw new IllegalArgumentException("Unable to parse Class-Path entry '"+classpath+"' in "+baseURI);
139                 }
140             } else
141                 result = ref.toURI();
142             classPath.add(result);
143         }
144     }
145 
146     /***
147      * Extension point to manipulate the classpath element before it's parsed.
148      * @see #parseClassPath(Attributes,URI) 
149      */
150     protected String decorateClassPath(String classpathElement) {
151         return classpathElement;
152     }
153 
154     /***
155      * Extensibility point to parse more information from Manifest attributes.
156      *
157      * @param attr
158      *      Main attributes of the manifest. Always non-null.
159      */
160     protected void parseAttributes(Attributes attr) {
161         // noop
162     }

165      * Returns the name of the module
166      * @return the name
167      */
168     public String getName() {

170     }

174      * public interfaces of this module.
175      * @return the list of public packages or classes

177     public String[] getPublicInterfaces() {

179     }

181     /***
182      * Returns the list of module dependencies
183      * @return the ModuleDependency
184      */
185     public ModuleDependency[] getDependencies() {
186         return dependencies.toArray(new ModuleDependency[dependencies.size()]);
187     }
188     
189     /***
190      * Returns the list of URI locations forming the classpath for this module.
191      *
192      * @return the list of URI locations for this module
193      */
194     public URI[] getLocations() {
195         return classPath.toArray(new URI[classPath.size()]);
196     }
197     
198     /***
199      * Returns the module's version
200      * @return the module's version
201      */
202     public String getVersion() {
203         return version;
204     }
205     
206     /***
207      * Returns the class name implementing the
208      * {@link com.sun.enterprise.module.ImportPolicy ImportPolicy} interface for this module or
209      * null if there is no such implementation
210      * @return the  {@link com.sun.enterprise.module.ImportPolicy ImportPolicy} implementation class name
211      */
212     public String getImportPolicyClassName() {
213         return importPolicy;
214     }
215     
216     /***
217      * Returns the class name implementing the
218      * {@link com.sun.enterprise.module.LifecyclePolicy LifecyclePolicy} interface for this module or
219      * null if there is no such implementation
220      * @return the  {@link com.sun.enterprise.module.LifecyclePolicy LifecyclePolicy} implementation class name
221      */
222     public String getLifecyclePolicyClassName() {
223         return lifecyclePolicy;
224     }
225     
226     /***
227      * Returns the manifest file from the module's implementation jar file
228      * @return the manifest file
229      */
230     public Manifest getManifest() {
231         return manifest;
232     }
233 
234     public Collection<ServiceProviderInfo> getServiceProviders() {
235         return services;
236     }

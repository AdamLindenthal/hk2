75      /***
76       * Loads a jar file and builds a {@link ModuleDefinition}.
77       *
78       * <p>
79       * The system allows {@link ModuleDefinition}s to be built in any way,
80       * but in practice module jars need to be built in a way agnostic
81       * to {@link Repository} implementations (so that same module could
82       * be used in different {@link Repository}s), so it makes sense
83       * to try to stick to the "common" loading scheme.
84       *
85       * @param jar
86       *      Either a jar file or a directory that has the same structure as a jar file. 
87       */
88      protected ModuleDefinition loadJar(File jar) throws IOException {
89          Jar jarFile = Jar.create(jar);
90          Manifest manifest = jarFile.getManifest();
91          if (manifest==null || manifest.getMainAttributes().getValue(ManifestConstants.BUNDLE_NAME)==null) {
92              // we cannot find a manifest file in the bundle, so we look
93              // if there is a manifest file residing outside of the jar
94              // file with the same jar file name with a .mf extension,
95              // so for foo.jar, we look for foo.mf and if it is there,
96              // we use it as the manifest file.
97              String simpleName = jarFile.getBaseName();
98              File manifestFile = new File(jar.getParentFile(), simpleName+".mf");
99              if (manifestFile.exists()) {
100                 InputStream is=null;
101                 try {
102                     is = new BufferedInputStream(new FileInputStream(manifestFile));
103                     manifest = new Manifest(is);
104                 } finally {
105                     if (is!=null) {
106                         is.close();
107                     }
108                 }
109             }
110         }
111         if (manifest!=null) {
112             Attributes attr = manifest.getMainAttributes();
113             String bundleName = attr.getValue(ManifestConstants.BUNDLE_NAME);
114             if (bundleName!=null) {
115                 return newModuleDefinition(jar, attr);
116             }
117         }
118 
119         /* this has undesirable side effect of picking up jar files
120            referenced via class-path as modules. so just return null.
121 
122         // if we are here, that mean that either there was no co-bundled or
123         // external manifest file for this jar file OR the manifest file did
124         // not contain the manifest entries for module management
125         // in that case, I am just adding the jar file to the repository
126         // with no dependency management.
127         return new CookedModuleDefinition(jar, null);
128          */
129         return null;
130     }

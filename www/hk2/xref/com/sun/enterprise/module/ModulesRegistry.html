94      /***
95       * Add a new <code>Repository</code> to this registry. From now on
96       * the repository will be used to procure requested module not yet registered
97       * in this registry instance. Repository can be searched in a particular 
98       * order (to accomodate performance requirements like looking at local 
99       * repositories first), a search order (1 to 100) can be specified when 
100      * adding a repository to the registry (1 is highest priority). 
101      * @param repository new repository to attach to this registry
102      * @param weight int value from 1 to 100 to specify the search order
103      */
104     public synchronized void addRepository(Repository repository, int weight) {
105         while (repositories.containsKey(weight)) {
106             weight++;
107         }
108         repositories.put(weight, repository);
109     }

113      * the repository will be used to procure requested nodule not 
114      * registered in this instance.

116      */
117     public synchronized void addRepository(Repository repository) {
118         repositories.put(100+repositories.size(), repository);
119     }

121     /***
122      * Remove a repository from the list of attached repositories to 
123      * this instances. After this call, the <code>Repository</code>
124      * name will not be used to procure missing modules any 
125      * longer
126      * @param name name of the repository to remove
127      */
128     public synchronized void removeRepository(String name) {
129         for (Integer weight : repositories.keySet()) {
130             Repository repo = repositories.get(weight);
131             if (repo.getName().equals(name)) {
132                 repositories.remove(weight);
133                 return;
134             }
135         }
136     }
137     /***
138      * Returns the <code>Module</code> instance giving a name and version 
139      * constraints.
140      *
141      * @param name the module name
142      * @param version
143      *      the module version. Can be null if the caller doesn't care about the version.
144      * @return the module instance or null if none can be found
145      * @throws ResolveError if the module dependencies cannot be resolved
146      */
147     public Module makeModuleFor(String name, String version) throws ResolveError {
148         Module module;
149 
150         if(parent!=null) {
151             module = parent.makeModuleFor(name,version);
152             if(module!=null)        return module;
153         }
154 
155         module = modules.get(name);
156         if (module==null) {
157             module = loadFromRepository(name, version);
158             if (module!=null) {
159                 add(module);
160             }
161         }
162         if (module!=null) {
163             module.resolve();
164         }
165         return module;
166     }

168     /***
169      * Find and return a loaded Module that has the package name in its list 
170      * of exported interfaces.
171      *
172      * @param packageName the requested implementation package name. 
173      * @return the <code>Module</code> instance implementing the package
174      * name or null if not found.
175      * @throws ResolveError if the module dependencies cannot be resolved
176      */
177     public Module makeModuleFor(String packageName) throws ResolveError {
178         if(parent!=null) {
179             Module m = parent.makeModuleFor(packageName);
180             if(m!=null)     return m;
181         }
182 
183         for (Module module : modules.values()) {
184             String[] exportedPkgs = module.getModuleDefinition().getPublicInterfaces();
185             for (String exportedPkg : exportedPkgs) {
186                 if (exportedPkg.equals(packageName)) {
187                     module.resolve();
188                     return module;
189                 }
190             }
191         }
192         return null;
193     }

195     /***
196      * Creates and return a new private module implementation giving a name and
197      * version constraints. A private module is like any other module except 
198      * it is not registered to be shared by other potential module users.
199      *
200      * @param moduleName the module name
201      * @param version the desired version
202      * @return the new private module or null if cannot be found
203      * @throws ResolveError if the module dependencies cannot be resolved
204      */
205     public Module newPrivateModuleFor(String moduleName, String version) {
206         if(parent!=null) {
207             Module m = parent.newPrivateModuleFor(moduleName,version);
208             if(m!=null)     return m;
209         }
210 
211         if (modules.containsKey(moduleName)) {
212             Module module = modules.get(moduleName);
213             Module privateModule = new Module(this,module.getModuleDefinition());
214             privateModule.resolve();
215             return privateModule;
216         }
217         return loadFromRepository(moduleName, version);
218         
219     }

221     private Module loadFromRepository(String name, String version) {
222         Set<Integer> keys = repositories.keySet();
223         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
224         sortedKeys.addAll(keys);
225         for (Integer key : sortedKeys) {
226             Repository repo = repositories.get(key);
227             ModuleDefinition moduleDef = repo.find(name, version);
228             if (moduleDef!=null) {
229                 return new Module(this,moduleDef);
230             }
231         }
232         return null;
233     }

235     /***
236      * Add a new module to this registry. Once added, the module will be 
237      * available through one of the getServiceImplementor methods.
238      * @param newModule the new module
239      */
240     private void add(Module newModule) {
241         //if (Utils.isLoggable(Level.INFO)) {
242         //    Utils.getDefaultLogger().info("New module " + newModule);
243         //}
244         assert newModule.getRegistry()==this;
245         modules.put(newModule.getModuleDefinition().getName(), newModule);
246 
247         // pick up providers from this module
248         for( ServiceProviderInfoList.Entry spi : newModule.getServiceProviders().getEntries() ) {
249             for( String name : spi.providerNames )
250                 providers.put(name,newModule);
251         }
252     }

272     /*** 
273      * Returns the list of shared Modules registered in this instance.
274      *
275      * <p>
276      * The returned list will not include the modules defined in the ancestor
277      * {@link ModulesRegistry}s.
278      *
279      * @return an umodifiable list of loaded modules
280      */
281     public Collection<Module> getModules() {
282         // force repository extraction
283         Set<Integer> keys = repositories.keySet();
284         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
285         sortedKeys.addAll(keys);
286         for (Integer key : sortedKeys) {
287             Repository repo = repositories.get(key);
288             for (ModuleDefinition moduleDef : repo.findAll()) {
289                 if (modules.get(moduleDef.getName())==null) {
290                     add(new Module(this,moduleDef));
291                     // don't resolve such modules, we just want to know about them
292                 }
293             }
294         }
295         return modules.values();
296     }

325      * Registers a new DefaultModuleDefinition in this registry. Using this module
326      * definition, the registry will be capable of created shared and private
327      * <code>Module</code> instances. 
328      */
329     public synchronized Module add(ModuleDefinition info) throws ResolveError {
330         
331         // it may have already been created
332         Module service = makeModuleFor(info.getName(), info.getVersion());
333         if (service!=null) {
334         //    Utils.getDefaultLogger().info("Service " + info.getName() 
335         //       + " already registered");
336         } else {
337             // create the service instance
338             service = new Module(this,info);
339             add(service);   
340         }
341         return service;
342     }

427     /***
428      * Registers a running service, this is useful when other components need
429      * to have access to a provider of a service without having to create
430      * a new instance and initialize it.
431      * @param serviceClass the service interface
432      * @param provider the provider of that service.
433      */
434     public <T> void registerRunningService(Class<T> serviceClass, T provider) {
435         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
436         if (rs==null) {
437             rs = new CopyOnWriteArrayList<T>();
438             CopyOnWriteArrayList existing = runningServices.putIfAbsent(serviceClass, rs);
439             if(existing!=null)
440                 rs = existing;
441         }
442         rs.add(provider);
443     }
444 
445     /***
446      * Removes a running service, this is useful when a service instance is no longer
447      * available as a provider of a service.
448      */
449     public <T> boolean unregisterRunningService(Class<T> serviceClass, T provider) {
450         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
451         if (rs==null) {
452             return false;
453         }
454         return rs.remove(provider);
455     }

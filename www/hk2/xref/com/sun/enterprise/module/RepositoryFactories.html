34  public class RepositoryFactories {

38      private ArrayList<RepositoryFactory> factories = new ArrayList();
39      
40      /***
41       * Return the instance holding registered repository factories
42       * @return the instance holding factories
43       */
44      public static RepositoryFactories getInstance() {

46      }

49      private RepositoryFactories() {
50      }

53       * Add a new <code> RepositoryFactory </code> to the list of 
54       * repository factories. 
55       * @param factory the new factory to add
56       */
57      public void addRepositoryFactory(RepositoryFactory factory) {        
58          factories.add(factory);
59      }
60      
61      /***
62       * Returns an interator of registered <code>ReposistoryFactory</code>
63       * @return an iterator or registered factories
64       */
65      public Iterator<RepositoryFactory> getFactories() {
66          return factories.iterator();
67      }

24  package com.sun.enterprise.module;
25  
26  import java.net.URI;
27  
28  /***
29   * Factory to create repositories. 
30   *
31   * @author Jerome Dochez
32   */
33  public abstract class RepositoryFactory {
34          
35      /***
36       * Returns true if this factory can handle this type of repository
37       * @param type repository type
38       */
39      public abstract boolean handleType(String type);
40      
41      /***
42       * Creates a new <code>Repository</code> with a parent (for delegating 
43       * module resolutions) and a name. The URI source identifies the repository 
44       * location.
45       * @param parent the parent <code>Repository</code> to delegate module 
46       * resolution
47       * @param name the repository name
48       * @param source the location of the repository
49       */
50      public abstract Repository createRepository(Repository parent, String name, URI source);
51      
52      /***
53       * Creates a new <code>Repository</code>. The URI source identifies the 
54       * repository location.
55       * @param name the repository name
56       * @param source the location of the repository
57       */    
58      public abstract Repository createRepository(String name, URI source);
59  }

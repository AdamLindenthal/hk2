56  public interface ModulesRegistry extends ModuleChangeListener, HabitatFactory {
57      /***
58       * Creates a new child {@link ModulesRegistry} in this {@link ModulesRegistry}.
59       */
60      ModulesRegistry createChild();
61  
62      /***
63       * Creates an uninitialized {@link Habitat}
64       *
65       */
66      Habitat newHabitat() throws ComponentException;
67  
68      /***
69       * Creates a {@link Habitat} from all the modules in this registry
70       *
71       * @param name
72       *      Determines which inhabitants descriptors are loaded.
73       *      (so that different parallel habitats can be
74       *      created over the same modules registry.)
75       */
76      Habitat createHabitat(String name) throws ComponentException;
77  
78      /***
79       * Creates a {@link Habitat} from all the modules in this registry.
80       *
81       * <p>
82       * Default {@link InhabitantsParser} is used.
83       *
84       * @param name
85       *      Determines which inhabitants descriptors are loaded.
86       *      (so that different parallel habitats can be
87       *      created over the same modules registry.)
88       * @param h
89       *      Habitat to initialize, null if it should be created
90       *
91       * @return initialized Habitat
92       */
93      Habitat createHabitat(String name, Habitat h) throws ComponentException;
94  
95      /***
96       * Creates a {@link Habitat} from all the modules in this registry,
97       * by using the given {@link InhabitantsParser}.
98       *
99       * @param name
100      *      Determines which inhabitants descriptors are loaded.
101      *      (so that different parallel habitats can be
102      *      created over the same modules registry.)
103      * @param parser
104      *      Parser to load inhabitants file. Sub types can be used
105      *      to customize the parsing behavior, like ignoring some inhabitants deliberately.
106      *
107      * @return initialized Habitat
108      */
109     Habitat createHabitat(String name, InhabitantsParser parser) throws ComponentException;
110 
111     /***
112      * Add a new <code>Repository</code> to this registry. From now on
113      * the repository will be used to procure requested module not yet registered
114      * in this registry instance. Repository can be searched in a particular
115      * order (to accomodate performance requirements like looking at local
116      * repositories first), a search order (1 to 100) can be specified when
117      * adding a repository to the registry (1 is highest priority).
118      * @param repository new repository to attach to this registry
119      * @param weight int value from 1 to 100 to specify the search order
120      */
121     void addRepository(Repository repository, int weight);
122 
123     /***
124      * Add a new <code>Repository</code> to this registry. From now on
125      * the repository will be used to procure requested nodule not
126      * registered in this instance.
127      * @param repository new repository to attach to this registry
128      */
129     void addRepository(Repository repository);
130 
131     /***
132      * Remove a repository from the list of attached repositories to
133      * this instances. After this call, the <code>Repository</code>
134      * name will not be used to procure missing modules any
135      * longer
136      * @param name name of the repository to remove
137      */
138     void removeRepository(String name);
139 
140     /***
141      * Get a repository from the list of attached repositories
142      *
143      * @param name name of the repository to return
144      * @return the repository or null if not found
145      */
146     Repository getRepository(String name);
147 
148     /***
149      * Returns the <code>Module</code> instance giving a name and version
150      * constraints.
151      *
152      * @param name the module name
153      * @param version the module version. Caller should specify a correct version.
154      * @return the module instance or null if none can be found
155      * @throws ResolveError if the module dependencies cannot be resolved
156      */
157     Module makeModuleFor(String name, String version) throws ResolveError;
158 
159     /***
160      * Returns the <code>Module</code> instance giving a name and version
161      * constraints.
162      *
163      * @param name the module name
164      * @param version the module version. Caller should specify a correct version.
165      * @param resolve should the module be resolved or not
166      * @return the module instance or null if none can be found
167      * @throws ResolveError if the module dependencies cannot be resolved
168      */
169     Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError;
170 
171     /***
172      * Find and return a loaded Module that has the package name in its list
173      * of exported interfaces.
174      *
175      * @param packageName the requested implementation package name.
176      * @return the <code>Module</code> instance implementing the package
177      * name or null if not found.
178      * @throws ResolveError if the module dependencies cannot be resolved
179      */
180     Module makeModuleFor(String packageName) throws ResolveError;
181 
182     /***
183      * Returns the list of shared Modules registered in this instance.
184      *
185      * <p>
186      * The returned list will not include the modules defined in the ancestor
187      * {@link ModulesRegistry}s.
188      *
189      * @return an umodifiable list of loaded modules
190      */
191     Collection<Module> getModules();
192 
193     /***
194      * Returns the list of shared Modules registered in this instance whose name
195      * matches the given name
196      *
197      * <p>
198      * The returned list will not include the modules defined in the ancestor
199      * {@link ModulesRegistry}s.
200      *
201      * @return an umodifiable list of loaded modules having names that match
202      * the given name
203      */
204     Collection<Module> getModules(String moduleName);
205 
206     /***
207      * Detaches all the modules from this registry. The modules are not
208      * deconstructed when calling this method.
209      */
210     void detachAll();
211 
212     /***
213      * Registers a new DefaultModuleDefinition in this registry. Using this module
214      * definition, the registry will be capable of created shared and private
215      * <code>Module</code> instances.
216      */
217     Module add(ModuleDefinition info) throws ResolveError;
218 
219     /***
220      * Registers a new DefaultModuleDefinition in this registry. Using this module
221      * definition, the registry will be capable of created shared and private
222      * <code>Module</code> instances.
223      * @param info ModuleDefinition representing the new module content
224      * @param resolve should the new module be resolved or not
225      */
226     Module add(ModuleDefinition info, boolean resolve) throws ResolveError;
227 
228     /***
229      * Print a Registry dump to the logger
230      * @param logger the logger to dump on
231      */
232     void print(Logger logger);
233 
234     /***
235      * Add a <code>ModuleLifecycleListener</code> to this registry. The listener
236      * will be notified for each module startup and shutdown.
237      * @param listener the listener implementation
238      */
239     void register(ModuleLifecycleListener listener);
240 
241     /***
242      * Removes an <code>ModuleLifecycleListener</code> from this registry.
243      * Notification of module startup and shutdown will not be emitted to this
244      * listener any longer.
245      * @param listener the listener to unregister
246      */
247     void unregister(ModuleLifecycleListener listener);
248 
249     /***
250      * Shuts down this module's registry, apply housekeeping tasks
251      *
252      */
253     void shutdown();
254 
255     void dumpState(PrintStream writer);

258 
259     /***
260      * Returns a collection of Module containing at least one implementation
261      * of the passed service interface class.
262      *
263      * @param serviceClass the service interface class
264      * @return a collection of module
265      */
266     Iterable<Module> getModulesProvider(Class serviceClass);
267 
268     /***
269      * Registers a running service, this is useful when other components need
270      * to have access to a provider of a service without having to create
271      * a new instance and initialize it.
272      * @param serviceClass the service interface
273      * @param provider the provider of that service.
274      */
275     <T> void registerRunningService(Class<T> serviceClass, T provider);
276 
277     /***
278      * Removes a running service, this is useful when a service instance is no longer
279      * available as a provider of a service.
280      */
281     <T> boolean unregisterRunningService(Class<T> serviceClass, T provider);
282 
283     /***
284      * Returns all running services implementation of the passed service
285      * interface
286      * @param serviceClass the service interface
287      * @return the list of providers of that service.
288      */
289     <T> List<T> getRunningServices(Class<T> serviceClass);

291     void setParentClassLoader(ClassLoader parent);
292 
293     ClassLoader getParentClassLoader();
294 
295     /***
296      * Returns a ClassLoader capable of loading classes from a set of modules identified

298      *
299      * @param parent the parent class loader for the returned class loader instance

301      *        capable of loading classes from
302      * @return class loader instance
303      * @throws com.sun.enterprise.module.ResolveError if one of the provided module

305      */
306     ClassLoader getModulesClassLoader(ClassLoader parent,
307                                       Collection<ModuleDefinition> defs)
308         throws ResolveError;
309 
310     /***
311      * Returns a ClassLoader capable of loading classes from a set of modules identified
312      * by their module definition and also load new urls.
313      *
314      * @param parent the parent class loader for the returned class loader instance
315      * @param defs module definitions for all modules this classloader should be
316      *        capable of loading
317      * @param urls urls to be added to the module classloader
318      * @return class loader instance
319      * @throws com.sun.enterprise.module.ResolveError if one of the provided module
320      *         definition cannot be resolved
321      */
322     ClassLoader getModulesClassLoader(ClassLoader parent,
323                                       Collection<ModuleDefinition> defs,
324                                       URL[] urls) throws ResolveError;

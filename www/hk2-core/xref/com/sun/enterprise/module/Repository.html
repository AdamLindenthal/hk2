38  package com.sun.enterprise.module;
39  
40  import java.io.IOException;
41  import java.net.URI;
42  import java.util.List;
43  
44  /***
45   * A Repository instance is an abstraction of a set of accessible
46   * modules. Repository can be local or remote and are used to
47   * procure modules implementation based on constraints like name or
48   * version.
49   *
50   * @author Jerome Dochez
51   */
52  public interface Repository {
53      
54      /***
55       * Returns the repository name
56       * @return repository name
57       */
58      public String getName();
59      
60      /***
61       * Returns the repository location
62       * @return the URI for the repository location
63       */
64      public URI getLocation();
65      
66      /***
67       * Finds and returns a <code>DefaultModuleDefinition</code> instance
68       * for a module given the name and version constraints.
69       * @param name the requested module name
70       * @param version
71       *      the module version. Can be null if the caller doesn't care about the version.
72       * @return a <code>DefaultModuleDefinition</code> or null if not found
73       * in this repository.
74       */
75      public ModuleDefinition find(String name, String version);
76      
77      /***
78       * Returns a list of all modules available in this repository 
79       * @return a list of available modules
80       */
81      public List<ModuleDefinition> findAll();
82      
83      /***
84       * Finds and returns a list of all the available versions of a 
85       * module given its name.
86       * @param name the requested module name
87       */
88      public List<ModuleDefinition> findAll(String name);
89      
90      /***
91       * Initialize the repository for use. This need to be called at least
92       * once before any find methods is invoked.  
93       * @throws IOException if an error occur accessing the repository
94       */
95      public void initialize() throws IOException;
96      
97      /***
98       * Shutdown the repository. After this call return, the find methods cannot 
99       * be used until initialize() is called again.
100      * @throws IOException if an error occur accessing the repository
101      */
102     public void shutdown() throws IOException;
103 
104     /***
105      * Returns the plain jar files installed in this repository. Plain jar files
106      * are not modules, they do not have the module's metadata and can only be used
107      * when referenced from a module dependency list or when added to a class
108      * loader directly
109      *
110      * @return jar files location stored in this repository.
111      */
112     public List<URI> getJarLocations();
113 
114     /***
115      * Add a listener to changes happening to this repository. Repository can
116      * change during the lifetime of an execution (files added/removed/changed)
117      *
118      * @param listener implementation listening to this repository changes
119      * @return true if the listener was added successfully
120      */
121     public boolean addListener(RepositoryChangeListener listener);
122 
123     /***
124      * Removes a previously registered listener
125      *
126      * @param listener the previously registered listener
127      * @return true if the listener was successfully unregistered
128      */
129     public boolean removeListener(RepositoryChangeListener listener);
130 
131 }

142     public Habitat createHabitat(String name, InhabitantsParser parser) throws ComponentException {
143         try {
144             Habitat habitat = parser.habitat;
145 
146             for (final Module module : getModules())
147                 parseInhabitants(module, name,parser);
148 
149             populateConfig(habitat);
150 
151             // default modules registry is the one that created the habitat
152             habitat.addIndex(new ExistingSingletonInhabitant<ModulesRegistry>(this),
153                     ModulesRegistry.class.getName(), null);
154             habitats.put(name, habitat);
155 
156             habitat.initialized();
157             
158             return habitat;
159         } catch (Exception e) {
160             throw new ComponentException("Failed to create a habitat",e);
161         }
162     }

164     protected void populateConfig(Habitat habitat) throws BootException {
165         ConfigParser configParser = new ConfigParser(habitat);
166         for( Populator p : habitat.getAllByContract(Populator.class) )
167             p.run(configParser);
168     }

175     /***
176      * Add a new <code>Repository</code> to this registry. From now on
177      * the repository will be used to procure requested module not yet registered
178      * in this registry instance. Repository can be searched in a particular 
179      * order (to accomodate performance requirements like looking at local 
180      * repositories first), a search order (1 to 100) can be specified when 
181      * adding a repository to the registry (1 is highest priority). 
182      * @param repository new repository to attach to this registry
183      * @param weight int value from 1 to 100 to specify the search order
184      */
185     public synchronized void addRepository(Repository repository, int weight) {
186         // check that we don't already have this repository
187         for (Repository repo : repositories.values()) {
188             if (repo.getLocation().equals(repository.getLocation())) {
189                 throw new RuntimeException("repository at " + repository.getLocation() + " already registered");
190             }
191         }
192         while (repositories.containsKey(weight)) {
193             weight++;
194         }
195         repositories.put(weight, repository);
196     }

211      * name will not be used to procure missing modules any

213      * @param name name of the repository to remove

215     public synchronized void removeRepository(String name) {
216         for (Integer weight : repositories.keySet()) {
217             Repository repo = repositories.get(weight);

219                 repositories.remove(weight);

221             }
222         }

254     public Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError {
255         Module module;
256         Logger.getAnonymousLogger().fine("this.makeModuleFor("+name+ ", " +
257                 version + ", " + resolve + ") called.");
258         if(parent!=null) {
259             module = parent.makeModuleFor(name,version, resolve);
260             if(module!=null)        return module;
261         }
262 
263         module = modules.get(AbstractFactory.getInstance().createModuleId(name, version));
264         if (module==null) {
265             module = loadFromRepository(name, version);
266             if (module!=null) {
267                 add(module);
268             }
269         }
270         if (module!=null && resolve) {
271             try {
272                 module.resolve();
273             } catch(Throwable e) {
274                 module.uninstall();
275                 throw new ResolveError(e);
276             }
277         }
278         Logger.getAnonymousLogger().fine("this.makeModuleFor("+name+ ", " + version + ") returned " + module);
279         return module;
280     }

282     /***
283      * Find and return a loaded Module that has the package name in its list
284      * of exported interfaces.
285      *
286      * @param packageName the requested implementation package name.
287      * @return the <code>Module</code> instance implementing the package
288      * name or null if not found.
289      * @throws ResolveError if the module dependencies cannot be resolved
290      */
291     public Module makeModuleFor(String packageName) throws ResolveError {
292         if(parent!=null) {
293             Module m = parent.makeModuleFor(packageName);
294             if(m!=null)     return m;
295         }
296 
297         for (Module module : modules.values()) {
298             String[] exportedPkgs = module.getModuleDefinition().getPublicInterfaces();
299             for (String exportedPkg : exportedPkgs) {
300                 if (exportedPkg.equals(packageName)) {
301                     module.resolve();
302                     return module;
303                 }
304             }
305         }
306         return null;
307     }

309     protected Module loadFromRepository(String name, String version) {
310         Set<Integer> keys = repositories.keySet();
311         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
312         sortedKeys.addAll(keys);
313         for (Integer key : sortedKeys) {
314             Repository repo = repositories.get(key);
315             ModuleDefinition moduleDef = repo.find(name, version);
316             if (moduleDef!=null) {
317                 return newModule(moduleDef);
318             }
319         }
320         return null;
321     }

330     /***
331      * Add a new module to this registry. Once added, the module will be 
332      * available through one of the getServiceImplementor methods.
333      * @param newModule the new module
334      */
335     protected void add(Module newModule) {
336         //if (Utils.isLoggable(Level.INFO)) {
337         //    Utils.getDefaultLogger().info("New module " + newModule);
338         //}
339         assert newModule.getRegistry()==this;
340         // see if this module is already added. This can happen
341         // in case of OSGiModulesRegistryImpl which uses SynchronousBundleListener.
342         ModuleId id = AbstractFactory.getInstance().createModuleId(
343                 newModule.getModuleDefinition());
344         if (modules.get(id) != null) return;
345         modules.put(id, newModule);
346 
347         // pick up providers from this module
348         for( ModuleMetadata.Entry spi : newModule.getMetadata().getEntries() ) {
349             for( String name : spi.providerNames )
350                 providers.put(name,newModule);
351         }
352         for (Map.Entry<String, Habitat> entry : habitats.entrySet()) {
353             String name = entry.getKey();
354             Habitat h = entry.getValue();
355             try
356             {
357                 parseInhabitants(newModule, name, new InhabitantsParser(h));
358             }
359             catch (IOException e)
360             {
361                 throw new RuntimeException("Not able to parse inhabitants information");
362             }
363         }
364     }

381     /*** 
382      * Returns the list of shared Modules registered in this instance.
383      *
384      * <p>
385      * The returned list will not include the modules defined in the ancestor
386      * {@link AbstractModulesRegistryImpl}s.
387      *
388      * @return an umodifiable list of loaded modules
389      */
390     public Collection<Module> getModules() {
391 
392         // make a copy to avoid synchronizing since this API can be called while
393         // modules are added or removed by other threads.
394         Map<Integer,Repository> repos = new TreeMap<Integer,Repository>();
395         repos.putAll(repositories);
396 
397         // force repository extraction
398         Set<Integer> keys = repos.keySet();
399         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
400         sortedKeys.addAll(keys);
401         for (Integer key : sortedKeys) {
402             Repository repo = repos.get(key);
403             for (ModuleDefinition moduleDef : repo.findAll()) {
404                 if (modules.get(AbstractFactory.getInstance().createModuleId(moduleDef))==null) {
405                     Module newModule = newModule(moduleDef);
406                     if (newModule!=null) {
407                         // When some module can't get installed,
408                         // don't halt proceeding, instead continue
409                         add(newModule);
410                         // don't resolve such modules, we just want to know about them
411                     }
412                 }
413             }
414         }
415         return modules.values();
416     }

454     public Module add(ModuleDefinition info, boolean resolve) throws ResolveError {
455         // it may have already been created

457         if (service!=null) {

460         } else {
461             // create the service instance
462             service = newModule(info);
463             if (service != null){
464                 add(service);
465             }
466         }
467         return service;

516     /***
517      * Registers a running service, this is useful when other components need
518      * to have access to a provider of a service without having to create
519      * a new instance and initialize it.
520      * @param serviceClass the service interface
521      * @param provider the provider of that service.
522      */
523     public <T> void registerRunningService(Class<T> serviceClass, T provider) {
524         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
525         if (rs==null) {
526             rs = new CopyOnWriteArrayList<T>();
527             CopyOnWriteArrayList existing = runningServices.putIfAbsent(serviceClass, rs);
528             if(existing!=null)
529                 rs = existing;
530         }
531         rs.add(provider);
532     }

534     /***
535      * Removes a running service, this is useful when a service instance is no longer
536      * available as a provider of a service.
537      */
538     public <T> boolean unregisterRunningService(Class<T> serviceClass, T provider) {
539         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
540         if (rs==null) {
541             return false;
542         }
543         return rs.remove(provider);
544     }

55  /***
56   * {@link ModuleDefinition} implementation that picks up most of the module
57   * properties from the manifest file of the jar, as baked in by
58   * the hk2-maven-plugin.

60   * @author Jerome Dochez

68      protected final List<URI> classPath = new ArrayList<URI>();
69      private final String importPolicy;

80      private final ModuleMetadata metadata = new ModuleMetadata();

83      public DefaultModuleDefinition(File location) throws IOException {

87      // Sahoo changed it to public so that it can be used by osgi-adapter
88      public DefaultModuleDefinition(File location, Attributes attr) throws IOException {
89  
90          classPath.add(location.toURI());
91          
92          Jar jarFile = Jar.create(location);
93          Manifest m = jarFile.getManifest();
94          if(m==null) m = EMPTY_MANIFEST;
95          manifest = m;
96  
97          if (attr==null)
98              attr = manifest.getMainAttributes();
99  
100         // no attributes whatsoever, I just use an empty collection to avoid 
101         // testing for null all the time.
102         if (attr==null) {
103             attr = new Attributes();
104         }
105         this.mainAttributes = attr;
106                 
107         // name
108         if (attr.getValue(ManifestConstants.BUNDLE_NAME)!=null) {
109             name = attr.getValue(ManifestConstants.BUNDLE_NAME);
110         } else {
111             name = location.getName();
112         }
113 
114         // classpath
115         parseClassPath(attr, location.toURI());
116         
117         // class exported...
118         String exported = attr.getValue(ManifestConstants.PKG_EXPORT_NAME);
119         ArrayList<String> tmpList = new ArrayList<String>();
120         for( String token : new Tokenizer(exported,",")) {
121             tmpList.add(token);
122         }
123         publicPkgs = tmpList.toArray(new String[tmpList.size()]);
124         
125         importPolicy = attr.getValue(ManifestConstants.IMPORT_POLICY);
126         lifecyclePolicy = attr.getValue(ManifestConstants.LIFECYLE_POLICY);
127         
128         version = "1.0"; // for now ;-)
129 
130         parseAttributes(attr);
131 
132         jarFile.loadMetadata(metadata);
133     }

135     /***
136      * Parses <tt>{@value ManifestConstants#CLASS_PATH}</tt> from manifest attributes
137      * and updates URI list.
138      */
139     protected void parseClassPath(Attributes attr, URI baseURI) throws IOException {
140         String classpath = attr.getValue(ManifestConstants.CLASS_PATH);
141         for( String classpathElement : new Tokenizer(classpath," ")) {
142             classpathElement = decorateClassPath(classpathElement);
143             URI result;
144             File ref = new File(classpathElement);
145             if (!ref.isAbsolute()) {
146                 try {
147                     result = baseURI.resolve(classpathElement);
148                 } catch (IllegalArgumentException e) {
149                     throw new IllegalArgumentException("Unable to parse Class-Path entry '"+classpath+"' in "+baseURI);
150                 }
151             } else
152                 result = ref.toURI();
153 
154             assert testClassPath(result);
155 
156             // look for /META-INF/services for classloader punch-in
157             File file = new File(result);
158             if (file.exists()) {
159                 Jar jar = Jar.create(file);
160                 Manifest m = jar.getManifest();
161                 if ((m != null) && (m.getMainAttributes().getValue("HK2-Bundle-Name")==null)) {
162                     // if Class-Path refers to other HK2 modules, ignore loading metadata
163                     // When someone makes HK2 module bootable from Java (that is, to have the main method),
164                     // it often needs to refer to other jars. The parent-first classloader delegation
165                     // helps us avoid loading dupliates, but metadata needs to be ignored here.
166                     jar.loadMetadata(metadata);
167                 }
168                 classPath.add(result);
169             } else {
170                 // even if the pointed resource doesn't exist, don't complain by default,
171                 // which is consistent with ClassLoader behavior of silently ignoring URLs
172                 // that doesn't resolve.
173                 // When the assertion is on, testClassPath above would do the warning.
174             }
175         }
176     }
177 
178     /***
179      * Optional error diagnostics performed during the development time
180      * to check if the URL pointed by the path actually exists.
181      */
182     private boolean testClassPath(URI uri) {
183         try {
184             if(uri.getScheme().equals("file")) {
185                 URLConnection c = uri.toURL().openConnection();
186                 if(c.getContentLength()==-1)
187                     LOGGER.warning(uri+" pointed from "+name+" in classpath doesn't exist");
188             }
189         } catch (IOException e) {
190             throw new AssertionError(e); // file is a valid URL
191         }
192         return true; // assertion should always succeed.
193     }

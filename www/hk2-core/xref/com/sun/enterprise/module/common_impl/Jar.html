57  /***
58   * Abstraction of {@link JarFile} so that we can handle
59   * both a jar file and a directory image transparently.
60   *
61   * @author Kohsuke Kawaguchi
62   */
63  public abstract class Jar {
64      protected Jar() {}
65  
66      /***
67       * See {@link JarFile#getManifest()} for the contract.
68       */
69      public abstract Manifest getManifest() throws IOException;
70  
71      /***
72       * Loads all <tt>META-INF/habitats</tt> entries and store them to the list.
73       */
74      public abstract void loadMetadata(ModuleMetadata result);
75  
76      /***
77       * Gets the base name of the jar.
78       *
79       * <p>
80       * For example, "bar" for "bar.jar".
81       */
82      public abstract String getBaseName();
83  
84      public static Jar create(File file) throws IOException {
85          if(file.isDirectory())
86              return new Directory(file);
87          else
88              return new Archive(file);
89      }

91      private static final class Directory extends Jar {
92          private final File dir;
93  
94          public Directory(File dir) {
95              this.dir = dir;
96          }
97  
98          public Manifest getManifest() throws IOException {
99              File mf = new File(dir,JarFile.MANIFEST_NAME);
100             if(mf.exists()) {
101                 FileInputStream in = new FileInputStream(mf);
102                 try {
103                     return new Manifest(in);
104                 } finally {
105                     in.close();
106                 }
107             } else {
108                 return null;
109             }
110         }
111 
112         private File[] fixNull(File[] f) {
113             if(f==null) return new File[0];
114             else        return f;
115         }
116 
117         public void loadMetadata(ModuleMetadata result) {
118             File inhabitantFileLocation = new File(dir, InhabitantsFile.PATH);
119             if (!inhabitantFileLocation.exists()) {
120                 // rely on introspection...
121                 result.addHabitat("default", new InhabitantsDescriptor() {
122                     public String getSystemId() {
123                         return null;  //To change body of implemented methods use File | Settings | File Templates.
124                     }
125 
126 
127 
128                     public Iterable<InhabitantParser> createScanner() throws IOException {
129                         ParsingContext context = (new ParsingContext.Builder()).build();
130                         Parser parser = new Parser(context);
131                         parser.parse(dir, null);
132                         try {
133                             parser.awaitTermination();
134                         } catch (InterruptedException e) {
135                             throw new IOException(e);
136                         }
137 
138                         return new IntrospectionScanner(context);
139                     }                    
140                 });
141             }
142             for( File svc : fixNull(new File(dir, InhabitantsFile.PATH).listFiles())) {
143                 if(svc.isDirectory())
144                     continue;
145 
146                 try {
147                     result.addHabitat(svc.getName(),
148                         new ByteArrayInhabitantsDescriptor(svc.getPath(), readFully(svc))
149                     );
150                 } catch(IOException e) {
151                     LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading habitats file from " + svc, e);
152                 }
153             }
154 
155             for( File svc : fixNull(new File(dir, SERVICE_LOCATION).listFiles()) ) {
156                 if(svc.isDirectory())
157                     continue;
158 
159                 try {
160                     result.load( svc.toURL(), svc.getName() );
161                 } catch(IOException e) {
162                     LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading service provider from " + svc, e);
163                 }
164             }
165         }
166 
167         public String getBaseName() {
168             return dir.getName();
169         }
170 
171         private byte[] readFully(File f) throws IOException {
172             byte[] buf = new byte[(int)f.length()];
173             DataInputStream in = new DataInputStream(new FileInputStream(f));
174             try {
175                 in.readFully(buf);
176                 return buf;
177             } finally {
178                 in.close();
179             }
180         }
181     }

183     private static final class Archive extends Jar {
184         private final JarFile jar;
185         private final File file;
186 
187         public Archive(File jar) throws IOException {
188             try {
189                 this.jar = new JarFile(jar);
190                 this.file = jar;
191             } catch (ZipException e) {
192                 // ZipException doesn't include this crucial information, so rewrap
193                 IOException x = new IOException("Failed to open " + jar);
194                 x.initCause(e);
195                 throw x;
196             }
197         }
198 
199         public Manifest getManifest() throws IOException {
200             return jar.getManifest();
201         }
202 
203         public void loadMetadata(ModuleMetadata result) {
204 
205             if (jar.getJarEntry(InhabitantsFile.PATH)==null
206                     && jar.getJarEntry(SERVICE_LOCATION)==null) {
207                 return;
208             }
209             Enumeration<JarEntry> entries = jar.entries();
210             while (entries.hasMoreElements()) {
211                 JarEntry entry = entries.nextElement();
212                 if (!entry.isDirectory() && entry.getName().startsWith(InhabitantsFile.PATH)) {
213                     String habitatName = entry.getName().substring(InhabitantsFile.PATH.length()+1);
214 
215                     try {
216                         result.addHabitat(habitatName,new ByteArrayInhabitantsDescriptor(
217                             "jar:"+file.toURL()+"!/"+entry.getName(),
218                             loadFully(entry)
219                         ));
220                     } catch(IOException e) {
221                         LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading inhabitants list in " + jar.getName(), e);
222                     }
223                 } else {
224                     if (!entry.isDirectory() && entry.getName().startsWith(SERVICE_LOCATION)) {
225                         String serviceName = entry.getName().substring(SERVICE_LOCATION.length()+1);
226 
227                         try {
228                             result.load( new URL("jar:"+file.toURL()+"!/"+entry.getName()), serviceName, jar.getInputStream(entry));
229                         } catch(IOException e) {
230                             LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading service provider in " + jar.getName(), e);
231                         }
232                     }
233                 }
234             }
235         }
236 
237         public String getBaseName() {
238             String name = file.getName();
239             int idx = name.lastIndexOf('.');
240             if(idx>=0)
241                 name = name.substring(0,idx);
242             return name;
243         }
244 
245         private byte[] loadFully(JarEntry e) throws IOException {
246             byte[] buf = new byte[(int)e.getSize()];
247             DataInputStream in = new DataInputStream(jar.getInputStream(e));
248             try {
249                 in.readFully(buf);
250                 return buf;
251             } finally {
252                 in.close();
253             }
254         }
255     }

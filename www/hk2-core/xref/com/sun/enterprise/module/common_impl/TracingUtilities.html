67      public static File getLocation() {
68          String location = System.getProperty("hk2.module.loglocation");
69          if (location==null) {
70              location = System.getProperty("user.dir");
71          }
72          File f = new File(location);
73          if (f.isAbsolute()) {
74              return f;
75          } else {
76              return new File(System.getProperty("user.dir"), location);
77          }
78      }

80      public static void traceResolution(ModulesRegistry registry, long bundleId, String bundleName) {
81          traceState(registry, "resolved", bundleId, bundleName);
82      }

84      public static void traceStarted(ModulesRegistry registry, long bundleId, String bundleName) {
85          traceState(registry, "started", bundleId, bundleName);
86      }

88      public static void traceState(ModulesRegistry registry, String state, long bundleId, String bundleName) {
89          File out = new File(getLocation(), state + "-" + bundleId+".log");
90          Writer w = null;
91          try {
92              w = new FileWriter(out);
93              w.append("\n");
94              w.append("Module ["+ bundleId + "] " + state + " " + bundleName+"\n");
95              String prefix="-";
96              StackTraceElement[] stack = Thread.currentThread().getStackTrace();
97              InhabitantTracing tracing = TracingThreadLocal.get();
98              Iterator<Inhabitant> itr = tracing.inOrder();
99  
100             w.append("\n");
101             w.append("-----------------------------------\n");
102             w.append("Inhabitants / stack combination\n");
103             w.append("-----------------------------------\n");
104 
105             String currentBundleName = bundleName;
106             
107             for (int i=0;i<stack.length;i++) {
108                 StackTraceElement element = stack[i];
109                 if (element.getClassName().equals(AbstractInhabitantImpl.class.getName()) &&
110                         element.getMethodName().equals("get"))  {
111                     // now let's find out the first non hk2 class asking for this...
112                     int j=i+1;
113                     for (;j<stack.length;j++) {
114                         StackTraceElement caller = stack[j];
115                         if (!caller.getClassName().contains("hk2")) {
116                             break;
117                         }
118                     }
119                     if (itr.hasNext()) {
120                         Inhabitant reason = itr.next();
121                         StackTraceElement caller = stack[j];
122                         Module m = registry.find(reason.type());
123                         if (m!=null && !m.getModuleDefinition().getName().equals(currentBundleName)) {
124                             w.append("Looks like module " + currentBundleName + " was " + state + " because "+
125                                     m.getName() + " was " + state + "\nSince ");
126                         } else {
127                             w.append("Requested by ");
128                         }
129                         if (m!=null)
130                             currentBundleName = m.getName();
131 
132                         w.append(m.getName() + " contains " + reason + " called from " + caller.getClassName()
133                                 + "." + caller.getMethodName() + ":" + caller.getLineNumber()+" metadata \n" + reason.metadata()+"\n");
134                         w.append("\n");
135 
136                     }
137 
138                 }
139 
140             }
141             w.append("\n");
142 
143             w.append("-----------------------------------\n");
144             w.append("Inhabitants initialization stack\n");
145             w.append("-----------------------------------\n");
146             itr = tracing.inOrder();
147             while (itr.hasNext()) {
148                 Inhabitant i = itr.next();
149                 w.append(prefix +"> requested from "+ i+"\n");
150                 prefix=prefix+"-";
151             }
152 
153             w.append("\n");
154 
155             w.append("---------------------------\n");
156             w.append("Complete thread stack Trace\n");
157             w.append("---------------------------\n");
158 
159             for (StackTraceElement element : stack) {
160                 w.append(element.toString()+"\n");
161             }
162 
163 
164         } catch (IOException e) {
165             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
166         } finally {
167             if (w!=null) {
168                 try {
169                     w.close();
170                 } catch (IOException e) {
171                     e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
172                 }
173             }
174         }
175 
176     }

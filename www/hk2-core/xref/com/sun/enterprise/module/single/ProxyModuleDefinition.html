73          public ProxyModuleDefinition(ClassLoader classLoader, List<ManifestProxy.SeparatorMappings> mappings,
74                                       Collection<String> habitatNames) throws IOException {
75              manifest = new ManifestProxy(classLoader, mappings);
76              for (String habitatName : habitatNames) {
77                  Enumeration<URL> inhabitants = classLoader.getResources(InhabitantsFile.PATH+'/'+habitatName);
78                  Set<File> processedFiles = new HashSet<File>();
79                  while (inhabitants.hasMoreElements()) {
80                      URL url = inhabitants.nextElement();
81                      if (url.getProtocol().equals("jar")) {
82                          int index = url.getPath().indexOf("!");
83                          if (index>0 && url.getPath().length()>5) {
84                              String path = url.getPath().substring(5, url.getPath().indexOf("!"));
85                              File f = new File(path);
86                              if (f.exists()) {
87                                  f = f.getCanonicalFile();
88                                  if (processedFiles.contains(f)) {
89                                      continue;
90                                  }
91                                  processedFiles.add(f);
92                              }   
93                          }
94                      }
95                      metadata.addHabitat(habitatName,
96                          new ByteArrayInhabitantsDescriptor(url, readFully(url))
97                      );
98                  }
99              }
100         }
101 
102         private static byte[] readFully(URL url) throws IOException {
103             DataInputStream dis=null;
104             try {
105                 URLConnection con = url.openConnection();
106                 int len = con.getContentLength();
107                 InputStream in = con.getInputStream();
108                 dis = new DataInputStream(in);
109                 byte[] bytes = new byte[len];
110                 dis.readFully(bytes);
111                 return bytes;
112             } catch (IOException e) {
113                 IOException x = new IOException("Failed to read " + url);
114                 x.initCause(e);
115                 throw x;
116             } finally {
117                 if (dis!=null)
118                     dis.close();
119             }
120         }

158         private static boolean ok(String s) {
159             return s != null && s.length() > 0;

162         private static boolean ok(String[] ss) {
163             return ss != null && ss.length > 0;

165         private static final ModuleDependency[] EMPTY_MODULE_DEFINITIONS_ARRAY = new ModuleDependency[0];
166         private static URI[] uris = new URI[0];
167 
168     static {
169         // It is impossible to change java.class.path after the JVM starts --
170         // so cache away a copy...
171 
172         String cp = System.getProperty("java.class.path");
173         if(ok(cp)) {
174             String[] paths = cp.split(System.getProperty("path.separator"));
175 
176             if(ok(paths)) {
177                 uris = new URI[paths.length];
178 
179                 for(int i = 0; i < paths.length; i++) {
180                     uris[i] = new File(paths[i]).toURI();
181                 }
182             }
183         }

19  /***
20   * Implements a modules registry based on a class-path style of module
21   * description using a single class loader (capable of loading the entire
22   * class-path)
23   *
24   * @author Jerome Dochez
25   */
26  public class ClassPathBasedModulesRegistry extends ModulesRegistryImpl {
27  
28      final ClassLoader cLoader;
29      final List<ModuleDefinition> moduleDefs = new ArrayList<ModuleDefinition>();
30      final List<Module> modules = new ArrayList<Module>();
31  
32  
33      public ClassPathBasedModulesRegistry(ClassLoader singleCL, String classPath) throws IOException {
34  
35          super(null);
36          this.cLoader = singleCL;
37          setParentClassLoader(cLoader);
38  
39          StringTokenizer st = new StringTokenizer(classPath, File.pathSeparator);
40          while (st.hasMoreTokens()) {
41              String classPathElement = st.nextToken();
42              File f = new File(classPathElement);
43              if (f.exists()) {
44                  ModuleDefinition md = new DefaultModuleDefinition(f);
45                  moduleDefs.add(md);
46                  add(md);
47              }
48          }
49  
50          // now create fake modules
51          for (ModuleDefinition md : moduleDefs) {
52              // they all use the same class loader since they are not really modules
53              // and we don't run in a modular environment
54              modules.add(new ProxyModule(this, md, cLoader));
55          }
56      }
57  
58      @Override
59      public Module find(Class clazz) {
60          Module m = super.find(clazz);
61          // all modules can load all classes
62          if (m == null)
63              return modules.get(0);
64          return m;
65      }
66  
67      @Override
68      public Collection<Module> getModules(String moduleName) {
69          // I could not care less about the modules names
70          return getModules();
71      }
72  
73      @Override
74      public Collection<Module> getModules() {
75          ArrayList<Module> list = new ArrayList<Module>();
76          list.addAll(modules);
77          return list;
78      }
79  
80      @Override
81      public Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError {
82          for (int i=0;i<moduleDefs.size();i++) {
83              ModuleDefinition md = moduleDefs.get(i);
84              if (md.getName().equals(name)) {
85                  return modules.get(i); 
86              }
87          }
88          return null;
89      }
90  
91      @Override
92      public void parseInhabitants(Module module, String name, InhabitantsParser inhabitantsParser)
93              throws IOException {
94  
95          Holder<ClassLoader> holder = new Holder<ClassLoader>() {
96              public ClassLoader get() {
97                  return cLoader;
98              }
99          };
100 
101         for (Module m : modules) {
102             // each module can have a different way of representing the inhabitant meta-data
103             // some may use the inhabitant file, others may rely on introspection 
104             for (InhabitantsDescriptor d :m.getMetadata().getHabitats(name))
105                 inhabitantsParser.parse(d.createScanner(), holder);
106         }
107     }

94      public void run(final String[] args) {
95          try {
96              final Main main = this;
97              Thread thread = new Thread() {
98                  public void run() {
99                      try {
100                         main.start(args);
101                     } catch(BootException e) {
102                         e.printStackTrace();
103                     } catch (RuntimeException e) {
104                         e.printStackTrace();
105                     }
106                 }
107             };
108             thread.start();
109             try {
110                 thread.join();
111             }
112             catch (InterruptedException e) {
113                 e.printStackTrace();
114             }
115         } catch (Throwable t) {
116             t.printStackTrace();
117         }

140     /***
141      * Start the server from the command line
142      * @param args the command line arguments
143      */
144     public void start(String[] args) throws BootException {
145         
146         File bootstrap = this.getBootstrapFile();
147         File root = bootstrap.getAbsoluteFile().getParentFile();
148 
149         // root is the directory in which this bootstrap.jar is located
150         // For most cases, this is the lib directory although this is completely
151         // dependent on the usage of this facility.
152         if (root==null) {
153             throw new BootException("Cannot find root installation from "+bootstrap);
154         }
155 
156         String targetModule = findMainModuleName(bootstrap);
157 
158         // get the ModuleStartup implementation.
159         ModulesRegistry mr = AbstractFactory.getInstance().createModulesRegistry();
160         Manifest mf;
161         try {
162             mf = (new JarFile(bootstrap)).getManifest();
163         } catch (IOException e) {
164             throw new BootException("Failed to read manifest from "+bootstrap);
165         }
166 
167         createRepository(root,bootstrap,mf,mr);
168 
169         StartupContext context = new StartupContext(ArgumentManager.argsToMap(args));        
170         launch(mr, targetModule, context);
171     }

178     /***
179      * Creates repositories needed for the launch and 
180      * adds the repositories to {@link ModulesRegistry}
181      *
182      * @param bootstrapJar
183      *      The file from which manifest entries are loaded. Used for error reporting
184      */
185     protected void createRepository(File root, File bootstrapJar, Manifest mf, ModulesRegistry mr) throws BootException {
186         String repos = mf.getMainAttributes().getValue(ManifestConstants.REPOSITORIES);
187         if (repos!=null) {
188             StringTokenizer st = new StringTokenizer(repos);
189             while (st.hasMoreTokens()) {
190                 final String repoId = st.nextToken();
191                 final String repoKey = "HK2-Repository-"+repoId;
192                 final String repoInfo;
193                 try {
194                     repoInfo = mf.getMainAttributes().getValue(repoKey);
195                 } catch (Exception e) {
196                     throw new BootException("Invalid repository id " + repoId+" in "+bootstrapJar, e);
197                 }
198                 if (repoInfo!=null) {
199                     addRepo(root, repoId, repoInfo, mr);
200                 }
201             }
202         } else {
203             // by default, adding the boot archive directory
204             addRepo(root, "lib", "uri=. type=directory", mr);
205         }
206     }

208     private void addRepo(File root, String repoId, String repoInfo, ModulesRegistry mr) throws BootException {
209 
210         StringTokenizer st = new StringTokenizer(repoInfo);
211         Properties props = new Properties();
212         Pattern p = Pattern.compile("([^=]*)=(.*)");
213         
214         while(st.hasMoreTokens()) {
215             Matcher m = p.matcher(st.nextToken());
216             if (m.matches()) {
217                 props.put(m.group(1), m.group(2));
218             }
219         }
220         
221         String uri = props.getProperty("uri");
222         if (uri==null) {
223             uri = ".";
224         }
225         String type = props.getProperty("type");
226         String weight = props.getProperty("weight");
227 
228         // need a plugability layer here...
229         if ("directory".equalsIgnoreCase(type)) {
230 
231             File location = new File(uri);
232             if (!location.isAbsolute()) {
233                 location = new File(root, uri);
234             }
235             if (!location.exists())
236                 throw new BootException("Non-existent directory: "+location);
237         
238             /* bnevins 3/21/08
239              * location might be something like "/gf/modules/."
240              * The "." can cause all sorts of trouble later, so sanitize
241              * the name now!
242              * here's an example of the trouble:            
243              * new File("/foo/.").getParentFile().getPath() --> "/foo", not "/"
244              * JDK treats the dot as a file in the foo directory!!
245              */
246             
247             try {
248                 location = location.getCanonicalFile();
249             }
250             catch(Exception e) {
251                 // I've never seen this happen!
252                 location = location.getAbsoluteFile();
253             }
254             
255             try {
256                 Repository repo = new DirectoryBasedRepository(repoId, location);
257                 addRepo(repo, mr, weight);
258             } catch (IOException e) {
259                 throw new BootException("Exception while adding " + repoId + " repository", e);
260 
261             }
262         } else {
263             throw new BootException("Invalid attributes for modules repository " + repoId + " : " + repoInfo);
264         }
265     }

267     protected void addRepo(Repository repo, ModulesRegistry mr, String weight)
268         throws IOException {
269         
270         repo.initialize();
271         int iWeight=50;
272         if (weight!=null) {
273             try {
274                 iWeight = Integer.parseInt(weight);
275             } catch (NumberFormatException e) {
276                 // ignore
277             }
278         }
279         mr.addRepository(repo, iWeight);
280     }

330     /***
331      * Return the ModuleStartup service configured to be used to start the system.
332      * @param registry
333      * @param habitat
334      * @param mainModuleName
335      * @param context
336      * @return
337      * @throws BootException
338      */
339     public ModuleStartup findStartupService(ModulesRegistry registry, Habitat habitat, String mainModuleName, StartupContext context) throws BootException {
340         ModuleStartup startupCode=null;
341         final Module mainModule;
342 
343         if(mainModuleName!=null) {
344             // instantiate the main module, this is the entry point of the application
345             // code. it is supposed to have 1 ModuleStartup implementation.
346             Collection<Module> modules = registry.getModules(mainModuleName);
347             if (modules.size() != 1) {
348                 if(registry.getModules().isEmpty())
349                     throw new BootException("Registry has no module at all");
350                 else
351                     throw new BootException("Cannot find main module " + mainModuleName+" : no such module");
352             }
353             mainModule = modules.iterator().next();
354             String targetClassName = findModuleStartupClassName(mainModule,context.getPlatformMainServiceName(), HABITAT_NAME);
355             if (targetClassName==null) {
356                 throw new BootException("Cannot find a ModuleStartup implementation in the META-INF/services/com.sun.enterprise.v3.ModuleStartup file, aborting");
357             }
358 
359             Class<? extends ModuleStartup> targetClass=null;
360             ClassLoader currentCL = Thread.currentThread().getContextClassLoader();
361             Thread.currentThread().setContextClassLoader(mainModule.getClassLoader());
362             try {
363                 targetClass = mainModule.getClassLoader().loadClass(targetClassName).asSubclass(ModuleStartup.class);
364                 startupCode = habitat.getComponent(targetClass);
365             } catch (ClassNotFoundException e) {
366                 throw new BootException("Unable to load component of type " + targetClassName,e);
367             } catch (ComponentException e) {
368                 throw new BootException("Unable to load component of type " + targetClassName,e);
369             } finally {
370                 Thread.currentThread().setContextClassLoader(currentCL);
371             }
372         } else {
373             Collection<Inhabitant<? extends ModuleStartup>> startups = habitat.getInhabitants(ModuleStartup.class);
374 
375             if(startups.isEmpty())
376                 throw new BootException("No module has a ModuleStartup implementation");
377             if(startups.size()>1) {
378                 // maybe the user specified a main
379                 String mainServiceName = context.getPlatformMainServiceName();
380                 for (Inhabitant<? extends ModuleStartup> startup : startups) {
381                     Collection<String> regNames = Inhabitants.getNamesFor(startup, ModuleStartup.class.getName());
382                     if (regNames.isEmpty() && mainServiceName==null) {
383                         startupCode = startup.get();
384                     } else {
385                         for (String regName : regNames) {
386                             if (regName.equals(mainServiceName)) {
387                                 startupCode = startup.get();
388                             }
389                         }
390                     }
391 
392                 }
393                 if (startupCode==null) {
394                     if (mainServiceName==null) {
395                         Iterator<Inhabitant<? extends ModuleStartup>> itr = startups.iterator();
396                         ModuleStartup a = itr.next().get();
397                         ModuleStartup b = itr.next().get();
398                         Module am = registry.find(a.getClass());
399                         Module bm = registry.find(b.getClass());
400                         throw new BootException(String.format("Multiple ModuleStartup found: %s from %s and %s from %s",a,am,b,bm));
401                     } else {
402                         throw new BootException(String.format("Cannot find %s ModuleStartup", mainServiceName));
403                     }
404                 }
405             } else {
406                 startupCode = startups.iterator().next().get();
407             }
408             mainModule = registry.find(startupCode.getClass());
409         }
410 
411         habitat.addIndex(Inhabitants.create(startupCode),
412                 ModuleStartup.class.getName(), habitat.DEFAULT_NAME);
413         mainModule.setSticky(true);
414         return startupCode;
415     }

417     public Habitat createHabitat(ModulesRegistry registry, StartupContext context) throws BootException {
418         // set the parent class loader before we start loading modules
419         setParentClassLoader(context, registry);
420 
421         // create a habitat and initialize them
422         Habitat habitat = registry.newHabitat();
423         habitat.add(new ExistingSingletonInhabitant<StartupContext>(context));
424         habitat.add(new ExistingSingletonInhabitant<Logger>(Logger.global));
425         // the root registry must be added as other components sometimes inject it
426         habitat.add(new ExistingSingletonInhabitant(ModulesRegistry.class, registry));
427         ClassLoader oldCL = Thread.currentThread().getContextClassLoader();
428         Thread.currentThread().setContextClassLoader(getClass().getClassLoader());
429         try {
430             registry.createHabitat(HABITAT_NAME, createInhabitantsParser(habitat));
431         } finally {
432             Thread.currentThread().setContextClassLoader(oldCL);
433         }
434         return habitat;
435     }

445     protected String findMainModuleName(File bootstrap) throws BootException {
446         String targetModule;
447         try {
448             JarFile jarFile = new JarFile(bootstrap);
449             Manifest manifest = jarFile.getManifest();
450                           
451             Attributes attr = manifest.getMainAttributes();
452             targetModule = attr.getValue(ManifestConstants.MAIN_BUNDLE);
453             if (targetModule==null) {
454                 LogHelper.getDefaultLogger().warning(
455                         "No Main-Bundle module found in manifest of " +
456                         bootstrap.getAbsoluteFile());
457             }
458         } catch(IOException ioe) {
459             throw new BootException("Cannot get manifest from " + bootstrap.getAbsolutePath(), ioe);
460         }
461         return targetModule;
462     }

469     /***
470      * Finds {@link ModuleStartup} implementation class name to perform the launch.
471      *
472      * <p>
473      * This implementation does so by looking it up from services.
474      */
475     protected String findModuleStartupClassName(Module mainModule, String serviceName, String habitatName) throws BootException {
476         String index = (serviceName==null || serviceName.isEmpty()?ModuleStartup.class.getName():ModuleStartup.class.getName()+":"+serviceName);
477         for(InhabitantsDescriptor d : mainModule.getMetadata().getHabitats(habitatName)) {
478             try {
479                 for (InhabitantParser parser : d.createScanner()) {
480                     for (String v : parser.getIndexes()) {
481                         if(v.equals(index)) {
482                             parser.rewind();
483                             return parser.getImplName();
484                         }
485                     }
486                 }
487             } catch (IOException e) {
488                 throw new BootException("Failed to parse "+d.getSystemId(),e);
489             }
490         }
491 
492         throw new BootException("No "+ModuleStartup.class.getName()+" in "+mainModule);
493     }

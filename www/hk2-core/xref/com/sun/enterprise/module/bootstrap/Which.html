14  public class Which {
15      public static File jarFile(Class clazz) throws IOException {
16          String resourceName = clazz.getName().replace(".","/")+".class";
17          URL resource = clazz.getClassLoader().getResource(resourceName);
18          if (resource==null) {
19              throw new IllegalArgumentException("Cannot get bootstrap path from "+ clazz + " class location, aborting");
20          }
21  
22          if (resource.getProtocol().equals("jar")) {
23              try {
24                  JarURLConnection c = (JarURLConnection) resource.openConnection();
25                  URL jarFile = c.getJarFileURL();
26                  try {
27                      return new File(jarFile.toURI());
28                  } catch (URISyntaxException e) {
29                      return new File(jarFile.getPath());
30                  }
31              } catch (IOException e) {
32                  throw new IllegalArgumentException("Cannot open jar file "+resource, e);
33              }
34          } else
35              throw new IllegalArgumentException("Don't support packaging "+resource+" , please contribute !");
36      }

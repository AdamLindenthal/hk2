52  /***
53   * {@link ClassLoader} that loads classes for a module.
54   *
55   * @author dochez
56   */
57  final class ModuleClassLoader extends ClassLoaderProxy {
58      
59      private final ModuleImpl module;
60  
61      /***
62       * Module will be initialized when this classloader is consulted for the first time.
63       */
64      private volatile boolean initialized = false;
65      private StackTraceElement[] initializerThread;
66      private String initializerClassName;
67  
68      /*** Creates a new instance of ClassLoader */
69      public ModuleClassLoader(ModuleImpl owner, URL[] shared, ClassLoader parent) {
70          super(shared, parent);
71          this.module = owner;
72      }
73      
74      protected void finalize() throws Throwable {
75          super.finalize();
76          LogHelper.getDefaultLogger().info("ModuleClassLoader gc'ed " + module.getModuleDefinition().getName());
77      }
78  
79  
80      protected synchronized Class<?> loadClass(String name, boolean resolve) throws ClassNotFoundException {
81          initialize(name);
82          return super.loadClass(name, resolve);
83      }
84  
85      protected Class<?> findClass(String name) throws ClassNotFoundException {
86          try {
87              return super.findClass(name);
88          } catch (ClassNotFoundException e) {
89              // punch in. find the provider class, no matter where we are.
90              ModuleImpl m = module.getRegistry().getProvidingModule(name);
91              if(m!=null)
92                  return m.getPrivateClassLoader().loadClass(name);
93              throw e;
94          }
95      }
96  
97      public URL getResource(String name) {
98          initialize(name);
99          if(name.startsWith(META_INF_SERVICES)) {
100             // punch in. find the service loader from any module
101             String serviceName = name.substring(META_INF_SERVICES.length());
102 
103             ModulesRegistry reg = module.getRegistry();
104             for( Module m : reg.getModules() ) {
105                 List<URL> list = m.getMetadata().getDescriptors(serviceName);
106                 if(!list.isEmpty())     return list.get(0);
107             }
108 
109             // no such resource
110             return super.getResource(name);
111         } else {
112             // normal service look up
113             URL url = super.getResource(name);
114             if(url!=null)
115                 return url;
116 
117             // commons-logging looks for a class file resource for providers,
118             // so check for those
119             if(name.endsWith(".class")) {
120                 String className = name.replace('/', '.').substring(0, name.length() - 6);
121                 ModuleImpl m = module.getRegistry().getProvidingModule(className);
122                 if(m!=null)
123                     return m.getPrivateClassLoader().getResource(name);
124             }
125 
126             return null;
127         }
128     }
129 
130     public Enumeration<URL> getResources(String name) throws IOException {
131         initialize(name);
132         if(name.startsWith(META_INF_SERVICES)) {
133             // punch in. find the service loader from any module
134             String serviceName = name.substring(META_INF_SERVICES.length());
135 
136             Vector<URL> urls = new Vector<URL>();
137 
138             ModulesRegistry reg = module.getRegistry();
139             for( Module m : reg.getModules() )
140                 urls.addAll(m.getMetadata().getDescriptors(serviceName));
141 
142             return urls.elements();
143         } else {
144             // normal look up
145             return super.getResources(name);
146         }
147     }
148 
149     /***
150      * Lazily initialize this module if not done so yet.
151      */
152     private void initialize(String name) {
153         if (initialized)    return;
154 
155         synchronized(this) {
156             if(!initialized) {
157                 // if we are preparing, we should just not initiate initialization.
158                 if (module.getState().equals(ModuleState.PREPARING)) {
159                     return;
160                 }
161                 initialized = true;
162 
163                 module.start();
164                 // remember who started us to assist debugging.
165                 initializerThread = Thread.currentThread().getStackTrace();
166                 initializerClassName = name;
167             }
168         }
169     }
170 
171     /*package*/ void dumpState(PrintStream writer) {
172         writer.println("Class-Path:");
173         for (URL url : getURLs())
174             writer.println("  "+url);
175         if (initializerThread!=null) {
176             writer.println("Initialized when " + initializerClassName + " was requested by :");
177             for (StackTraceElement e : initializerThread) {
178                 writer.println("  "+e.toString());
179             }
180         }
181     }

184     /***
185      * called by the facade class loader when it is garbage collected. 
186      * this is a good time to see if this module should be unloaded.
187      */
188     @Override
189     public void stop() {
190         
191         // we should only detach if the sticky flag is not set
192         if (!module.isSticky()) {
193             
194             LogHelper.getDefaultLogger().info("ModuleClassLoader stopped " + module.getModuleDefinition().getName());
195             super.stop();
196             module.stop();
197         }
198     }
199         
200     public ModuleImpl getOwner() {
201         return module;
202     }

206     public String toString() {
207         StringBuffer s= new StringBuffer(); 
208         s.append("ModuleClassLoader(name=").append(module.getName());
209         s.append(", parent=").append(super.toString());
210         s.append(",init=").append(initialized);
211         s.append(",URls[]=");
212         for (URL url : getURLs()) {
213             s.append(url).append(",");
214         }
215         s.append(")");
216                
217         for (ClassLoader surrogate : super.getDelegates()) {
218             s.append("\n ref : ").append(surrogate.toString());
219         }
220         return s.toString();
221     }

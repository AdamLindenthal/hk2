92      /**
93       * Creates and return a new private module implementation giving a name and
94       * version constraints. A private module is like any other module except
95       * it is not registered to be shared by other potential module users.
96       *
97       * @param moduleName the module name
98       * @param version the desired version
99       * @return the new private module or null if cannot be found
100      * @throws com.sun.enterprise.module.ResolveError if the module dependencies cannot be resolved
101      */
102     /*package*/ ModuleImpl newPrivateModuleFor(String moduleName, String version) {
103         if(parent!=null) {
104             ModuleImpl m = ModulesRegistryImpl.class.cast(parent).newPrivateModuleFor(moduleName,version);
105             if(m!=null)     return m;
106         }
107 
108         ModuleId id = AbstractFactory.getInstance().createModuleId(moduleName, version);
109         Module module = modules.get(id);
110         if (module!=null) {
111             ModuleImpl privateModule =
112                     (ModuleImpl)newModule(module.getModuleDefinition());
113             privateModule.resolve();
114             return privateModule;
115         }
116         return (ModuleImpl)loadFromRepository(moduleName, version);
117 
118     }

143     /**
144      * Returns a ClassLoader capable of loading classes from a set of modules identified
145      * by their module definition and also load new urls.
146      *
147      * @param parent the parent class loader for the returned class loader instance
148      * @param defs module definitions for all modules this classloader should be
149      *        capable of loading
150      * @param urls urls to be added to the module classloader
151      * @return class loader instance
152      * @throws com.sun.enterprise.module.ResolveError if one of the provided module
153      *         definition cannot be resolved
154      */
155     public ClassLoader getModulesClassLoader(ClassLoader parent,
156                                              Collection<ModuleDefinition> defs,
157                                              URL[] urls) throws ResolveError {
158 
159         if (parent==null) {
160             parent = getParentClassLoader();
161         }
162         ClassLoaderProxy cl = new ClassLoaderProxy(new URL[0], parent);
163         for (ModuleDefinition def : defs) {
164             Module module = this.makeModuleFor(def.getName(), def.getVersion());
165             cl.addDelegate(module.getClassLoader());
166         }
167         
168         if (urls != null) {
169             for (URL url : urls) {
170                 cl.addURL(url);
171             }
172         }
173         return cl;
174     }

195     public Module find(Class clazz) {
196         ClassLoader cl = clazz.getClassLoader();
197         if(cl==null)    return null;
198         if (cl instanceof ModuleClassLoader)
199             return ((ModuleClassLoader) cl).getOwner();
200         return null;
201     }

234     /**
235      * Shuts down this module's registry, apply housekeeping tasks
236      *
237      */
238     public void shutdown() {
239         detachAll();
240         for (Repository repo : repositories.values()) {
241             try {
242                 repo.shutdown();
243             } catch(Exception e) {
244                 Logger.getAnonymousLogger().log(Level.SEVERE, "Error while closing repository " + repo, e);
245                 // swallows
246             }
247         }

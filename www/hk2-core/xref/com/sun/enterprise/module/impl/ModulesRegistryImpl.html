90      /***
91       * Creates and return a new private module implementation giving a name and
92       * version constraints. A private module is like any other module except
93       * it is not registered to be shared by other potential module users.
94       *
95       * @param moduleName the module name
96       * @param version the desired version
97       * @return the new private module or null if cannot be found
98       * @throws com.sun.enterprise.module.ResolveError if the module dependencies cannot be resolved
99       */
100     /*package*/ ModuleImpl newPrivateModuleFor(String moduleName, String version) {
101         if(parent!=null) {
102             ModuleImpl m = ModulesRegistryImpl.class.cast(parent).newPrivateModuleFor(moduleName,version);
103             if(m!=null)     return m;
104         }
105 
106         ModuleId id = AbstractFactory.getInstance().createModuleId(moduleName, version);
107         Module module = modules.get(id);
108         if (module!=null) {
109             ModuleImpl privateModule =
110                     (ModuleImpl)newModule(module.getModuleDefinition());
111             privateModule.resolve();
112             return privateModule;
113         }
114         return (ModuleImpl)loadFromRepository(moduleName, version);
115 
116     }

141     /***
142      * Returns a ClassLoader capable of loading classes from a set of modules identified
143      * by their module definition and also load new urls.
144      *
145      * @param parent the parent class loader for the returned class loader instance
146      * @param defs module definitions for all modules this classloader should be
147      *        capable of loading
148      * @param urls urls to be added to the module classloader
149      * @return class loader instance
150      * @throws com.sun.enterprise.module.ResolveError if one of the provided module
151      *         definition cannot be resolved
152      */
153     public ClassLoader getModulesClassLoader(ClassLoader parent,
154                                              Collection<ModuleDefinition> defs,
155                                              URL[] urls) throws ResolveError {
156 
157         if (parent==null) {
158             parent = getParentClassLoader();
159         }
160         ClassLoaderProxy cl = new ClassLoaderProxy(new URL[0], parent);
161         for (ModuleDefinition def : defs) {
162             Module module = this.makeModuleFor(def.getName(), def.getVersion());
163             cl.addDelegate(module.getClassLoader());
164         }
165         
166         if (urls != null) {
167             for (URL url : urls) {
168                 cl.addURL(url);
169             }
170         }
171         return cl;
172     }

193     public Module find(Class clazz) {
194         ClassLoader cl = clazz.getClassLoader();
195         if(cl==null)    return null;
196         if (cl instanceof ModuleClassLoader)
197             return ((ModuleClassLoader) cl).getOwner();
198         return null;
199     }

232     /***
233      * Shuts down this module's registry, apply housekeeping tasks
234      *
235      */
236     public void shutdown() {
237         detachAll();
238         for (Repository repo : repositories.values()) {
239             try {
240                 repo.shutdown();
241             } catch(Exception e) {
242                 Logger.getAnonymousLogger().log(Level.SEVERE, "Error while closing repository " + repo, e);
243                 // swallows
244             }
245         }

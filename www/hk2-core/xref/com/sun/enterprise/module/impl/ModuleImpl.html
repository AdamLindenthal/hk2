77  /***
78   * A module represents a set of resources accessible to third party modules. 
79   * Each module has a module definition which defines its name, its list of 
80   * exported resources and its dependencies to other modules. 
81   * A module instance stores references to the class loader instances giving
82   * access to the module's implementation. 
83   * A module instance belongs to a <code>ModuleRegistry</code> which can be used
84   * to get the list of available modules and/or get particular module
85   * implementation. 
86   * Modules can only satisfy their dependencies within the <code>ModuleRegistry
87   * </code> instance they are registered in.
88   *
89   * @author Jerome Dochez
90   */

117     /* package */ ModuleImpl(ModulesRegistryImpl registry, ModuleDefinition info) {
118         assert registry!=null && info!=null;
119         this.registry = registry;
120         moduleDef = info;
121         for (URI lib : info.getLocations()) {
122             File f = new File(lib);
123             if (f.exists()) {
124                 lastModifieds.put(f.getAbsolutePath(), f.lastModified());
125             }

127         }
128         state = ModuleState.NEW;

138     public ClassLoaderFacade getClassLoader() {
139         ClassLoaderFacade r=null;
140         if (publicCL!=null)
141             r = publicCL.get();
142         if (r!=null)
143             return r;
144 
145         ClassLoaderFacade facade = AccessController.doPrivileged(new PrivilegedAction<ClassLoaderFacade>() {
146             public ClassLoaderFacade run() {
147                 return new ClassLoaderFacade(getPrivateClassLoader());
148             }
149         });
150         facade.setPublicPkgs(moduleDef.getPublicInterfaces());
151         publicCL = new WeakReference<ClassLoaderFacade>(facade);
152         return facade;
153     }

155     /***
156      * Return the private class loader for this module. This class loader will
157      * be loading all the classes which are not explicitely exported in the 
158      * module definition
159      * @return the private <code>ClassLoader</code> instance
160      */
161     /*package*/ ModuleClassLoader getPrivateClassLoader() {
162         if (privateCL==null) {
163             synchronized(this) {
164                 if(privateCL==null) {
165                     URI[] locations = moduleDef.getLocations();
166                     URL[] urlLocations = new URL[locations.length];
167                     for (int i=0;i<locations.length;i++) {
168                         try {
169                             urlLocations[i] = locations[i].toURL();
170                         } catch(MalformedURLException e) {
171                             e.printStackTrace();
172                             return null;
173                         }
174                     }
175                     final URL[] urls = urlLocations.clone();
176                     privateCL = AccessController.doPrivileged(new PrivilegedAction<ModuleClassLoader>() {
177                         public ModuleClassLoader run() {
178                             return new ModuleClassLoader(
179                                 ModuleImpl.this,
180                                 urls,
181                                 registry.getParentClassLoader());
182                         }
183                     });
184                 }
185             }
186         }
187         return privateCL;
188     }

253     /***
254      * Trigger manual refresh mechanism, the module will check all its 
255      * URLs and generate change events if any of them has changed. This 
256      * will allow the owning registry to force a module upgrade at next 
257      * module request.
258      */
259     public void refresh() {
260         URI[] urls = moduleDef.getLocations();
261         boolean notify = false;
262         for (URI lib : urls) {
263             File f = new File(lib);
264             if (f.exists() && lastModifieds.containsKey(f.getAbsolutePath())) {
265                 if (lastModifieds.get(f.getAbsolutePath()) !=f.lastModified()) {
266                     //Utils.getDefaultLogger().info("Changed : " + this);
267                     notify = true;
268                     break;
269                 }
270             }
271         }
272         if(notify) {
273             fireChangeEvent();
274         }
275     }

298     /***
299      * Ensure that this module is {@link ModuleState#RESOLVED resolved}.
300      *
301      * <p>
302      * If the module is already resolved, this method does nothing.
303      * Otherwise, iterate over all declared ModuleDependency instances and use the
304      * associated <code>ModuleRegistry</code> to resolve it. After successful
305      * completion of this method, the module state is 
306      * {@link ModuleState#RESOLVED}.
307      *
308      * @throws com.sun.enterprise.module.ResolveError if any of the declared dependency of this module
309      * cannot be satisfied
310      */
311     public synchronized void resolve() throws ResolveError {
312         
313         // already resolved ?
314         if (state==ModuleState.ERROR)
315             throw new ResolveError("Module " + getName() + " is in ERROR state");
316         if (state.compareTo(ModuleState.RESOLVED)>=0)
317             return;
318 
319         if (state==ModuleState.PREPARING) {
320             Utils.identifyCyclicDependency(this, Logger.getAnonymousLogger());
321             throw new ResolveError("Cyclic dependency with " + getName());
322         }
323         state = ModuleState.PREPARING;
324         
325         if (moduleDef.getImportPolicyClassName()!=null) {
326             try {
327                 Class<ImportPolicy> importPolicyClass = (Class<ImportPolicy>) getPrivateClassLoader().loadClass(moduleDef.getImportPolicyClassName());
328                 ImportPolicy importPolicy = importPolicyClass.newInstance();
329                 importPolicy.prepare(this);
330             } catch(ClassNotFoundException e) {
331                 state = ModuleState.ERROR;
332                 throw new ResolveError(e);
333             } catch(java.lang.InstantiationException e) {
334                 state = ModuleState.ERROR;
335                 throw new ResolveError(e);
336             } catch(IllegalAccessException e) {
337                 state = ModuleState.ERROR;
338                 throw new ResolveError(e);
339             }
340         }
341         for (ModuleDependency dependency : moduleDef.getDependencies()) {
342             ModuleImpl depModule = (ModuleImpl)registry.makeModuleFor(dependency.getName(), dependency.getVersion());
343             if (depModule==null) {
344                 state = ModuleState.ERROR;                
345                 throw new ResolveError(dependency + " referenced from " 
346                         + moduleDef.getName() + " is not resolved");
347             }
348 
349             //if (Utils.isLoggable(Level.INFO)) {
350             //    Utils.getDefaultLogger().info("For module" + getName() + " adding new dependent " + module.getName());
351             //}
352             dependencies.add(depModule);
353         }
354 
355         // once we have proper import/export filtering for modules, we can
356         // build a look-up table to improve performance
357 
358         // build up the complete list of transitive dependency modules, without any duplication,
359         // in a breadth-first fashion. The reason we do this in breadth-first is to reduce
360         // the search time based on the assumption that classes tend to be discovered in close dependencies. 
361         List<ModuleImpl> transitiveDependencies = new ArrayList<ModuleImpl>();
362         Set<ModuleImpl> transitiveDependenciesSet = new HashSet<ModuleImpl>();
363         LinkedList<ModuleImpl> q = new LinkedList<ModuleImpl>();
364         q.addAll(dependencies);
365         while(!q.isEmpty()) {
366             ModuleImpl m = q.removeFirst();
367             if(transitiveDependenciesSet.add(m)) {
368                 // first time visited
369                 transitiveDependencies.add(m);
370                 m.resolve();
371                 q.addAll(m.dependencies);
372             }
373         }
374 
375         for (ModuleImpl m : transitiveDependencies) {
376             getPrivateClassLoader().addDelegate(m.getClassLoader());
377         }
378 
379         //Logger.global.info("Module " + getName() + " resolved");
380         state = ModuleState.RESOLVED;
381         for (ModuleLifecycleListener l : registry.getLifecycleListeners()) {
382             l.moduleResolved(this);
383         }
384     }

386     /***
387      * Forces module startup. In most cases, the runtime will take care 
388      * of starting modules when they are first used. There could be cases where
389      * code need to manually start a sub module. Invoking this method will 
390      * move the module to the {@link ModuleState#READY ModuleState.READY}, the 
391      * {@link LifecyclePolicy#start Lifecycle.start} method will be invoked.
392      */
393     public void start() throws ResolveError {
394         
395         if (state==ModuleState.READY)
396             return;
397         
398         // ensure RESOLVED state
399         resolve();      
400         
401         for (Module subModules : dependencies) {
402             subModules.start();
403         }
404 
405         // time to initialize the lifecycle instance
406         if (moduleDef.getLifecyclePolicyClassName()!=null) {
407             try {
408                 Class<LifecyclePolicy> lifecyclePolicyClass = (Class<LifecyclePolicy>) getPrivateClassLoader().loadClass(moduleDef.getLifecyclePolicyClassName());
409                 lifecyclePolicy = lifecyclePolicyClass.newInstance();
410             } catch(ClassNotFoundException e) {
411                 state = ModuleState.ERROR;
412                 throw new ResolveError("ClassNotFound : " + e.getMessage(), e);
413             } catch(java.lang.InstantiationException e) {
414                 state = ModuleState.ERROR;
415                 throw new ResolveError(e);
416             } catch(IllegalAccessException e) {
417                 state = ModuleState.ERROR;
418                 throw new ResolveError(e);
419             }
420         }
421         if (lifecyclePolicy!=null) {
422             lifecyclePolicy.start(this);
423         }
424         state = ModuleState.READY;
425 
426         //Logger.global.info("Module " + getName() + " started");
427 
428         // module started. notify listeners
429         for (ModuleLifecycleListener listener : registry.getLifecycleListeners())
430             listener.moduleStarted(this);
431     }

433     /*** 
434      * Forces module stop. In most cases, the runtime will take care of stopping
435      * modules when the last module user released its interest. However, in 
436      * certain cases, it may be interesting to manually stop the module. 
437      * Stopping the module means that the module is removed from the registry, 
438      * the class loader references are released (note : the class loaders will
439      * only be released if all instances of any class loaded by them are gc'ed). 
440      * If a <code>LifecyclePolicy</code> for this module is defined, the 
441      * {@link LifecyclePolicy#stop(Module) Lifecycle.stop(Module)}
442      * method will be called and finally the module state will be 
443      * returned to {@link ModuleState#NEW ModuleState.NEW}.
444      *
445      * @return true if unloading was successful
446      */
447     public boolean stop() {
448 
449         if (sticky) {
450             return false;
451         } 
452         
453         if (lifecyclePolicy!=null) {
454             lifecyclePolicy.stop(this);
455             lifecyclePolicy=null;
456         }
457         
458         detach();
459         
460         // we do NOT stop our sub modules which are shared...
461         for (ModuleImpl subModule : dependencies) {
462             if (!subModule.isShared()) {
463                 subModule.stop();
464             }
465         }
466         // release all sub modules class loaders
467         privateCL = null;
468         publicCL = null;
469         dependencies.clear();
470         
471         state = ModuleState.NEW;
472 
473         // notify interested listeners that I have stopped.
474         for (ModuleLifecycleListener listener : registry.getLifecycleListeners()) {
475             listener.moduleStopped(this);
476         }
477         return true;
478     }

588     public Iterable<Class> getProvidersClass(String name) {
589         List<Class> r = serviceClasses.get(name);
590         if(r!=null) return r;
591 
592         // the worst case scenario in the race situation is we end up creating the same list twice,
593         // which is not a big deal.
594 
595         for( String provider : getMetadata().getEntry(name).providerNames) {
596             if(r==null)
597                 r = new ArrayList<Class>();
598             try {
599                 r.add(getPrivateClassLoader().loadClass(provider));
600             } catch (ClassNotFoundException e) {
601                 LogHelper.getDefaultLogger().log(Level.SEVERE, "Failed to load "+provider+" from "+getName(),e);
602             }
603         }
604 
605         if(r==null)
606             r = Collections.emptyList();
607 
608         serviceClasses.put(name, r);
609         return r;
610     }

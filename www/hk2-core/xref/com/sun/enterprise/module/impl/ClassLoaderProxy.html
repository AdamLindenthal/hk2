48  /***
49   * ClassLoaderProxy capable of loading classes from itself but also from other class loaders
50   *
51   * @author Jerome Dochez
52   */
53  public class ClassLoaderProxy extends URLClassLoader {
54  
55      private final List<ClassLoader> surrogates = new CopyOnWriteArrayList<ClassLoader>();
56      private final List<ClassLoaderFacade> facadeSurrogates = new CopyOnWriteArrayList<ClassLoaderFacade>();
57  
58      /*** Creates a new instance of ClassLoader */
59      public ClassLoaderProxy(URL[] shared, ClassLoader parent) {
60          super(shared, parent);
61      }

63      protected void finalize() throws Throwable {
64          super.finalize();
65          stop();
66      }

68      protected Class<?> loadClass(String name, boolean resolve, boolean followImports)
69              throws ClassNotFoundException {
70          // First, check if the class has already been loaded
71          Class c = findLoadedClass(name);
72          if (c == null) {
73              try {
74                  if (getParent() != null) {
75                      c = getParent().loadClass(name);
76                  }
77              } catch (ClassNotFoundException e) {
78  
79              }
80              if (c == null) {
81                  c = findClass(name, followImports);
82              }
83              if (resolve) {
84                  resolveClass(c);
85              }
86          } else {
87              if (c.getClassLoader() == this) {
88                  return c;
89              } else throw new ClassNotFoundException(name);
90          }
91  
92          return c;
93      }
94  
95      protected Class<?> findClass(String name) throws ClassNotFoundException {
96          return findClass(name, true);
97      }
98  
99      protected Class<?> findClass(String name, boolean followImports) throws ClassNotFoundException {
100         try {
101             // try to find it within this module first.
102             // this potentially causes a problem when two modules have the same jar in the classpath,
103             // but because the classes are most often found locally, this has a tremendous performance boost.
104             // so we knowingly make this decision to do child-first loading.
105 
106             // the pain of duplicate jars are somewhat mitigated by the fact that dependencies tend to be
107             // defined between HK2 modules, and those will not show up in the classpath of this module.
108             return findClassDirect(name);
109         } catch(ClassNotFoundException cfne) {
110             if (followImports) {
111                 Class c=null;
112                 for (ClassLoaderFacade classLoader : facadeSurrogates) {
113                     try {
114                         c = classLoader.getClass(name);
115                     } catch(ClassNotFoundException e) {
116                         // ignored.
117                     }
118                     if (c!=null) {
119                         return c;
120                     }
121                 }
122                 for (ClassLoader classLoader : surrogates) {
123                     try {
124                         c = classLoader.loadClass(name);
125                     } catch(ClassNotFoundException e) {
126                         // ignored.
127                     }
128                     if (c!=null) {
129                         return c;
130                     }
131                 }
132             }
133             throw cfne;
134         }
135     }

137     /***
138      * {@link #findClass(String)} except the classloader punch-in hack.
139      */
140     /*package*/ synchronized Class findClassDirect(String name) throws ClassNotFoundException {
141         Class c = findLoadedClass(name);
142         if(c!=null) return c;
143         try {
144             return super.findClass(name);
145         } catch (NoClassDefFoundError e) {
146             throw new ClassNotFoundException(e.getMessage());
147         }
148     }

150     public URL findResource(String name) {
151         URL url = super.findResource(name);
152         if (url!=null)  return url;
153 
154         for (ClassLoaderFacade classLoader : facadeSurrogates) {
155             url = classLoader.findResourceDirect(name);
156             if (url!=null) {
157                 return url;
158             }
159         }
160         for (ClassLoader classLoader : surrogates) {
161             url = classLoader.getResource(name);
162             if (url!=null) {
163                 return url;
164             }
165         }
166         return null;
167     }
168 
169     /***
170      * Works like {@link #findResource(String)} but only looks at
171      * this module, without delegating to ancestors.
172      */
173     public URL findResourceDirect(String name) {
174         return super.findResource(name);
175     }

177     public Enumeration<URL> findResources(String name) throws IOException {
178         // Let's build our list of enumerations first.
179         Vector<Enumeration<URL>> sources = new Vector<Enumeration<URL>>();
180         Enumeration< URL> enumerat = super.findResources(name);
181         if (enumerat!=null && enumerat.hasMoreElements()) {
182              sources.add(enumerat);
183         }
184         for (ClassLoaderFacade classLoader : facadeSurrogates) {
185             enumerat = classLoader.getResources(name);
186             if (enumerat!=null && enumerat.hasMoreElements()) {
187                 sources.add(enumerat);
188             }
189         }
190         for (ClassLoader classLoader : surrogates) {
191             enumerat = classLoader.getResources(name);
192             if (enumerat!=null && enumerat.hasMoreElements()) {
193                 sources.add(enumerat);
194             }
195         }
196         // return a flattened enumeration now.
197         return new FlattenEnumeration<URL>(sources.elements());
198     }

200     public void addDelegate(ClassLoader cl) {
201         if (cl instanceof ClassLoaderFacade) {
202             facadeSurrogates.add((ClassLoaderFacade) cl);
203         } else {
204             surrogates.add(cl);
205         }
206     }
207 
208     public void removeDelegate(ClassLoader cl) {
209         if (cl instanceof ClassLoaderFacade) {
210             facadeSurrogates.remove(cl);
211         } else {
212             surrogates.remove(cl);
213         }
214     }
215 
216     public Collection<ClassLoader> getDelegates() {
217         return new ArrayList<ClassLoader>(surrogates);
218     }
219 
220 
221     /***
222      * called by the facade class loader when it is garbage collected.
223      * this is a good time to see if this module should be unloaded.
224      */
225     public void stop() {
226        surrogates.clear();
227        facadeSurrogates.clear();
228     }

230     public String toString() {
231         StringBuffer s= new StringBuffer();
232         s.append(",URls[]=");
233         for (URL url : getURLs()) {
234             s.append(url).append(",");
235         }
236         s.append(")");
237 
238         for (ClassLoader surrogate : surrogates) {
239             s.append("\n ref : ").append(surrogate.toString());
240         }
241         return s.toString();
242     }

39  package com.sun.enterprise.module;
40  
41  import java.io.PrintStream;
42  import java.util.List;
43  
44  /**
45   * @author Sanjeeb.Sahoo@Sun.COM
46   */
47  public interface Module {
48      /**
49       * Returns the module definition for this module instance
50       * @return the module definition
51       */
52      ModuleDefinition getModuleDefinition();
53  
54      /**
55       * Short-cut for {@code getModuleDefinition().getName()}.
56       */
57      String getName();
58  
59      /**
60       * Returns the registry owning this module
61       * @return the registry owning the module
62       */
63      ModulesRegistry getRegistry();
64  
65      /**
66       * Returns the module's state
67       * @return the module's state
68       */
69      ModuleState getState();
70  
71      /**
72       * Ensure that this module is {@link ModuleState#RESOLVED resolved}.
73       *
74       * <p>
75       * If the module is already resolved, this method does nothing.
76       * Otherwise, iterate over all declared ModuleDependency instances and use the
77       * associated <code>ModuleRegistry</code> to resolve it. After successful
78       * completion of this method, the module state is
79       * {@link ModuleState#RESOLVED}.
80       *
81       * @throws ResolveError if any of the declared dependency of this module
82       * cannot be satisfied
83       */
84      void resolve() throws ResolveError;
85  
86      /**
87       * Forces module startup. In most cases, the runtime will take care
88       * of starting modules when they are first used. There could be cases where
89       * code need to manually start a sub module. Invoking this method will
90       * move the module to the {@link ModuleState#READY ModuleState.READY}, the
91       * {@link LifecyclePolicy#start Lifecycle.start} method will be invoked.
92       */
93      void start() throws ResolveError;
94  
95      /**
96       * Forces module stop. In most cases, the runtime will take care of stopping
97       * modules when the last module user released its interest. However, in
98       * certain cases, it may be interesting to manually stop the module.
99       * Stopping the module means that the module is removed from the registry,
100      * the class loader references are released (note : the class loaders will
101      * only be released if all instances of any class loaded by them are gc'ed).
102      * If a <code>LifecyclePolicy</code> for this module is defined, the
103      * {@link LifecyclePolicy#stop(Module) Lifecycle.stop(Module)}
104      * method will be called and finally the module state will be
105      * returned to {@link ModuleState#NEW ModuleState.NEW}.
106      *
107      * @return true if unloading was successful
108      */
109     boolean stop();
110 
111     /**
112      * Detach this module from its registry. This does not free any of the
113      * loaded resources. Only proper release of all references to the public
114      * class loader will ensure module being garbage collected.
115      * Detached modules are orphan and will be garbage collected if resources
116      * are properly disposed.
117      */
118     void detach();
119 
120     /**
121      * Trigger manual refresh mechanism, the module will check all its
122      * URLs and generate change events if any of them has changed. This
123      * will allow the owning registry to force a module upgrade at next
124      * module request.
125      */
126     void refresh();
127 
128     /**
129      * Gets the metadata of this module.
130      */
131     ModuleMetadata getMetadata();
132 
133     /**
134      * Add a new module change listener
135      * @param listener the listener
136      */
137     void addListener(ModuleChangeListener listener);
138 
139     /**
140      * Unregister a module change listener
141      * @param listener the listener to unregister
142      */
143     void removeListener(ModuleChangeListener listener);
144 
145     /**
146      * Return the <code>ClassLoader</code>  instance associated with this module.
147      * Only designated public interfaces will be loaded and returned by
148      * this classloader
149      * @return the public <code>ClassLoader</code>
150      */
151     ClassLoader getClassLoader();
152 
153     /**
154      * Returns the list of imported modules.
155      *
156      * <p>
157      * This is the module version of {@link ModuleDefinition#getDependencies()},
158      * but after fully resolved.
159      *
160      * <p>
161      * To enforce the stable class visibility, once {@link Module} is
162      * created, dependencies cannot be changed &mdash; that is, we
163      * don't want "a.b.C" to suddenly mean something different once
164      * the code starts running.
165      *
166      * @return
167      *      Can be empty but never null. Read-only.
168      */
169     List<Module> getImports();
170 
171     void addImport(Module module);
172 
173     /**
174      * Create and add a new module to this module's list of
175      * imports.
176      * @param dependency new module's definition
177      */
178     Module addImport(ModuleDependency dependency);
179 
180     /**
181      * Returns true if this module is sharable. A sharable module means that
182      * onlu one instance of the module classloader will be used by all users.
183      *
184      * @return true if this module is sharable.
185      */
186     boolean isShared();
187 
188     /**
189      * Returns true if the module is sticky. A sticky module cannot be stopped or
190      * unloaded. Once a sticky module is loaded or started, it will stay in the
191      * JVM until it exists.
192      * @return true is the module is sticky
193      */
194     boolean isSticky();
195 
196     /**
197      * Sets the sticky flag.
198      * @param sticky true if the module should stick around
199      */
200     void setSticky(boolean sticky);
201 
202     <T> Iterable<Class<? extends T>> getProvidersClass(Class<T> serviceClass);
203 
204     Iterable<Class> getProvidersClass(String name);
205 
206     /**
207      * Returns true if this module has any provider for the given service class.
208      */
209     boolean hasProvider(Class serviceClass);
210 
211     void dumpState(PrintStream writer);
212 
213     /**
214      * Removes the module from the registry backing store, the module will need
215      * be reinstalled to be loaded.
216      */
217     void uninstall();
218 }

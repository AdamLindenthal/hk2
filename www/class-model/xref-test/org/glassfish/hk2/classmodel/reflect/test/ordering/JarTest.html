72      public void process(String path) throws IOException {
73          File f = new File(path);
74          if (!f.exists()) {
75              System.out.println("File not found : " + path);
76              return;
77          }
78          Logger logger = Logger.getAnonymousLogger();
79          logger.setLevel(Level.FINE);
80          ParsingContext pc = new ParsingContext.Builder().logger(logger).build();
81          Parser parser = new Parser(pc);
82          long start = System.currentTimeMillis();
83  
84          parser.parse(f, new Runnable() {
85              @Override
86              public void run() {
87                  System.out.println("Done parsing");
88              }
89          });
90          try {
91              parser.awaitTermination();
92          } catch (InterruptedException e) {
93              e.printStackTrace();
94          }
95          Collection<Type> types = pc.getTypes().getAllTypes();
96          System.out.println("Finished parsing " + types.size() + " classes in " + (System.currentTimeMillis() - start) + "ms");
97  
98      }

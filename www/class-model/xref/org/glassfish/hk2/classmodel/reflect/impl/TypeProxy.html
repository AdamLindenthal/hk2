37  package org.glassfish.hk2.classmodel.reflect.impl;
38  
39  import org.glassfish.hk2.classmodel.reflect.*;
40  
41  import java.util.*;
42  
43  /**
44   * Proxy for types, used in place until the type can be properly instantiated.
45   * Proxy type also holds all incoming reference to the type.
46   *
47   * @author Jerome Dochez
48   */
49  public class TypeProxy<T extends Type> {
50  
51      private T value = null;
52      private volatile boolean visited = false;
53      private final String name;
54      private final Notifier<T> notifier;
55      private final Set<Member> fieldRefs;
56      private final Set<Type> subTypeRefs;
57      private final Set<ClassModel> implementations = Collections.synchronizedSet(new HashSet<ClassModel>());
58  
59  
60      /**
61       * Creates a new type proxy, this ctor is package private as many
62       * other activities must be performed when a new proxy type is created.
63       *
64       * @param notifier notification handle to notify receiver the proxied
65       * type has been resolved
66       * @param name type name
67       */
68      TypeProxy(Notifier<T> notifier, String name) {
69          this.notifier = notifier;
70          this.name = name;
71          fieldRefs = Collections.synchronizedSet(new HashSet<Member>());
72          subTypeRefs = Collections.synchronizedSet(new HashSet<Type>());
73      }
74      
75      @Override
76      public String toString() {
77        return "TypeProxy:" + name;
78      }
79  
80      public void set(T  value) {
81          this.value = value;
82          if (notifier!=null) {
83              notifier.valueSet(value);
84          }
85  
86      }
87  
88      public T get() {
89          return value;
90      }
91  
92      public String getName() {
93          if (value!=null) return value.getName();
94          return name;
95      }
96      
97      public interface Notifier<T> {
98          public void valueSet(T value);
99      }
100 
101     public Set<Member> getRefs() {
102         return fieldRefs;
103     }
104 
105     public Set<Type> getSubTypeRefs() {
106         return subTypeRefs;
107     }
108 
109     public Set<ClassModel> getImplementations() {
110         return implementations;
111     }
112 
113     public static <U extends Type> Collection<U> adapter(final Collection<TypeProxy<U>> source) {
114         return new AbstractCollection<U>() {
115 
116             @Override
117             public Iterator<U> iterator() {
118                 final Iterator<TypeProxy<U>> itr = source.iterator();
119                 return new Iterator<U>() {
120                     @Override
121                     public boolean hasNext() {
122                         return itr.hasNext();
123                     }
124 
125                     @Override
126                     public U next() {
127                         return itr.next().get();
128                     }
129 
130                     @Override
131                     public void remove() {
132                         throw new UnsupportedOperationException();
133                     }
134                 };
135             }
136 
137             @Override
138             public int size() {
139                 return source.size();
140             }
141         };
142     }
143 
144     public void visited() {
145         visited=true;
146     }
147 
148     public boolean isVisited() {
149         return visited;
150     }
151 }

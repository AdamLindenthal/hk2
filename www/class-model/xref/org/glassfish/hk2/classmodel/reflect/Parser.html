84      public synchronized Exception[] awaitTermination(int timeOut, TimeUnit unit) throws InterruptedException {
85  
86          List<Exception> exceptions = new ArrayList<Exception>();
87          if (context.logger.isLoggable(Level.FINE)) {
88              context.logger.log(Level.FINE, "awaiting termination of " + futures.size() + " tasks");
89          }
90  
91          for (Future<Result> f : futures) {
92              try {
93                  Result result = f.get(timeOut, unit);
94                  if (context.logger.isLoggable(Level.FINER)) {
95                      context.logger.log(Level.FINER, "result " + result);
96                      if (result!=null && result.fault!=null) {
97                          context.logger.log(Level.FINER, "result fault" + result);
98                      }
99                  }
100                 if (result!=null && result.fault!=null) {
101                     exceptions.add(result.fault);
102                 }
103             } catch (TimeoutException e) {
104                 exceptions.add(e);
105             } catch (ExecutionException e) {
106                 exceptions.add(e);
107             }
108             if (executorService!=null) {
109                 executorService.shutdown();
110                 executorService=null;
111             }
112         }
113         return exceptions.toArray(new Exception[exceptions.size()]);
114     }

116     public void parse(final File source, final Runnable doneHook) throws IOException {
117         // todo : use protocol to lookup implementation
118         final ArchiveAdapter adapter = source.isFile()?new JarArchive(source.toURI()):new DirectoryArchive(source);
119         final Runnable cleanUpAndNotify = new Runnable() {
120           @Override
121           public void run() {
122             try {
123               try {
124                 adapter.close();
125               } catch (IOException e) {
126                 throw new RuntimeException(e);
127               }
128             } finally {
129                 if (doneHook!=null)
130                     doneHook.run();
131             }
132           }
133         };
134         parse(adapter, cleanUpAndNotify);

137     public synchronized void parse(final ArchiveAdapter source, final Runnable doneHook) throws IOException {
138 
139         final Logger logger = context.logger;
140         Types types = getResult(source.getURI());
141         if (types!=null) {
142             if (!processedURI.containsKey(source.getURI())) {
143                 processedURI.put(source.getURI(), types);    
144             }
145             if (logger.isLoggable(Level.FINE)) {
146                 logger.fine("Skipping reparsing..." + source.getURI());
147             }
148             return;
149         }
150         
151         if (logger.isLoggable(Level.FINE)) {
152             logger.log(Level.FINE, "submitting file " + source.getURI().getPath());
153         }
154         futures.add(getExecutorService().submit(new Callable<Result>() {
155             @Override
156             public Result call() throws Exception {
157                 try {
158                     if (logger.isLoggable(Level.FINE)) {
159                         logger.log(Level.FINE, "elected file " + source.getURI().getPath());
160                     }
161                     doJob(source, doneHook);
162                     
163                     return new Result(source.getURI().getPath(), null);
164                 } catch (Exception e) {
165                     logger.log(Level.SEVERE, "Exception while parsing file " + source, e);
166                     return new Result(source.getURI().getPath(), e);
167                 }
168             }
169         }));
170     }

172     private Types getResult(URI uri) {
173         Types types = processedURI.get(uri);
174         if (types==null && parent!=null) {
175             types = parent.getResult(uri);
176         }
177         return types;
178     }

180     private void saveResult(URI uri, Types types) {
181         this.processedURI.put(uri, types);
182         if (parent!=null) {
183             parent.saveResult(uri, types);
184         }
185     }

187     private void doJob(final ArchiveAdapter adapter, final Runnable doneHook) throws Exception {
188         final Logger logger = context.logger;
189         if (logger.isLoggable(Level.FINE)) {
190             logger.fine("Parsing " + adapter.getURI() + " on thread " + Thread.currentThread().getName());
191         }
192         if (context.archiveSelector == null || context.archiveSelector.selects(adapter)) {
193             if (logger.isLoggable(Level.FINE)) {
194                 logger.log(Level.FINE, "Parsing file " + adapter.getURI().getPath());
195             }
196             final URI uri = adapter.getURI();
197 
198             adapter.onSelectedEntries(
199                     new ArchiveAdapter.Selector() {
200                         @Override
201                         public boolean isSelected(ArchiveAdapter.Entry entry) {
202                             return entry.name.endsWith(".class");
203                         }
204                     },
205                     new ArchiveAdapter.EntryTask() {
206                         @Override
207                         public void on(ArchiveAdapter.Entry entry, byte[] bytes) throws IOException {
208                             if (logger.isLoggable(Level.FINER)) {
209                                 logger.log(Level.FINER, "Parsing class " + entry.name);
210                             }
211 
212                             ClassReader cr = new ClassReader(bytes, 0, (int) entry.size);
213                             try {
214                                 cr.accept(context.getClassVisitor(uri, entry.name), ClassReader.SKIP_DEBUG);
215                             } catch (Throwable e) {
216                                 logger.log(Level.SEVERE, "Exception while visiting " + entry.name
217                                         + " of size " + entry.size, e);
218                             }
219                         }
220                     },
221                     logger
222             );
223             saveResult(uri, context.getTypes());
224         }
225         if (logger.isLoggable(Level.FINE)) {
226             logger.log(Level.FINE, "before running doneHook" + adapter.getURI().getPath());
227         }
228         if (doneHook != null)
229             doneHook.run();
230         if (logger.isLoggable(Level.FINE)) {
231             logger.log(Level.FINE, "after running doneHook " + adapter.getURI().getPath());
232         }
233     }

246     private synchronized ExecutorService getExecutorService() {
247         if (executorService==null) {
248             Runtime runtime = Runtime.getRuntime();
249             int nrOfProcessors = runtime.availableProcessors();
250             executorService = Executors.newFixedThreadPool(nrOfProcessors, new ThreadFactory() {
251                     @Override
252                     public Thread newThread(Runnable r) {
253                         Thread t = new Thread(r);
254                         t.setName("Hk2-jar-scanner");
255                         return t;
256                     }
257                 });
258         }
259         return executorService;
260     }

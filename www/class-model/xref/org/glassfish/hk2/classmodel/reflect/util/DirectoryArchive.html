36  package org.glassfish.hk2.classmodel.reflect.util;

38  import org.glassfish.hk2.classmodel.reflect.ArchiveAdapter;
39  
40  import java.io.*;
41  import java.nio.ByteBuffer;
42  import java.nio.channels.Channels;
43  import java.net.URI;
44  import java.nio.channels.FileChannel;
45  import java.util.jar.JarFile;
46  import java.util.jar.Manifest;
47  import java.util.logging.Level;
48  import java.util.logging.Logger;
49  
50  /***
51   * Directory base archive abstraction
52   */
53  public class DirectoryArchive extends AbstractAdapter {
54      public final File directory;
55  
56      public DirectoryArchive(File directory) {
57          this.directory = directory;
58      }
59  
60      @Override
61      public URI getURI() {
62          return directory.toURI();
63      }
64  
65  
66      @Override
67      public Manifest getManifest() throws IOException {
68          File manifest = new File(directory, JarFile.MANIFEST_NAME);
69          if (manifest.exists()) {
70              InputStream is = new BufferedInputStream(new FileInputStream(manifest));
71              try {
72                  return new Manifest(is);
73              } finally {
74                  is.close();
75              }
76          }
77          return null;
78      }
79  
80      @Override
81      public void onSelectedEntries(Selector selector, EntryTask task, Logger logger) throws IOException {
82          parse(directory, selector, task, logger);
83      }
84  
85      private void parse(File dir, Selector selector, EntryTask task, Logger logger) throws IOException {
86  
87          ByteBuffer buffer = ByteBuffer.allocate(52000);
88  
89          for (File f : dir.listFiles()) {
90              Entry ae = new Entry(mangle(f), f.length(), f.isDirectory());
91              if (!f.isDirectory()) {
92                  if (ae.name.endsWith(".jar")) {
93                      JarArchive ja = null;
94                      try {
95                          ja = new JarArchive(f.toURI());
96                          ja.onSelectedEntries(selector, task, logger);
97                      } finally {
98                          ja.close();
99                      }
100                     continue;
101                 }
102                 if (!selector.isSelected(ae))
103                     continue;
104                 FileChannel fc = null;
105                 try {
106                     try {
107                         if (buffer.capacity()<f.length()) {
108                             buffer = ByteBuffer.allocate((int) f.length());
109                         }
110                         fc = (new FileInputStream(f)).getChannel();
111 
112                         int read = fc.read(buffer);
113                         if (read!=f.length()) {
114                              logger.severe("Incorrect file length while reading " + f.getName() +
115                                      " of size " + f.length() + " reported is " + read);
116                         }
117                         buffer.rewind();
118                         task.on(ae, buffer.array());
119                     } catch(Exception e) {
120                         logger.log(Level.SEVERE, "Exception while processing " + f.getName() +
121                                 " of size " + f.length(), e);
122                     }
123 
124                 } finally {
125                     if (fc!=null) {
126                         fc.close();
127                     }
128                 }
129             } else {
130                 parse(f, selector, task, logger);
131             }
132 
133         }
134     }
135 
136     private String mangle(File f) {
137         String relativePath = f.getAbsolutePath().substring(directory.getAbsolutePath().length()+1);
138         return relativePath.replace(File.separatorChar, '/');
139     }
140 
141     @Override
142     public void close() throws IOException {
143     }
144 }

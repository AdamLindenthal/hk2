37  package org.glassfish.hk2.classmodel.reflect;
38  
39  import java.io.Closeable;
40  import java.io.IOException;
41  import java.io.InputStream;
42  import java.net.URI;
43  import java.util.jar.Manifest;
44  import java.util.logging.Logger;
45  
46  /***
47   * adapter for reading archive style structure
48   *
49   * @author Jerome Dochez
50   */
51  public interface ArchiveAdapter extends Closeable {
52  
53      /***
54       * Returns the URI of the archive
55       *
56       * @return URI of the archive
57       */
58      public URI getURI();
59  
60      /***
61       * Returns the manifest instance for the archive.
62       *
63       * @return the archive's manifest
64       * @throws IOException if the manifest cannot be loaded.
65       */
66      public Manifest getManifest() throws IOException ;
67  
68      /***
69       * defines the notion of an archive entry task which is a task
70       * aimed to be run on particular archive entry.
71       */
72      public interface EntryTask {
73          
74          /***
75           * callback to do some processing on an archive entry.
76           *
77           * @param e the archive entry information such as its name, size...
78           * @param bytes the archive entry content.
79           * @throws IOException if the input stream reading generates a failure
80           */
81          public void on(final Entry e, byte[] bytes) throws IOException;
82      }
83  
84      public interface Selector {
85  
86          /***
87           * callback to select an archive for processing
88           * @param entry the archive entry information
89           * @return true if the archive entry has been selected for processing
90           */
91          public boolean isSelected(final Entry entry);
92      }
93      
94      /***
95       * perform a task on each archive entry
96       *
97       * @param task the task to perform
98       * @param logger for any logging activity
99       * @throws IOException can be generated while reading the archive entries
100      */
101     public void onAllEntries(EntryTask task, Logger logger) throws IOException;
102 
103     /***
104      * perform a task on selected archive entries
105      *
106      * @param selector implementation to select the archive archive entries on
107      * which the task should be performed.
108      * @param task the task to perform
109      * @param logger for any logging activity
110      * @throws IOException can be generated while reading the archive entries
111      */
112     public void onSelectedEntries(Selector selector, EntryTask task, Logger logger) throws IOException;
113 
114 
115     /***
116      * Definition of an archive entry
117      */
118     public final class Entry {
119         final public String name;
120         final public long size;
121         final boolean isDirectory;
122 
123         /***
124          * creates a new archive entry
125          * @param name the entry name
126          * @param size the entry size
127          * @param isDirectory true if this entry is a directory
128          */
129         public Entry(String name, long size, boolean isDirectory) {
130             this.name = name;
131             this.size = size;
132             this.isDirectory = isDirectory;
133         }
134     }
135 }

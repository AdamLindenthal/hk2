39  import org.glassfish.hk2.classmodel.reflect.ArchiveAdapter;

41  import java.io.BufferedInputStream;
42  import java.io.IOException;
43  import java.io.InputStream;
44  import java.net.URI;
45  import java.util.Arrays;
46  import java.util.jar.JarEntry;
47  import java.util.jar.JarInputStream;
48  import java.util.jar.Manifest;
49  import java.util.logging.Level;
50  import java.util.logging.Logger;
51  
52  /***
53   * Archive adapter based on a single InputStream instance.
54   * @author Jerome Dochez
55   */
56  public class InputStreamArchiveAdapter extends AbstractAdapter {
57      
58      final private InputStream is;
59      final private URI uri;
60  
61      public InputStreamArchiveAdapter(URI uri, InputStream is) {
62          this.uri = uri;
63          this.is = is;
64      }
65  
66      @Override
67      public URI getURI() {
68          return uri;
69      }
70  
71      @Override
72      public Manifest getManifest() throws IOException {
73          throw new IOException("Not Implemented");
74      }
75  
76      @Override
77      public void onSelectedEntries(Selector selector, EntryTask task, Logger logger) throws IOException {
78          JarInputStream jis = new JarInputStream(new BufferedInputStream(is));
79          byte[] bytes = new byte[52000];
80          JarEntry ja;
81          while ((ja=jis.getNextJarEntry())!=null) {
82              try {
83                  Entry je = new Entry(ja.getName(), ja.getSize(), ja.isDirectory());
84                  if (!selector.isSelected(je))
85                      continue;
86  
87                  try {
88                      if (ja.getSize()>bytes.length) {
89                          bytes = new byte[(int) ja.getSize()];
90                      }
91                      if (ja.getSize()!=0) {
92                          // beware, ja.getSize() can be equal to -1 if the size cannot be determined.
93  
94                          int read = 0;
95                          int allRead=0;
96                          do {
97                              read = jis.read(bytes, allRead, bytes.length-allRead);
98                              allRead+=read;
99                              if (allRead==bytes.length) {
100                                 // oh crap !
101                                 bytes = Arrays.copyOf(bytes, bytes.length*2);
102                             }
103 
104                         } while (read!=-1);
105 
106                         if (ja.getSize()!=-1 && ja.getSize()!=(allRead+1)) {
107                             logger.severe("Incorrect file length while processing " + ja.getName() + " of size " + ja.getSize() + " got " + allRead);
108                         }
109 
110                         // if the size was not known, let's reset it now.
111                         if (je.size==-1) {
112                             je = new Entry(ja.getName(), allRead+1, ja.isDirectory());
113                         }
114                     }
115                     task.on(je, bytes);
116 
117                 } catch (Exception e) {
118                     logger.log(Level.SEVERE, "Exception while processing " + ja.getName()
119                             + " of size " + ja.getSize(), e);
120                 }
121             } finally {
122                 jis.closeEntry();
123             }
124         }
125         jis.close();
126     }
127 
128     @Override
129     public void close() throws IOException {
130         is.close();
131     }
132 }

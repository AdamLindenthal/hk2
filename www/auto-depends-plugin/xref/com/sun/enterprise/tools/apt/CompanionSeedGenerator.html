54  import java.io.IOException;

56  /***
57   * Generates {@link CompanionSeed} classes for each {@link CompanionOf} components.
58   * @author Kohsuke Kawaguchi
59   */
60  public class CompanionSeedGenerator extends SimpleDeclarationVisitor implements AnnotationProcessor {
61      private final AnnotationProcessorEnvironment env;
62      private JCodeModel cm;
63  
64      public CompanionSeedGenerator(AnnotationProcessorEnvironment env) {
65          this.env = env;
66      }
67  
68      public void process() {
69          cm = new JCodeModel();
70  
71          AnnotationTypeDeclaration ann = (AnnotationTypeDeclaration) env.getTypeDeclaration(CompanionOf.class.getName());
72          for(Declaration d : env.getDeclarationsAnnotatedWith(ann))
73              d.accept(this);
74  
75          try {
76              cm.build(new FilerCodeWriter(env.getFiler()));
77          } catch (IOException e) {
78              throw new Error(e);
79          }
80          cm = null;
81      }
82  
83      /***
84       * For each class annotated with {@link CompanionOf}.
85       */
86      public void visitClassDeclaration(ClassDeclaration clz) {
87          try {
88              CompanionOf companionOf = clz.getAnnotation(CompanionOf.class);
89              String lead;
90              try {
91                  companionOf.value();
92                  throw new AssertionError();
93              } catch (MirroredTypeException e) {
94                  lead = e.getTypeMirror().toString();
95              }
96  
97              JDefinedClass seed = cm._class(clz.getQualifiedName()+"Seed");
98              JAnnotationUse a = seed.annotate(CompanionSeed.class);
99  
100             a.param("lead",cm.ref(lead));
101             a.param("companion",cm.ref(clz.getQualifiedName()));
102         } catch (JClassAlreadyExistsException e) {
103             env.getMessager().printError(clz.getPosition(),e.toString());
104         }
105     }
106 }

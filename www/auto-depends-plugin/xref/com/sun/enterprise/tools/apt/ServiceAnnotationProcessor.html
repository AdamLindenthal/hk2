67  /***
68   * This class is proccessing the @{@link Service} annotation and generates
69   * META-INF/services style text file for each interface annotated with @Contract
70   *
71   * @author Jerome Dochez
72   */
73  public class ServiceAnnotationProcessor implements AnnotationProcessor, RoundCompleteListener {
74  
75      private final boolean debug;
76      private final AnnotationProcessorEnvironment env;
77      private Map<String, ServiceFileInfo> serviceFiles = new HashMap<String, ServiceFileInfo>();
78  
79      /***
80       * Creates a new instance of ServiceAnnotationProcessor
81       */
82      public ServiceAnnotationProcessor(AnnotationProcessorEnvironment env) {
83          this.env = env;
84          env.addListener(this);
85  
86          debug = env.getOptions().containsKey("-Adebug");
87          if (debug) {
88              env.getMessager().printNotice(env.getOptions().toString());
89          }
90          
91          // load all existing meta-inf files...
92          loadExistingMetaInfFiles();
93      }

96      /***
97       * Loads all existing META-INF/services file from our destination directory
98       * This is usuful because during incremental builds, not all source files
99       * are recompiled, henve we cannot rewrite the META-INF/services file from
100      * scratch each time, but append/remove entries from it as necessary.
101      */
102     protected void loadExistingMetaInfFiles() {
103         String outDirectory = env.getOptions().get("-s");
104         if (outDirectory==null) {
105             outDirectory = System.getProperty("user.dir");
106         }
107         File outDir = new File(new File(outDirectory),"META-INF/services").getAbsoluteFile();
108         if (debug) {
109             env.getMessager().printNotice("Output dir is " + outDir.getAbsolutePath());
110         }
111 
112         if (!outDir.exists()) {
113             return;
114         }
115         for (File file : outDir.listFiles()) {
116             if(file.isDirectory())  continue;
117             Set<String> entries = new HashSet<String>();
118             try {
119                 FileReader reader = new FileReader(file);
120                 LineNumberReader lineReader = new LineNumberReader(reader);
121                 String line = lineReader.readLine();
122                 while (line != null) {
123                     entries.add(line);
124                     line = lineReader.readLine();
125                 }
126             } catch (IOException e) {
127                 env.getMessager().printError(e.getMessage());
128             }
129             ServiceFileInfo info = new ServiceFileInfo(file.getName(), entries);
130             serviceFiles.put(file.getName(), info);
131         }
132     }

134     /***
135      * Annotation processor entry point, we are using a visitor pattern the visit
136      * only the class declaration.
137      */
138     public void process() {
139         DeclarationVisitor scanner = DeclarationVisitors.getDeclarationScanner(
140             new ListClassVisitor(),
141             DeclarationVisitors.NO_OP);
142 
143         for (TypeDeclaration typeDecl : env.getSpecifiedTypeDeclarations())
144             typeDecl.accept(scanner);
145     }
146 
147     /***
148      * Invoked on all class declaration, use the cached META-INF/servies
149      * information and the mirror APIs to find out if classes need to be
150      * added or removed from the generated service file.
151      */
152     private class ListClassVisitor extends SimpleDeclarationVisitor {
153         public void visitClassDeclaration(ClassDeclaration d) {
154             if (debug) {
155                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
156             }
157             Service service = d.getAnnotation(Service.class);
158             if (debug) {
159                 env.getMessager().printNotice("Service annotation = " + service);
160             }
161             if (service != null) {
162 
163                 // look for @ContractProvided interface
164                 checkContractProvided(d);
165                 
166                 // look for contract in interfaces
167                 for (InterfaceType intf : d.getSuperinterfaces()) {
168                     checkContract(intf.getDeclaration(), d);
169                 }
170 
171                 // look for contract in super classes
172                 ClassDeclaration sd = d;
173                 while(sd.getSuperclass()!=null) {
174                     sd = sd.getSuperclass().getDeclaration();
175                     checkContract(sd,d);
176                 }
177             } else {
178                 // we need to check if that class previously add an @Service annotation so
179                 // we remove the entry from the META-INF file
180                 for (ServiceFileInfo info : serviceFiles.values()) {
181                     if (debug) {
182                         env.getMessager().printNotice("Checking against " + info.getServiceName());
183                     }
184                     for (String implementor : info.getImplementors()) {
185                         if (implementor.equals(d.getQualifiedName())) {
186                             if (debug) {
187                                 env.getMessager().printNotice("Need to remove " + implementor);
188                             }
189                             info.getImplementors().remove(implementor);
190                             try {
191                                 info.createFile(env);
192                             } catch(IOException ioe) {
193                                 env.getMessager().printError(ioe.getMessage());
194                             }
195                             return;    
196                         }
197                     }
198                 }
199             }
200 
201             // check for meta annotations
202             for(AnnotationMirror a : d.getAnnotationMirrors()) {
203                 AnnotationTypeDeclaration atd = a.getAnnotationType().getDeclaration();
204                 Contract c = atd.getAnnotation(Contract.class);
205                 if(c!=null) {
206                     // this is a contract annotation
207                     createContractImplementation(atd.getQualifiedName(),d);
208                 }
209             }
210         }
211 
212 
213         private void checkContractProvided(ClassDeclaration impl) {
214             ContractProvided provided = impl.getAnnotation(ContractProvided.class);
215 
216             String intfName = null;
217             if (provided != null) {
218 
219                 try {
220                     provided.value();
221                 } catch (MirroredTypeException e) {
222                     intfName = e.getQualifiedName();
223                 }
224                 if (debug) {
225                     System.out.println("Provided is " + intfName);
226                 }
227                 createContractImplementation(intfName, impl);
228             }
229 
230         }
231 
232         private void checkContract(TypeDeclaration type, ClassDeclaration impl) {
233             Contract contract = type.getAnnotation(Contract.class);
234 
235             if (contract != null)
236                 createContractImplementation(type.getQualifiedName(), impl);
237         }
238     }

240     /***
241      * Records the contract&lt;->service relationship.
242      *
243      * @param contractName
244      *      FQCN of the contract type/annotation.
245      * @param impl
246      *      Implementation class.
247      */
248     private void createContractImplementation(String contractName, ClassDeclaration impl) {
249         ServiceFileInfo info;
250         if (!serviceFiles.containsKey(contractName)) {
251             info = new ServiceFileInfo(contractName, new HashSet<String>());
252             serviceFiles.put(contractName, info);
253         } else {
254             info = serviceFiles.get(contractName);
255         }
256 
257         if (info.getImplementors().add(impl.getQualifiedName())) {
258             try {
259                 info.createFile(env);
260             } catch (IOException ioe) {
261                 env.getMessager().printError(ioe.getMessage());
262             }
263         }
264     }

266     /***
267      * Notification of round complete, I use this to actually write the
268      * service file content and close the file.
269      * 
270      * @param e
271      */
272     public void roundComplete(RoundCompleteEvent e) {
273         for (ServiceFileInfo info : serviceFiles.values()) {
274             if (info.isDirty()) {
275                 if (debug) {
276                     env.getMessager().printNotice("Creating META-INF/services "+ info.getServiceName() + " file");
277                 }
278                 PrintWriter writer = info.getWriter();
279                 for (String implementor : info.getImplementors()) {
280                     if (debug) {
281                         env.getMessager().printNotice(" Implementor " + implementor);
282                     }
283                     writer.println(implementor);
284                 }
285                 writer.close();
286             }
287         }
288     }

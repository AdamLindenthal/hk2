39  import com.sun.mirror.declaration.AnnotationMirror;
40  import com.sun.mirror.declaration.AnnotationTypeElementDeclaration;
41  import com.sun.mirror.declaration.AnnotationValue;
42  import com.sun.mirror.declaration.TypeDeclaration;
43  import com.sun.mirror.type.AnnotationType;
44  import com.sun.mirror.type.DeclaredType;
45  import com.sun.mirror.type.TypeMirror;
46  import org.jvnet.hk2.annotations.InhabitantMetadata;
47  
48  import java.lang.annotation.Annotation;
49  import java.util.HashMap;
50  import java.util.Map;

52  /***
53   * Discoveres all {@link InhabitantMetadata} and puts them into the bag.
54   *
55   * @author Kohsuke Kawaguchi
56   */
57  public class InhabitantMetadataProcessor extends TypeHierarchyVisitor<Map<String,String>> {
58  
59      private final Map<AnnotationType,Model> models = new HashMap<AnnotationType, Model>();
60  
61      /***
62       * For a particular {@link AnnotationType}, remember what properties are to be added as metadata.
63       */
64      private static final class Model {
65          private final AnnotationType type;
66          private final Map<AnnotationTypeElementDeclaration,String> metadataProperties = new HashMap<AnnotationTypeElementDeclaration, String>();
67  
68          public Model(AnnotationType type) {
69              this.type = type;
70              for (AnnotationTypeElementDeclaration e : type.getDeclaration().getMethods()) {
71                  InhabitantMetadata im = e.getAnnotation(InhabitantMetadata.class);
72                  if(im==null)    continue;
73  
74                  String name = im.value();
75                  if(name.length()==0)    name=type.getDeclaration().getQualifiedName()+'.'+e.getSimpleName();
76  
77                  metadataProperties.put(e,name);
78              }
79          }
80  
81          /***
82           * Based on the model, parse the annotation mirror and updates the metadata bag by adding
83           * discovered values.
84           */
85          public void parse(AnnotationMirror a, Map<String,String> metadataBag) {
86              assert a.getAnnotationType().equals(type);
87  
88              for (Map.Entry<AnnotationTypeElementDeclaration, String> e : metadataProperties.entrySet()) {
89                  AnnotationValue value = a.getElementValues().get(e.getKey());
90                  if(value!=null)
91                      metadataBag.put(e.getValue(),toString(value));
92              }
93          }
94  
95          private String toString(AnnotationValue value) {
96              if (value.getValue() instanceof TypeMirror) {
97                  TypeMirror tm = (TypeMirror) value.getValue();
98                  // TODO: needs to be more robust
99                  if (tm instanceof DeclaredType) {
100                     DeclaredType dt = (DeclaredType) tm;
101                     return getClassName(dt.getDeclaration());
102                 }
103             }
104             return value.toString();
105         }
106 
107         /***
108          * Returns the fully qualified class name.
109          * The difference between this and {@link TypeDeclaration#getQualifiedName()}
110          * is that this method returns the same format as {@link Class#getName()}.
111          *
112          * Notably, separator for nested classes is '$', not '.'
113          */
114         private String getClassName(TypeDeclaration d) {
115             if(d.getDeclaringType()!=null)
116                 return getClassName(d.getDeclaringType())+'$'+d.getSimpleName();
117             else
118                 return d.getQualifiedName();
119         }
120     }
121 
122     public Map<String,String> process(TypeDeclaration d) {
123         visited.clear();
124         Map<String,String> r = new HashMap<String, String>();
125         check(d,r);
126         return r;
127     }
128 
129     protected void check(TypeDeclaration d, Map<String,String> result) {
130         checkAnnotations(d, result);
131         super.check(d,result);
132     }
133 
134     private void checkAnnotations(TypeDeclaration d, Map<String, String> result) {
135         for (AnnotationMirror a : d.getAnnotationMirrors()) {
136             getModel(a.getAnnotationType()).parse(a,result);
137             // check meta-annotations
138             for (AnnotationMirror b : a.getAnnotationType().getDeclaration().getAnnotationMirrors()) {
139                 getModel(b.getAnnotationType()).parse(b,result);
140             }
141         }
142     }
143 
144     /***
145      * Checks if the given annotation mirror has the given meta-annotation on it.
146      */
147     private boolean hasMetaAnnotation(AnnotationMirror a, Class<? extends Annotation> type) {
148         return a.getAnnotationType().getDeclaration().getAnnotation(type)!=null;
149     }
150 
151     private Model getModel(AnnotationType type) {
152         Model model = models.get(type);
153         if(model==null)
154             models.put(type,model=new Model(type));
155         return model;
156     }

70      public InhabitantsGenerator(AnnotationProcessorEnvironment env, DescriptorList list) {
71          this.env = env;
72          debug = env.getOptions().containsKey("-Adebug");
73          env.addListener(this);
74  
75          this.list = list;

79      public void process() {
80          // env.getTypeDeclarations() only contain types that are newly introduced in the round
81          DeclarationVisitor scanner = DeclarationVisitors.getDeclarationScanner(
82              new ListClassVisitor(),
83              DeclarationVisitors.NO_OP);
84          for (TypeDeclaration typeDecl : env.getTypeDeclarations())
85              typeDecl.accept(scanner);
86      }
87  
88      /***
89       * Defer the output till the very end, because otherwise we'll end up writing
90       * files multiple times, which APT doesn't like.
91       *
92       * But APT also doesn't like us creating files after the rounds are completed,
93       * so we need a hack to figure out where the output directory is, and use that.
94       */
95      public void roundComplete(RoundCompleteEvent event) {
96          if(!event.getRoundState().finalRound())
97              return;
98  
99          list.write(env);
100     }

114         private AnnotationMirror find(Declaration d, Class<? extends Annotation> annType) {
115             for(AnnotationMirror am : d.getAnnotationMirrors()) {
116                 if(am.getAnnotationType().getDeclaration().getQualifiedName().equals(annType.getName()))
117                     return am;
118             }
119             return null;
120         }

122         public void visitClassDeclaration(ClassDeclaration d) {
123             if (debug) {
124                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
125             }
126 
127             // check if this class has annotation that has InhabitantAnnotation meta-annotation.
128             InhabitantAnnotation ia=null;
129             AnnotationMirror a=null;
130             for (AnnotationMirror am : d.getAnnotationMirrors()) {
131                 ia = am.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
132                 if(ia!=null) {
133                     a = am;
134                     break;
135                 }
136             }
137             if(ia==null) {
138                 // doesn't have any component annotation.
139                 // in case annotations are removed from them, update the descriptors
140                 for (InhabitantsDescriptor id : list.descriptors.values())
141                     id.remove(d.getQualifiedName());
142                 return;
143             }
144 
145             if (debug) {
146                 env.getMessager().printWarning("Found component annotation " + a + " on "+d.getQualifiedName());
147             }
148             
149             // update the descriptor
150             InhabitantsDescriptor descriptor = list.get(ia.value());
151             descriptor.put(d.getQualifiedName(),getInhabitantDeclaration(a,d));
152         }

154         public void visitInterfaceDeclaration(InterfaceDeclaration d) {
155             if (debug) {
156                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
157             }
158 
159             // check if this class has annotation that has InhabitantAnnotation meta-annotation.
160             InhabitantAnnotation ia=null;
161             AnnotationMirror a=null;
162             for (AnnotationMirror am : d.getAnnotationMirrors()) {
163                 ia = am.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
164                 if(ia!=null) {
165                     a = am;
166                     if (debug) {
167                         env.getMessager().printWarning("Found component annotation " + a + " on "+d.getQualifiedName());
168                     }
169                     String service=null;
170                     ServiceProvider sp = a.getAnnotationType().getDeclaration().getAnnotation(ServiceProvider.class);
171                     if (sp!=null) {
172                         try {
173                             sp.value();
174                         } catch (MirroredTypeException e) {
175                             service = ((DeclaredType)e.getTypeMirror()).getDeclaration().getQualifiedName();
176                         }
177                     }
178 
179                     InhabitantsDescriptor descriptor = list.get(ia.value());
180 
181                     if (service!=null) {
182                         processGenericImpl(service, descriptor, d, a);
183                     }
184                 }
185             }
186         }

232         /***
233          * Visits a method declaration.
234          * The implementation simply invokes
235          * {@link #visitExecutableDeclaration visitExecutableDeclaration}.
236          * @param d the declaration to visit
237          */
238         public void visitMethodDeclaration(MethodDeclaration d) {
239             if (debug) {
240                 env.getMessager().printWarning("Visiting " + d.getSimpleName());
241             }
242             for (AnnotationMirror am : d.getAnnotationMirrors()) {
243                 Multiple multiple = am.getAnnotationType().getDeclaration().getAnnotation(Multiple.class);
244                 if (multiple!=null) {
245                     Collection<AnnotationTypeElementDeclaration> members = am.getAnnotationType().getDeclaration().getMethods();
246 
247                     AnnotationTypeElementDeclaration valueMember=null;
248                     for (AnnotationTypeElementDeclaration aMember : members) {
249                         if (aMember.getSimpleName().endsWith("value")) {
250                             valueMember = aMember;
251                             break;
252                         }
253                     }
254                     if (valueMember==null) return;
255 
256                     AnnotationValue valueVal = am.getElementValues().get(valueMember);
257                     Object annotationValue = valueVal.getValue();
258                     if (annotationValue instanceof Collection) {
259                         Collection<AnnotationValue> values = (Collection<AnnotationValue>) annotationValue;
260                         for (AnnotationValue aVal : values) {
261                             AnnotationMirror aValMirror = (AnnotationMirror) aVal.getValue();
262                             // check if the annotation contained in the multiple declaration is itself annotated
263                             // with InhabitantAnnotation
264                             InhabitantAnnotation ia = aValMirror.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
265                             if (ia!=null) {
266                                 generateInhabitantEntry(d, aValMirror, ia);
267                             }
268                         }
269                     } else {
270                         System.err.println("Ignoring " + am + " which is annotated with @Multiple, yet it's value() method is not collection");                        
271                     }
272                 }
273 
274                 InhabitantAnnotation ia = am.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
275                 if (ia!=null) {
276                     generateInhabitantEntry(d, am, ia);
277                 }
278             }
279         }

355         /***
356          * Computes the metadata line for the given class declaration. 
357          *
358          * @param a
359          *      The annotation which is meta-annotated with {@link InhabitantAnnotation}.
360          *      Used to extract the index.
361          */
362         public String getInhabitantDeclaration(AnnotationMirror a, ClassDeclaration d) {
363             indices.clear();
364             checkedInterfaces.clear();
365 
366             // check for Contract supertypes.
367             String name = getIndexValue(a);
368             for (TypeDeclaration t : ContractFinder.find(d)) {
369                 enforceContractLevelScope(t,d);
370                 addIndex(t.getQualifiedName(),name);
371             }
372 
373             // check for contract annotations
374             for(AnnotationMirror am : d.getAnnotationMirrors()) {
375                 AnnotationTypeDeclaration atd = am.getAnnotationType().getDeclaration();
376                 Contract c = atd.getAnnotation(Contract.class);
377                 if(c!=null) {
378                     // this is a contract annotation
379                     enforceContractLevelScope(atd,d);
380                     addIndex(atd.getQualifiedName(), getIndexValue(am));
381                 }
382 
383                 // check for meta-annotations
384                 for(AnnotationMirror mam : atd.getAnnotationMirrors()) {
385                     AnnotationTypeDeclaration matd = mam.getAnnotationType().getDeclaration();
386                     Contract mc = matd.getAnnotation(Contract.class);
387                     if(mc!=null)
388                         // meta-contract annotation
389                         addIndex(matd.getQualifiedName(), getIndexValue(mam));
390                 }
391             }
392 
393             StringBuilder buf = new StringBuilder();
394             buf.append(InhabitantsFile.CLASS_KEY).append('=').append(getClassName(d));
395             for (String contract : indices)
396                 addMetadata(buf, INDEX_KEY, contract);
397 
398             findInhabitantMetadata(d, buf);
399 
400             // for seed, capture the metadata for the actual companion object
401             CompanionSeed seed = d.getAnnotation(CompanionSeed.class);
402             if(seed!=null) {
403                 try {
404                     seed.companion();
405                 } catch (MirroredTypeException e) {
406                     ClassDeclaration companion = ((ClassType) e.getTypeMirror()).getDeclaration();
407                     addMetadata(buf, COMPANION_CLASS_METADATA_KEY,
408                         quote(getInhabitantDeclaration(find(companion, CompanionOf.class),companion)));
409                 }
410             }
411 
412             // TODO: should be deprecated and replaced with InhabitantMetadata
413             String metadata = getStringValue(a,"metadata");
414             if(metadata!=null && metadata.length()>0)
415                 buf.append(',').append(metadata);
416 
417             return buf.toString();
418         }

420         /***
421          * Returns the fully qualified class name.
422          * The difference between this and {@link TypeDeclaration#getQualifiedName()}
423          * is that this method returns the same format as {@link Class#getName()}.
424          *
425          * Notably, separator for nested classes is '$', not '.'
426          */
427         private String getClassName(TypeDeclaration d) {
428             if(d.getDeclaringType()!=null)
429                 return getClassName(d.getDeclaringType())+'$'+d.getSimpleName();
430             else
431                 return d.getQualifiedName();
432         }

438         private String quote(String s) {
439             StringBuilder buf = new StringBuilder();
440             buf.append('"');
441             for( int i=0; i<s.length(); i++ ) {
442                 char ch=s.charAt(i);
443                 switch(ch) {
444                 case '//':
445                     buf.append("////");
446                     break;
447                 case '"':
448                     buf.append("//\"");
449                     break;
450                 default:
451                     buf.append(ch);
452                 }
453             }
454             buf.append('"');
455             return buf.toString();
456         }

466         /***
467          * Finds the value of the annotation element annotated with {@link Index}.
468          */
469         private String getIndexValue(AnnotationMirror a) {
470             AnnotationTypeDeclaration decl = a.getAnnotationType().getDeclaration();
471             for(AnnotationTypeElementDeclaration e : decl.getMethods()) {
472                 if(e.getAnnotation(Index.class)!=null) {
473                     AnnotationValue v = a.getElementValues().get(e);
474                     if(v!=null) // explicitly given
475                         return v.getValue().toString();
476                     else // defaulted
477                         return e.getDefaultValue().getValue().toString();
478                 }
479             }
480             return null;
481         }

483         /***
484          * Given a use of annotation like <tt>@abc(def="ghi",jkl="mno")</tt> and a property name 'def',
485          * obtain the value of the annotation (in this example "ghi")
486          */
487         private String getStringValue(AnnotationMirror a, String name) {
488             AnnotationTypeDeclaration decl = a.getAnnotationType().getDeclaration();
489             for(AnnotationTypeElementDeclaration e : decl.getMethods()) {
490                 if(e.getSimpleName().equals(name)) {
491                     AnnotationValue v = a.getElementValues().get(e);
492                     if(v!=null) // explicitly given
493                         return v.getValue().toString();
494                     else // defaulted
495                         return e.getDefaultValue().getValue().toString();
496                 }
497             }
498             return null;
499         }

502         private void addIndex(String primary, String secondary) {
503             if(secondary==null || secondary.length()==0)
504                 indices.add(primary);    // unnamed
505             else
506                 indices.add(primary+':'+secondary); // named
507         }

509         /***
510          * Records the contract&lt;->service relationship.
511          *
512          * @param ctrct
513          *      The contract type declaration. Null in case of {@link ContractProvided}.
514          * @param impl
515          *      Implementation class.
516          */
517         private void enforceContractLevelScope(TypeDeclaration ctrct, ClassDeclaration impl) {
518             // if @Scoped is on the contract, that means we are forcing a certain scope type.
519             Scoped s = ctrct.getAnnotation(Scoped.class);
520             if(s==null)     return;
521 
522             try {
523                 s.value();
524                 throw new AssertionError();
525             } catch (MirroredTypeException e) {
526                 TypeMirror forcedScope = e.getTypeMirror();
527 
528                 s = impl.getAnnotation(Scoped.class);
529                 if(s!=null) {
530                     try {
531                         s.value();
532                         throw new AssertionError();
533                     } catch (MirroredTypeException f) {
534                         if(forcedScope.equals(f.getTypeMirror()))
535                             return; // forced scope and the actual scope are consistent
536                     }
537                 }
538 
539                 env.getMessager().printError(impl.getPosition(),"@Scoped("+forcedScope+") is required because of the contract "+ctrct.getQualifiedName());
540             }
541         }

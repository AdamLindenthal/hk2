83      public void process() {
84          // env.getTypeDeclarations() only contain types that are newly introduced in the round
85          DeclarationVisitor scanner = DeclarationVisitors.getDeclarationScanner(
86              new ListClassVisitor(),
87              DeclarationVisitors.NO_OP);
88          for (TypeDeclaration typeDecl : env.getTypeDeclarations())
89              typeDecl.accept(scanner);
90      }

92      /***
93       * Defer the output till the very end, because otherwise we'll end up writing
94       * files multiple times, which APT doesn't like.
95       *
96       * But APT also doesn't like us creating files after the rounds are completed,
97       * so we need a hack to figure out where the output directory is, and use that.
98       */
99      public void roundComplete(RoundCompleteEvent event) {
100         if(!event.getRoundState().finalRound())
101             return;
102 
103         list.write(env);
104     }

118         private AnnotationMirror find(Declaration d, Class<? extends Annotation> annType) {
119             for(AnnotationMirror am : d.getAnnotationMirrors()) {
120                 if(am.getAnnotationType().getDeclaration().getQualifiedName().equals(annType.getName()))
121                     return am;
122             }
123             return null;
124         }

126         public void visitClassDeclaration(ClassDeclaration d) {
127             if (debug) {
128                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
129             }
130 
131             // check if this class has annotation that has InhabitantAnnotation meta-annotation.
132             InhabitantAnnotation ia=null;
133             AnnotationMirror a=null;
134             for (AnnotationMirror am : d.getAnnotationMirrors()) {
135                 ia = am.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
136                 if(ia!=null) {
137                     a = am;
138                     break;
139                 }
140             }
141             if(ia==null) {
142                 // doesn't have any component annotation.
143                 // in case annotations are removed from them, update the descriptors
144                 for (InhabitantsDescriptor id : list.descriptors.values())
145                     id.remove(d.getQualifiedName());
146                 return;
147             }
148 
149             if (debug) {
150                 env.getMessager().printWarning("Found component annotation " + a + " on "+d.getQualifiedName());
151             }
152             
153             // update the descriptor
154             InhabitantsDescriptor descriptor = list.get(ia.value());
155             descriptor.put(d.getQualifiedName(),getInhabitantDeclaration(a,d));
156         }

158         public void visitInterfaceDeclaration(InterfaceDeclaration d) {
159             if (debug) {
160                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
161             }
162 
163             // check if this class has annotation that has InhabitantAnnotation meta-annotation.
164             InhabitantAnnotation ia=null;
165             AnnotationMirror a=null;
166             for (AnnotationMirror am : d.getAnnotationMirrors()) {
167                 ia = am.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
168                 if(ia!=null) {
169                     a = am;
170                     if (debug) {
171                         env.getMessager().printWarning("Found component annotation " + a + " on "+d.getQualifiedName());
172                     }
173                     String service=null;
174                     ServiceProvider sp = a.getAnnotationType().getDeclaration().getAnnotation(ServiceProvider.class);
175                     if (sp!=null) {
176                         try {
177                             sp.value();
178                         } catch (MirroredTypeException e) {
179                             service = ((DeclaredType)e.getTypeMirror()).getDeclaration().getQualifiedName();
180                         }
181                     }
182 
183                     InhabitantsDescriptor descriptor = list.get(ia.value());
184 
185                     if (service!=null) {
186                         processGenericImpl(service, descriptor, d, a);
187                     }
188                 }
189             }
190         }

236         /***
237          * Visits a method declaration.
238          * The implementation simply invokes
239          * {@link #visitExecutableDeclaration visitExecutableDeclaration}.
240          * @param d the declaration to visit
241          */
242         public void visitMethodDeclaration(MethodDeclaration d) {
243             if (debug) {
244                 env.getMessager().printWarning("Visiting " + d.getSimpleName());
245             }
246             InhabitantAnnotation ia=null;
247             AnnotationMirror a=null;
248             for (AnnotationMirror am : d.getAnnotationMirrors()) {
249                 ia = am.getAnnotationType().getDeclaration().getAnnotation(InhabitantAnnotation.class);
250                 if (ia!=null) {
251                     a=am;
252                 }
253             }
254             if (a==null)
255                 return;
256             InhabitantsDescriptor descriptor = list.get(ia.value());
257 
258             String service=null;
259             // I cannot use a.getAnnotationType().getDeclaration().getAnnotation() because the value() is a class
260             // which cannot be loaded by the classloader at compile time.
261             for (AnnotationMirror am : a.getAnnotationType().getDeclaration().getAnnotationMirrors()) {
262                 if (am.getAnnotationType().getDeclaration().getSimpleName().equals(ServiceProvider.class.getSimpleName())) {
263                     for (Map.Entry<AnnotationTypeElementDeclaration, AnnotationValue> entry : am.getElementValues().entrySet()) {
264                         service = entry.getValue().toString();
265                     }
266                 }
267             }
268             if (service!=null) {
269                 String name = getIndexValue(a);
270                 String contract="";
271                 // we should support gettting the @ContractProvided from the ServiceProvider
272                 for (AnnotationMirror am : a.getAnnotationType().getDeclaration().getAnnotationMirrors()) {
273                     if (am.getAnnotationType().getDeclaration().getSimpleName().equals(ContractProvided.class.getSimpleName())) {
274                         for (Map.Entry<AnnotationTypeElementDeclaration, AnnotationValue> entry : am.getElementValues().entrySet()) {
275                             contract = entry.getValue().toString();
276                         }
277                     }
278                 }
279 
280                 StringBuilder buf = new StringBuilder();
281                 buf.append(InhabitantsFile.CLASS_KEY).append('=').append(service);
282                 buf.append(",").append(INDEX_KEY).append("=").append(contract).append(":").append(name);
283                 buf.append(",").append(TARGET_TYPE).append("=").append(d.getDeclaringType().getQualifiedName());
284                 buf.append(",").append("method-name").append('=').append(d.getSimpleName());
285                 for (AnnotationTypeElementDeclaration ated : a.getAnnotationType().getDeclaration().getMethods()) {
286                     for (AnnotationMirror am : ated.getAnnotationMirrors()) {
287                         if (am.getAnnotationType().getDeclaration().getSimpleName().equals(InhabitantMetadata.class.getSimpleName())) {
288                             for (Map.Entry<AnnotationTypeElementDeclaration, AnnotationValue> entry : a.getElementValues().entrySet()) {
289                                 if (entry.getKey().getSimpleName().equals(ated.getSimpleName())) {
290                                     buf.append(",").append(ated.getSimpleName()).append("=").append(entry.getValue().toString());
291                                 }
292                             }
293                         }
294                     }
295                 }
296 
297                 descriptor.put(contract+":"+name, buf.toString());
298             } else {
299                 descriptor.put(d.getDeclaringType().getQualifiedName(),
300                         getInhabitantDeclaration(a, (ClassDeclaration) d.getDeclaringType()));
301             }
302         }

305         /***
306          * Computes the metadata line for the given class declaration. 
307          *
308          * @param a
309          *      The annotation which is meta-annotated with {@link InhabitantAnnotation}.
310          *      Used to extract the index.
311          */
312         public String getInhabitantDeclaration(AnnotationMirror a, ClassDeclaration d) {
313             indices.clear();
314             checkedInterfaces.clear();
315 
316             // check for Contract supertypes.
317             String name = getIndexValue(a);
318             for (TypeDeclaration t : ContractFinder.find(d)) {
319                 enforceContractLevelScope(t,d);
320                 addIndex(t.getQualifiedName(),name);
321             }
322 
323             // check for contract annotations
324             for(AnnotationMirror am : d.getAnnotationMirrors()) {
325                 AnnotationTypeDeclaration atd = am.getAnnotationType().getDeclaration();
326                 Contract c = atd.getAnnotation(Contract.class);
327                 if(c!=null) {
328                     // this is a contract annotation
329                     enforceContractLevelScope(atd,d);
330                     addIndex(atd.getQualifiedName(), getIndexValue(am));
331                 }
332 
333                 // check for meta-annotations
334                 for(AnnotationMirror mam : atd.getAnnotationMirrors()) {
335                     AnnotationTypeDeclaration matd = mam.getAnnotationType().getDeclaration();
336                     Contract mc = matd.getAnnotation(Contract.class);
337                     if(mc!=null)
338                         // meta-contract annotation
339                         addIndex(matd.getQualifiedName(), getIndexValue(mam));
340                 }
341             }
342 
343             StringBuilder buf = new StringBuilder();
344             buf.append(InhabitantsFile.CLASS_KEY).append('=').append(getClassName(d));
345             for (String contract : indices)
346                 addMetadata(buf, INDEX_KEY, contract);
347 
348             findInhabitantMetadata(d, buf);
349 
350             // for seed, capture the metadata for the actual companion object
351             CompanionSeed seed = d.getAnnotation(CompanionSeed.class);
352             if(seed!=null) {
353                 try {
354                     seed.companion();
355                 } catch (MirroredTypeException e) {
356                     ClassDeclaration companion = ((ClassType) e.getTypeMirror()).getDeclaration();
357                     addMetadata(buf, COMPANION_CLASS_METADATA_KEY,
358                         quote(getInhabitantDeclaration(find(companion, CompanionOf.class),companion)));
359                 }
360             }
361 
362             // TODO: should be deprecated and replaced with InhabitantMetadata
363             String metadata = getStringValue(a,"metadata");
364             if(metadata!=null && metadata.length()>0)
365                 buf.append(',').append(metadata);
366 
367             return buf.toString();
368         }

370         /***
371          * Returns the fully qualified class name.
372          * The difference between this and {@link TypeDeclaration#getQualifiedName()}
373          * is that this method returns the same format as {@link Class#getName()}.
374          *
375          * Notably, separator for nested classes is '$', not '.'
376          */
377         private String getClassName(TypeDeclaration d) {
378             if(d.getDeclaringType()!=null)
379                 return getClassName(d.getDeclaringType())+'$'+d.getSimpleName();
380             else
381                 return d.getQualifiedName();
382         }

384         private void addMetadata(StringBuilder buf, String key, String value) {
385             buf.append(',').append(key).append('=').append(value);

388         private String quote(String s) {
389             StringBuilder buf = new StringBuilder();
390             buf.append('"');
391             for( int i=0; i<s.length(); i++ ) {
392                 char ch=s.charAt(i);
393                 switch(ch) {
394                 case '//':
395                     buf.append("////");
396                     break;
397                 case '"':
398                     buf.append("//\"");
399                     break;
400                 default:
401                     buf.append(ch);
402                 }
403             }
404             buf.append('"');
405             return buf.toString();
406         }

416         /***
417          * Finds the value of the annotation element annotated with {@link Index}.
418          */
419         private String getIndexValue(AnnotationMirror a) {
420             AnnotationTypeDeclaration decl = a.getAnnotationType().getDeclaration();
421             for(AnnotationTypeElementDeclaration e : decl.getMethods()) {
422                 if(e.getAnnotation(Index.class)!=null) {
423                     AnnotationValue v = a.getElementValues().get(e);
424                     if(v!=null) // explicitly given
425                         return v.getValue().toString();
426                     else // defaulted
427                         return e.getDefaultValue().getValue().toString();
428                 }
429             }
430             return null;
431         }

433         /***
434          * Given a use of annotation like <tt>@abc(def="ghi",jkl="mno")</tt> and a property name 'def',
435          * obtain the value of the annotation (in this example "ghi")
436          */
437         private String getStringValue(AnnotationMirror a, String name) {
438             AnnotationTypeDeclaration decl = a.getAnnotationType().getDeclaration();
439             for(AnnotationTypeElementDeclaration e : decl.getMethods()) {
440                 if(e.getSimpleName().equals(name)) {
441                     AnnotationValue v = a.getElementValues().get(e);
442                     if(v!=null) // explicitly given
443                         return v.getValue().toString();
444                     else // defaulted
445                         return e.getDefaultValue().getValue().toString();
446                 }
447             }
448             return null;
449         }

452         private void addIndex(String primary, String secondary) {
453             if(secondary==null || secondary.length()==0)
454                 indices.add(primary);    // unnamed
455             else
456                 indices.add(primary+':'+secondary); // named
457         }

459         /***
460          * Records the contract&lt;->service relationship.
461          *
462          * @param ctrct
463          *      The contract type declaration. Null in case of {@link ContractProvided}.
464          * @param impl
465          *      Implementation class.
466          */
467         private void enforceContractLevelScope(TypeDeclaration ctrct, ClassDeclaration impl) {
468             // if @Scoped is on the contract, that means we are forcing a certain scope type.
469             Scoped s = ctrct.getAnnotation(Scoped.class);
470             if(s==null)     return;
471 
472             try {
473                 s.value();
474                 throw new AssertionError();
475             } catch (MirroredTypeException e) {
476                 TypeMirror forcedScope = e.getTypeMirror();
477 
478                 s = impl.getAnnotation(Scoped.class);
479                 if(s!=null) {
480                     try {
481                         s.value();
482                         throw new AssertionError();
483                     } catch (MirroredTypeException f) {
484                         if(forcedScope.equals(f.getTypeMirror()))
485                             return; // forced scope and the actual scope are consistent
486                     }
487                 }
488 
489                 env.getMessager().printError(impl.getPosition(),"@Scoped("+forcedScope+") is required because of the contract "+ctrct.getQualifiedName());
490             }
491         }

24  package com.sun.enterprise.tools.apt;
25  
26  import com.sun.mirror.apt.AnnotationProcessor;
27  import com.sun.mirror.apt.AnnotationProcessorEnvironment;
28  import com.sun.mirror.apt.AnnotationProcessorFactory;
29  import com.sun.mirror.declaration.AnnotationTypeDeclaration;
30  
31  import java.util.Arrays;
32  import java.util.Collection;
33  import java.util.Collections;
34  import java.util.Set;
35  
36  /***
37   * {@link AnnotationProcessorFactory} implementation to generate <tt>META-INF/services</tt>
38   * by scanning {@link org.jvnet.hk2.annotations.Service}.
39   *
40   * @author Jerome Dochez
41   */
42  public class MetainfServiceGenerator implements AnnotationProcessorFactory {
43  
44      /***
45       * Although I should theorically only return @Service annotation, I
46       * use this trick of returning '*' so I have my annoation processor called
47       * even if the @Service class is not present on the source file.
48       * This is very useful when a class that used to be annotated with @Service
49       * is not anymore so that I can remove it from the existing generated
50       * META-INF/services file.
51       */
52      private static final Collection<String> supportedAnnotations
53          = Collections.unmodifiableCollection(Arrays.asList("*"));
54  
55      // We only support debugging option
56      private static final Collection<String> supportedOptions = Arrays.asList("-Adebug");
57  
58      public Collection<String> supportedAnnotationTypes() {
59          return supportedAnnotations;
60      }
61  
62      public Collection<String> supportedOptions() {
63          return supportedOptions;
64      }
65  
66      public AnnotationProcessor getProcessorFor(
67              Set<AnnotationTypeDeclaration> atds,
68              AnnotationProcessorEnvironment env) {
69          
70              return new ServiceAnnotationProcessor(env);
71          
72      }    
73  }

89  	private void initXML() throws Exception {
90  		File file = new File(this.options.input);
91  		debug("file " + file);
92  		DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
93  		DocumentBuilder db = dbf.newDocumentBuilder();
94  		doc = db.parse(file);
95  		doc.getDocumentElement().normalize();

98  	private PackageInfo[] getPackageXML() throws IOException {
99  		NodeList pkgLst = doc.getElementsByTagName("Package");
100 		debug("Package count:" + pkgLst.getLength());
101 		List<PackageInfo> pkgInfos = new ArrayList<PackageInfo>();
102 		for (int i = 0; i < pkgLst.getLength(); i++) {
103 			Node pkgNode = pkgLst.item(i);
104 			if (pkgNode.getNodeType() == Node.ELEMENT_NODE) {
105 				Element packageElement = (Element) pkgNode;
106 				// exports
107 				NodeList ExportersList = packageElement
108 						.getElementsByTagName("Exporters");
109 				Element exporterElt = (Element) ExportersList.item(0);
110 				String exporter = ((Node) exporterElt.getChildNodes().item(0))
111 						.getNodeValue().trim();
112 				debug("Exporter : " + exporter);
113 
114 				// importers
115 				NodeList importersList = packageElement
116 						.getElementsByTagName("Importers");
117 				Element importerElt = (Element) importersList.item(0);
118 				String importers = ((Node) importerElt.getChildNodes().item(0))
119 						.getNodeValue().trim();
120 				debug("Importers : " + importers);
121 
122 				// Get package name and return PackageInfos
123 				String pkgName = packageElement.getAttribute("name").trim();
124 				debug("Package Name : " + pkgName);
125 				PackageInfo pkgInfo = new PackageInfo(pkgName, exporter,
126 						split(importers));
127 				pkgInfos.add(pkgInfo);
128 			}
129 		}
130 		return pkgInfos.toArray(new PackageInfo[] {});
131 	}

133 	private void generate() throws Exception {
134 		generateDotStart();
135 		PackageInfo[] pkgInfos = getPackageXML();
136 		for (PackageInfo pkgInfo : pkgInfos) {
137 			// Match if needed
138 			String matchString = this.options.match.trim();
139 			boolean matchNeeded = !matchString.isEmpty();
140 			if (matchNeeded) {
141 				if (pkgInfo.exportedBy.contains(matchString)) {
142 					generateDotEdge(pkgInfo.importedBy, pkgInfo.exportedBy,
143 							pkgInfo.packageName);
144 				}
145 			} else {
146 				generateDotEdge(pkgInfo.importedBy, pkgInfo.exportedBy,
147 						pkgInfo.packageName);
148 			}
149 		}
150 		generateDotEnd();
151 	}

204 	// Generate a Dot representation for each edge in the graph
205 	private void generateDotEdge(String[] importedBy, String exportedBy,
206 			String pkg) {
207 		if (importedBy.length == 0)
208 			return;
209 		for (String s : importedBy) {
210 			if (!s.equals(exportedBy)) { // remove self-loops for readability
211 				this.wireOut.println("\"" + s + "\" -> \"" + exportedBy
212 						+ "\" [label =\"" + pkg + "\"" + "]");
213 			}
214 		}
215 	}

222 	// Utility class to split the importers representation (space separated) 
223 	//in wires.xml
224 	private String[] split(String s) {
225 		StringTokenizer st = new StringTokenizer(s);
226 		List<String> l = new ArrayList<String>();
227 		while (st.hasMoreTokens()) {
228 			l.add(st.nextToken());
229 		}
230 		return l.toArray(new String[] {});

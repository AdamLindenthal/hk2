87              logger.logp(Level.INFO, "ObrHandler", "addingService",
88                      "We already have a repository admin service, so ignoring {0}", new Object[]{reference});

93          repositories.add(repositoryAdmin.getLocalRepository());

113         if (isDirectory(obrUri)) {
114             setupRepository(new File(obrUri), isSynchronous());

145     private boolean isSynchronous() {
146         String property = context.getProperty(Constants.INITIALIZE_OBR_SYNCHRONOUSLY);
147         // default is synchronous as we are not sure if we have covered every race condition in asynchronous path
148         return property == null || Boolean.TRUE.toString().equalsIgnoreCase(property);

180     private File getRepositoryFile(File repoDir) {
181         String extn = ".xml";
182         String cacheDir = context.getProperty(Constants.HK2_CACHE_DIR);
183         if (cacheDir == null) {
184             return null; // caching is disabled, so don't do it.
185         }
186         return new File(cacheDir, Constants.OBR_FILE_NAME_PREFIX + repoDir.getName() + extn);

189     /**
190      * Create a new Repository from a directory by recurssively traversing all the jar files found there.
191      *
192      * @param repoFile
193      * @param repoDir
194      * @return
195      * @throws IOException
196      */
197     private Repository createRepository(File repoFile, File repoDir) throws IOException {
198         DataModelHelper dmh = getRepositoryAdmin().getHelper();
199         List<Resource> resources = new ArrayList<Resource>();
200         for (File jar : findAllJars(repoDir)) {
201             Resource r = dmh.createResource(jar.toURI().toURL());
202             resources.add(r);
203         }
204         Repository repository = dmh.repository(resources.toArray(new Resource[resources.size()]));
205         logger.logp(Level.INFO, "ObrHandler", "createRepository", "Created {0} containing {1} resources.",
206                 new Object[]{repoFile, resources.size()});
207         if (repoFile != null) {
208             saveRepository(repoFile, repository);
209         }
210         return repository;
211     }

213     private void saveRepository(File repoFile, Repository repository) throws IOException {
214         assert (repoFile != null);
215         final FileWriter writer = new FileWriter(repoFile);
216         getRepositoryAdmin().getHelper().writeRepository(repository, writer);
217         writer.flush();
218     }

225     private Repository updateRepository(File repoFile, File repoDir) throws Exception {
226         Repository repository = loadRepository(repoFile);
227         if (isObsoleteRepo(repository, repoFile, repoDir)) {
228             if (!repoFile.delete()) {
229                 throw new IOException("Failed to delete " + repoFile.getAbsolutePath());
230             }
231             logger.logp(Level.INFO, "ObrHandler", "updateRepository", "Recreating {0}", new Object[]{repoFile});
232             repository = createRepository(repoFile, repoDir);
233         }
234         return repository;
235     }

237     private boolean isObsoleteRepo(Repository repository, File repoFile, File repoDir) {
238         // TODO(Sahoo): Revisit this...
239         // This method assumes that the cached repoFile has been created before a newer jar is created.
240         // So, this method does not always detect stale repoFile. Imagine the following situation:
241         // time t1: v1 version of jar is released.
242         // time t2: v2 version of jar is released.
243         // time t3: repo.xml is populated using v1 version of jar, so repo.xml records a timestamp of t3 > t2.
244         // time t4: v2 version of jar is unzipped on modules/ and unzip maintains the timestamp of jar as t2.
245         // Next time when we compare timestamp, we will see that repo.xml is newer than this jar, when it is not.
246         // So, we include a size check. We go for the total size check...
247 
248         long lastModifiedTime = repoFile.lastModified();
249         // optimistic: see if the repoDir has been touched. dir timestamp changes when files are added or removed.
250         if (repoDir.lastModified() > lastModifiedTime) {
251             return true;
252         }
253         long totalSize = 0;
254         // now compare timestamp of each jar and take a sum of size of all jars.
255         for (File jar : findAllJars(repoDir)) {
256             if (jar.lastModified() > lastModifiedTime) {
257                 logger.logp(Level.INFO, "ObrHandler", "isObsoleteRepo", "{0} is newer than {1}", new Object[]{jar, repoFile});
258                 return true;
259             }
260             totalSize += jar.length();
261         }
262         // time stamps didn't identify any difference, so check sizes. The probabibility of sizes of all jars being same
263         // when some jars have changed is very very low.
264         for (Resource r : repository.getResources()) {
265             totalSize -= r.getSize();
266         }
267         if (totalSize != 0) {
268             logger.logp(Level.INFO, "ObrHandler", "isObsoleteRepo", "Change in size detected by {0} bytes", new Object[]{totalSize});
269             return true;
270         }
271         return false;
272     }

274     private List<File> findAllJars(File repo) {
275         final List<File> files = new ArrayList<File>();
276         repo.listFiles(new FileFilter() {
277             @Override
278             public boolean accept(File pathname) {
279                 if (pathname.isDirectory()) {
280                     pathname.listFiles(this);
281                 } else if (pathname.getName().endsWith("jar")) {
282                     files.add(pathname);
283                 }
284                 return true;
285             }
286         });
287         return files;
288     }

291     synchronized Bundle deploy(Resource resource) {
292         final Resolver resolver = getRepositoryAdmin().resolver(getRepositories());
293         boolean resolved = resolve(resolver, resource);
294         if (resolved) {
295             final int flags = !deployOptionalRequirements ? Resolver.NO_OPTIONAL_RESOURCES : 0;
296             resolver.deploy(flags);
297             return getBundle(resource);
298         } else {
299             Reason[] reqs = resolver.getUnsatisfiedRequirements();
300             logger.logp(Level.WARNING, "ObrHandler", "deploy",
301                     "Unable to satisfy the requirements: {0}", new Object[]{Arrays.toString(reqs)});
302             return null;
303         }
304     }

333     synchronized Bundle deploy(String name, String version) {
334         Resource resource = findResource(name, version);
335         if (resource == null) {
336             logger.logp(Level.INFO, "ObrHandler", "deploy",
337                     "No resource matching name = {0} and version = {1} ", new Object[]{name, version});
338             return null;
339         }
340         if (resource.isLocal()) {
341             return getBundle(resource);
342         }
343         return deploy(resource);
344     }

346     private Bundle getBundle(Resource resource) {
347         for (Bundle b : context.getBundles()) {
348             final String bsn = b.getSymbolicName();
349             final Version bv = b.getVersion();
350             final String rsn = resource.getSymbolicName();
351             final Version rv = resource.getVersion();
352             boolean versionMatching = (rv == bv) || (rv != null && rv.equals(bv));
353             boolean nameMatching = (bsn == rsn) || (bsn != null && bsn.equals(rsn));
354             if (nameMatching && versionMatching) return b;
355         }
356         return null;
357     }

359     private Resource findResource(String name, String version) {
360         final RepositoryAdmin repositoryAdmin = getRepositoryAdmin();
361         if (repositoryAdmin == null) {
362             logger.logp(Level.WARNING, "ObrHandler", "findResource",
363                     "OBR is not yet available, so can't find resource with name = {0} and version = {1} from repository",
364                     new Object[]{name, version});
365             return null;
366         }
367         String s1 = "(symbolicname=" + name + ")";
368         String s2 = "(version=" + version + ")";
369         String query = (version != null) ? "(&" + s1 + s2 + ")" : s1;
370         try {
371             Resource[] resources = discoverResources(query);
372             logger.logp(Level.INFO, "ObrHandler", "findResource",
373                     "Using the first one from the list of {0} discovered bundles shown below: {1}",
374                     new Object[]{resources.length, Arrays.toString(resources)});
375             return resources.length > 0 ? resources[0] : null;
376         } catch (InvalidSyntaxException e) {
377             throw new RuntimeException(e); // TODO(Sahoo): Proper Exception Handling
378         }
379     }

381     private Resource[] discoverResources(String filterExpr) throws InvalidSyntaxException {
382         // TODO(Sahoo): File a bug against Obr to add a suitable method to Repository interface.
383         // We can't use the following method, because we can't rely on the RepositoryAdmin to have the correct
384         // list of repositories. So, we do the discovery ourselves.
385         // return getRepositoryAdmin().discoverResources(query);
386         Filter filter = filterExpr != null ? getRepositoryAdmin().getHelper().filter(filterExpr) : null;
387         Resource[] resources;
388         Repository[] repos = getRepositories();
389         List<Resource> matchList = new ArrayList<Resource>();
390         for (int repoIdx = 0; (repos != null) && (repoIdx < repos.length); repoIdx++) {
391             resources = repos[repoIdx].getResources();
392             for (int resIdx = 0; (resources != null) && (resIdx < resources.length); resIdx++) {
393                 Properties dict = new Properties();
394                 dict.putAll(resources[resIdx].getProperties());
395                 if (filter == null || filter.match(dict)) {
396                     matchList.add(resources[resIdx]);
397                 }
398             }
399         }
400 
401         return matchList.toArray(new Resource[matchList.size()]);
402     }

409     private StringBuffer getResolverOutput(Resolver resolver) {
410         Resource[] addedResources = resolver.getAddedResources();
411         Resource[] requiredResources = resolver.getRequiredResources();
412         Resource[] optionalResources = resolver.getOptionalResources();
413         Reason[] unsatisfiedRequirements = resolver.getUnsatisfiedRequirements();
414         StringBuffer sb = new StringBuffer("Added resources: [");
415         for (Resource r : addedResources) {
416             sb.append("\n").append(r.getSymbolicName()).append(", ").append(r.getVersion()).append(", ").append(r.getURI());
417         }
418         sb.append("]\nRequired Resources: [");
419         for (Resource r : requiredResources) {
420             sb.append("\n").append(r.getURI());
421         }
422         String optionalRequirementsDeployed = deployOptionalRequirements ? "deployed" : "not deployed";
423         sb.append("]\nOptional resources (" + optionalRequirementsDeployed + "): [");
424         for (Resource r : optionalResources) {
425             sb.append("\n").append(r.getURI());
426         }
427         sb.append("]\nUnsatisfied requirements: [");
428         for (Reason r : unsatisfiedRequirements) {
429             sb.append("\n").append(r.getRequirement());
430         }
431         sb.append("]");
432         return sb;
433     }

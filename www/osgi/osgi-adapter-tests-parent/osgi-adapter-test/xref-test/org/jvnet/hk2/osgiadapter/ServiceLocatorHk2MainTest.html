232 	    
233         ServiceTracker hk2Tracker = new ServiceTracker(
234                 this.bundleContext,
235                 InstallSDPService.class.getName(),
236                 null);
237         hk2Tracker.open();
238         InstallSDPService installationService = (InstallSDPService)
239                 hk2Tracker.waitForService(0);
240         
241         hk2Tracker.close();
242         
243         FooContract fooC = serviceLocator.getService(FooContract.class);
244         Assert.assertNull(fooC);
245         
246         /**
247          * First install and uninstall
248          */
249         installationService.install();
250         
251         List<ActiveDescriptor<?>> descriptors =
252                 serviceLocator.getDescriptors(
253                         BuilderHelper.createContractFilter(
254                                 FooContract.class.getName()));
255         Assert.assertEquals(1, descriptors.size());
256         
257         Assert.assertTrue(installationService.uninstall());
258         
259         descriptors =
260                 serviceLocator.getDescriptors(
261                         BuilderHelper.createContractFilter(
262                                 FooContract.class.getName()));
263         
264         Assert.assertEquals(0, descriptors.size());
265         
266         fooC = serviceLocator.getService(FooContract.class);
267         Assert.assertNull(fooC);
268         
269         /**
270          * then install again
271          */
272         installationService.install();
273         
274         descriptors =
275                 serviceLocator.getDescriptors(
276                         BuilderHelper.createContractFilter(
277                                 FooContract.class.getName()));
278         
279         Assert.assertEquals(1, descriptors.size());
280         
281         fooC = serviceLocator.getService(FooContract.class);

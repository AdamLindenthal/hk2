74  @Service

80    @Inject
81    private Habitat habitat;

83    @Inject @Optional
84    private InhabitantActivator activator;

88    private Map<Object, MultiMap<String, ConfigByInhabitant>> beanToManagedConfigBy = new HashMap<Object, MultiMap<String, ConfigByInhabitant>>();

90    @Override
91    public synchronized ConfigTransaction getActiveTransaction(boolean create) {
92      if (create && null == theActiveOne) {
93        theActiveOne = new ConfigTransactionImpl(this);
94      }
95      
96      return theActiveOne;
97    }
98    
99    synchronized void finishedTransaction(ConfigTransaction txn) {
100     assert(theActiveOne == txn);
101     theActiveOne = null;
102   }
103 
104   @Override
105   public void preDestroy() {
106     if (null != theActiveOne) {
107       theActiveOne.rollback();
108       assert(null == theActiveOne);
109     }
110   }

112   boolean has(Object bean) {
113     return beanToManagedConfigBy.containsKey(bean);
114   }

116   MultiMap<String, ConfigByInhabitant> get(Object bean) {
117     // return a copy so that Transaction has there own working area
118     MultiMap<String, ConfigByInhabitant> orig = beanToManagedConfigBy.get(bean);
119     return (null == orig) ? null : new MultiMap<String, ConfigByInhabitant>();
120   }

122   // TODO: look for a faster implementation
123   @SuppressWarnings("unchecked")
124   Collection<ConfigByMetaInhabitant> getConfiguredByInhabitants(Class<?> clazz) {
125     HashSet<Class<?>> configured = new HashSet<Class<?>>();
126 
127     if (Proxy.isProxyClass(clazz)) {
128       for (Class<?> iface : clazz.getInterfaces()) {
129         if (null != iface.getAnnotation(Configured.class)) {
130           configured.add(iface);
131         }
132       }
133     }
134     
135     while (Object.class != clazz) {
136       String name = clazz.getName();
137       if (null != name && null != clazz.getAnnotation(Configured.class)) {
138         configured.add(clazz);
139       }
140       clazz = clazz.getSuperclass();
141     }
142     
143     ArrayList<ConfigByMetaInhabitant> configuredByList = null;
144     for (Inhabitant<?> i : habitat.getInhabitantsByContract(ConfiguredBy.class.getName())) {
145       ConfigByMetaInhabitant configuredBy = (ConfigByMetaInhabitant)i;
146       if (configured.contains(configuredBy.getConfiguredBy())) {
147         if (null == configuredByList) {
148           configuredByList = new ArrayList<ConfigByMetaInhabitant>();
149         }
150         configuredByList.add(configuredBy);
151       }
152     }
153     
154     return (null == configuredByList) ? Collections.EMPTY_LIST : Collections.unmodifiableList(configuredByList);
155   }

157   /**
158    * Called when we are committing the txn to manage new beans and the services inhabitants for those beans
159    */
160   void manage(Object bean, Set<Class<?>> beanContracts, String name, MultiMap<String, ConfigByInhabitant> managed) {
161     if (logger.isLoggable(Level.FINE)) {
162       logger.log(Level.FINE, "managing: {0}, {1}, {2}, {3}", new Object[] {bean, name, beanContracts, managed});
163     }
164     
165     // put the bean itself into the habitat
166     if (null != beanContracts && !beanContracts.isEmpty()) {
167       Inhabitant<?> configBeanInhabitant = new ExistingSingletonInhabitant<Object>(bean);
168       for (Class<?> beanContract : beanContracts) {
169         habitat.addIndex(configBeanInhabitant, beanContract.getName(), name);
170       }
171     }
172 
173     // put the configured-by instances into the habitat
174     for (Entry<String, List<ConfigByInhabitant>> cbi : managed.entrySet()) {
175       String index = cbi.getKey();
176       if (null != index) {
177         StringBuilder sb = new StringBuilder();
178         index = InhabitantsParser.parseIndex(index, sb);
179         if (null == name) {
180           name = (0 == sb.length()) ? null : sb.toString();
181         }
182       }
183       
184       for (ConfigByInhabitant i : cbi.getValue()) {
185         if (null == index) {
186           // intentionally do not register byType since that is reserved for the meta inhabitant entry
187 //          habitat.add(i);
188           habitat.addIndex(i, i.typeName(), name);
189         } else {
190           habitat.addIndex(i, index, name);
191         }
192 
193         // force activation
194         if (null == activator) {
195           Object service = i.get();
196           assert(null != service);
197         } else {
198           activator.activate(i);
199         }
200       }
201     }
202 
203     beanToManagedConfigBy.put(bean, managed);
204 
205     if (null != activator) {
206       try {
207         activator.awaitCompletion();
208       } catch (Exception e) {
209         throw new ComponentException(e);
210       }
211     }
212   }

214   /**
215    * Called when we are committing a txn to remove the beans and the associated services for those beans from the habitat
216    */
217   void unmanage(Object bean, Set<Class<?>> beanContracts, String name) {
218     if (logger.isLoggable(Level.FINE)) {
219       logger.log(Level.FINE, "unmanaging: {0}, {1}, {2}", new Object[] {bean, name, beanContracts});
220     }
221     
222     // remove the configured-by services related to this bean
223     MultiMap<String, ConfigByInhabitant> managed = beanToManagedConfigBy.remove(bean);
224     for (Entry<String, List<ConfigByInhabitant>> cbi : managed.entrySet()) {
225       String index = cbi.getKey();
226       if (null != index) {
227         index = InhabitantsParser.parseIndex(index, null);
228       }
229       
230       for (ConfigByInhabitant i : cbi.getValue()) {
231         boolean removed;
232         if (null == index) {
233           // intentionally do not register byType since that is reserved for the meta inhabitant entry
234 //          habitat.add(i);
235           removed = habitat.removeIndex(i.typeName(), i);
236         } else {
237           removed = habitat.removeIndex(index, i);
238         }
239         assert(removed);
240         
241         i.release();
242       }
243     }
244     
245     // remove the bean itself from the habitat
246     if (null != beanContracts) {
247       for (Class<?> beanContract : beanContracts) {
248         boolean removed = habitat.removeIndex(beanContract.getName(), bean);
249         assert(removed);
250         
251         // TODO: should we release it too?
252       }
253     }
254   }

57  /**
58    * <p>
59    * Helper class to execute some code on configuration objects while taking
60    * care of the transaction boiler plate code.
61    * </p>
62    * <p>
63    * Programmers that wish to apply some changes to configuration objects
64    * can use these convenience methods to reduce the complexity of handling
65    * transactions.
66    * </p>
67    * <p>
68    * For instance, say a programmer need to change the HttpListener port from
69    * 8080 to 8989, it just needs to do :
70    * </p>
71    * <pre>
72    *     ... in his code somewhere ...
73    *     HttpListener httpListener = domain.get...
74    *
75    *     // If the programmer tries to modify the httpListener directly
76    *     // it will get an exception
77    *     httpListener.setPort("8989"); // will generate a PropertyVetoException
78    *
79    *     // instead he needs to use a transaction and can use the helper services
80    *     ConfigSupport.apply(new SingleConfigCode<HttpListener>() {
81    *         public Object run(HttpListener okToChange) throws PropertyException {
82    *             okToChange.setPort("8989"); // good...
83    *             httpListener.setPort("7878"); // not good, exceptions still raised...
84    *             return null;
85    *         });
86    *
87    *     // Note that after this code
88    *     System.out.println("Port is " + httpListener.getPort());
89    *     // will display 8989
90    * }
91    * </pre>
92    * @author Jerome Dochez
93    */
94  @Service

138     /**
139      * Executes some logic on some config beans protected by a transaction.
140      *
141      * @param code code to execute
142      * @param objects config beans participating to the transaction
143      * @return list of property change events
144      * @throws TransactionFailure when the code did run successfully due to a
145      * transaction exception
146      */    
147     public Object _apply(ConfigCode code, ConfigBeanProxy... objects)
148                 throws TransactionFailure {
149 
150         // the fools think they operate on the "real" object while I am
151         // feeding them with writeable view. Only if the transaction succeed
152         // will I apply the "changes" to the real ones.
153         WriteableView[] views = new WriteableView[objects.length];
154 
155         ConfigBeanProxy[] proxies = new ConfigBeanProxy[objects.length];
156 
157         // create writeable views.
158         for (int i=0;i<objects.length;i++) {
159             proxies[i] = getWriteableView(objects[i]);
160             views[i] = (WriteableView) Proxy.getInvocationHandler(proxies[i]);
161         }
162 
163         // Of course I am not locking the live objects but the writable views.
164         // if the user try to massage the real objects, he will get
165         // a well deserved nasty exception
166         Transaction t = new Transaction();
167         for (WriteableView view : views) {
168             if (!view.join(t)) {
169                 t.rollback();
170                 throw new TransactionFailure("Cannot enlist " + view.getMasterView().getProxyType()
171                     + " in transaction", null);
172             }
173         }
174         
175         try {
176             final Object toReturn = code.run(proxies);
177             try {
178                 t.commit();
179                 if (toReturn instanceof WriteableView) {
180                     return ((WriteableView) toReturn).getMasterView();
181                 } else {
182                     return toReturn;
183                 }
184             } catch (RetryableException e) {
185                 System.out.println("Retryable...");
186                 // TODO : do something meaninful here
187                 t.rollback();
188                 return null;
189             } catch (TransactionFailure e) {
190                 t.rollback();
191                 throw e;
192             }
193 
194         } catch (java.lang.reflect.UndeclaredThrowableException e) {
195             t.rollback();
196             Throwable throwable = e.getCause();
197             if (throwable instanceof PropertyVetoException) throw new TransactionFailure(throwable.toString(), throwable);
198             throw new TransactionFailure(e.toString(), e);
199 
200         } catch(TransactionFailure e) {
201             t.rollback();
202             throw e;
203         } catch (Exception e) {
204             t.rollback();
205             throw new TransactionFailure(e.getMessage(), e);
206         }
207 
208     }

210     static <T extends ConfigBeanProxy> WriteableView getWriteableView(T s, ConfigBean sourceBean)
211         throws TransactionFailure {
212 
213         WriteableView f = new WriteableView(s);
214         try {
215             if (sourceBean.getLock().tryLock(lockTimeOutInSeconds, TimeUnit.SECONDS)) {
216                 return f;
217             }
218         } catch(InterruptedException e) {
219             // ignore, will throw a TransactionFailure exception
220         }
221         throw new TransactionFailure("Config bean already locked " + sourceBean, null);
222     }

225      * Returns a writeable view of a configuration object
226      * @param source the configured interface implementation
227      * @return the new interface implementation providing write access
228      * @throws TransactionFailure if the object cannot be enrolled (probably already enrolled in
229      * another transaction).
230      */
231     public <T extends ConfigBeanProxy> T getWriteableView(final T source)
232         throws TransactionFailure {
233         T configBeanProxy = revealProxy(source);
234         ConfigView sourceBean = (ConfigView) Proxy.getInvocationHandler(configBeanProxy);
235         WriteableView writeableView = getWriteableView(configBeanProxy, (ConfigBean) sourceBean.getMasterView());
236         return (T) writeableView.getProxy(sourceBean.getProxyType());

265     /**
266      * sort events and dispatch the changes. There will be only one notification of event
267      * per event type, per object, meaning that if an object has had 3 attributes changes, the
268      * Changed interface implementation will get notified only once.
269      *
270      * @param events of events that resulted of a successful configuration transaction
271      * @param target the intended receiver of the changes notification
272      * @param logger to log any issues.
273      */
274     public static UnprocessedChangeEvents sortAndDispatch(PropertyChangeEvent[] events, Changed target, Logger logger) {
275         if ( logger == null ) throw new IllegalArgumentException();
276         
277         List<UnprocessedChangeEvent> unprocessed = new ArrayList<UnprocessedChangeEvent>();
278         List<Dom> added = new ArrayList<Dom>();
279         List<Dom> changed = new ArrayList<Dom>();
280 
281         for (PropertyChangeEvent event : events) {
282 
283             if (event.getOldValue()==null && event.getNewValue() instanceof ConfigBeanProxy) {
284                 // something was added
285                 try {
286                     final ConfigBeanProxy proxy =  ConfigBeanProxy.class.cast(event.getNewValue());
287                     added.add(Dom.unwrap(proxy));
288                     final NotProcessed nc = target.changed(Changed.TYPE.ADD, proxyType(proxy), proxy);
289                     if ( nc != null ) {
290                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
291                     }
292                 } catch (Exception e) {
293                     logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
294                 }
295             }
296         }
297 
298         for (PropertyChangeEvent event : events) {
299 
300             try {
301                 Dom eventSource = Dom.unwrap((ConfigBeanProxy) event.getSource());
302                 if (added.contains(eventSource)) {
303                     // we don't really send the changed events for new comers.
304                     continue;
305                 }
306                 ConfigBeanProxy proxy =  null;
307                 if (event.getNewValue()==null) {
308                     try {
309                         // getOldValue() can be null, we will notify a CHANGE event
310                         proxy = ConfigBeanProxy.class.cast(event.getOldValue());
311                     } catch (ClassCastException e) {
312                         // this is ok, the old value was probably a string or something like this...
313                         // we will notify the event.getSource() that it changed.
314                     }
315                     // new value is null, but not old value, we removed something
316                     if (proxy!=null) {
317                         final NotProcessed nc = target.changed(Changed.TYPE.REMOVE, proxyType(proxy), proxy );
318                         if ( nc != null ) {
319                             unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
320                         }
321                         continue;
322                     }
323                 }
324                 // we fall back in this case, the newValue was nullm the old value was also null  or was not a ConfigBean,
325                 // we revert to just notify of the change.
326                 // when the new value is not null, we also send the CHANGE  on the source all the time, unless this was
327                 // and added config bean.
328                 if (!changed.contains(eventSource)) {
329                     proxy =  ConfigBeanProxy.class.cast(event.getSource());
330                     changed.add(eventSource);
331                     final NotProcessed nc = target.changed(Changed.TYPE.CHANGE, proxyType(proxy), proxy);
332                     if ( nc != null ) {
333                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
334                     }
335                 }
336             } catch (Exception e) {
337                 logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
338             }
339         }
340         
341         return new UnprocessedChangeEvents( unprocessed );
342     }

349     public void apply(Map<ConfigBean, Map<String, String>> mapOfChanges) throws TransactionFailure {
350 
351 
352         Transaction t = new Transaction();
353         try {
354             for (Map.Entry<ConfigBean, Map<String, String>> configBeanChange : mapOfChanges.entrySet()) {
355 
356                 ConfigBean source = configBeanChange.getKey();
357                 ConfigBeanProxy readableView = source.getProxy(source.getProxyType());
358                 WriteableView writeable = getWriteableView(readableView, source);
359                 if (!writeable.join(t)) {
360                     t.rollback();
361                     throw new TransactionFailure("Cannot enlist " + source.getProxyType() + " in transaction", null);
362                 }
363                 for (Map.Entry<String, String> change : configBeanChange.getValue().entrySet()) {
364                     String xmlName = change.getKey();
365                     ConfigModel.Property prop = writeable.getProperty(xmlName);
366                     if (prop == null) {
367                         throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
368                     }
369                     if (prop.isCollection()) {
370                         try {
371                             List<String> values = (List<String>) writeable.getter(prop,
372                                     ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
373                             values.add(change.getValue());
374                         } catch (NoSuchMethodException e) {
375                             throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
376                         }
377                     } else {
378                         writeable.setter(prop, change.getValue(), String.class);
379                     }
380                 }
381             }
382         } catch(TransactionFailure e) {
383             t.rollback();
384             throw e;
385         } catch (Exception e) {
386             t.rollback();
387             throw new TransactionFailure(e.getMessage(), e);
388         }
389         try {
390             t.commit();
391         } catch (RetryableException e) {
392             System.out.println("Retryable...");
393             // TODO : do something meaninful here
394             t.rollback();
395             throw new TransactionFailure(e.getMessage(), e);
396         } catch (TransactionFailure e) {
397             System.out.println("failure, not retryable...");
398             t.rollback();
399             throw e;
400         }
401     }

403     /**
404      * Returns the list of sub-elements supported by a ConfigBean
405      * @return array of classes reprensenting the sub elements of a particular
406      * @throws ClassNotFoundException for severe errors with the model associated
407      * with the passed config bean.
408      */
409     public static Class<?>[] getSubElementsTypes(ConfigBean bean)
410         throws ClassNotFoundException {
411 
412         List<Class<?>> subTypes = new ArrayList<Class<?>>();
413         for (ConfigModel.Property element : bean.model.elements.values()) {
414             if (!element.isLeaf()) {
415                 ConfigModel elementModel =  ((ConfigModel.Node) element).model;
416                 Class<?> subType = elementModel.classLoaderHolder.loadClass(elementModel.targetTypeName);
417                 subTypes.add(subType);
418             } else {
419                 if (element.isCollection()) {
420                     subTypes.add(List.class);
421                 }
422             }
423         }
424         return subTypes.toArray(new Class[subTypes.size()]);
425     }

521     ConfigBean _createAndSet(
522                     final ConfigBean parent,
523                     final Class<? extends ConfigBeanProxy> childType,
524                     final List<AttributeChanges> attributes,
525                     final TransactionCallBack<WriteableView> runnable)
526             throws TransactionFailure {
527 
528         ConfigBeanProxy readableView = parent.getProxy(parent.getProxyType());
529         ConfigBeanProxy readableChild = (ConfigBeanProxy)
530                 apply(new SingleConfigCode<ConfigBeanProxy>() {
531             /**
532              * Runs the following command passing the configration object. The code will be run
533              * within a transaction, returning true will commit the transaction, false will abort
534              * it.
535              *
536              * @param param is the configuration object protected by the transaction
537              * @return any object that should be returned from within the transaction code
538              * @throws java.beans.PropertyVetoException
539              *          if the changes cannot be applied
540              *          to the configuration
541              */
542             public Object run(ConfigBeanProxy param) throws PropertyVetoException, TransactionFailure {
543 
544                 // create the child
545                 ConfigBeanProxy child = param.createChild(childType);
546                 Dom dom = Dom.unwrap(child);
547 
548                 // add the child to the parent.
549                 WriteableView writeableParent = (WriteableView) Proxy.getInvocationHandler(param);
550                 Class parentProxyType = parent.getProxyType();
551 
552                 Class<?> targetClass = null;
553                 // first we need to find the element associated with this type
554                 ConfigModel.Property element = null;
555                 for (ConfigModel.Property e : parent.model.elements.values()) {
556                     if (e.isLeaf()) {
557                         continue;
558                     }
559                     ConfigModel elementModel =  ((ConfigModel.Node) e).model;
560 
561                     if (Logger.getAnonymousLogger().isLoggable(Level.FINE)) {
562                         Logger.getAnonymousLogger().fine( "elementModel.targetTypeName = " + elementModel.targetTypeName +
563                             ", collection: " + e.isCollection() + ", childType.getName() = " + childType.getName() );
564                     }
565                             
566                     if (elementModel.targetTypeName.equals(childType.getName())) {
567                         element = e;
568                         break;
569                     }
570                     else if ( e.isCollection() ) {
571                         try {
572                             final Class<?> tempClass = elementModel.classLoaderHolder.loadClass(elementModel.targetTypeName);
573                             if ( tempClass.isAssignableFrom( childType ) ) {
574                                 element = e;
575                                 targetClass = tempClass;
576                                 break;
577                             }
578                         } catch (Exception ex ) { 
579                             throw new TransactionFailure("EXCEPTION getting class for " + elementModel.targetTypeName, ex);
580                         }
581                     }
582                 }
583                 
584                 // now depending whether this is a collection or a single leaf,
585                 // we need to process this setting differently
586                 if (element != null) {
587                     if (element.isCollection()) {
588                         // this is kind of nasty, I have to find the method that returns the collection
589                         // object because trying to do a element.get without having the parametized List
590                         // type will not work.
591                         for (Method m : parentProxyType.getMethods()) {
592                             final Class returnType = m.getReturnType();
593                             if (Collection.class.isAssignableFrom(returnType) && (m.getParameterTypes().length == 0)) {
594                                 // this could be it...
595                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
596                                     throw new IllegalArgumentException("List needs to be parameterized");
597                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
598                                 if (itemType.isAssignableFrom(childType)) {
599                                     List list = null;
600                                     try {
601                                         list = (List) m.invoke(param, null);
602                                     } catch (IllegalAccessException e) {
603                                         throw new TransactionFailure("Exception while adding to the parent", e);
604                                     } catch (InvocationTargetException e) {
605                                         throw new TransactionFailure("Exception while adding to the parent", e);
606                                     }
607                                     if (list != null) {
608                                         list.add(child);
609                                         break;
610                                     }
611                                 }
612                             }
613                         }
614                     } else {
615                         // much simpler, I can use the setter directly.
616                         writeableParent.setter(element, dom.<ConfigBeanProxy>createProxy(), childType);
617                     }
618                 } else {
619                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
620                 }
621 
622                 WriteableView writeableChild = (WriteableView) Proxy.getInvocationHandler(child);
623                 applyProperties(writeableChild, attributes);
624 
625                 dom.addDefaultChildren();
626                 dom.register();
627 
628                 if (runnable!=null) {
629                     runnable.performOn(writeableChild);
630                 }
631                 return child;
632             }
633         }, readableView);
634         return (ConfigBean) Dom.unwrap(readableChild);
635     }

637     private void applyProperties(WriteableView target, List<? extends AttributeChanges> changes)
638             throws TransactionFailure {
639 
640         if (changes != null) {
641             for (AttributeChanges change : changes) {
642 
643                 ConfigModel.Property prop = target.getProperty(change.name);
644                 if (prop == null) {
645                     throw new TransactionFailure("Unknown property name " + change.name + " on " + target.getProxyType());
646                 }
647                 if (prop.isCollection()) {
648                     // we need access to the List
649                     try {
650                         List list = (List) target.getter(prop, ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
651                         for (String value : change.values()) {
652                             list.add(value);
653                         }
654                     } catch (NoSuchMethodException e) {
655                         throw new TransactionFailure(e.getMessage(), e);
656                     }
657                 } else {
658                     target.setter(prop, change.values()[0], String.class);
659                 }
660             }
661         }
662 
663     }

737     /**
738      * Unprotected child deletion, caller must start a transaction before calling this
739      * method.
740      *
741      * @param parent the parent element
742      * @param writeableParent the writeable view of the parent element
743      * @param child the child to delete
744      * @throws TransactionFailure if something goes wrong.
745      */
746     public static void _deleteChild(
747                 final ConfigBean parent,
748                 final WriteableView writeableParent,
749                 final ConfigBean child)
750         throws TransactionFailure {
751 
752 
753         final Class<? extends ConfigBeanProxy> childType = child.getProxyType();
754 
755 
756                 // get the child
757                 ConfigBeanProxy childProxy = child.getProxy(childType);
758 
759                 // get the parent type
760                 Class parentProxyType = parent.getProxyType();
761 
762                 // first we need to find the element associated with this type
763                 ConfigModel.Property element = null;
764                 for (ConfigModel.Property e : parent.model.elements.values()) {
765                     if (!(e instanceof ConfigModel.Node)) {
766                         continue;
767                     }
768 
769                     ConfigModel elementModel = ((ConfigModel.Node) e).model;
770                     try {
771                         final Class<?> targetClass = parent.model.classLoaderHolder.loadClass(elementModel.targetTypeName);
772                         if (targetClass.isAssignableFrom(childType)) {
773                             element = e;
774                             break;
775                         }
776                     } catch(Exception ex) {
777                         // ok.
778                     }
779                 }
780                 // now depending whether this is a collection or a single leaf,
781                 // we need to process this setting differently
782                 if (element != null) {
783                     if (element.isCollection()) {
784                         // this is kind of nasty, I have to find the method that returns the collection
785                         // object because trying to do a element.get without having the parametized List
786                         // type will not work.
787                         for (Method m : parentProxyType.getMethods()) {
788                             final Class returnType = m.getReturnType();
789                             if (Collection.class.isAssignableFrom(returnType)  && (m.getParameterTypes().length == 0)) {
790                                 // this could be it...
791                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
792                                     throw new IllegalArgumentException("List needs to be parameterized");
793                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
794                                 if (itemType.isAssignableFrom(childType)) {
795                                     List list = null;
796                                     try {
797                                         list = (List) m.invoke(writeableParent.getProxy(writeableParent.<ConfigBeanProxy>getProxyType()), null);
798                                     } catch (IllegalAccessException e) {
799                                         throw new TransactionFailure("Exception while adding to the parent", e);
800                                     } catch (InvocationTargetException e) {
801                                         throw new TransactionFailure("Exception while adding to the parent", e);
802                                     }
803                                     if (list != null) {
804                                         list.remove(childProxy);
805                                         break;
806                                     }
807                                 }
808                             }
809                         }
810                     } else {
811                         // much simpler, I can use the setter directly.
812                         writeableParent.setter(element, null, childType);
813                     }
814                 } else {
815                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
816                 }
817     }

870     public static Class<? extends ConfigBeanProxy> getElementTypeByName(ConfigBeanProxy parent, String elementName)
871         throws ClassNotFoundException {
872 
873         final Dom parentDom = Dom.unwrap(parent);
874         DomDocument document = parentDom.document;
875 
876 
877         ConfigModel.Property a = parentDom.model.elements.get(elementName);
878         if (a!=null) {
879             if (a.isLeaf()) {
880                 // dochez : I am not too sure, but that should be a String @Element
881                 return null;
882             } else {
883                 ConfigModel childModel = ((ConfigModel.Node) a).model;
884                 return (Class<? extends ConfigBeanProxy>) childModel.classLoaderHolder.loadClass(childModel.targetTypeName);
885             }
886         }
887         // global lookup
888         ConfigModel model = document.getModelByElementName(elementName);
889         if (model!=null) {
890             return (Class<? extends ConfigBeanProxy>) model.classLoaderHolder.loadClass(model.targetTypeName);
891         }
892 
893         return null;
894     }

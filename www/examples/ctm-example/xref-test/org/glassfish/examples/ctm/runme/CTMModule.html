44  import org.glassfish.examples.ctm.Environment;

54  /**
55   * TODO:  This should be done via auto-depends (via Service and contract
56   * and all that).  However, since those don't work yet with the new
57   * API, we must code this up by hand.
58   * 
59   * @author jwells
60   *
61   */
62  public class CTMModule {
63  
64      /**
65       * Configures the HK2 instance
66       * 
67       * @param configurator
68       */
69      public void configure(DynamicConfiguration configurator) {
70          // Bind our custom scope
71          configurator.bind(
72                  BuilderHelper.link(TenantScopedContext.class).
73                                to(Context.class).
74                                in(Singleton.class.getName()).
75                                build());
76          
77          // Bind our factory
78          configurator.bind(
79                  BuilderHelper.link(EnvironmentFactory.class).
80                                to(Environment.class).
81                                in(TenantScoped.class.getName()).
82                                buildFactory(Singleton.class.getName()));
83          
84          // We implemented the TenantManager as a service (nice!)
85          configurator.bind(
86                  BuilderHelper.link(TenantManager.class).
87                                in(Singleton.class.getName()).
88                                build());
89          
90          // And of course, ServiceProviderEngine is a service
91          configurator.bind(
92                  BuilderHelper.link(ServiceProviderEngine.class).
93                                in(Singleton.class.getName()).
94                                build());
95  
96      }
97  
98  }

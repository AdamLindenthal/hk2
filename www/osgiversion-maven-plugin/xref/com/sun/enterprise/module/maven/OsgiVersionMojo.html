38  package com.sun.enterprise.module.maven;
39  
40  import org.apache.maven.plugin.AbstractMojo;
41  import org.apache.maven.plugin.MojoExecutionException;
42  import org.apache.maven.plugin.MojoFailureException;
43  import org.apache.maven.project.MavenProject;
44  import org.apache.maven.shared.osgi.Maven2OsgiConverter;
45  import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
46  
47  /**
48   * Converts the project version into the OSGi format and
49   * set that to "project.osgi.version" property.
50   * It can be configured to drop certain portions from the
51   * version. See {@link #dropVersionComponent}.
52   *
53   * @author Kohsuke Kawaguchi
54   * @author Sanjeeb.Sahoo@Sun.COM
55   * @goal compute-osgi-version
56   * @phase validate
57   * @requiresProject
58   */
59  public class OsgiVersionMojo extends AbstractMojo {
60      /**
61       * The maven project.
62       *
63       * @parameter expression="${project}"
64       * @required
65       * @readonly
66       */
67      protected MavenProject project;
68  
69      /**
70       * Flag used to determine what components of the version will be used
71       * in OSGi version.
72       * An OSGi version has four parts as shown below:
73       * major.minor.micro.qualifer.
74       * It is not always desirable to use all four parts while
75       * exporting packages. In fact, maven version and OSGi version
76       * behave just opposite during version comparison as shown below:
77       * a maven version 1.2.3-SNAPSHOT is mapped to OSGi version 1.2.3.SNAPSHOT.
78       * In maven, 1.2.3 > 1.2.3-SNAPSHOT, but in OSGi, 1.2.3 < 1.2.3.SNAPSHOT.
79       * So, it is highly desirable to drop qualifier while computing the version.
80       * Instead of hardcoding the policy, we let user tell us what portions will
81       * be used in the OSGi version. If they ask us to drop minor, then only
82       * major will be used. Similarly, if they ask us to drop qualifier, then
83       * major, minor and micro portions will be used.
84       * @parameter
85       */
86      protected String dropVersionComponent;
87  
88      /**
89       * @parameter default-value="project.osgi.version"
90       */
91      protected String versionPropertyName;
92  
93      private enum VERSION_COMPONENT {major, minor, micro, qualifier};
94  
95      /**
96       * @component
97       */
98      protected Maven2OsgiConverter converter;
99  
100     public void execute() throws MojoExecutionException, MojoFailureException {
101         DefaultArtifactVersion projectVersion =
102                 new DefaultArtifactVersion(project.getVersion());
103         VERSION_COMPONENT compToDrop = dropVersionComponent == null ?
104                 null : VERSION_COMPONENT.valueOf(dropVersionComponent);
105 
106         DefaultArtifactVersion newVersion = projectVersion;
107         if (compToDrop != null) {
108             switch (compToDrop) {
109                 case major: {
110                     newVersion = new DefaultArtifactVersion("0.0.0");
111                     break;
112                 }
113                 case minor: {
114                     final int major = projectVersion.getMajorVersion();
115                     newVersion = new DefaultArtifactVersion(major +"");
116                     break;
117                 }
118                 case micro: {
119                     final int major = projectVersion.getMajorVersion();
120                     final int minor = projectVersion.getMinorVersion();
121                     newVersion = new DefaultArtifactVersion(major + "." + minor);
122                     break;
123                 }
124                 case qualifier: {
125                     final int major = projectVersion.getMajorVersion();
126                     final int minor = projectVersion.getMinorVersion();
127                     final int micro = projectVersion.getIncrementalVersion();
128                     newVersion = new DefaultArtifactVersion(major + "." + minor + "." + micro);
129                     break;
130                 }
131             }
132         }
133         String v = converter.getVersion(newVersion.toString());
134 
135         getLog().debug("OSGi Version for "+project.getVersion()+" is "+v);
136         getLog().debug("It is set in project property called "+ versionPropertyName);
137         project.getProperties().put(versionPropertyName,v);
138     }

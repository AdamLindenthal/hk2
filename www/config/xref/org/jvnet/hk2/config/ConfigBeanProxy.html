37  package org.jvnet.hk2.config;
38  
39  import java.lang.reflect.Proxy;
40  
41  /***
42   * Marker interface that signifies that the interface
43   * is meant to be used as a strongly-typed proxy to
44   * {@link Dom}. 
45   *
46   * <p>
47   * To obtain the Dom object, use {@link Dom#unwrap(ConfigBeanProxy)}.
48   * This design allows the interfaces to be implemented by other code
49   * outside DOM more easily.
50   *
51   * @author Kohsuke Kawaguchi
52   * @see Dom#unwrap(ConfigBeanProxy)
53   * @see DuckTyped
54   * @see Element
55   * @see Attribute
56   */
57  public interface ConfigBeanProxy {
58  
59      /***
60       * Returns the parent element of this configuration element.
61       *
62       * It is possible to return a not null parent while the parent knows nothing of this
63       * child element. This could happen when the child element was removed
64       * from the configuration tree, yet it's parent would not have been reset.
65       *
66       * @return the parent configuration node.
67       */
68      @DuckTyped
69      public ConfigBeanProxy getParent();
70  
71      /***
72       * Returns the typed parent element of this configuration element.
73       *
74       * It is possible to return a not null parent while the parent knows nothing of this
75       * child element. This could happen when the child element was removed
76       * from the configuration tree, yet it's parent would not have been reset.
77       *
78       * @param type parent's type
79       * @return the parent configuration node.
80       */
81      @DuckTyped
82      public <T extends ConfigBeanProxy> T getParent(Class<T> type);
83  
84      /***
85       * Creates a child element of this configuration element
86       *
87       * @param type the child element type
88       * @return the newly created child instance
89       * @throws TransactionFailure when called outside the boundaries of a transaction 
90       */
91      @DuckTyped
92      public <T extends ConfigBeanProxy> T createChild(Class<T> type) throws TransactionFailure;
93  
94  
95      /***
96       * Performs a deep copy of this configuration element and returns it.
97       *
98       * @return a deep copy of itself.
99       */
100     @DuckTyped
101     public ConfigBeanProxy deepCopy();
102 
103     public class Duck {
104 
105         public static ConfigBeanProxy getParent(ConfigBeanProxy self) {
106             Dom dom = Dom.unwrap(self);
107             return dom.parent().createProxy();
108         }
109 
110         public static <T extends ConfigBeanProxy> T getParent(ConfigBeanProxy self, Class<T> c) {
111              Dom dom = Dom.unwrap(self);
112             return dom.parent().createProxy(c);
113         }
114 
115         public static <T extends ConfigBeanProxy> T createChild(ConfigBeanProxy self, Class<T> c)
116             throws TransactionFailure {
117             
118              try {
119                  WriteableView bean = WriteableView.class.cast(Proxy.getInvocationHandler(Proxy.class.cast(self)));
120                  return bean.allocateProxy(c);
121              } catch (ClassCastException e) {
122                  throw new TransactionFailure("Must use a locked parent config object for instantiating new config object", e);
123              }
124 
125         }
126 
127         public static ConfigBeanProxy deepCopy(ConfigBeanProxy self) {
128             ConfigBean configBean = (ConfigBean) Dom.unwrap(self);
129             ConfigBean copy = new ConfigBean(configBean);
130             return copy.createProxy();
131         }
132 
133     }

54       * Loads the configuration file.

56      public void postConstruct() {
57          try {
58              final XMLStreamReader xsr = xif.createXMLStreamReader(getConfigFile());
59  
60              // first skip to the start element
61              while(xsr.getEventType()!= XMLStreamConstants.START_ELEMENT)
62                  xsr.nextTag();
63  
64              Context context = new Context() {
65                  public Reader createReader() {throw new UnsupportedOperationException();}
66                  public Writer createWriter() {throw new UnsupportedOperationException();}
67              };
68              context.put(ComponentManager.class.getName(),cm);
69              
70              load(new XoXMLStreamReaderImpl(xsr), context,null);
71          } catch (XMLStreamException e) {
72              e.printStackTrace();
73          } catch (IOException e) {
74              e.printStackTrace();
75          }
76      }

78      private Object load(XoXMLStreamReader xsr, Context context, Map<String, Object> properties) throws XMLStreamException {
79          assert xsr.getEventType()==XMLStreamConstants.START_ELEMENT;
80  
81          String name = xsr.getLocalName();
82          try {
83              // figure out the class to be loaded from this element
84              Class component = cm.getComponentClass(new ResourceLocator(name, Configured.class));
85              if(component==null)
86                  throw new XMLStreamException("Unrecognized element name: "+name, xsr.getLocation());
87  
88              Class<?> readerClass = component.getClassLoader().loadClass(component.getName() + "Reader");
89              com.envoisolutions.sxc.Reader reader = (com.envoisolutions.sxc.Reader) readerClass.getConstructor(Context.class).newInstance(context);
90              Object result = reader.read(xsr, properties);
91              assert component.isInstance(result);
92  
93              // push back to the scope
94              // TODO: we'd like to do this selectively
95              instance.store.add(new ResourceLocator(component),result);
96  
97              return result;
98          } catch (ComponentException e) {
99              throw new XMLStreamException2("Failed to handle <"+name+">",xsr.getLocation());
100         } catch (NoSuchMethodException e) {
101             throw new XMLStreamException2("Unable to load the reader class",xsr.getLocation(), e);
102         } catch (IllegalAccessException e) {
103             throw new XMLStreamException2("Unable to access the reader class",xsr.getLocation(), e);
104         } catch (InvocationTargetException e) {
105             throw new XMLStreamException2("Unable to create the reader class",xsr.getLocation(), e);
106         } catch (InstantiationException e) {
107             throw new XMLStreamException2("Unable to load the reader class",xsr.getLocation(), e);
108         } catch (ClassNotFoundException e) {
109             throw new XMLStreamException2("Unable to find the reader class",xsr.getLocation(), e);
110         } catch (Exception e) {
111             throw new XMLStreamException2("Failed to handle <"+name+">",xsr.getLocation(), e);
112         }
113     }

37  package org.jvnet.hk2.config;
38  
39  import java.lang.annotation.Retention;
40  import java.lang.annotation.Target;
41  import static java.lang.annotation.ElementType.FIELD;
42  import static java.lang.annotation.ElementType.METHOD;
43  import static java.lang.annotation.RetentionPolicy.RUNTIME;
44  
45  /***
46   * Indicates that this property or the field value must be injected from
47   * an XML element in a configuration file.
48   *
49   * @author Kohsuke Kawaguchi
50   * @see Attribute
51   */
52  @Retention(RUNTIME)
53  @Target({FIELD,METHOD})
54  public @interface Element {
55      /***
56       * Element name.
57       *
58       * See {@link Attribute#value()} for how the default value is inferred.
59       */
60      String value() default "";
61  
62      /***
63       * Indicates that this property becomes the name of the component.
64       * There can be only one key on a class.
65       */
66      boolean key() default false;
67  
68      /***
69       * Indicates that this element is required.
70       *
71       * <p>
72       * To specify the default value, simply use the field initializer
73       * to set it to a certain value. The field/method values are only
74       * set when the value is present.
75       */
76      boolean required() default false;
77  
78      /***
79       * Indicates that this property is a reference to another
80       * configured inhabitant.
81       *
82       * <p>
83       * On XML, this is represented as a string value that points
84       * to the {@link #key() value of the key property} of the target
85       * inhabitant. See the following example:
86       *
87       * <pre>
88       * &#x40;Configured
89       * class VirtualHost {
90       *   &#x40;Attribute(key=true)
91       *   String name;
92       * }
93       *
94       * &#x40;Configured
95       * class HttpListener {
96       *   &#x40;Attribute(reference=true)
97       *   VirtualHost host;
98       * }
99       * </pre>
100      *
101      * <pre><xmp>
102      * <virtual-host name="foo" />
103      * <http-listener host="foo" />
104      * </xmp></pre>
105      */
106     boolean reference() default false;
107 
108     /***
109      * Indicates that the variable expansion should be performed on this proeprty.
110      *
111      * <p>
112      * The configuration mechanism supports the Ant/Maven like {@link VariableResolver variable expansion}
113      * in the configuration XML out of the box. Normally this happens transparently to objects in modules,
114      * hence this property is set to true by default.
115      *
116      * <p>
117      * However, in a rare circumstance you might want to get values injected before the variables
118      * are expanded, in which case you can set this property to false to indicate so. Note that such
119      * property must be of type {@link String} (or its collection/array.)
120      *
121      * <p>
122      * Also note the inhabitants can always access the XML infoset by talking to {@link Dom} directly.
123      */
124     boolean variableExpansion() default true;
125 }

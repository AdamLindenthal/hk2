39  /***
40   * Used to perform string pre-processing on values found in the configuration file.
41   *
42   * <p>
43   * This hook allows applications to support variable
44   * expansions like Ant in the configuration file.
45   *
46   * @author Kohsuke Kawaguchi
47   */
48  public interface Translator {
49      String translate(String str) throws TranslationException;
50  
51      /***
52       * {@link Translator} that does nothing.
53       */
54      public static final Translator NOOP = new Translator() {
55          public String translate(String str) {
56              return str;
57          }
58      };
59      /***
60       * A translator that does translation from the system properties. Thus, any reference to an existing
61       * System Property like "${name}" will be replaced by its value, i.e. System.getProperty("name"). All
62       * found references are translated. If a System Property is not defined, its reference is returned verbatim.
63       * No escape sequences are handled.
64       */
65      public static final Translator SYS_PROP_TR = new Translator() {
66          public String translate(String s) {
67              StringBuilder sb = new StringBuilder();
68              int length = s.length();
69              int i = 0;
70              while(i < length) {
71                  char c = s.charAt(i);
72                  if (c == '$' && (i+1) < length && s.charAt(i+1) == '{') {
73                  i += 2;
74                  char cc='\0';
75                  StringBuilder prop = new StringBuilder();
76                  while (i < length && (cc=s.charAt(i)) != '}') {
77                    prop.append(cc);
78                    i++;
79                  }
80                  if (cc == '}') {
81                    String value = System.getProperty(prop.toString());
82                    if (value != null)
83                        sb.append(value);
84                    else //return reference to non-existent system-property verbatim
85                        sb.append("${" + prop + "}");
86                    i++;
87                  } else { //we reached the end, no } found
88                    sb.append("${").append(prop);
89                  }
90                } else {
91                  sb.append(c);
92                  i++;
93                }
94            }
95            return sb.toString();
96          }
97      };
98  }

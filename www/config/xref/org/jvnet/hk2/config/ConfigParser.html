53  /***
54   * Parses configuration files, builds {@link Inhabitant}s,
55   * and add them to {@link Habitat}.
56   *
57   * <p>
58   * This class also maintains the model of various elements in the configuration file.
59   *
60   * <p>
61   * This class can be sub-classed to create a {@link ConfigParser} with a custom non-standard behavior.
62   *
63   * @author Kohsuke Kawaguchi
64   */
65  public class ConfigParser {
66      /***
67       * This is where we put parsed inhabitants into.
68       */
69      protected final Habitat habitat;
70  
71  
72      public ConfigParser(Habitat habitat) {
73          this.habitat = habitat;
74      }
75  
76  
77      public DomDocument parse(XMLStreamReader in) throws XMLStreamException {
78          DomDocument document = new DomDocument(habitat);
79          parse(in, document);
80          return document;
81      }
82  
83      public void parse(XMLStreamReader in, DomDocument document) throws XMLStreamException {
84          in.nextTag();
85          document.root = handleElement(in,document, null);
86      }
87  
88      /***
89       * Parses the given source as a config file, and adds resulting
90       * {@link Dom}s into {@link Habitat} as {@link Inhabitant}s.
91       */
92      public DomDocument parse(URL source) {
93          return parse(source, new DomDocument(habitat));
94      }
95                                    
96  
97      public DomDocument parse(URL source, DomDocument document) {
98          try {
99              parse(xif.createXMLStreamReader(new StreamSource(source.toString())), document);
100             return document;
101         } catch (XMLStreamException e) {
102             throw new ComponentException("Failed to parse "+source,e);
103         }
104     }

106     /***
107      * Parses a whole XML tree and builds a {@link Dom} tree.
108      *
109      * <p>
110      * This is the entry point for the root element of a configuration tree.
111      *
112      * @param in
113      *      pre-condition:  'in' is at the start element.
114      *      post-condition: 'in' is at the end element.
115      * @param document
116      *      The document that we are building right now.
117      *      Newly created {@link Dom} will belong to this document.
118      * @param parent
119      *      The parent element
120      * @return
121      *      Null if the XML element didn't yield anything (which can happen if the element is skipped.)
122      *      Otherwise fully parsed valid {@link Dom} object.
123      */
124     protected Dom handleElement(XMLStreamReader in,DomDocument document, Dom parent) throws XMLStreamException {
125         ConfigModel model = document.getModelByElementName(in.getLocalName());
126         if(model==null)
127             throw new XMLStreamException("Unrecognized element "+in.getLocalName(),in.getLocation());
128         return handleElement(in,document,parent,model);
129     }
130 
131     /***
132      * Parses a whole XML tree and builds a {@link Dom} tree, by using the given model
133      * for the top-level element.
134      *
135      * <p>
136      * This is the entry point for recursively parsing inside a configuration tree.
137      * Since not every element is global, you don't always want to infer the model
138      * just from the element name (as is the case with {@link #handleElement(XMLStreamReader, DomDocument, Dom)}.
139      * 
140      * @param in
141      *      pre-condition:  'in' is at the start element.
142      *      post-condition: 'in' is at the end element.
143      * @param document
144      *      The document that we are building right now.
145      *      Newly created {@link Dom} will belong to this document.
146      * @param parent
147      *      The parent element
148      * @return
149      *      Null if the XML element didn't yield anything (which can happen if the element is skipped.)
150      *      Otherwise fully parsed valid {@link Dom} object.
151      */
152     protected Dom handleElement(XMLStreamReader in, DomDocument document, Dom parent, ConfigModel model) throws XMLStreamException {
153         final Dom dom = document.make(habitat, in, parent, model);
154 
155         // read values and fill DOM
156         dom.fillAttributes(in);
157 
158         List<Child> children=null;
159 
160         while(in.nextTag()==START_ELEMENT) {
161             String name = in.getLocalName();
162             ConfigModel.Property a = model.elements.get(name);
163 
164             if(children==null)
165                 children = new ArrayList<Child>();
166 
167             if(a==null) {
168                 // global look up
169                 Dom child = handleElement(in, document, dom);
170                 if(child!=null)
171                     children.add(new Dom.NodeChild(name, child));
172             } else
173             if(a.isLeaf()) {
174                 children.add(new Dom.LeafChild(name,in.getElementText()));
175             } else {
176                 Dom child = handleElement(in, document, dom, ((ConfigModel.Node) a).model);
177                 if(child!=null)
178                     children.add(new Dom.NodeChild(name, child));
179             }
180         }
181 
182         if(children!=null)
183             dom.setChildren(children);
184 
185         habitat.add(dom);
186 
187         // register 'dom' under indices
188         String key = dom.getKey();
189         for (String contract : model.contracts)
190             habitat.addIndex(dom,contract,key);
191         if(key!=null)
192             // if this is named component, register under its own FQCN too
193             // to support look up.
194             habitat.addIndex(dom,model.targetTypeName,key);
195 
196         dom.initializationCompleted();
197         return dom;
198     }

200     // In JDK 1.6, StAX is part of JRE, so we use no argument variant of
201     // newInstance(), where as on JDK 1.5, we use two argument version of
202     // newInstance() so that we can pass the classloader that loads
203     // XMLInputFactory to load the factory, otherwise by default StAX uses
204     // Thread's context class loader to locate the factory. See:
205     // https://glassfish.dev.java.net/issues/show_bug.cgi?id=6428
206     private static final XMLInputFactory xif =
207             XMLInputFactory.class.getClassLoader() == null ?
208                     XMLInputFactory.newInstance() :
209                     XMLInputFactory.newInstance(XMLInputFactory.class.getName(),
210                             XMLInputFactory.class.getClassLoader());

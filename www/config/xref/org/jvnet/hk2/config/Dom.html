86      static final class NodeChild extends Child {
87          final Dom dom;
88  
89          NodeChild(String name, Dom dom) {
90              super(name);
91              this.dom = dom;
92          }
93  
94          protected void writeTo(XMLStreamWriter w) throws XMLStreamException {
95              dom.writeTo(name,w);

100         /***
101          * Raw element text value before {@link Translator} processing.
102          */

105         LeafChild(String name, String value) {
106             super(name);
107             this.value = value;
108         }
109 
110         protected void writeTo(XMLStreamWriter w) throws XMLStreamException {
111             w.writeStartElement(name);
112             w.writeCharacters(value);
113             w.writeEndElement();
114         }

176     /***
177      * Obtains the actual key value from this {@link Dom}.
178      */
179     public String getKey() {
180         String k = model.key;
181         if(k==null) return null;
182 
183         switch(k.charAt(0)) {
184         case '@':
185             return attribute(k.substring(1));
186         case '<':
187             return leafElement(k.substring(1,k.length()-1));
188         default:
189             throw new IllegalStateException("Invalid key value:"+k);
190         }
191     }

201     /*package*/ void fillAttributes(XMLStreamReader in) {
202         for( int i=in.getAttributeCount()-1; i>=0; i-- ) {
203             String n = in.getAttributeLocalName(i);
204             if(model.attributes.containsKey(n)) {
205                 if(attributes==null)
206                     attributes = new HashMap<String, String>();
207                 attributes.put(n,in.getAttributeValue(i));
208             }
209         }
210         if(attributes==null)
211             attributes = Collections.emptyMap();
212     }

279     /***
280      * Obtains the plural attribute value. Values are separate by ',' and surrounding whitespaces are ignored.
281      *
282      * @return
283      *      null if the attribute doesn't exist. This is a distinct state from the empty list,
284      *      which indicates that the attribute was there but no values were found.
285      */
286     public List<String> attributes(String name) {
287         String v = attribute(name);
288         if(v==null)     return null;
289         List<String> r = new ArrayList<String>();
290         StringTokenizer tokens = new StringTokenizer(v,",");
291         while(tokens.hasMoreTokens())
292             r.add(tokens.nextToken().trim());
293         return r;
294     }

339     /***
340      * Inserts a new {@link Dom} node right after the given DOM element.
341      *
342      * @param reference
343      *      If null, the new element will be inserted at the very beginning.
344      * @param name
345      *      The element name of the newly inserted item. "*" to indicate that the element
346      *      name be determined by the model of the new node.
347      */
348     public synchronized void insertAfter(Dom reference, String name, Dom newNode) {
349         // TODO: reparent newNode
350         if(name.equals("*"))    name=newNode.model.tagName;
351         NodeChild newChild = new NodeChild(name, newNode);
352 
353         if (children.size()==0) {
354             children = new ArrayList<Child>();
355         }
356         if(reference==null) {
357             children.add(0, newChild);
358             habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
359             return;
360         }
361 
362         ListIterator<Child> itr = children.listIterator();
363         while(itr.hasNext()) {
364             Child child = itr.next();
365             if (child instanceof NodeChild) {
366                 NodeChild nc = (NodeChild) child;
367                 if(nc.dom==reference) {
368                     itr.add(newChild);
369                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
370                     //habitat.add(newNode);
371                     return;
372                 }
373             }
374         }
375         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
376     }

378     /***
379      * Replaces an existing {@link NodeChild} with another one.
380      *
381      * @see #insertAfter(Dom, String, Dom)
382      */
383     public synchronized void replaceChild(Dom reference, String name, Dom newNode) {
384         ListIterator<Child> itr = children.listIterator();
385         while(itr.hasNext()) {
386             Child child = itr.next();   
387             if (child instanceof NodeChild) {
388                 NodeChild nc = (NodeChild) child;
389                 if(nc.dom==reference) {
390                     habitat.removeIndex(reference.getProxyType().getName(), reference.getKey());
391                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());                    
392                     itr.set(new NodeChild(name,newNode));
393                     return;
394                 }
395             }
396         }
397         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
398     }

401      * Removes an existing {@link NodeChild}
402      *
403      */
404     public synchronized void removeChild(Dom reference) {
405         ListIterator<Child> itr = children.listIterator();
406         while(itr.hasNext()) {
407             Child child = itr.next();
408             if (child instanceof NodeChild) {
409                 NodeChild nc = (NodeChild) child;
410                 if(nc.dom==reference) {
411                     itr.remove();
412                     habitat.removeIndex(reference.getProxyType().getName(), reference.getKey());
413                     reference.release();
414                     return;
415                 }
416             }
417         }
418         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);

472     /***
473      * Given a master list and the new sub list, replace the items in the master list with the matching items
474      * from the new sub list. This process works even if the length of the new sublist is different.
475      *
476      * <p>
477      * For example, givn:
478      *
479      * <pre>
480      * replace A by A':
481      *   M=[A,B,C], S=[A'] => [A',B,C]
482      *   M=[A,B,A,B,C], S=[A',A'] => [A',B,A',B,C]
483      *
484      * when list length is different:
485      *   M=[A,A,B,C], S=[] => [B,C]
486      *   M=[A,B,C], S=[A',A'] => [A',A',B,C]
487      *   M=[B,C], S=[A',A'] => [B,C,A',A']
488      * </pre>
489      */
490     private static void stitchList(List<Child> list, String name, List<? extends Child> newSubList) {
491         // to preserve order, try to put new itesm where old items are found.
492         // if the new list is longer than the current list, we put all the extra
493         // after the last item in the sequence. That is,
494         // given [A,A,B,C] and [A',A',A'], we'll update the list to [A',A',A',B,C]
495         // The 'last' variable remembers the insertion position.
496         int last = list.size();
497 
498         ListIterator<Child> itr = list.listIterator();
499         ListIterator<? extends Child> jtr = newSubList.listIterator();
500         while(itr.hasNext()) {
501             Child child = itr.next();
502             if(child.name.equals(name)) {
503                 if(jtr.hasNext()) {
504                     itr.set(jtr.next());    // replace
505                     last = itr.nextIndex();
506                 } else
507                     itr.remove();   // remove
508             }
509         }
510 
511         // new list is longer than the current one
512         if(jtr.hasNext())
513             list.addAll(last,newSubList.subList(jtr.nextIndex(),newSubList.size()));
514     }

521     public synchronized void setLeafElements(final String name, String... values) {
522         List<Child> newChildren = new ArrayList<Child>(children);
523 
524         LeafChild[] leaves = new LeafChild[values.length];
525         for (int i = 0; i < values.length; i++)
526             leaves[i] = new LeafChild(name,values[i]);

528         stitchList(newChildren,name,Arrays.asList(leaves));
529         children = newChildren;
530

532         getInjector().injectElement(this,name,get());

627     /***
628      * Picks up all node elements that are assignable to the given type,
629      * except those who are matched by other named elements in the model.
630      *
631      * Used to implement {@code FromElement("*")}.
632      */
633     public  List<Dom> domNodeByTypeElements(Class baseType) {
634         List<Dom> r = new ArrayList<Dom>();
635 
636         int len = children.size();
637         for( int i=0; i<len; i++ ) {
638             Child child = children.get(i);
639             if (child instanceof NodeChild) {
640                 NodeChild nc = (NodeChild) child;
641                 if(model.elements.containsKey(nc.name))
642                     continue;   // match with named
643                 if(baseType.isAssignableFrom(nc.dom.type()))
644                     r.add(nc.dom);
645             }
646         }
647         return r;
648     }

650     public <T> List<T> nodeByTypeElements(final Class<T> baseType) {
651         final List<Dom> elements = domNodeByTypeElements(baseType);
652         return new AbstractList<T>() {
653             public T get(int index) {
654                 return baseType.cast(elements.get(index).get());
655             }
656             public int size() {
657                 return elements.size();
658             }
659         };
660     }

707     /***
708      * Recursively decends the DOM tree and finds a DOM that has the given key
709      * and the type name.
710      *
711      * <p>
712      * TODO: the current algorithm does a full tree scan. Expand the model
713      * so that we can detect deadends that are statically known not to contain
714      * the kind we are looking for, and use that to cut the search space.
715      */
716     public Dom resolveReference(String key, String typeName) {
717         String keyedAs = model.keyedAs;
718         if(keyedAs!=null && keyedAs.equals(typeName) && getKey().equals(key))
719             return this; // found it
720 
721         for (Child child : children) {

763     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
764         // serve java.lang.Object methods by ourselves
765         Class<?> clazz = method.getDeclaringClass();
766         if(clazz ==Object.class) {
767             try {

783             throw new IllegalArgumentException("No corresponding property found for method: "+method);

786             // getter
787             return getter(p, method.getGenericReturnType());
788         } else {
789             // setter

844     public static String convertName(String name) {
845         // first, trim off the prefix
846         for (String p : PROPERTY_PREFIX) {
847             if(name.startsWith(p)) {
848                 name = name.substring(p.length());
849                 break;
850             }
851         }
852 
853         // tokenize by finding 'x|X' and 'X|Xx' then insert '-'.
854         StringBuilder buf = new StringBuilder(name.length()+5);

856             if(buf.length()>0)  buf.append('-');
857             buf.append(t.toLowerCase());
858         }
859         return buf.toString();

92      static final class NodeChild extends Child {
93          final Dom dom;
94  
95          NodeChild(String name, Dom dom) {
96              super(name);
97              this.dom = dom;
98          }
99  
100         protected void writeTo(XMLStreamWriter w) throws XMLStreamException {
101             dom.writeTo(name,w);
102         }
103 
104         @Override
105         protected Child deepCopy() {
106             return new NodeChild(name, new Dom(dom));
107         }

118             super(name);
119             this.value = value;

207     /***
208      * Obtains the actual key value from this {@link Dom}.
209      */
210     public String getKey() {
211         String k = model.key;
212         if(k==null) return null;
213 
214         switch(k.charAt(0)) {
215         case '@':
216             return attribute(k.substring(1));
217         case '<':
218             return leafElement(k.substring(1,k.length()-1));
219         default:
220             throw new IllegalStateException("Invalid key value:"+k);
221         }
222     }

232     /*package*/ void fillAttributes(XMLStreamReader in) {
233         for( int i=in.getAttributeCount()-1; i>=0; i-- ) {
234             String n = in.getAttributeLocalName(i);
235             if(model.attributes.containsKey(n)) {
236                 if(attributes==null)
237                     attributes = new HashMap<String, String>();
238                 attributes.put(n,in.getAttributeValue(i));
239             }
240         }
241         if(attributes==null)
242             attributes = Collections.emptyMap();
243     }

310     /***
311      * Obtains the plural attribute value. Values are separate by ',' and surrounding whitespaces are ignored.
312      *
313      * @return
314      *      null if the attribute doesn't exist. This is a distinct state from the empty list,
315      *      which indicates that the attribute was there but no values were found.
316      */
317     public List<String> attributes(String name) {
318         String v = attribute(name);
319         if(v==null)     return null;
320         List<String> r = new ArrayList<String>();
321         StringTokenizer tokens = new StringTokenizer(v,",");
322         while(tokens.hasMoreTokens())
323             r.add(tokens.nextToken().trim());
324         return r;
325     }

370     /***
371      * Inserts a new {@link Dom} node right after the given DOM element.
372      *
373      * @param reference
374      *      If null, the new element will be inserted at the very beginning.
375      * @param name
376      *      The element name of the newly inserted item. "*" to indicate that the element
377      *      name be determined by the model of the new node.
378      */
379     public synchronized void insertAfter(Dom reference, String name, Dom newNode) {
380         // TODO: reparent newNode
381         if(name.equals("*"))    name=newNode.model.tagName;
382         NodeChild newChild = new NodeChild(name, newNode);
383 
384         if (children.size()==0) {
385             children = new ArrayList<Child>();
386         }
387         if(reference==null) {
388             children.add(0, newChild);
389             habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
390             return;
391         }
392 
393         ListIterator<Child> itr = children.listIterator();
394         while(itr.hasNext()) {
395             Child child = itr.next();
396             if (child instanceof NodeChild) {
397                 NodeChild nc = (NodeChild) child;
398                 if(nc.dom==reference) {
399                     itr.add(newChild);
400                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
401                     //habitat.add(newNode);
402                     return;
403                 }
404             }
405         }
406         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
407     }

409     /***
410      * Replaces an existing {@link NodeChild} with another one.
411      *
412      * @see #insertAfter(Dom, String, Dom)
413      */
414     public synchronized void replaceChild(Dom reference, String name, Dom newNode) {
415         ListIterator<Child> itr = children.listIterator();
416         while(itr.hasNext()) {
417             Child child = itr.next();   
418             if (child instanceof NodeChild) {
419                 NodeChild nc = (NodeChild) child;
420                 if(nc.dom==reference) {
421                     habitat.removeIndex(reference.getProxyType().getName(), reference.getKey());
422                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());                    
423                     itr.set(new NodeChild(name,newNode));
424                     return;
425                 }
426             }
427         }
428         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
429     }

432      * Removes an existing {@link NodeChild}
433      *
434      */
435     public synchronized void removeChild(Dom reference) {
436         ListIterator<Child> itr = children.listIterator();
437         while(itr.hasNext()) {
438             Child child = itr.next();
439             if (child instanceof NodeChild) {
440                 NodeChild nc = (NodeChild) child;

503     /***
504      * Given a master list and the new sub list, replace the items in the master list with the matching items
505      * from the new sub list. This process works even if the length of the new sublist is different.
506      *
507      * <p>
508      * For example, givn:
509      *
510      * <pre>
511      * replace A by A':
512      *   M=[A,B,C], S=[A'] => [A',B,C]
513      *   M=[A,B,A,B,C], S=[A',A'] => [A',B,A',B,C]
514      *
515      * when list length is different:
516      *   M=[A,A,B,C], S=[] => [B,C]
517      *   M=[A,B,C], S=[A',A'] => [A',A',B,C]
518      *   M=[B,C], S=[A',A'] => [B,C,A',A']
519      * </pre>
520      */
521     private static void stitchList(List<Child> list, String name, List<? extends Child> newSubList) {
522         // to preserve order, try to put new itesm where old items are found.
523         // if the new list is longer than the current list, we put all the extra
524         // after the last item in the sequence. That is,
525         // given [A,A,B,C] and [A',A',A'], we'll update the list to [A',A',A',B,C]
526         // The 'last' variable remembers the insertion position.
527         int last = list.size();
528 
529         ListIterator<Child> itr = list.listIterator();
530         ListIterator<? extends Child> jtr = newSubList.listIterator();
531         while(itr.hasNext()) {
532             Child child = itr.next();
533             if(child.name.equals(name)) {
534                 if(jtr.hasNext()) {
535                     itr.set(jtr.next());    // replace
536                     last = itr.nextIndex();
537                 } else
538                     itr.remove();   // remove
539             }
540         }
541 
542         // new list is longer than the current one
543         if(jtr.hasNext())
544             list.addAll(last,newSubList.subList(jtr.nextIndex(),newSubList.size()));
545     }

552     public synchronized void setLeafElements(final String name, String... values) {
553         List<Child> newChildren = new ArrayList<Child>(children);

556         for (int i = 0; i < values.length; i++)

658     /***
659      * Picks up all node elements that are assignable to the given type,
660      * except those who are matched by other named elements in the model.
661      *
662      * Used to implement {@code FromElement("*")}.
663      */
664     public  List<Dom> domNodeByTypeElements(Class baseType) {
665         List<Dom> r = new ArrayList<Dom>();
666 
667         int len = children.size();
668         for( int i=0; i<len; i++ ) {
669             Child child = children.get(i);
670             if (child instanceof NodeChild) {
671                 NodeChild nc = (NodeChild) child;
672                 if(model.elements.containsKey(nc.name))
673                     continue;   // match with named
674                 if(baseType.isAssignableFrom(nc.dom.type()))
675                     r.add(nc.dom);
676             }
677         }
678         return r;
679     }

682         final List<Dom> elements = domNodeByTypeElements(baseType);
683         return new AbstractList<T>() {
684             public T get(int index) {
685                 return baseType.cast(elements.get(index).get());
686             }
687             public int size() {
688                 return elements.size();
689             }

723     /***
724      * Locates the DOM that serves as the symbol space root.
725      *
726      * @return always non-null.
727      */
728     public Dom getSymbolSpaceRoot(String typeName) {

794     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
795         // serve java.lang.Object methods by ourselves
796         Class<?> clazz = method.getDeclaringClass();
797         if(clazz ==Object.class) {
798             try {

813         if(p==null)
814             throw new IllegalArgumentException("No corresponding property found for method: "+method);
815

818             return getter(p, method.getGenericReturnType());
819         } else {
820             // setter
821             setter(p, args[0]);

875     public static String convertName(String name) {
876         // first, trim off the prefix
877         for (String p : PROPERTY_PREFIX) {
878             if(name.startsWith(p)) {
879                 name = name.substring(p.length());
880                 break;
881             }

884         // tokenize by finding 'x|X' and 'X|Xx' then insert '-'.
885         StringBuilder buf = new StringBuilder(name.length()+5);

887             if(buf.length()>0)  buf.append('-');

889         }
890         return buf.toString();

104         NodeChild(String name, Dom dom) {
105             super(name);
106             this.dom = dom;

110             dom.writeTo(name,w);

181     /* package */ void ensureConstraints(List<Child> children) {
182         Set<String> nullElements = new HashSet<String>(model.getElementNames());
183         for (Child child : children) {
184             nullElements.remove(child.name);
185         }
186 
187         for (String s : nullElements) {
188             ConfigModel.Property p = model.getElement(s);
189             for (String annotation : p.getAnnotations()) {
190                 if (annotation.equals(NotNull.class.getName())) {
191                     if (p instanceof ConfigModel.Node) {
192                         ConfigModel childModel = ((ConfigModel.Node) p).model;
193                         Dom child = document.make(habitat, null, this, childModel);
194                         child.register();
195 
196                         children.add(new Dom.NodeChild(s, child));
197 
198                         // recursive call to ensure the children constraints are also respected
199                         List<Child> grandChildren = new ArrayList<Child>();
200                         child.ensureConstraints(grandChildren);
201                         if (!grandChildren.isEmpty()) {
202                             child.setChildren(grandChildren);
203                         }
204 
205                         child.initializationCompleted();
206                     }
207                 }
208             }
209         }
210 
211     }

293     /***
294      * Obtains the actual key value from this {@link Dom}.
295      */
296     public String getKey() {
297         String k = model.key;
298         if(k==null) return null;
299 
300         switch(k.charAt(0)) {
301         case '@':
302             return attribute(k.substring(1));
303         case '<':
304             return leafElement(k.substring(1,k.length()-1));
305         default:
306             throw new IllegalStateException("Invalid key value:"+k);
307         }
308     }

318     /*package*/ void fillAttributes(XMLStreamReader in) {
319         for( int i=in.getAttributeCount()-1; i>=0; i-- ) {
320             String n = in.getAttributeLocalName(i);
321             if(model.attributes.containsKey(n)) {
322                 if(attributes==null)
323                     attributes = new HashMap<String, String>();
324                 attributes.put(n,in.getAttributeValue(i));
325             }
326         }
327         if(attributes==null)
328             attributes = Collections.emptyMap();
329     }

396     /***
397      * Obtains the plural attribute value. Values are separate by ',' and surrounding whitespaces are ignored.
398      *
399      * @return
400      *      null if the attribute doesn't exist. This is a distinct state from the empty list,
401      *      which indicates that the attribute was there but no values were found.
402      */
403     public List<String> attributes(String name) {
404         String v = attribute(name);
405         if(v==null)     return null;
406         List<String> r = new ArrayList<String>();
407         StringTokenizer tokens = new StringTokenizer(v,",");
408         while(tokens.hasMoreTokens())
409             r.add(tokens.nextToken().trim());
410         return r;
411     }

456     /***
457      * Inserts a new {@link Dom} node right after the given DOM element.
458      *
459      * @param reference
460      *      If null, the new element will be inserted at the very beginning.
461      * @param name
462      *      The element name of the newly inserted item. "*" to indicate that the element
463      *      name be determined by the model of the new node.
464      */
465     public synchronized void insertAfter(Dom reference, String name, Dom newNode) {
466         // TODO: reparent newNode
467         if(name.equals("*"))    name=newNode.model.tagName;
468         NodeChild newChild = new NodeChild(name, newNode);
469 
470         if (children.size()==0) {
471             children = new ArrayList<Child>();
472         }
473         if(reference==null) {
474             children.add(0, newChild);
475             habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
476             return;
477         }
478 
479         ListIterator<Child> itr = children.listIterator();
480         while(itr.hasNext()) {
481             Child child = itr.next();
482             if (child instanceof NodeChild) {
483                 NodeChild nc = (NodeChild) child;
484                 if(nc.dom==reference) {
485                     itr.add(newChild);
486                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
487                     //habitat.add(newNode);
488                     return;
489                 }
490             }
491         }
492         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
493     }

495     /***
496      * Replaces an existing {@link NodeChild} with another one.
497      *
498      * @see #insertAfter(Dom, String, Dom)
499      */
500     public synchronized void replaceChild(Dom reference, String name, Dom newNode) {
501         ListIterator<Child> itr = children.listIterator();
502         while(itr.hasNext()) {
503             Child child = itr.next();   
504             if (child instanceof NodeChild) {
505                 NodeChild nc = (NodeChild) child;
506                 if(nc.dom==reference) {
507                     habitat.removeIndex(reference.getProxyType().getName(), reference.getKey());
508                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());                    
509                     itr.set(new NodeChild(name,newNode));
510                     return;
511                 }
512             }
513         }
514         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
515     }

518      * Removes an existing {@link NodeChild}
519      *
520      */

523         while(itr.hasNext()) {
524             Child child = itr.next();

526                 NodeChild nc = (NodeChild) child;

529                     habitat.removeIndex(reference.getProxyType().getName(), reference.getKey());
530                     reference.release();
531                     return;
532                 }
533             }
534         }
535         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);

589     /***
590      * Given a master list and the new sub list, replace the items in the master list with the matching items
591      * from the new sub list. This process works even if the length of the new sublist is different.
592      *
593      * <p>
594      * For example, givn:
595      *
596      * <pre>
597      * replace A by A':
598      *   M=[A,B,C], S=[A'] => [A',B,C]
599      *   M=[A,B,A,B,C], S=[A',A'] => [A',B,A',B,C]
600      *
601      * when list length is different:
602      *   M=[A,A,B,C], S=[] => [B,C]
603      *   M=[A,B,C], S=[A',A'] => [A',A',B,C]
604      *   M=[B,C], S=[A',A'] => [B,C,A',A']
605      * </pre>
606      */
607     private static void stitchList(List<Child> list, String name, List<? extends Child> newSubList) {
608         // to preserve order, try to put new itesm where old items are found.
609         // if the new list is longer than the current list, we put all the extra
610         // after the last item in the sequence. That is,
611         // given [A,A,B,C] and [A',A',A'], we'll update the list to [A',A',A',B,C]
612         // The 'last' variable remembers the insertion position.
613         int last = list.size();
614 
615         ListIterator<Child> itr = list.listIterator();
616         ListIterator<? extends Child> jtr = newSubList.listIterator();
617         while(itr.hasNext()) {
618             Child child = itr.next();
619             if(child.name.equals(name)) {
620                 if(jtr.hasNext()) {
621                     itr.set(jtr.next());    // replace
622                     last = itr.nextIndex();
623                 } else
624                     itr.remove();   // remove
625             }
626         }
627 
628         // new list is longer than the current one
629         if(jtr.hasNext())
630             list.addAll(last,newSubList.subList(jtr.nextIndex(),newSubList.size()));
631     }

734         for( int i=0; i<len; i++ ) {

963         // first, trim off the prefix

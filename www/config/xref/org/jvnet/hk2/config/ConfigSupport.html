53  /***
54    * <p>
55    * Helper class to execute some code on configuration objects while taking
56    * care of the transaction boiler plate code.
57    * </p>
58    * <p>
59    * Programmers that wish to apply some changes to configuration objects
60    * can use these convenience methods to reduce the complexity of handling
61    * transactions.
62    * </p>
63    * <p>
64    * For instance, say a programmer need to change the HttpListener port from
65    * 8080 to 8989, it just needs to do :
66    * </p>
67    * <pre>
68    *     ... in his code somewhere ...
69    *     HttpListener httpListener = domain.get...
70    *
71    *     // If the programmer tries to modify the httpListener directly
72    *     // it will get an exception
73    *     httpListener.setPort("8989"); // will generate a PropertyVetoException
74    *
75    *     // instead he needs to use a transaction and can use the helper services
76    *     ConfigSupport.apply(new SingleConfigCode<HttpListener>() {
77    *         public Object run(HttpListener okToChange) throws PropertyException {
78    *             okToChange.setPort("8989"); // good...
79    *             httpListener.setPort("7878"); // not good, exceptions still raised...
80    *             return null;
81    *         });
82    *
83    *     // Note that after this code
84    *     System.out.println("Port is " + httpListener.getPort());
85    *     // will display 8989
86    * }
87    * </pre>
88    * @author Jerome Dochez
89    */
90  @Service

97      /***
98       * Execute some logic on one config bean of type T protected by a transaction
99       *
100      * @param code code to execute
101      * @param param config object participating in the transaction
102      * @return list of events that represents the modified config elements.
103      * @throws TransactionFailure when code did not run successfully
104      */
105     public static <T extends ConfigBeanProxy> Object apply(final SingleConfigCode<T> code, T param)
106         throws TransactionFailure {
107         
108         ConfigBeanProxy[] objects = { param };
109         return apply((new ConfigCode() {
110             @SuppressWarnings("unchecked")
111             public Object run(ConfigBeanProxy... objects) throws PropertyVetoException, TransactionFailure {
112                 return code.run((T) objects[0]);
113             }
114         }), objects);
115     }

133     /***
134      * Executes some logic on some config beans protected by a transaction.
135      *
136      * @param code code to execute
137      * @param objects config beans participating to the transaction
138      * @return list of property change events
139      * @throws TransactionFailure when the code did run successfully due to a
140      * transaction exception
141      */    
142     public Object _apply(ConfigCode code, ConfigBeanProxy... objects)
143                 throws TransactionFailure {
144 
145         // the fools think they operate on the "real" object while I am
146         // feeding them with writeable view. Only if the transaction succeed
147         // will I apply the "changes" to the real ones.
148         WriteableView[] views = new WriteableView[objects.length];
149 
150         ConfigBeanProxy[] proxies = new ConfigBeanProxy[objects.length];
151 
152         // create writeable views.
153         for (int i=0;i<objects.length;i++) {
154             proxies[i] = getWriteableView(objects[i]);
155             views[i] = (WriteableView) Proxy.getInvocationHandler(proxies[i]);
156         }
157 
158         // Of course I am not locking the live objects but the writable views.
159         // if the user try to massage the real objects, he will get
160         // a well deserved nasty exception
161         Transaction t = new Transaction();
162         for (WriteableView view : views) {
163             if (!view.join(t)) {
164                 t.rollback();
165                 throw new TransactionFailure("Cannot enlist " + view.getMasterView().getProxyType()
166                     + " in transaction", null);
167             }
168         }
169         
170         try {
171             final Object toReturn = code.run(proxies);
172             try {
173                 t.commit();
174                 if (toReturn instanceof WriteableView) {
175                     return ((WriteableView) toReturn).getMasterView();
176                 } else {
177                     return toReturn;
178                 }
179             } catch (RetryableException e) {
180                 System.out.println("Retryable...");
181                 // TODO : do something meaninful here
182                 t.rollback();
183                 return null;
184             } catch (TransactionFailure e) {
185                 t.rollback();
186                 throw e;
187             }
188 
189         } catch(TransactionFailure e) {
190             t.rollback();
191             throw e;
192         } catch (Exception e) {
193             t.rollback();
194             throw new TransactionFailure(e.getMessage(), e);
195         }
196 
197     }

199     <T extends ConfigBeanProxy> WriteableView getWriteableView(T s, ConfigBean sourceBean)
200         throws TransactionFailure {
201 
202         WriteableView f = new WriteableView(s);
203         if (sourceBean.getLock().tryLock()) {
204             return f;
205         }
206         throw new TransactionFailure("Config bean already locked " + sourceBean, null);
207     }

248     /***
249      * sort events and dispatch the changes. There will be only one notification of event
250      * per event type, per object, meaning that if an object has had 3 attributes changes, the
251      * Changed interface implementation will get notified only once.
252      *
253      * @param events of events that resulted of a successful configuration transaction
254      * @param target the intended receiver of the changes notification
255      * @param logger to log any issues.
256      */
257     public static UnprocessedChangeEvents sortAndDispatch(PropertyChangeEvent[] events, Changed target, Logger logger) {
258         if ( logger == null ) throw new IllegalArgumentException();
259         
260         List<UnprocessedChangeEvent> unprocessed = new ArrayList<UnprocessedChangeEvent>();
261         List<Dom> added = new ArrayList<Dom>();
262         List<Dom> changed = new ArrayList<Dom>();
263 
264         for (PropertyChangeEvent event : events) {
265 
266             if (event.getOldValue()==null && event.getNewValue() instanceof ConfigBeanProxy) {
267                 // something was added
268                 try {
269                     final ConfigBeanProxy proxy =  ConfigBeanProxy.class.cast(event.getNewValue());
270                     added.add(Dom.unwrap(proxy));
271                     final NotProcessed nc = target.changed(Changed.TYPE.ADD, proxyType(proxy), proxy);
272                     if ( nc != null ) {
273                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
274                     }
275                 } catch (Exception e) {
276                     logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
277                 }
278             }
279         }
280 
281         for (PropertyChangeEvent event : events) {
282 
283             try {
284                 Dom eventSource = Dom.unwrap((ConfigBeanProxy) event.getSource());
285                 if (added.contains(eventSource)) {
286                     // we don't really send the changed events for new comers.
287                     continue;
288                 }
289                 ConfigBeanProxy proxy =  null;
290                 if (event.getNewValue()==null) {
291                     try {
292                         // getOldValue() can be null, we will notify a CHANGE event
293                         proxy = ConfigBeanProxy.class.cast(event.getOldValue());
294                     } catch (ClassCastException e) {
295                         // this is ok, the old value was probably a string or something like this...
296                         // we will notify the event.getSource() that it changed.
297                     }
298                     // new value is null, but not old value, we removed something
299                     if (proxy!=null) {
300                         final NotProcessed nc = target.changed(Changed.TYPE.REMOVE, proxyType(proxy), proxy );
301                         if ( nc != null ) {
302                             unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
303                         }
304                         continue;
305                     }
306                 }
307                 // we fall back in this case, the newValue was nullm the old value was also null  or was not a ConfigBean,
308                 // we revert to just notify of the change.
309                 // when the new value is not null, we also send the CHANGE  on the source all the time, unless this was
310                 // and added config bean.
311                 if (!changed.contains(eventSource)) {
312                     proxy =  ConfigBeanProxy.class.cast(event.getSource());
313                     changed.add(eventSource);
314                     final NotProcessed nc = target.changed(Changed.TYPE.CHANGE, proxyType(proxy), proxy);
315                     if ( nc != null ) {
316                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
317                     }
318                 }
319             } catch (Exception e) {
320                 logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
321             }
322         }
323         
324         return new UnprocessedChangeEvents( unprocessed );
325     }

332     public void apply(Map<ConfigBean, Map<String, String>> mapOfChanges) throws TransactionFailure {
333 
334 
335         Transaction t = new Transaction();
336         try {
337             for (Map.Entry<ConfigBean, Map<String, String>> configBeanChange : mapOfChanges.entrySet()) {
338 
339                 ConfigBean source = configBeanChange.getKey();
340                 ConfigBeanProxy readableView = source.getProxy(source.getProxyType());
341                 WriteableView writeable = getWriteableView(readableView, source);
342                 if (!writeable.join(t)) {
343                     t.rollback();
344                     throw new TransactionFailure("Cannot enlist " + source.getProxyType() + " in transaction", null);
345                 }
346                 for (Map.Entry<String, String> change : configBeanChange.getValue().entrySet()) {
347                     String xmlName = change.getKey();
348                     ConfigModel.Property prop = writeable.getProperty(xmlName);
349                     if (prop == null) {
350                         throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
351                     }
352                     if (prop.isCollection()) {
353                         try {
354                             List<String> values = (List<String>) writeable.getter(prop,
355                                     ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
356                             values.add(change.getValue());
357                         } catch (NoSuchMethodException e) {
358                             throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
359                         }
360                     } else {
361                         writeable.setter(prop, change.getValue(), String.class);
362                     }
363                 }
364             }
365         } catch(TransactionFailure e) {
366             t.rollback();
367             throw e;
368         } catch (Exception e) {
369             t.rollback();
370             throw new TransactionFailure(e.getMessage(), e);
371         }
372         try {
373             t.commit();
374         } catch (RetryableException e) {
375             System.out.println("Retryable...");
376             // TODO : do something meaninful here
377             t.rollback();
378             throw new TransactionFailure(e.getMessage(), e);
379         } catch (TransactionFailure e) {
380             System.out.println("failure, not retryable...");
381             t.rollback();
382             throw e;
383         }
384     }

386     /***
387      * Returns the list of sub-elements supported by a ConfigBean
388      * @return array of classes reprensenting the sub elements of a particular
389      * @throws ClassNotFoundException for severe errors with the model associated
390      * with the passed config bean.
391      */
392     public static Class<?>[] getSubElementsTypes(ConfigBean bean)
393         throws ClassNotFoundException {
394 
395         List<Class<?>> subTypes = new ArrayList<Class<?>>();
396         for (ConfigModel.Property element : bean.model.elements.values()) {
397             if (!element.isLeaf()) {
398                 ConfigModel elementModel =  ((ConfigModel.Node) element).model;
399                 Class<?> subType = elementModel.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
400                 subTypes.add(subType);
401             } else {
402                 if (element.isCollection()) {
403                     subTypes.add(List.class);
404                 }
405             }
406         }
407         return subTypes.toArray(new Class[subTypes.size()]);
408     }

504     ConfigBean _createAndSet(
505                     final ConfigBean parent,
506                     final Class<? extends ConfigBeanProxy> childType,
507                     final List<AttributeChanges> attributes,
508                     final TransactionCallBack<WriteableView> runnable)
509             throws TransactionFailure {
510 
511         ConfigBeanProxy readableView = parent.getProxy(parent.getProxyType());
512         ConfigBeanProxy readableChild = (ConfigBeanProxy)
513                 apply(new SingleConfigCode<ConfigBeanProxy>() {
514             /***
515              * Runs the following command passing the configration object. The code will be run
516              * within a transaction, returning true will commit the transaction, false will abort
517              * it.
518              *
519              * @param param is the configuration object protected by the transaction
520              * @return any object that should be returned from within the transaction code
521              * @throws java.beans.PropertyVetoException
522              *          if the changes cannot be applied
523              *          to the configuration
524              */
525             public Object run(ConfigBeanProxy param) throws PropertyVetoException, TransactionFailure {
526 
527                 // create the child
528                 ConfigBeanProxy child = param.createChild(childType);
529 
530                 // add the child to the parent.
531                 WriteableView writeableParent = (WriteableView) Proxy.getInvocationHandler(param);
532                 Class parentProxyType = parent.getProxyType();
533 
534                 Class<?> targetClass = null;
535                 // first we need to find the element associated with this type
536                 ConfigModel.Property element = null;
537                 for (ConfigModel.Property e : parent.model.elements.values()) {
538                     if (e.isLeaf()) {
539                         continue;
540                     }
541                     ConfigModel elementModel =  ((ConfigModel.Node) e).model;
542 
543                     if (Logger.getAnonymousLogger().isLoggable(Level.FINE)) {
544                         Logger.getAnonymousLogger().fine( "elementModel.targetTypeName = " + elementModel.targetTypeName +
545                             ", collection: " + e.isCollection() + ", childType.getName() = " + childType.getName() );
546                     }
547                             
548                     if (elementModel.targetTypeName.equals(childType.getName())) {
549                         element = e;
550                         break;
551                     }
552                     else if ( e.isCollection() ) {
553                         try {
554                             final Class<?> tempClass = elementModel.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
555                             if ( tempClass.isAssignableFrom( childType ) ) {
556                                 element = e;
557                                 targetClass = tempClass;
558                                 break;
559                             }
560                         } catch (Exception ex ) { 
561                             throw new TransactionFailure("EXCEPTION getting class for " + elementModel.targetTypeName, ex);
562                         }
563                     }
564                 }
565                 
566                 // now depending whether this is a collection or a single leaf,
567                 // we need to process this setting differently
568                 if (element != null) {
569                     if (element.isCollection()) {
570                         // this is kind of nasty, I have to find the method that returns the collection
571                         // object because trying to do a element.get without having the parametized List
572                         // type will not work.
573                         for (Method m : parentProxyType.getMethods()) {
574                             final Class returnType = m.getReturnType();
575                             if (Collection.class.isAssignableFrom(returnType)) {
576                                 // this could be it...
577                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
578                                     throw new IllegalArgumentException("List needs to be parameterized");
579                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
580                                 if (itemType.isAssignableFrom(childType)) {
581                                     List list = null;
582                                     try {
583                                         list = (List) m.invoke(param, null);
584                                     } catch (IllegalAccessException e) {
585                                         throw new TransactionFailure("Exception while adding to the parent", e);
586                                     } catch (InvocationTargetException e) {
587                                         throw new TransactionFailure("Exception while adding to the parent", e);
588                                     }
589                                     if (list != null) {
590                                         list.add(child);
591                                         break;
592                                     }
593                                 }
594                             }
595                         }
596                     } else {
597                         // much simpler, I can use the setter directly.
598                         writeableParent.setter(element, child, childType);
599                     }
600                 } else {
601                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
602                 }
603 
604                 WriteableView writeableChild = (WriteableView) Proxy.getInvocationHandler(child);
605                 applyProperties(writeableChild, attributes);
606                 
607                 if (runnable!=null) {
608                     runnable.performOn(writeableChild);
609                 }
610                 return child;
611             }
612         }, readableView);
613         return (ConfigBean) Dom.unwrap(readableChild);
614     }

616     private void applyProperties(WriteableView target, List<? extends AttributeChanges> changes)
617             throws TransactionFailure {
618 
619         if (changes != null) {
620             for (AttributeChanges change : changes) {
621 
622                 ConfigModel.Property prop = target.getProperty(change.name);
623                 if (prop == null) {
624                     throw new TransactionFailure("Unknown property name " + change.name + " on " + target.getProxyType());
625                 }
626                 if (prop.isCollection()) {
627                     // we need access to the List
628                     try {
629                         List list = (List) target.getter(prop, ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
630                         for (String value : change.values()) {
631                             list.add(value);
632                         }
633                     } catch (NoSuchMethodException e) {
634                         throw new TransactionFailure(e.getMessage(), e);
635                     }
636                 } else {
637                     target.setter(prop, change.values()[0], String.class);
638                 }
639             }
640         }
641 
642     }

645     /***
646      * Creates a new child of the passed child and add it to the parent's live
647      * list of elements. The child is also initialized with the attributes passed
648      * where each key represent the xml property name for the attribute and the value
649      * represent the attribute's value.
650      *
651      * This code will be executed within a Transaction and can therefore throw
652      * a TransactionFailure when the creation or settings of attributes failed.
653      *
654      * Example creating a new http-listener element under http-service
655      *      ConfigBean httpService = ... // got it from somwhere.
656      *      Map<String, String> attributes = new HashMap<String, String>();
657      *      attributes.put("id", "jerome-listener");
658      *      attributes.put("enabled", "true");
659      *      ConfigSupport.createAndSet(httpService, HttpListener.class, attributes);
660      *
661      * @param parent parent config bean to which the child will be added.

684     public static void deleteChild(
685                 final ConfigBean parent,
686                 final ConfigBean child)
687         throws TransactionFailure {
688 
689 
690         ConfigBeanProxy readableView = parent.getProxy(parent.getProxyType());
691         final Class<? extends ConfigBeanProxy> childType = child.getProxyType();
692 
693         apply(new SingleConfigCode<ConfigBeanProxy>() {
694 
695             /***
696              * Runs the following command passing the configration object. The code will be run
697              * within a transaction, returning true will commit the transaction, false will abort
698              * it.
699              *
700              * @param param is the configuration object protected by the transaction
701              * @return any object that should be returned from within the transaction code
702              * @throws java.beans.PropertyVetoException
703              *          if the changes cannot be applied
704              *          to the configuration
705              */
706             public Object run(ConfigBeanProxy param) throws PropertyVetoException, TransactionFailure {
707 
708                 // get the child
709                 ConfigBeanProxy childProxy = child.getProxy(childType);
710 
711                 // remove the child from the parent.
712                 WriteableView writeableParent = (WriteableView) Proxy.getInvocationHandler(param);
713                 Class parentProxyType = parent.getProxyType();
714 
715                 // first we need to find the element associated with this type
716                 ConfigModel.Property element = null;
717                 for (ConfigModel.Property e : parent.model.elements.values()) {
718                     if (!(e instanceof ConfigModel.Node)) {
719                         continue;
720                     }
721                         
722                     ConfigModel elementModel = ((ConfigModel.Node) e).model;
723                     try {
724                         final Class<?> targetClass = parent.model.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
725                         if (targetClass.isAssignableFrom(childType)) {
726                             element = e;
727                             break;
728                         }
729                     } catch(Exception ex) {
730                         // ok.
731                     }
732                 }
733                 // now depending whether this is a collection or a single leaf,
734                 // we need to process this setting differently
735                 if (element != null) {
736                     if (element.isCollection()) {
737                         // this is kind of nasty, I have to find the method that returns the collection
738                         // object because trying to do a element.get without having the parametized List
739                         // type will not work.
740                         for (Method m : parentProxyType.getMethods()) {
741                             final Class returnType = m.getReturnType();
742                             if (Collection.class.isAssignableFrom(returnType)) {
743                                 // this could be it...
744                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
745                                     throw new IllegalArgumentException("List needs to be parameterized");
746                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
747                                 if (itemType.isAssignableFrom(childType)) {
748                                     List list = null;
749                                     try {
750                                         list = (List) m.invoke(param, null);
751                                     } catch (IllegalAccessException e) {
752                                         throw new TransactionFailure("Exception while adding to the parent", e);
753                                     } catch (InvocationTargetException e) {
754                                         throw new TransactionFailure("Exception while adding to the parent", e);
755                                     }
756                                     if (list != null) {
757                                         list.remove(childProxy);
758                                         break;
759                                     }
760                                 }
761                             }
762                         }
763                     } else {
764                         // much simpler, I can use the setter directly.
765                         writeableParent.setter(element, null, childType);
766                     }
767                 } else {
768                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
769                 }
770 
771                 return child;
772             }
773         }, readableView);
774     }

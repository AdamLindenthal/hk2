10  /**
11   * Denotes the <code>type</code> of the data a particular config
12   * element (attribute, element) should have. This interface should be 
13   * implemented whenever a need arises to check if
14   * an abstract data type can be represented as a given <code> String </code>.
15   * The implementations of a DataType are mapped by their <code> names </code> elsewhere.
16   * Implementations should provide functional implementation of the #validate method
17   * and must have a public parameterless constructor (except possibly for primitives).
18   * 
19   * @author &#2325;&#2375;&#2342;&#2366;&#2352 (km@dev.java.net)
20   * @see PrimitiveDataType
21   * @see WriteableView
22   * @since hk2 0.3.10
23   */
24  @Contract
25  public interface DataType {
26  
27      /** Checks if given value can be had by the abstract data type represented
28       *  by this implementation.

31       * not represent this data type.
32       */
33      public void validate(String value) throws ValidationException;

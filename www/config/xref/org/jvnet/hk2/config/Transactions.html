84                      }

87              }
88          }

97      public void preDestroy() {
98         for (Holder<ListenerNotifier<TransactionListener,  ?, Void>> listener : listeners) {
99             listener.get().stop();
100        }
101        configListenerNotifier.get().stop();
102        executor.shutdown();
103     }

105     /***
106      * Abstract notification pump, it adds jobs to the queue and process them in the order
107      * jobs were added.
108      *
109      * Jobs are just a wrapper for events of type <U> and a notification mechanism for
110      * completion notification
111      *
112      * @param <T> type of listener interface
113      * @param <U> type of events the listener methods are expecting
114      * @param <V> return type of the listener interface methods.
115      */
116     private abstract class Notifier<T, U, V> {
117         
118         private final BlockingQueue<FutureTask> pendingJobs = new ArrayBlockingQueue<FutureTask>(50);
119         private CountDownLatch latch = new CountDownLatch(1);
120 
121         /***
122          * Creates the task that will notify the listeners of a particular job.
123          * @param job contains the specifics of the notification like the events that need to be notified.
124          * @return a task that can be run and return an optional value.
125          */
126         protected abstract FutureTask<V> prepare(final Job<T, U, V> job);
127 
128         /***
129          * Adds a job to the notification pump. This job will be processed as soon as all other pending
130          * jobs have completed.
131          *
132          * @param job new notification job.
133          * @return a future on the return value.
134          */
135         public Future<V> add(final Job<T, U, V> job) {
136                 
137             // NOTE that this is put() which blocks, *not* add() which will not block and will
138             // throw an IllegalStateException if the queue is full.
139             if (latch.getCount()==0) {
140                 throw new RuntimeException("TransactionListener is inactive, yet jobs are published to it");
141             }
142             try {
143                 pendingJobs.put(prepare(job));
144             } catch (InterruptedException e ) {
145                 throw new RuntimeException(e);
146             }
147             return null;
148         }
149         
150         protected void start() {
151 
152             executor.submit(new Runnable() {
153 
154                 public void run() {
155                     while (latch.getCount()>0) {
156                         try {
157                             final FutureTask job = pendingJobs.take();
158                             job.run();                            
159                         }
160 
161                         catch (InterruptedException e) {
162                             // do anything here?
163                         }
164                     }
165                 }
166                 
167             });
168         }
169 
170         void stop() {
171             latch.countDown();
172             // last event to force the close
173             pendingJobs.add(prepare(new Job<T, U, V>(null, null) {
174                 public V process(T target) {
175                     return null;
176                 }
177             }));
178         }
179     }

181     /***
182      * Default listener notification pump. One thread per listener, jobs processed in
183      * the order it was received.
184      *
185      * @param <T> type of listener interface
186      * @param <U> type of events the listener methods are expecting
187      * @param <V> return type of the listener interface methods.
188      */
189     public class ListenerNotifier<T,U,V> extends Notifier<T,U,V> {
190         
191         final T listener;
192 
193         public ListenerNotifier(T listener) {
194             this.listener = listener;
195         }
196 
197         protected FutureTask<V> prepare(final Job<T, U, V> job) {
198             return new FutureTask<V>(new Callable<V>() {
199                     public V call() throws Exception {
200                         try {
201                             if ( job.mEvents.size() != 0 ) {
202                                 return job.process(listener);
203                            }
204                         } finally {
205                             job.releaseLatch();
206                         }
207                         return null;
208                     }
209                 });
210         }   
211         
212     }

214     /***
215      * Configuration listener notification pump. All Listeners are notified within their own thread, only on thread
216      * takes care of the job pump.
217      * 
218      */
219     public class ConfigListenerNotifier extends Notifier<ConfigListener,PropertyChangeEvent,UnprocessedChangeEvents> {
220 
221         protected FutureTask<UnprocessedChangeEvents>
222             prepare(final Job<ConfigListener, PropertyChangeEvent, UnprocessedChangeEvents> job) {
223 
224         // first, calculate the recipients.
225         final Set<ConfigListener> configListeners = new HashSet<ConfigListener>();
226         for (PropertyChangeEvent event : job.mEvents) {
227             final Dom dom = (Dom) ((ConfigView) Proxy.getInvocationHandler(event.getSource())).getMasterView();
228             configListeners.addAll(dom.getListeners());
229 
230             // we also notify the parent.
231             if (dom.parent()!=null) {
232                 configListeners.addAll(dom.parent().getListeners());
233             }
234         }
235 
236         return new FutureTask<UnprocessedChangeEvents>(new Callable<UnprocessedChangeEvents>() {
237             public UnprocessedChangeEvents call() throws Exception {
238 
239                 try {
240                     // temporary structure to store our future notifications
241                     List<Future<UnprocessedChangeEvents>> futures = new ArrayList<Future<UnprocessedChangeEvents>>();
242 
243                     for (final ConfigListener listener : configListeners) {
244                         // each listener is notified in it's own thread.
245 
246                         futures.add(executor.submit(new Callable<UnprocessedChangeEvents>() {
247                             public UnprocessedChangeEvents call() throws Exception {
248                                 return job.process(listener);
249                             }
250                         }));
251                     }
252                     List<UnprocessedChangeEvents> unprocessed = new ArrayList<UnprocessedChangeEvents>(futures.size());
253                     for (Future<UnprocessedChangeEvents> future : futures) {
254                         try {
255                             UnprocessedChangeEvents result = future.get(200, TimeUnit.SECONDS);
256                             if (result.getUnprocessed()!=null && result.getUnprocessed().size()>0) {
257                                 for (UnprocessedChangeEvent event : result.getUnprocessed()) {
258                                     Logger.getAnonymousLogger().log(Level.WARNING, "Unprocessed event : " + event);
259                                 }
260                                 unprocessed.add(result);
261                             }
262                         } catch (InterruptedException e) {
263                             Logger.getAnonymousLogger().log(Level.SEVERE, "Config Listener notification got interruped", e);
264                         } catch (ExecutionException e) {
265                             Logger.getAnonymousLogger().log(Level.SEVERE, "Config Listener notification got interruped", e);
266                         } catch (TimeoutException e) {
267                             Logger.getAnonymousLogger().log(Level.SEVERE, "Config Listener notification took too long", e);
268                         }
269                     }
270 
271                     // all notification have been successful, I just need to notify the unprocessed events.
272                     // note these events are always synchronous so far.
273                     if (!unprocessed.isEmpty()) {
274                         Job unprocessedJob = new UnprocessedEventsJob(unprocessed, null);
275                         for (Holder<ListenerNotifier<TransactionListener, ?, Void>> listener : Transactions.this.listeners) {
276                             listener.get().add(unprocessedJob);
277                         }
278                     }
279                 } finally {
280                     job.releaseLatch();
281                 }
282 
283                 // in theory I should aggregate my unprocessed events but nobody cares.
284                 return null;
285             }
286         });
287     }

292     /***
293         A job contains an optional CountdownLatch so that a caller can learn when the
294         transaction has "cleared" by blocking until that time.
295      */
296     private abstract static class Job<T,U,V> {
297 
298         private final CountDownLatch mLatch;
299         protected final List<U> mEvents;
300         
301         public Job(List<U> events, final CountDownLatch latch ) {
302             mLatch  = latch;
303             mEvents = events;
304         }
305         
306         public void waitForLatch() throws InterruptedException {
307             if ( mLatch != null ) {
308                 mLatch.await();
309             }
310         }
311         
312         public void releaseLatch() {
313             if ( mLatch != null ) {
314                 mLatch.countDown();
315             }
316         }
317         
318         public abstract V process(T target);
319     }
320     
321     private static class TransactionListenerJob extends Job<TransactionListener, PropertyChangeEvent, Void> {
322 
323         public TransactionListenerJob(List<PropertyChangeEvent> events, CountDownLatch latch) {
324             super(events,  latch);
325         }
326         
327         @Override
328         public Void process(TransactionListener listener) {
329             try {
330                 listener.transactionCommited(mEvents);
331             } catch(Exception e) {
332                 e.printStackTrace();
333             }
334             return null;
335         }
336     }
337 
338     private static class UnprocessedEventsJob extends Job<TransactionListener, UnprocessedChangeEvents, Void> {
339 
340         public UnprocessedEventsJob(List<UnprocessedChangeEvents> events, CountDownLatch latch) {
341             super(events, latch);
342         }
343 
344         @Override
345         public Void process(TransactionListener listener) {
346             try {
347                 listener.unprocessedTransactedEvents(mEvents);
348             } catch(Exception e) {
349                 e.printStackTrace();
350             }
351             return null;
352         }
353     }

355     public class ConfigListenerJob extends Job<ConfigListener, PropertyChangeEvent, UnprocessedChangeEvents> {
356 
357         final PropertyChangeEvent[] eventsArray;
358 
359         public ConfigListenerJob(List<PropertyChangeEvent> events, CountDownLatch latch) {
360             super(events, latch);
361             eventsArray = mEvents.toArray(new PropertyChangeEvent[mEvents.size()]);            
362         }
363 
364         public UnprocessedChangeEvents process(ConfigListener target) {
365             return target.changed(eventsArray);
366         }

369     /***
370      * add a new listener to all transaction events.
371      *
372      * @param listener to be added.
373      */
374     public void addTransactionsListener(final TransactionListener listener) {
375         synchronized(listeners) {
376             listeners.add(new Holder<ListenerNotifier<TransactionListener, ?, Void>>() {
377 
378                 final ListenerNotifier tsListener = new ListenerNotifier<TransactionListener, PropertyChangeEvent, Void>(listener);
379                 final CountDownLatch initialized = new CountDownLatch(1);
380 
381                 public ListenerNotifier<TransactionListener, PropertyChangeEvent, Void> get() {
382                     synchronized(initialized) {
383                         if (initialized.getCount()>0) {
384                             tsListener.start();
385                             initialized.countDown();
386                         }
387                     }
388                     return tsListener;
389                 }
390             });
391         }
392     }

394     /***
395      * Removes an existing listener for transaction events
396      * @param listener the registered listener
397      * @return true if the listener unregistration was successful
398      */
399     public boolean removeTransactionsListener(TransactionListener listener) {
400         synchronized(listeners) {
401             for (Holder<ListenerNotifier<TransactionListener, ?, Void>> holder : listeners) {
402                 ListenerNotifier info = holder.get();
403                 if (info.listener==listener) {
404                     info.stop();
405                     return listeners.remove(holder);
406                 }
407             }
408         }
409         return false;
410     }

412     public List<TransactionListener> currentListeners() {
413         synchronized(listeners) {            
414             List<TransactionListener> l = new ArrayList<TransactionListener>();
415             for (Holder<ListenerNotifier<TransactionListener, ?, Void>> holder : listeners) {
416                 ListenerNotifier<TransactionListener, ?, Void> info = holder.get();
417                 l.add(info.listener);
418             }
419             return l;

433     /***
434      * Notification of a new transation completion
435      *
436      * @param events accumulated list of changes
437      * @param waitTillCleared  synchronous semantics; wait until all change events are sent
438      */
439     @SuppressWarnings("cast")
440     void addTransaction(
441         final List<PropertyChangeEvent> events,
442         final boolean waitTillCleared ) {
443         
444         final List<ListenerNotifier<TransactionListener, ?, Void>> listInfos = new ArrayList<ListenerNotifier<TransactionListener, ?, Void>>();
445         for (Holder<ListenerNotifier<TransactionListener, ?, Void>> holder : listeners) {
446             ListenerNotifier<TransactionListener, ?, Void> info = holder.get();
447             listInfos.add(info);
448         }
449         
450         // create a CountDownLatch to implement waiting for events to actually be sent
451         final Job<TransactionListener, ?, Void> job = new TransactionListenerJob( events,
452                                 waitTillCleared ? new CountDownLatch(listInfos.size()) : null);
453         
454         final ConfigListenerJob configJob = new ConfigListenerJob(events,
455                 waitTillCleared? new CountDownLatch(1):null);
456         
457         // NOTE that this is put() which blocks, *not* add() which will not block and will
458         // throw an IllegalStateException if the queue is full.
459         try {
460             for (ListenerNotifier listener : listInfos) {
461                 listener.add(job);
462             }
463 
464             configListenerNotifier.get().add(configJob);
465 
466             job.waitForLatch();
467             configJob.waitForLatch();
468         } catch (InterruptedException e ) {
469             throw new RuntimeException(e);
470         }
471     }

473     public void waitForDrain() {
474         // insert a dummy Job and block until is has been processed.  This guarantees
475         // that all prior jobs have finished
476         addTransaction( new ArrayList<PropertyChangeEvent>(), true );
477         // at this point all prior transactions are guaranteed to have cleared
478     }

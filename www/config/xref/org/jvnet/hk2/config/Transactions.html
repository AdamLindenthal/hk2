53  /***
54   * Transactions is a singleton service that receives transaction notifications and dispatch these
55   * notifications asynchronously to listeners.
56   *
57   * @author Jerome Dochez
58   */
59  
60  @Service
61  public final class Transactions {
62  
63      private static Transactions singleton;
64      
65      // NOTE: synchronization on the object itself
66      List<ListenerInfo<TransactionListener>> listeners;
67  
68      ExecutorService executor;
69          
70      private ListenerInfo<Object> configListeners;
71      
72      private final class ListenerInfo<T> {
73          
74          private final T listener;
75          
76          private final BlockingQueue<Job> pendingJobs = new ArrayBlockingQueue<Job>(50);
77          private CountDownLatch latch = new CountDownLatch(1);
78          
79          public ListenerInfo(T listener) {
80              this.listener = listener;
81              start();
82          }
83          
84          public void addTransaction(Job job) {
85                  
86              // NOTE that this is put() which blocks, *not* add() which will not block and will
87              // throw an IllegalStateException if the queue is full.
88              if (latch.getCount()==0) {
89                  throw new RuntimeException("TransactionListener is inactive, yet jobs are published to it");
90              }
91              try {
92                  pendingJobs.put(job);
93              } catch (InterruptedException e ) {
94                  throw new RuntimeException(e);
95              }            
96              
97          }
98          
99          private void start() {
100 
101             executor.submit(new Runnable() {
102 
103                 public void run() {
104                     while (latch.getCount()>0) {
105                         try {
106                             final Job job  = pendingJobs.take();
107                             try {
108                                 if ( job.mEvents.size() != 0 ) {
109                                     job.process(listener);
110                                }
111                             } finally {
112                                 job.releaseLatch();
113                             }
114                         }
115                         catch (InterruptedException e) {
116                             // do anything here?
117                         }
118                     }
119                 }
120                 
121             });
122         }
123 
124         void stop() {
125             latch.countDown();
126             // last event to force the close
127             pendingJobs.add(new TransactionListenerJob(new ArrayList<PropertyChangeEvent>(), null));
128         }
129     }
130     /***
131         A job contains an optional CountdownLatch so that a caller can learn when the
132         transaction has "cleared" by blocking until that time.
133      */
134     private abstract static class Job<T,U> {
135 
136         private final CountDownLatch mLatch;
137         protected final List<U> mEvents;
138         
139         public Job(List events, final CountDownLatch latch ) {
140             mLatch  = latch;
141             mEvents = events;
142         }
143         
144         public void waitForLatch() throws InterruptedException {
145             if ( mLatch != null ) {
146                 mLatch.await();
147             }
148         }
149         
150         public void releaseLatch() {
151             if ( mLatch != null ) {
152                 mLatch.countDown();
153             }
154         }
155         
156         public abstract void process(T target);
157     }

159     private static class TransactionListenerJob extends Job<TransactionListener, PropertyChangeEvent> {
160 
161         public TransactionListenerJob(List<PropertyChangeEvent> events, CountDownLatch latch) {
162             super(events,  latch);
163         }
164         
165         @Override
166         public void process(TransactionListener listener) {
167             try {
168                 listener.transactionCommited(mEvents);
169             } catch(Exception e) {
170                 e.printStackTrace();
171             }            
172         }
173     }

175     private static class UnprocessedEventsJob extends Job<TransactionListener, UnprocessedChangeEvents> {
176 
177         public UnprocessedEventsJob(List<UnprocessedChangeEvents> events, CountDownLatch latch) {
178             super(events, latch);
179         }
180 
181         @Override
182         public void process(TransactionListener listener) {
183             try {
184                 listener.unprocessedTransactedEvents(mEvents);
185             } catch(Exception e) {
186                 e.printStackTrace();
187             }
188         }
189     }

191     private   class ConfigListenerJob extends Job<Object, PropertyChangeEvent> {
192 
193 
194         public ConfigListenerJob(List events, CountDownLatch latch) {
195             super(events, latch);
196         }
197 
198         @Override
199         public void process(Object target) {
200             Set<ConfigListener> notifiedListeners = new HashSet<ConfigListener>();
201             List<UnprocessedChangeEvents> unprocessedEvents  = new ArrayList<UnprocessedChangeEvents>();
202             for (final PropertyChangeEvent evt : mEvents) {
203                 final Dom dom = (Dom) ((ConfigView) Proxy.getInvocationHandler(evt.getSource())).getMasterView();
204                 if (dom.getListeners() != null && dom.getListeners().size() != 0 ) {
205                     List<ConfigListener> listeners = new ArrayList<ConfigListener>(dom.getListeners());
206                     for (final ConfigListener listener :listeners) {                       
207                         if (listener==null) {
208                             Logger.getAnonymousLogger().warning("Null config listener is registered to " + dom);
209                             continue;
210                         }
211                         if (!notifiedListeners.contains(listener)) {
212                             try {
213                                 // create a new array each time to avoid any potential array changes?
214                                 UnprocessedChangeEvents unprocessed = listener.changed(mEvents.toArray(new PropertyChangeEvent[mEvents.size()]));
215                                 if (unprocessed != null && unprocessed.size() != 0 ) {
216                                     unprocessedEvents.add(unprocessed);
217                                 }
218                             } catch (Exception e) {
219                                 e.printStackTrace();
220                             }
221                         }
222                         notifiedListeners.add(listener);
223                     }
224                 }
225                 // we notify the immediate parent.
226                 // dochez : should we notify the parent chain up to the root or stop at the first parent.
227                 if (dom.parent()!=null && dom.parent().getListeners()!=null) {
228                     List<ConfigListener> parentListeners = new ArrayList<ConfigListener>(dom.parent().getListeners());
229                     for (ConfigListener parentListener : parentListeners) {
230                         if (parentListener==null) {
231                             Logger.getAnonymousLogger().warning("Null config listener is registered to " + dom);
232                             continue;
233                         }
234                         if (!notifiedListeners.contains(parentListener)) {
235                             try {
236                                 // create a new array each time to avoid any potential array changes?
237                                 UnprocessedChangeEvents unprocessed = parentListener.changed(mEvents.toArray(new PropertyChangeEvent[mEvents.size()]));
238                                 if (unprocessed != null && unprocessed.size() != 0 ) {
239                                     unprocessedEvents.add(unprocessed);
240                                 }
241                             } catch (Exception e) {
242                                 e.printStackTrace();
243                             }
244                             notifiedListeners.add(parentListener);
245                         }
246                     }
247                 }
248             }
249             // all the config listeners have been notified, let's see if we have
250             // some unprocessed events to notifiy the transation listeners.
251             if (!unprocessedEvents.isEmpty()) {
252                 synchronized(listeners) {
253                     UnprocessedEventsJob job = new UnprocessedEventsJob(unprocessedEvents, null);
254                     for (ListenerInfo listener : listeners) {
255                         listener.addTransaction(job);
256                     }
257                 }
258             }
259         }        
260     }

262     /***
263      * add a new listener to all transaction events.
264      *
265      * @param listener to be added.

267     public void addTransactionsListener(TransactionListener listener) {
268         synchronized(listeners) {

273     /***
274      * Removes an existing listener for transaction events
275      * @param listener the registered listener
276      * @return true if the listener unregistration was successful
277      */
278     public boolean removeTransactionsListener(TransactionListener listener) {
279         synchronized(listeners) {
280             for (ListenerInfo info : listeners) {
281                 if (info.listener==listener) {
282                     info.stop();
283                     return listeners.remove(info);
284                 }
285             }
286         }
287         return false;
288     }

290     public List<TransactionListener> currentListeners() {
291         synchronized(listeners) {            
292             List<TransactionListener> l = new ArrayList<TransactionListener>();
293             for (ListenerInfo<TransactionListener> info : listeners) {
294                 l.add(info.listener);
295             }
296             return l;
297         }

302     void addTransaction( final List<PropertyChangeEvent> events) {
303         addTransaction(events, false);
304     }

306     /***
307      * Notification of a new transation completion
308      *
309      * @param events accumulated list of changes
310      * @param waitTillCleared  synchronous semantics; wait until all change events are sent
311      */
312     void addTransaction(
313         final List<PropertyChangeEvent> events,
314         final boolean waitTillCleared ) {
315         
316         final List<ListenerInfo> listInfos = new ArrayList<ListenerInfo>();
317         listInfos.addAll(listeners);
318         
319         // create a CountDownLatch to implement waiting for events to actually be sent
320         final CountDownLatch latch = waitTillCleared ? new CountDownLatch(listInfos.size()+1) : null;
321         
322         final Job job = new TransactionListenerJob( events, latch );
323         
324         // NOTE that this is put() which blocks, *not* add() which will not block and will
325         // throw an IllegalStateException if the queue is full.
326         try {
327             for (ListenerInfo listener : listInfos) {
328                 listener.addTransaction(job);
329             }
330             // the config listener job
331             configListeners.addTransaction(new ConfigListenerJob(events, latch));
332             
333             job.waitForLatch();
334         } catch (InterruptedException e ) {
335             throw new RuntimeException(e);
336         }
337     }

339     public void waitForDrain() {
340         // insert a dummy Job and block until is has been processed.  This guarantees
341         // that all prior jobs have finished
342         addTransaction( new ArrayList<PropertyChangeEvent>(), true );
343         // at this point all prior transactions are guaranteed to have cleared
344     }

346     private Transactions(ExecutorService executor) {
347         this.executor = executor;
348         listeners = new ArrayList<ListenerInfo<TransactionListener>>();
349         configListeners = new ListenerInfo<Object>(null);
350     }

353         if (singleton==null) {
354             singleton=new Transactions(executor);
355         }

360         return singleton;

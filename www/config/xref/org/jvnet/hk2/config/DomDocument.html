56       * A hook to perform variable replacements on the attribute/element values
57       * that are found in the configuration file.
58       * The translation happens lazily when objects are actually created, not when
59       * configuration is parsed, so this allows circular references &mdash;
60       * {@link Translator} may refer to objects in the configuration file being read.

95      /***
96       * Creates {@link ConfigModel} for the given {@link ConfigInjector} if we haven't done so.

98      /*package*/ ConfigModel buildModel(Inhabitant<? extends ConfigInjector> i) {
99          ConfigModel m = models.get(i);
100         if(m==null)
101             m = new ConfigModel(this,i,i.metadata());

112     /***
113      * Obtains a {@link ConfigModel} for the given class (Which should have {@link Configured} annotation on it.)
114      */
115     public ConfigModel buildModel(String fullyQualifiedClassName) {
116         Inhabitant i = habitat.getInhabitantByAnnotation(InjectionTarget.class, fullyQualifiedClassName);
117         if(i==null)
118             throw new ComponentException("ConfigInjector for %s is not found",fullyQualifiedClassName);
119         return buildModel(i);
120     }

122     /***
123      * Obtains the {@link ConfigModel} from the "global" element name.
124      *
125      * <p>
126      * This method uses {@link #buildModel} to lazily build models if necessary.
127      * 
128      * @return
129      *      Null if no configurable component is registered under the given global element name.

131     public ConfigModel getModelByElementName(String elementName) {
132         Inhabitant<? extends ConfigInjector> i = habitat.getInhabitant(ConfigInjector.class, elementName);
133         if(i==null) return null;
134         return buildModel(i);
135     }

137     /***
138      * Calculates all @Configured interfaces subclassing the passed interface type.
139      *
140      * @param intf a @Configured interface
141      * @return List of all @Configured subclasses
142      * @throws ClassNotFoundException
143      */
144     public synchronized List<ConfigModel> getAllModelsImplementing(Class intf) throws ClassNotFoundException {
145         if (implementorsOf.size()==0) {
146             initXRef();
147         }
148         return implementorsOf.getOne(intf);   
149     }
150 
151     /***
152      * probably a bit slow, calculates all the @Configured interfaces subclassing, useful
153      * to find all possible subclasses of a type.
154      * 
155      * @throws ClassNotFoundException
156      */
157     private void initXRef() throws ClassNotFoundException {
158 
159         // force initialization of all the config models.
160         for (Inhabitant<? extends ConfigInjector> i : habitat.getInhabitants(ConfigInjector.class)) {
161             buildModel(i);
162         }
163         List<ConfigModel> result = new ArrayList<ConfigModel>();
164         for (ConfigModel cm : models.values()) {
165             Class targetType = cm.classLoaderHolder.get().loadClass(cm.targetTypeName);
166             do {
167                 Class[] intfs = targetType.getInterfaces();
168                 for (Class intf : intfs) {
169                     if (intf.isAnnotationPresent(Configured.class)) {
170 
171                         List<ConfigModel> models = implementorsOf.getOne(intf);
172                         if (models==null) {
173                             models = new ArrayList<ConfigModel>();
174                             implementorsOf.add(intf, models);
175                         }
176                         models.add(cm);
177                     }
178                 }
179                 targetType = targetType.getSuperclass();
180             } while (targetType!=null);
181         }
182     }

209     /*package*/
210     DataType getValidator(String dataType) {
211         synchronized(validators) {
212             DataType validator = validators.get(dataType);
213             if (validator != null)
214                 return (validator);
215         }
216         Collection<DataType> dtfh = habitat.getAllByContract(DataType.class);
217         synchronized(validators) {
218             for (DataType dt : dtfh) {
219                 validators.put(dt.getClass().getCanonicalName(), dt);
220             }
221             return (validators.get(dataType));
222         }
223     }

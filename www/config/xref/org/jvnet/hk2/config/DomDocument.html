54   * Represents a whole DOM tree.

58  public class DomDocument {
59      /***
60       * A hook to perform variable replacements on the attribute/element values
61       * that are found in the configuration file.
62       * The translation happens lazily when objects are actually created, not when
63       * configuration is parsed, so this allows circular references &mdash;
64       * {@link Translator} may refer to objects in the configuration file being read.
65       */
66      private volatile Translator translator = Translator.NOOP;
67  
68      protected final Map<Inhabitant<? extends ConfigInjector>,ConfigModel> models = new HashMap<Inhabitant<? extends ConfigInjector>, ConfigModel>();
69  
70      /*package*/ final Habitat habitat;
71  
72      /*package*/ Dom root;
73  
74      private final Map<String, DataType> validators = new HashMap<String, DataType>();
75      
76      /*package*/ static final List<String> PRIMS = Collections.unmodifiableList(Arrays.asList(
77      "boolean", "char", "int", "java.lang.Boolean", "java.lang.Character", "java.lang.Integer"));
78      
79      public DomDocument(Habitat habitat) {
80          this.habitat = habitat;
81          for (String prim : PRIMS) {
82              validators.put(prim, new PrimitiveDataType(prim) );
83          }
84      }
85  
86      public Dom getRoot() {
87          return root;
88      }
89  
90      public Translator getTranslator() {
91          return translator;
92      }
93  
94      public void setTranslator(Translator translator) {
95          this.translator = translator;
96      }

98      /***
99       * Creates {@link ConfigModel} for the given {@link ConfigInjector} if we haven't done so.
100      */
101     /*package*/ ConfigModel buildModel(Inhabitant<? extends ConfigInjector> i) {
102         ConfigModel m = models.get(i);
103         if(m==null)
104             m = new ConfigModel(this,i,i.metadata());
105         return m;
106     }
107 
108     /***
109      * Obtains a {@link ConfigModel} for the given class (Which should have {@link Configured} annotation on it.)
110      */
111     public ConfigModel buildModel(Class<?> clazz) {
112         return buildModel(clazz.getName());
113     }
114 
115     /***
116      * Obtains a {@link ConfigModel} for the given class (Which should have {@link Configured} annotation on it.)
117      */
118     public ConfigModel buildModel(String fullyQualifiedClassName) {
119         Inhabitant i = habitat.getInhabitantByAnnotation(InjectionTarget.class, fullyQualifiedClassName);
120         if(i==null)
121             throw new ComponentException("ConfigInjector for %s is not found",fullyQualifiedClassName);
122         return buildModel(i);
123     }
124 
125     /***
126      * Obtains the {@link ConfigModel} from the "global" element name.
127      *
128      * <p>
129      * This method uses {@link #buildModel} to lazily build models if necessary.
130      * 
131      * @return
132      *      Null if no configurable component is registered under the given global element name.
133      */
134     public ConfigModel getModelByElementName(String elementName) {
135         Inhabitant<? extends ConfigInjector> i = habitat.getInhabitant(ConfigInjector.class, elementName);
136         if(i==null) return null;
137         return buildModel(i);
138     }
139

148     }

150     /***
151      * Writes back the whole DOM tree as an XML document.
152      *
153      * <p>
154      * To support writing a subtree, this method doesn't invoke the start/endDocument
155      * events. Those are the responsibility of the caller.
156      *
157      * @param w
158      *      Receives XML infoset stream.
159      */
160     public void writeTo(XMLStreamWriter w) throws XMLStreamException {
161         root.writeTo(null,w);
162     }

164     /*package*/
165     DataType getValidator(String dataType) {
166         synchronized(validators) {
167             DataType validator = validators.get(dataType);
168             if (validator != null)
169                 return (validator);
170         }
171         Collection<DataType> dtfh = habitat.getAllByContract(DataType.class);
172         synchronized(validators) {
173             for (DataType dt : dtfh) {
174                 validators.put(dt.getClass().getCanonicalName(), dt);
175             }
176             return (validators.get(dataType));
177         }
178     }

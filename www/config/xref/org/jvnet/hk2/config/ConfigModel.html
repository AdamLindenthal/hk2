200     /***
201      * Obtains the duck method implementation from a method on the {@link ConfigBeanProxy}-derived interface.
202      */
203     public Method getDuckMethod(Method method) throws ClassNotFoundException, NoSuchMethodException {
204         synchronized (duckMethods) {
205             Method duckMethod = duckMethods.get(method);
206             if(duckMethod!=null)    return duckMethod;
207 
208             Class<?> clz = method.getDeclaringClass();
209             Class<?> duck = clz.getClassLoader().loadClass(clz.getName() + "$Duck");
210 
211             Class<?>[] types = method.getParameterTypes();
212             Class[] paramTypes = new Class[types.length+1];
213             System.arraycopy(types,0,paramTypes,1,types.length);
214             paramTypes[0] = clz;
215             duckMethod = duck.getMethod(method.getName(), paramTypes);
216             duckMethods.put(method,duckMethod);
217 
218             return duckMethod;
219         }
220     }

222     /***
223      * Obtain XML names (like "abc-def") from strings like "getAbcDef" and "hasAbcDef".
224      * <p>
225      * The conversion rule uses the model to find a good match.
226      */
227     public ConfigModel.Property toProperty(Method method) {
228         String name = method.getName();
229 
230         // check annotations first
231         Element e = method.getAnnotation(Element.class);
232         if(e!=null) {
233             String en = e.value();
234             if(en.length()>0)
235                 return elements.get(en);
236         }
237         Attribute a = method.getAnnotation(Attribute.class);
238         if(a!=null) {
239             String an = a.value();
240             if(an.length()>0)
241                 return attributes.get(an);
242         }
243         // TODO: check annotations on the getter/setter
244 
245         // first, trim off the prefix
246         for (String p : Dom.PROPERTY_PREFIX) {
247             if(name.startsWith(p)) {
248                 name = name.substring(p.length());
249                 break;
250             }
251         }
252 
253         // tokenize by finding 'x|X' and 'X|Xx' then insert '-'.
254         StringBuilder buf = new StringBuilder(name.length()+5);
255         for(String t : Dom.TOKENIZER.split(name)) {
256             if(buf.length()>0)  buf.append('-');
257             buf.append(t.toLowerCase());
258         }
259         name = buf.toString();
260 
261         // at this point name should match XML names in the model, modulo case.
262         return findIgnoreCase(name);
263     }

331         /***
332          * Coerce the given type to {@link Dom}.
333          * Only handles those types that are valid to the {@link #set(Dom, Object)} method.

335         protected final Dom toDom(Object arg) {
336             if(arg==null)
337                 return null;
338             if(arg instanceof Dom)

346     static final class CollectionNode extends Node {
347         CollectionNode(ConfigModel model, String xmlName) {
348             super(model, xmlName);
349         }
350 
351         public boolean isCollection() {
352             return true;
353         }
354 
355         public Object get(final Dom dom, Type returnType) {
356             // TODO: perhaps support more collection types?
357 
358 
359             if(!(returnType instanceof ParameterizedType))
360                 throw new IllegalArgumentException("List needs to be parameterized");
361             final Class itemType = Types.erasure(Types.getTypeArgument(returnType,0));
362 
363             final List<Dom> v = ("*".equals(xmlName)?dom.domNodeByTypeElements(itemType):dom.nodeElements(xmlName));
364 
365             if(itemType==Dom.class)
366                 // TODO: this returns a view, not a live list
367                 return v;
368             if(ConfigBeanProxy.class.isAssignableFrom(itemType)) {
369                 // return a live list
370                 return new AbstractList<Object>() {
371                     public Object get(int index) {
372                         return v.get(index).createProxy();
373                     }
374 
375                     public void add(int index, Object element) {
376                         // update the master children list, as well as this view 'v'
377                         Dom child = Dom.unwrap((ConfigBeanProxy) element);
378                         dom.insertAfter( index==0 ? null : v.get(index-1), xmlName, child);
379                         v.add(index,child);
380                     }
381 
382                     public Object remove(int index) {
383                         Dom child = v.get(index);
384                         dom.removeChild(child);
385                         v.remove(index);
386                         return child.createProxy();
387                     }
388 
389                     public Object set(int index, Object element) {
390                         Dom child = Dom.unwrap((ConfigBeanProxy) element);
391                         dom.replaceChild( v.get(index), xmlName, child );
392                         return v.set(index,child).createProxy();
393                     }
394 
395                     public int size() {
396                         return v.size();
397                     }
398                 };
399             }
400 
401             // TODO: error check needs to be improved,
402             // as itemType might be inconsistent with the actual type
403             return new AbstractList() {
404                 public Object get(int index) {
405                     return v.get(index).get();
406                 }
407 
408                 public int size() {
409                     return v.size();
410                 }
411             };
412         }
413 
414         public void set(Dom dom, Object _arg) {
415             if(!(_arg instanceof List))
416                 throw new IllegalArgumentException("Expecting a list but found "+_arg);
417             List arg = (List)_arg;
418 
419             Dom[] values = new Dom[arg.size()];
420             int i=0;
421             for (Object o : arg)
422                 values[i++] = toDom(o);
423 
424             dom.setNodeElements(xmlName,values);
425         }
426     }

470         /***
471          * Converts a single value from string to the specified target type.
472          *
473          * @return
474          *      Instance of the given 'returnType'
475          */
476         protected static Object convertLeafValue(Class<?> returnType, String v) {
477             if(v==null)
478                 // TODO: default value handling
479                 // TODO: if primitive types, report an error
480                 return null;
481 
482             if(returnType==String.class) {
483                 return v;
484             }
485             if(returnType==Integer.class || returnType==int.class) {
486                 return Integer.valueOf(v);
487             }
488             if(returnType==Boolean.class || returnType==boolean.class) {
489                 return BOOLEAN_TRUE.contains(v);
490             }
491             throw new IllegalArgumentException("Don't know how to handle "+returnType);
492         }
493

634     /***
635      * @param description
636      *      The description of the model as written in {@link InhabitantsFile the inhabitants file}.
637      */
638     public ConfigModel(DomDocument document, Inhabitant<? extends ConfigInjector> injector, MultiMap<String,String> description) {
639         if(description==null)
640             throw new ComponentException("%s doesn't have any metadata",injector.type());
641 
642         document.models.put(injector,this); // register now so that cyclic references are handled correctly.
643         this.injector = injector;
644         String targetTypeName=null,indexTypeName=null;
645         String key = null;
646         for (Map.Entry<String, List<String>> e : description.entrySet()) {
647             String name = e.getKey();
648             String value = e.getValue().size()>0 ? e.getValue().get(0) : null;
649             if(name.startsWith("@")) {
650                 // TODO: handle value.equals("optional") and value.equals("required") distinctively.
651                 String attributeName = name.substring(1);
652                 String dv = getMetadataFieldKeyedAs(e.getValue(), "default:");  //default value
653                 String dt = getMetadataFieldKeyedAs(e.getValue(), "datatype:"); //type
654                 AttributeLeaf leaf = null;
655                 if (dv == null)
656                     leaf = new AttributeLeaf(attributeName, dt);
657                 else
658                     leaf = new AttributeLeafWithDefaultValue(attributeName, dt, dv);
659                 attributes.put(attributeName, leaf);
660             } else
661             if(name.startsWith("<")) {
662                 String elementName = name.substring(1, name.length() - 1);
663                 elements.put(elementName,parseValue(elementName,document,value));
664             } else
665             if(name.equals(ConfigMetadata.TARGET))
666                 targetTypeName = value;
667             else
668             if(name.equals(ConfigMetadata.KEYED_AS))
669                 indexTypeName = value;
670             else
671             if(name.equals(ConfigMetadata.KEY))
672                 key = value;
673         }
674         if(targetTypeName==null)
675             throw new ComponentException("%s doesn't have the mandatory '%s' metadata", injector.type(), ConfigMetadata.TARGET);
676         if(key==null ^ indexTypeName==null)
677             throw new ComponentException("%s has inconsistent '%s=%s' and '%s=%s' metadata",
678                 ConfigMetadata.KEY, key, ConfigMetadata.TARGET, indexTypeName);
679         this.targetTypeName = targetTypeName;
680         this.keyedAs = indexTypeName;
681         this.key = key;
682         this.contracts = description.get(ConfigMetadata.TARGET_CONTRACTS);
683         this.symbolSpaces = new HashSet<String>(description.get("symbolSpaces"));
684 
685         String tagName = null;
686         for (String v : description.get(InhabitantsFile.INDEX_KEY)) {
687             if(v.startsWith(ELEMENT_NAME_PREFIX))
688                 tagName = v.substring(ELEMENT_NAME_PREFIX.length());
689         }
690         this.tagName = tagName;
691     }

693     private static final String ELEMENT_NAME_PREFIX = ConfigInjector.class.getName() + ':';

695     /***
696      * Finds the {@link Property} from either {@link #elements} or {@link #attributes}.
697      * @param xmlName
698      *      XML name to be searched.
699      * @return null
700      *      if none is found.
701      */
702     public Property findIgnoreCase(String xmlName) {
703         // first try the exact match to take our chance
704         Property a = attributes.get(xmlName);
705         if(a!=null)     return a;
706         a = elements.get(xmlName);
707         if(a!=null)     return a;
708 
709         // exhaustive search
710         a = _findIgnoreCase(xmlName, attributes);
711         if(a!=null)     return a;
712         return _findIgnoreCase(xmlName, elements);
713     }
714 
715     private Property _findIgnoreCase(String name, Map<String, ? extends Property> map) {
716         for (Map.Entry<String, ? extends Property> i : map.entrySet())
717             if(i.getKey().equalsIgnoreCase(name))
718                 return i.getValue();
719         return null;
720     }

722     /***
723      * Parses {@link Property} object from a value in the metadata description.
724      */
725     private Property parseValue(String elementName, DomDocument document, String value) {
726         boolean collection = false;
727         if(value.startsWith("collection:")) {
728             collection = true;
729             value = value.substring(11);
730         }
731 
732         if(value.equals("leaf")) {
733             if(collection)  return new CollectionLeaf(elementName);
734             else            return new SingleLeaf(elementName);
735         }
736 
737         // this element is a reference to another configured inhabitant.
738         // figure that out.
739         ConfigModel model = document.buildModel(value);
740         if(collection)
741             return new CollectionNode(model,elementName);
742         else
743             return new SingleNode(model,elementName);
744     }

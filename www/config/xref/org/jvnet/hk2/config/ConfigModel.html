143     /***
144      * Obtains the duck method implementation from a method on the {@link ConfigBeanProxy}-derived interface.
145      */
146     public Method getDuckMethod(Method method) throws ClassNotFoundException, NoSuchMethodException {
147         synchronized (duckMethods) {
148             Method duckMethod = duckMethods.get(method);
149             if(duckMethod!=null)    return duckMethod;
150 
151             Class<?> clz = method.getDeclaringClass();
152             Class<?> duck = clz.getClassLoader().loadClass(clz.getName() + "$Duck");
153 
154             Class<?>[] types = method.getParameterTypes();
155             Class[] paramTypes = new Class[types.length+1];
156             System.arraycopy(types,0,paramTypes,1,types.length);
157             paramTypes[0] = clz;
158             duckMethod = duck.getMethod(method.getName(), paramTypes);
159             duckMethods.put(method,duckMethod);
160 
161             return duckMethod;
162         }
163     }

238         CollectionNode(ConfigModel model, String xmlName) {
239             super(model, xmlName);

243             return true;
244         }

246         public Object get(final Dom dom, Type returnType) {
247             // TODO: perhaps support more collection types?
248 
249 
250             if(!(returnType instanceof ParameterizedType))
251                 throw new IllegalArgumentException("List needs to be parameterized");
252             final Class itemType = Types.erasure(Types.getTypeArgument(returnType,0));
253 
254             final List<Dom> v = ("*".equals(xmlName)?dom.domNodeByTypeElements(itemType):dom.nodeElements(xmlName));
255 
256             if(itemType==Dom.class)
257                 // TODO: this returns a view, not a live list
258                 return v;
259             if(ConfigBeanProxy.class.isAssignableFrom(itemType)) {
260                 // return a live list
261                 return new AbstractList<Object>() {
262                     public Object get(int index) {
263                         return v.get(index).createProxy();
264                     }
265 
266                     public void add(int index, Object element) {
267                         // update the master children list, as well as this view 'v'
268                         Dom child = Dom.unwrap((ConfigBeanProxy) element);
269                         dom.insertAfter( index==0 ? null : v.get(index-1), xmlName, child);
270                         v.add(index,child);
271                     }
272 
273                     public Object remove(int index) {
274                         Dom child = v.get(index);
275                         dom.removeChild(child);
276                         v.remove(index);
277                         return child.createProxy();
278                     }
279 
280                     public Object set(int index, Object element) {
281                         Dom child = Dom.unwrap((ConfigBeanProxy) element);
282                         dom.replaceChild( v.get(index), xmlName, child );
283                         return v.set(index,child).createProxy();
284                     }
285 
286                     public int size() {
287                         return v.size();
288                     }
289                 };
290             }
291 
292             // TODO: error check needs to be improved,
293             // as itemType might be inconsistent with the actual type
294             return new AbstractList() {
295                 public Object get(int index) {
296                     return v.get(index).get();
297                 }
298 
299                 public int size() {
300                     return v.size();
301                 }
302             };
303         }
304 
305         public void set(Dom dom, Object _arg) {
306             if(!(_arg instanceof List))
307                 throw new IllegalArgumentException("Expecting a list but found "+_arg);
308             List arg = (List)_arg;
309 
310             Dom[] values = new Dom[arg.size()];
311             int i=0;
312             for (Object o : arg)
313                 values[i++] = toDom(o);
314 
315             dom.setNodeElements(xmlName,values);
316         }

361         /***
362          * Converts a single value from string to the specified target type.
363          *
364          * @return
365          *      Instance of the given 'returnType'
366          */
367         protected static Object convertLeafValue(Class<?> returnType, String v) {
368             if(v==null)
369                 // TODO: default value handling
370                 // TODO: if primitive types, report an error
371                 return null;
372 
373             if(returnType==String.class) {
374                 return v;
375             }
376             if(returnType==Integer.class || returnType==int.class) {
377                 return Integer.valueOf(v);
378             }
379             if(returnType==Boolean.class || returnType==boolean.class) {
380                 return BOOLEAN_TRUE.contains(v);
381             }
382             throw new IllegalArgumentException("Don't know how to handle "+returnType);
383         }
384 
385         private static final Set<String> BOOLEAN_TRUE = new HashSet<String>(Arrays.asList("true","yes","on","1"));

525     /***
526      * @param description
527      *      The description of the model as written in {@link InhabitantsFile the inhabitants file}.
528      */
529     public ConfigModel(DomDocument document, Inhabitant<? extends ConfigInjector> injector, MultiMap<String,String> description) {
530         if(description==null)
531             throw new ComponentException("%s doesn't have any metadata",injector.type());
532 
533         document.models.put(injector,this); // register now so that cyclic references are handled correctly.
534         this.injector = injector;
535         String targetTypeName=null,indexTypeName=null;
536         String key = null;
537         for (Map.Entry<String, List<String>> e : description.entrySet()) {
538             String name = e.getKey();
539             String value = e.getValue().size()>0 ? e.getValue().get(0) : null;
540             if(name.startsWith("@")) {
541                 // TODO: handle value.equals("optional") and value.equals("required") distinctively.
542                 String attributeName = name.substring(1);
543                 String dv = getMetadataFieldKeyedAs(e.getValue(), "default:");  //default value
544                 String dt = getMetadataFieldKeyedAs(e.getValue(), "datatype:"); //type
545                 AttributeLeaf leaf = null;
546                 if (dv == null)
547                     leaf = new AttributeLeaf(attributeName, dt);
548                 else
549                     leaf = new AttributeLeafWithDefaultValue(attributeName, dt, dv);
550                 attributes.put(attributeName, leaf);
551             } else
552             if(name.startsWith("<")) {
553                 String elementName = name.substring(1, name.length() - 1);
554                 elements.put(elementName,parseValue(elementName,document,value));
555             } else
556             if(name.equals(ConfigMetadata.TARGET))
557                 targetTypeName = value;
558             else
559             if(name.equals(ConfigMetadata.KEYED_AS))
560                 indexTypeName = value;
561             else
562             if(name.equals(ConfigMetadata.KEY))
563                 key = value;
564         }
565         if(targetTypeName==null)
566             throw new ComponentException("%s doesn't have the mandatory '%s' metadata", injector.type(), ConfigMetadata.TARGET);
567         if(key==null ^ indexTypeName==null)
568             throw new ComponentException("%s has inconsistent '%s=%s' and '%s=%s' metadata",
569                 ConfigMetadata.KEY, key, ConfigMetadata.TARGET, indexTypeName);
570         this.targetTypeName = targetTypeName;
571         this.keyedAs = indexTypeName;
572         this.key = key;
573         this.contracts = description.get(ConfigMetadata.TARGET_CONTRACTS);
574         this.symbolSpaces = new HashSet<String>(description.get("symbolSpaces"));
575 
576         String tagName = null;
577         for (String v : description.get(InhabitantsFile.INDEX_KEY)) {
578             if(v.startsWith(ELEMENT_NAME_PREFIX))
579                 tagName = v.substring(ELEMENT_NAME_PREFIX.length());
580         }
581         this.tagName = tagName;
582     }

586     /***
587      * Finds the {@link Property} from either {@link #elements} or {@link #attributes}.
588      * @param xmlName
589      *      XML name to be searched.
590      * @return null
591      *      if none is found.
592      */
593     public Property findIgnoreCase(String xmlName) {
594         // first try the exact match to take our chance
595         Property a = attributes.get(xmlName);
596         if(a!=null)     return a;
597         a = elements.get(xmlName);
598         if(a!=null)     return a;
599 
600         // exhaustive search
601         a = _findIgnoreCase(xmlName, attributes);
602         if(a!=null)     return a;
603         return _findIgnoreCase(xmlName, elements);
604     }

606     private Property _findIgnoreCase(String name, Map<String, ? extends Property> map) {
607         for (Map.Entry<String, ? extends Property> i : map.entrySet())
608             if(i.getKey().equalsIgnoreCase(name))
609                 return i.getValue();
610         return null;
611     }

613     /***
614      * Parses {@link Property} object from a value in the metadata description.
615      */
616     private Property parseValue(String elementName, DomDocument document, String value) {
617         boolean collection = false;
618         if(value.startsWith("collection:")) {
619             collection = true;
620             value = value.substring(11);
621         }
622 
623         if(value.equals("leaf")) {
624             if(collection)  return new CollectionLeaf(elementName);
625             else            return new SingleLeaf(elementName);
626         }
627 
628         // this element is a reference to another configured inhabitant.
629         // figure that out.
630         ConfigModel model = document.buildModel(value);
631         if(collection)
632             return new CollectionNode(model,elementName);
633         else
634             return new SingleNode(model,elementName);
635     }

637     private static String getMetadataFieldKeyedAs(List<String> strings, String name) {
638         if (strings == null || strings.size() == 0 || name == null)
639             return ( null );
640         String dv = null;
641         for (String s : strings) {
642             if (s.startsWith(name)) {
643                 dv = s.substring(name.length());
644                 break;
645             }
646         }
647         return ( dv );        
648     }

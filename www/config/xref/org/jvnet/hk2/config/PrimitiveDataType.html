6   package org.jvnet.hk2.config;

8   import org.jvnet.hk2.annotations.Service;

10  /*** Represents a Java primitive (and its wrapper) data type. Not all Java primitives
11   *  are relevant from a configuration standpoint.
12   * @see DataType
13   * @see DomDocument#PRIMS
14   * @author &#2325;&#2375;&#2342;&#2366;&#2352 (km@dev.java.net)
15   * @since hk2 0.3.10
16   */
17  public final class PrimitiveDataType implements DataType {
18  
19      private final String realType;
20      PrimitiveDataType(String realType) {
21          assert DomDocument.PRIMS.contains(realType) : "This class can't validate: " + realType;
22          this.realType = realType;
23      }
24      
25      public void validate(String value) throws ValidationException {
26          if (value.startsWith("${") && value.endsWith("}")) //it's a token
27            return;
28          boolean match = false;
29          if ("int".equals(realType) || "java.lang.Integer".equals(realType))
30              match = representsInteger(value);
31          else if ("boolean".equals(realType) || "java.lang.Boolean".endsWith(realType))
32              match = representsBoolean(value);
33          else if ("char".equals(realType) || "java.lang.Character".equals(realType))
34              representsChar(value);
35          //no need for else as we are asserting it in the constructor
36          if (!match) {
37              String msg = "This value: " + value + " is not of type: " + realType + ", validation failed";
38              throw new ValidationException(msg);            
39          }
40      }
41      
42      private static boolean representsBoolean(String value) {
43          boolean isBoolean = "true".equalsIgnoreCase(value) || "false".equalsIgnoreCase(value);
44          return (isBoolean);
45      }
46      private static boolean representsChar(String value) {
47              if (value.length() == 1)
48                  return true;
49              return false;
50      }
51      private static boolean representsInteger(String value) {
52          try {
53              Integer.parseInt(value);
54              return true;
55          } catch(NumberFormatException ne) {
56              return false;
57          }
58      }

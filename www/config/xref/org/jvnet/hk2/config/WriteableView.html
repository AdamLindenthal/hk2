72      public WriteableView(ConfigBeanProxy readView) {
73          this.bean = (ConfigBean) ((ConfigView) Proxy.getInvocationHandler(readView)).getMasterView();
74          this.defaultView = bean.createProxy();
75          changedAttributes = new HashMap<String, PropertyChangeEvent>();
76          changedCollections = new HashMap<String, ProtectedList>();
77          if (beanValidator == null) {
78              ClassLoader cl = Thread.currentThread().getContextClassLoader();
79              try {
80                  Thread.currentThread().setContextClassLoader(null);
81                  TraversableResolver traversableResolver =
82                      new TraversableResolver() {
83                          public boolean isReachable(Object traversableObject,
84                              Path.Node traversableProperty, Class<?> rootBeanType,
85                              Path pathToTraversableObject, ElementType elementType) {

97                  ValidatorContext validatorContext = validatorFactory.usingContext();
98                      beanValidator = validatorContext.traversableResolver(
99                              traversableResolver).getValidator();
100             } finally {
101                 Thread.currentThread().setContextClassLoader(cl);
102             }
103         }

106     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
107 
108         if (method.getName().equals("hashCode"))
109             return super.hashCode();
110         
111         if (method.getName().equals("equals"))
112             return super.equals(args[0]);
113 
114         if(method.getAnnotation(DuckTyped.class)!=null) {
115             return bean.invokeDuckMethod(method,proxy,args);
116         }
117         
118         ConfigModel.Property property = bean.model.toProperty(method);
119 
120         if(property==null)
121              throw new IllegalArgumentException(
122                 "No corresponding property found for method: "+method);
123 
124         if(args==null || args.length==0) {
125             // getter, maybe one of our changed properties
126             if (changedAttributes.containsKey(property.xmlName())) {
127                 // serve masked changes.
128                 Object changedValue = changedAttributes.get(property.xmlName()).getNewValue();
129                 if (changedValue instanceof Dom) {
130                     return ((Dom) changedValue).createProxy();
131                 } else {
132                     return changedValue;
133                 }
134             } else {
135                 // pass through.
136                 return getter(property, method.getGenericReturnType());
137             }
138         } else {
139             setter(property, args[0], method.getGenericParameterTypes()[0]);
140             return null;
141         }
142     }

158     public Object getter(ConfigModel.Property property, java.lang.reflect.Type t) {
159         Object value =  bean._getter(property, t);
160         if (value instanceof List) {
161             if (!changedCollections.containsKey(property.xmlName())) {
162                 // wrap collections so we can record events on that collection mutation.
163                 changedCollections.put(property.xmlName(),
164                         new ProtectedList(List.class.cast(value), defaultView, property.xmlName()));
165             }
166             return changedCollections.get(property.xmlName());
167         }
168         return value;
169     }

171     public void setter(ConfigModel.Property property,
172         Object newValue, java.lang.reflect.Type t)  {
173         
174         // are we still in a transaction
175         if (currentTx==null) {
176             throw new IllegalStateException("Not part of a transation");
177         }
178         try {
179             if (newValue != null)
180                 handleValidation(property, newValue);
181         } catch(Exception v) {
182             bean.getLock().unlock();
183             throw new RuntimeException(v);
184         }
185 
186         // Following is a check to avoid duplication of elements with same key
187         // attribute values. See Issue 7956
188         if (property instanceof ConfigModel.AttributeLeaf) {
189 
190             // First check if the key leaf attribute is being set
191             ConfigModel.AttributeLeaf al =
192                 (ConfigModel.AttributeLeaf)property;
193 
194             ConfigBean master = getMasterView();
195             String key = master.model.key;
196 
197             // A key attribute may not exist at all if none of the attribs of
198             // an element are annotated with key=true. If one exists, make sure
199             // that attribute is actually the one being set
200             if ((key != null) && (key.substring(1).equals(property.xmlName))) {
201 
202                 // remove leading @
203                 key = key.substring(1);
204                 // Extract the old key value
205                 String oldKeyValue = getPropertyValue(key);
206 
207                 // Get the Parent Element which has the key attribute specified
208                 // through the input paramater 'property'. For e.g. in case of
209                 // Domain->Resources->ConnectorConnectionPool->name(key attrib)
210                 // thisview will equal ConnectorConnectionPool
211                 Dom thisview = Dom.unwrap(defaultView);
212 
213                 // parent will equal Resources
214                 Dom parent = thisview.parent();
215 
216                 // siblings will contain all ConnectorConnectionPools under
217                 // Resources
218                 List<Dom> siblings = 
219                     parent.domNodeByTypeElements(thisview.getProxyType());
220 
221                 // Iterate through each sibling element and see if anyone has
222                 // same key. If true throw an exception after unlocking this
223                 // element
224                 for (Dom sibling : siblings) {
225                     String siblingKey = sibling.getKey();
226                     if (newValue.equals(siblingKey)) {
227                         bean.getLock().unlock();
228                         throw new IllegalArgumentException(
229                             "Keys cannot be duplicate. Old value of this key " +
230                             "property, " + oldKeyValue + "will be retained");
231                     }
232                 }
233             }
234         }
235 
236         // setter
237         Object oldValue = bean.getter(property, t);
238         if (newValue instanceof ConfigBeanProxy) {
239             ConfigView bean = (ConfigView)
240                 Proxy.getInvocationHandler((ConfigBeanProxy) newValue);
241             newValue = bean.getMasterView();
242         }
243         PropertyChangeEvent evt = new PropertyChangeEvent(
244             defaultView,property.xmlName(), oldValue, newValue);
245         try {
246             for (ConfigBeanInterceptor interceptor : bean.getOptionalFeatures()) {
247                 interceptor.beforeChange(evt);
248             }
249         } catch(PropertyVetoException e) {
250             throw new RuntimeException(e);
251         }
252 
253         changedAttributes.put(property.xmlName(), evt);
254         for (ConfigBeanInterceptor interceptor : bean.getOptionalFeatures()) {
255             interceptor.afterChange(evt, System.currentTimeMillis());
256         }
257     }

264      * Enter a new Transaction, this method should return false if this object

266      * the passed transaction. If the object returns true, the object

274     public synchronized boolean join(Transaction t) {
275         if (currentTx==null) {
276             currentTx = t;
277             t.addParticipant(this);
278             return true;
279         }
280         return false;
281     }

283     /***
284      * Returns true of this Transaction can be committed on this object
285      *
286      * @param t is the transaction to commit, should be the same as the
287      *          one passed during the join(Transaction t) call.
288      * @return true if the trsaction commiting would be successful
289      */
290     public synchronized boolean canCommit(Transaction t) throws TransactionFailure {
291         Set constraintViolations =
292             beanValidator.validate(this.getProxy(this.getProxyType()));
293 
294         if (constraintViolations != null) {
295             Iterator it = constraintViolations.iterator();
296             boolean violated = false;
297             String msg = "Constraints for this bean " +
298                    "violated. \n Message = ";
299             while (it.hasNext()) {
300                 violated = true;
301                 ConstraintViolation cv = (ConstraintViolation) it.next();
302                 msg = msg + cv.getPropertyPath() + " " + cv.getMessage();
303             }
304             if (violated) {
305                 bean.getLock().unlock();
306                 throw new TransactionFailure(msg);
307             }
308         }
309         
310         return currentTx==t;
311     }

314     /*** remove @ or <> eg "@foo" => "foo" or "<foo>" => "foo" */
315     public static String stripMarkers(final String s ) {
316         if ( s.startsWith("@") ) {
317             return s.substring(1);
318         }
319         else if ( s.startsWith("<") ) {
320             return s.substring(1, s.length()-1);
321         }
322         return s;
323     }

325     /***
326      * Commit this Transaction.
327      *
328      * @param t the transaction commiting.
329      * @throws TransactionFailure
330      *          if the transaction commit failed
331      */
332     public synchronized List<PropertyChangeEvent> commit(Transaction t) throws TransactionFailure {
333         if (currentTx==t) {
334             currentTx=null;
335         }
336         
337         // a key attribute must be non-null and have length >= 1
338         final ConfigBean master = getMasterView();
339         final String keyStr = master.model.key;
340         if ( keyStr != null) {
341             final String key = stripMarkers(keyStr);
342             final String value = getPropertyValue(key);
343             if ( value == null ) {
344                 throw new TransactionFailure( "Key value cannot be null: " + key );
345             }
346             if ( value.length() == 0 ) {
347                 throw new TransactionFailure( "Key value cannot be empty string: " + key );
348             }
349         }
350 
351 
352         try {
353             List<PropertyChangeEvent> appliedChanges = new ArrayList<PropertyChangeEvent>();
354             for (PropertyChangeEvent event : changedAttributes.values()) {
355                 ConfigModel.Property property = bean.model.findIgnoreCase(event.getPropertyName());
356                 ConfigBeanInterceptor interceptor  = bean.getOptionalFeature(ConfigBeanInterceptor.class);
357                 try {
358                     if (interceptor!=null) {
359                         interceptor.beforeChange(event);
360                     }
361                 } catch (PropertyVetoException e) {
362                     throw new TransactionFailure(e.getMessage(), e);
363                 }
364                 property.set(bean, event.getNewValue());
365                 if (interceptor!=null) {
366                     interceptor.afterChange(event, System.currentTimeMillis());
367                 }
368                 appliedChanges.add(event);
369             }
370             for (ProtectedList entry :  changedCollections.values())  {
371                 List<Object> originalList = entry.readOnly;
372                 for (PropertyChangeEvent event : entry.changeEvents) {
373                     if (event.getOldValue()==null) {
374                         originalList.add(event.getNewValue());
375                     } else {
376                         final Object toBeRemovedObj = event.getOldValue();
377                         if ( toBeRemovedObj instanceof ConfigBeanProxy ) {
378                             final Dom toBeRemoved = Dom.unwrap((ConfigBeanProxy)toBeRemovedObj);
379                             for (int index=0;index<originalList.size();index++) {
380                                 Object element = originalList.get(index);
381                                 Dom dom = Dom.unwrap((ConfigBeanProxy) element);
382                                 if (dom==toBeRemoved) {
383                                     originalList.remove(index);
384                                 }
385                             }
386                         }
387                         else if ( toBeRemovedObj instanceof String ) {
388                             final String toBeRemoved = (String)toBeRemovedObj;
389                             for (int index=0;index<originalList.size();index++) {
390                                 final String item = (String)originalList.get(index);
391                                 if (item.equals(toBeRemoved)) {
392                                     originalList.remove(index);
393                                 }
394                             }
395                         }
396                         else {
397                               throw new IllegalArgumentException();
398                         }
399                     }
400                     appliedChanges.add(event);
401                 }
402             }
403             changedAttributes.clear();
404             return appliedChanges;
405         } catch(TransactionFailure e) {
406             throw e;
407         } catch(Exception e) {
408             throw new TransactionFailure(e.getMessage(), e);
409         } finally {
410             bean.getLock().unlock();
411         }
412 
413     }

463         if (!(type.getName().equals(sourceBean.model.targetTypeName))) {

515     @Override
516     public synchronized boolean add(Object object) {
517         Object param = object;
518         Object handler = null;
519         try {
520             handler = Proxy.getInvocationHandler(object);
521         } catch(IllegalArgumentException e) {
522             // ignore, this is a leaf
523         }
524         if (handler!=null && handler instanceof WriteableView) {
525             ConfigBean master = ((WriteableView) handler).getMasterView();
526             String key = master.model.key;
527             if (key!=null) {
528                 // remove leading @
529                 key = key.substring(1);
530                 // check that we are not adding a duplicate key element
531                 String keyValue = ((WriteableView) handler).getPropertyValue(key);
532                 for (Object o : proxied) {
533                     // the proxied object can be a read-only or a writeable view, we need
534                     // to be careful
535                     // ToDo : we need to encasulate this test.
536                     String value;
537                     if (Proxy.getInvocationHandler(o) instanceof WriteableView) {
538                         value = ((WriteableView) Proxy.getInvocationHandler(o)).getPropertyValue(key);
539                     }  else {
540                         Dom cbo = Dom.unwrap((ConfigBeanProxy) o);
541                         value = cbo.attribute(key);
542                     }
543                     if (keyValue!=null && keyValue.equals(value)) {
544                         Dom parent = Dom.unwrap(readView);
545                         throw new IllegalArgumentException("A " + master.getProxyType().getSimpleName() +
546                                 " with the same key \"" + keyValue + "\" already exists in " +
547                                 parent.getProxyType().getSimpleName() + " " + parent.getKey()) ;
548 
549                     }
550                 }
551             }
552             param = ((WriteableView) handler).getMasterView().createProxy(master.type());
553 
554         }
555         changeEvents.add(new PropertyChangeEvent(defaultView, id, null, param));
556         return proxied.add(object);
557     }

593     @Override
594     public synchronized boolean remove(Object object) {
595         changeEvents.add(new PropertyChangeEvent(defaultView, id, object, null));
596 
597         try {
598             ConfigView handler = ((ConfigView) Proxy.getInvocationHandler(object)).getMasterView();
599             for (int index = 0 ; index<proxied.size() ; index++) {
600                 Object target = proxied.get(index);
601                 try {
602                     ConfigView targetHandler = ((ConfigView) Proxy.getInvocationHandler(target)).getMasterView();
603                     if (targetHandler==handler) {
604                         return (proxied.remove(index)!=null);
605                     }
606                 } catch(IllegalArgumentException ex) {
607                     // ignore
608                 }
609             }
610         } catch(IllegalArgumentException e) {
611             // ignore, this is a leaf
612             return proxied.remove(object);
613 
614         }
615         return false;
616     }

620     private String toCamelCase(String xmlName) {
621         StringTokenizer st =  new StringTokenizer(xmlName, "-");
622         StringBuffer camelCaseName = null;
623         if (st.hasMoreTokens()) {
624             camelCaseName = new StringBuffer(st.nextToken());
625         }
626         StringBuffer sb = null;
627         while (st.hasMoreTokens()) {
628             sb = new StringBuffer(st.nextToken());
629             char startChar = sb.charAt(0);
630             sb.setCharAt(0,Character.toUpperCase(startChar));
631             camelCaseName.append(sb);
632         }
633         return camelCaseName.toString();
634     }

636     private void handleValidation(ConfigModel.Property property, Object value)
637     throws ValidationException { 
638 
639         // First check for dataType constraints -- as was done for v3 Prelude
640         // see PrimitiveDataType.java
641         // These validations could be transformed into BV custom annotations
642         // such as AssertBoolean, AssertInteger etc. But since GUI and other
643         // config clients such as AMX need dataType key in @Attribute it's been
644         // decided to validate using existing annotation information
645         if (property instanceof ConfigModel.AttributeLeaf) {
646             ConfigModel.AttributeLeaf al = (ConfigModel.AttributeLeaf)property;
647             validateDataType(al, value.toString());
648         }
649 
650         Set constraintViolations =
651             beanValidator.validateValue(
652                 bean.getProxyType(), toCamelCase(property.xmlName()), value);
653 
654         if (constraintViolations != null) {
655             Iterator it = constraintViolations.iterator();
656             boolean violated = false;
657             String msg = "Constraints for this bean " +
658                    "violated. \n Message = ";
659             while (it.hasNext()) {
660                 violated = true;
661                 ConstraintViolation cv = (ConstraintViolation) it.next();
662                 msg = msg + cv.getPropertyPath() + " " + cv.getMessage();
663             }
664             if (violated) {
665                 throw new ValidationException(msg);
666             }
667         }
668     }

670     private void validateDataType(ConfigModel.AttributeLeaf al, String value)
671     throws ValidationException {
672 
673         if (value.startsWith("${") && value.endsWith("}"))
674           return;
675 
676         boolean isValid = String.class.getName().equals(al.dataType);
677         if ("int".equals(al.dataType) ||
678             "java.lang.Integer".equals(al.dataType))
679             isValid = representsInteger(value);
680         else if ("boolean".equals(al.dataType) ||
681                  "java.lang.Boolean".endsWith(al.dataType))
682             isValid = representsBoolean(value);
683         else if ("char".equals(al.dataType) ||
684                  "java.lang.Character".equals(al.dataType))
685             isValid = representsChar(value);        
686         if (!isValid) {
687             String msg = "Validation Failed: " +
688                          value + " is not of data type: " + al.dataType;
689             throw new ValidationException(msg);            
690         }
691     }

693     private boolean representsBoolean(String value) {
694         boolean isBoolean = 
695            "true".equalsIgnoreCase(value) || "false".equalsIgnoreCase(value);
696         return (isBoolean);
697     }

710             return false;
711         }

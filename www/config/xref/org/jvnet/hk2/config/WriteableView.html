60      public WriteableView(ConfigBeanProxy readView) {

67      public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
68  
69          ConfigModel.Property property = bean.toProperty(method);
70          if(property==null)
71              throw new IllegalArgumentException("No corresponding property found for method: "+method);
72  
73          if(args==null || args.length==0) {
74              // getter, maybe one of our changed properties
75              if (changedAttributes.containsKey(property.xmlName())) {
76                  // serve masked changes.
77                  return changedAttributes.get(property.xmlName()).getNewValue();
78              } else {
79                  // pass through.
80                  return getter(property, method.getGenericReturnType());
81              }
82          } else {
83              setter(property, args[0], method.getGenericParameterTypes()[0]);
84              return null;
85          }
86      }

88      public String getPropertyValue(String propertyName) {
89  
90          ConfigModel.Property prop = this.getProperty(propertyName);
91          if (prop!=null) {
92              if (changedAttributes.containsKey(prop.xmlName())) {
93                  // serve masked changes.
94                  return (String) changedAttributes.get(prop.xmlName()).getNewValue();
95              } else {
96                  return (String) getter(prop, String.class);
97              }
98          }
99          return null;
100     }

102     public Object getter(ConfigModel.Property property, java.lang.reflect.Type t) {
103         Object value =  bean._getter(property, t);
104         if (value instanceof List) {
105             if (!changedCollections.containsKey(property.xmlName())) {
106                 // wrap collections so we can record events on that collection mutation.
107                 changedCollections.put(property.xmlName(),
108                         new ProtectedList((List) value, defaultView, property.xmlName()));
109             }
110             return changedCollections.get(property.xmlName());
111         }
112         return value;
113     }

115     public void setter(ConfigModel.Property property, Object newValue, java.lang.reflect.Type t)  {
116         // are we still in a transaction
117         if (currentTx==null) {
118             throw new IllegalStateException("Not part of a transation");
119         }
120         try {
121             if (newValue != null)
122                 handleValidation(property, newValue);
123         } catch(Exception v) {
124             bean.getLock().unlock();
125             throw new RuntimeException(v);
126         }
127         // setter
128         Object oldValue = bean.getter(property, t);
129         if (newValue instanceof ConfigBeanProxy) {
130             ConfigView bean = (ConfigView) Proxy.getInvocationHandler((ConfigBeanProxy) newValue);
131             newValue = bean.getMasterView();
132         }
133         PropertyChangeEvent evt = new PropertyChangeEvent(defaultView,property.xmlName(), oldValue, newValue);
134         changedAttributes.put(property.xmlName(), evt);
135     }

142      * Enter a new Transaction, this method should return false if this object
143      * is already enlisted in another transaction, or cannot be enlisted with
144      * the passed transaction. If the object returns true, the object
145      * is enlisted in the passed transaction and cannot be enlisted in another
146      * transaction until either commit or abort has been issued.

148      * @param t the transaction to enlist with
149      * @return true if the enlisting with the passed transaction was accepted,

152     public synchronized boolean join(Transaction t) {
153         if (currentTx==null) {
154             currentTx = t;
155             t.addParticipant(this);
156             return true;
157         }
158         return false;
159     }

172     /***
173      * Commit this Transaction.
174      *
175      * @param t the transaction commiting.
176      * @throws TransactionFailure
177      *          if the transaction commit failed
178      */
179     public synchronized List<PropertyChangeEvent> commit(Transaction t) throws TransactionFailure {
180         if (currentTx==t) {
181             currentTx=null;
182         }
183         try {
184             List<PropertyChangeEvent> appliedChanges = new ArrayList<PropertyChangeEvent>();
185             for (PropertyChangeEvent event : changedAttributes.values()) {
186                 ConfigModel.Property property = bean.model.findIgnoreCase(event.getPropertyName());
187                 ConfigBeanInterceptor interceptor  = bean.getOptionalFeature(ConfigBeanInterceptor.class);
188                 try {
189                     if (interceptor!=null) {
190                         interceptor.beforeChange(event);
191                     }
192                 } catch (PropertyVetoException e) {
193                     throw new TransactionFailure(e.getMessage(), e);
194                 }
195                 property.set(bean, event.getNewValue());
196                 if (interceptor!=null) {
197                     interceptor.afterChange(event, System.currentTimeMillis());
198                 }
199                 appliedChanges.add(event);
200             }
201             for (ProtectedList entry :  changedCollections.values())  {
202                 List originalList = entry.readOnly;
203                 for (PropertyChangeEvent event : (List<PropertyChangeEvent>) entry.changeEvents) {
204                     if (event.getOldValue()==null) {
205                         originalList.add(event.getNewValue());
206                     } else {
207                         final Object toBeRemovedObj = event.getOldValue();
208                         if ( toBeRemovedObj instanceof ConfigBeanProxy ) {
209                             final Dom toBeRemoved = Dom.unwrap((ConfigBeanProxy)toBeRemovedObj);
210                             for (int index=0;index<originalList.size();index++) {
211                                 Object element = originalList.get(index);
212                                 Dom dom = Dom.unwrap((ConfigBeanProxy) element);
213                                 if (dom==toBeRemoved) {
214                                     originalList.remove(index);
215                                 }
216                             }
217                         }
218                         else if ( toBeRemovedObj instanceof String ) {
219                             final String toBeRemoved = (String)toBeRemovedObj;
220                             for (int index=0;index<originalList.size();index++) {
221                                 final String item = (String)originalList.get(index);
222                                 if (item.equals(toBeRemoved)) {
223                                     originalList.remove(index);
224                                 }
225                             }
226                         }
227                         else {
228                             throw new IllegalArgumentException();
229                         }
230                     }
231                     appliedChanges.add(event);
232                 }
233             }
234             changedAttributes.clear();
235             return appliedChanges;
236         } catch(TransactionFailure e) {
237             throw e;
238         } catch(Exception e) {
239             throw new TransactionFailure(e.getMessage(), e);
240         } finally {
241             bean.getLock().unlock();
242         }
243 
244     }

268     public <T extends ConfigBeanProxy> T allocateProxy(Class<T> type) throws TransactionFailure {
269         if (currentTx==null) {
270             throw new TransactionFailure("Not part of a transaction", null);
271         }
272         ConfigBean newBean = bean.allocate(type);
273         WriteableView writeableView = ConfigSupport.getWriteableView(newBean.getProxy(type), newBean);
274         writeableView.join(currentTx);

291     @SuppressWarnings("unchecked")    
292     public <T extends ConfigBeanProxy> T getProxy(Class<T> type) {
293         final ConfigBean sourceBean = getMasterView();
294         if (!(type.getName().equals(sourceBean.model.targetTypeName))) {
295             throw new IllegalArgumentException("This config bean interface is " + sourceBean.model.targetTypeName
296                     + " not "  + type.getName());
297         }
298         Class[] interfacesClasses = { type };
299         return (T) Proxy.newProxyInstance(type.getClassLoader(), interfacesClasses, this);

316     ProtectedList(List readOnly, ConfigBeanProxy parent, String id) {

346     @Override
347     public synchronized boolean add(Object object) {
348         Object param = object;
349         Object handler = null;
350         try {
351             handler = Proxy.getInvocationHandler(object);
352         } catch(IllegalArgumentException e) {
353             // ignore, this is a leaf
354         }
355         if (handler!=null && handler instanceof WriteableView) {
356             ConfigBean master = ((WriteableView) handler).getMasterView();
357             String key = master.model.key;
358             if (key!=null) {
359                 // remove leading @
360                 key = key.substring(1);
361                 // check that we are not adding a duplicate key element
362                 String keyValue = ((WriteableView) handler).getPropertyValue(key);
363                 for (Object o : proxied) {
364                     // the proxied object can be a read-only or a writeable view, we need
365                     // to be careful
366                     // ToDo : we need to encasulate this test.
367                     String value;
368                     if (Proxy.getInvocationHandler(o) instanceof WriteableView) {
369                         value = ((WriteableView) Proxy.getInvocationHandler(o)).getPropertyValue(key);
370                     }  else {
371                         Dom cbo = Dom.unwrap((ConfigBeanProxy) o);
372                         value = cbo.attribute(key);
373                     }
374                     if (keyValue!=null && keyValue.equals(value)) {
375                         Dom parent = Dom.unwrap(readView);
376                         throw new IllegalArgumentException("A " + master.getProxyType().getSimpleName() +
377                                 " with the same key \"" + keyValue + "\" already exists in " +
378                                 parent.getProxyType().getSimpleName() + " " + parent.getKey()) ;
379 
380                     }
381                 }
382             }
383             param = ((WriteableView) handler).getMasterView().createProxy(master.type());
384 
385         }
386         changeEvents.add(new PropertyChangeEvent(defaultView, id, null, param));
387         return proxied.add(object);
388     }

390     @Override
391     public synchronized void clear() {
392         // make a temporary list, iterating while removing doesn't work
393         final List allItems = new ArrayList( proxied );
394         for( final Object item : allItems ) {
395             remove( item );
396         }
397     }

399     @Override
400     public synchronized boolean retainAll( final Collection keepers ) {
401         final List toRemoveList = new ArrayList();
402         for( final Object iffy : proxied ) {
403             if ( ! keepers.contains(iffy) ) {
404                 toRemoveList.add(iffy);
405             }
406         }
407         final boolean changed = removeAll(toRemoveList);
408         
409         return changed;
410     }
411     
412     @Override
413     public synchronized boolean removeAll( final Collection goners ) {
414         boolean listChanged = false;
415         for( final Object goner : goners ) {
416             if ( remove(goner) ) {
417                 listChanged = true;
418             }
419         }
420 
421         return listChanged;
422     }
423 
424     @Override
425     public synchronized boolean remove(Object object) {
426         changeEvents.add(new PropertyChangeEvent(defaultView, id, object, null));
427 
428         try {
429             ConfigView handler = ((ConfigView) Proxy.getInvocationHandler(object)).getMasterView();
430             for (int index = 0 ; index<proxied.size() ; index++) {
431                 Object target = proxied.get(index);
432                 try {
433                     ConfigView targetHandler = ((ConfigView) Proxy.getInvocationHandler(target)).getMasterView();
434                     if (targetHandler==handler) {
435                         return (proxied.remove(index)!=null);
436                     }
437                 } catch(IllegalArgumentException ex) {
438                     // ignore
439                 }
440             }
441         } catch(IllegalArgumentException e) {
442             // ignore, this is a leaf
443             return proxied.remove(object);
444 
445         }
446         return false;
447     }

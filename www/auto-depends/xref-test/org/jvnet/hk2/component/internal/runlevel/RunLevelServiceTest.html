408   /**
409    * This guy tests the underling Recorder Ordering.
410    * 
411    * Note that: ServiceA -> ServiceB -> ServiceC
412    * 
413    * So, if we active B, A, then C manually, we still expect
414    * the recorder to have A, B, C only in that order.
415    * 
416    * This takes some "rigging" on the runLevelService (to ignore proceedTo)
417    * @throws NoSuchFieldException 
418    * @throws Exception 
419    */
420   @Test
421   public void serviceABC_startUp_and_shutDown() throws Exception {
422     installTestRunLevelService(false);
423     
424     RunLevelServiceBase.count = 0;
425     rls.proceedTo(9);
426     assertEquals(0, RunLevelServiceBase.count);
427 
428     recorders.clear();
429 
430     RunLevelServiceBase.count = 0;
431     rls.proceedTo(10);
432     
433     assertNotNull(h.getComponent(ServiceB.class));
434     assertNotNull(h.getComponent(ServiceA.class));
435     assertNotNull(h.getComponent(ServiceC.class));
436 
437     assertEquals(recorders.toString(), 1, recorders.size());
438     assertEquals("count", 3, RunLevelServiceBase.count);
439 
440     Recorder recorder = recorders.get(10);
441     assertNotNull(recorder);
442 
443     List<Inhabitant<?>> activations = recorder.getActivations();
444     assertFalse("activations empty", activations.isEmpty());
445     try {
446       Iterator<Inhabitant<?>> iter = activations.iterator();
447       iter.remove();
448       fail("expected read-only collection");
449     } catch (UnsupportedOperationException e) {
450       // expected
451     }
452     
453     activations = new ArrayList<Inhabitant<?>>(activations);
454     Iterator<Inhabitant<?>> iter = activations.iterator();
455     while (iter.hasNext()) {
456       Inhabitant<?> i = iter.next();
457       if (!i.typeName().contains(".RunLevelService")) {
458         iter.remove();
459       }
460     }
461     assertEquals("activations: " + activations, 3, activations.size());
462     
463     Inhabitant<?> iB = h.getInhabitantByContract(ServiceB.class.getName(), null);
464     Inhabitant<?> iA = h.getInhabitantByContract(ServiceA.class.getName(), null);
465     Inhabitant<?> iC = h.getInhabitantByContract(ServiceC.class.getName(), null);
466 
467     assertTrue(iB.isActive());
468     assertTrue(iA.isActive());
469     assertTrue(iC.isActive());
470     
471     iter = activations.iterator();
472     assertSame("order is important", iC, iter.next());
473     assertSame("order is important", iB, iter.next());
474     assertSame("order is important", iA, iter.next());
475 
476     RunLevelServiceBase a = (RunLevelServiceBase) iA.get();
477     RunLevelServiceBase b = (RunLevelServiceBase) iB.get();
478     RunLevelServiceBase c = (RunLevelServiceBase) iC.get();
479 
480     RunLevelServiceBase.count = 0;
481     defRLS.proceedTo(0);
482     assertFalse(iB.isActive());
483     assertFalse(iA.isActive());
484     assertFalse(iC.isActive());
485 
486     assertEquals(recorders.toString(), 1, recorders.size());
487     assertNotNull(recorders.toString(), recorders.get(10));
488     assertTrue(recorders.toString(), recorders.get(10).getActivations().isEmpty());
489 
490     assertEquals("count", 3, RunLevelServiceBase.count);
491     assertEquals("order is important on shutdown too: A", 0, a.countStamp);
492     assertEquals("order is important on shutdown too: B", 1, b.countStamp);
493     assertEquals("order is important on shutdown too: C", 2, c.countStamp);
494 
495     assertListenerState(true, false, false);
496   }

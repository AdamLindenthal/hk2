168     @Test
169     public void testFactoryProvidedContractProvisioningViaServicesApi() {
170         // binding defined using (annonymous) factory instance 
171         final FactoryProvidedContractA a = services.forContract(FactoryProvidedContractA.class).get();
172         assertInjectedInstance(FactoryProvidedContractAImpl.class, a);
173 
174         final Provider<FactoryProvidedContractA> pa = services.forContract(FactoryProvidedContractA.class).getProvider();
175         assertInjectedProvider(FactoryProvidedContractAImpl.class, pa);
176 
177         // binding defined using factory class
178         final FactoryProvidedContractB b = services.forContract(FactoryProvidedContractB.class).get();
179         assertInjectedInstance(FactoryProvidedContractBImpl.class, b);
180 
181         final Provider<FactoryProvidedContractB> pb = services.forContract(FactoryProvidedContractB.class).getProvider();
182         assertInjectedProvider(FactoryProvidedContractBImpl.class, pb);
183 
184         // binding defined using factory class and qualifier annotation
185         final FactoryProvidedContractC c_a = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerA.class).get();
186         assertInjectedInstance(FactoryProvidedContractCAImpl.class, c_a);
187 
188         final Provider<FactoryProvidedContractC> pc_a = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerA.class).getProvider();
189         assertInjectedProvider(FactoryProvidedContractCAImpl.class, pc_a);
190 
191         // binding defined using factory instance and qualifier annotation
192         final FactoryProvidedContractC c_b = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerB.class).get();
193         assertInjectedInstance(FactoryProvidedContractCBImpl.class, c_b);
194 
195         final Provider<FactoryProvidedContractC> pc_b = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerB.class).getProvider();
196         assertInjectedProvider(FactoryProvidedContractCBImpl.class, pc_b);
197 
198         // verifying null is returned for non-annotated binding that was not defined
199         final FactoryProvidedContractC c_default = services.forContract(FactoryProvidedContractC.class).get();
200         assertNull("No binding defined for the non-annotated contract. Provisioned instance should be null.", c_default);
201 
202         final Provider<FactoryProvidedContractC> pc_default = services.forContract(FactoryProvidedContractC.class).getProvider();
203         assertTrue("No binding defined for the non-annotated contract. Provider or returned instance should be null.", pc_default == null || pc_default.get() == null);
204     }

238     @Test
239     public void testConstructorBasedInjectionOnNonStaticInnerClass() {
240         Services s = HK2.get().create(services, new ConstructorInjectionTestModule());
241         Injector injector = s.forContract(Injector.class).get();
242         class TestClass {
243 
244             final ContractA ca;
245             final Factory<ContractA> cap;
246 
247             @Context
248             public TestClass(
249                     ContractA ca,
250                     Factory<ContractA> cap) {
251                 this.ca = ca;
252                 this.cap = cap;
253             }
254         }
255         TestClass ci = injector.inject(TestClass.class);
256 
257         assertInjectedInstance(ServiceA.class, ci.ca);
258         assertInjectedInstance(ServiceB.class, ci.ca.getB());
259         assertInjectedInstance(ClassX.class, ci.ca.getB().getX());
260 
261         assertInjectedFactory(ServiceA.class, ci.cap);
262         assertInjectedInstance(ServiceB.class, ci.cap.get().getB());
263         assertInjectedInstance(ClassX.class, ci.cap.get().getB().getX());
264     }

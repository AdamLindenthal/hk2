12  /***
13   * {@link ResourceManager} implementation that stores values in memory
14   * by using {@link ConcurrentHashMap}.
15   * 
16   * @author Kohsuke Kawaguchi
17   */
18  public final class InMemoryResourceManager implements ResourceManager {
19      // give a hint to JIT that they can inline methods.
20      private final ConcurrentHashMap<ResourceLocator,Object> components = new ConcurrentHashMap<ResourceLocator,Object>();
21  
22      public <T> void add(ResourceLocator<T> resourceInfo, T value) {
23          components.put(resourceInfo,value);
24      }
25  
26      @SuppressWarnings("unchecked")
27      public <T> T lookup(ResourceLocator<T> resourceInfo) {
28          Object value = components.get(resourceInfo);
29          assert resourceInfo.getType().isInstance(value);
30          return (T) value;
31      }
32  
33      // TODO: consider creating a map if this needs to be faster
34      @SuppressWarnings("unchecked")
35      public <T> Collection<T> lookupAll(Class<T> type) {
36          List r = new ArrayList();
37          for (Entry<ResourceLocator,Object> e : components.entrySet()) {
38              if(e.getKey().getType()==type)
39                  r.add(e.getValue());
40          }
41  
42          return r;
43      }
44  
45      public void remove(Object value) {
46          components.values().remove(value);
47      }
48  }

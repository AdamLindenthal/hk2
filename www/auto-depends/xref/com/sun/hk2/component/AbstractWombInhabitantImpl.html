37  package com.sun.hk2.component;

39  import org.jvnet.hk2.component.MultiMap;
40  import org.jvnet.hk2.component.PreDestroy;
41  import org.jvnet.hk2.component.Womb;
42  import org.jvnet.hk2.component.Inhabitant;
43  
44  /***
45   * Partial implementation of {@link Inhabitant} that delegates to {@link Womb}
46   * for object creation.
47   * <p>
48   * Derived types are expected to implement the {@link #get()} method and
49   * choose when to create an object. 
50   *
51   * @author Kohsuke Kawaguchi
52   */
53  abstract class AbstractWombInhabitantImpl<T> extends AbstractInhabitantImpl<T> {
54      protected final Womb<T> womb;
55  
56      protected AbstractWombInhabitantImpl(Womb<T> womb) {
57          this.womb = womb;
58      }
59  
60      public final String typeName() {
61          return womb.typeName();
62      }
63  
64      public final Class<T> type() {
65          return womb.type();
66      }
67  
68      public MultiMap<String, String> metadata() {
69          return womb.metadata();
70      }
71  
72      protected final void dispose(T object) {
73          if (object instanceof PreDestroy)
74              ((PreDestroy)object).preDestroy();
75      }
76  }

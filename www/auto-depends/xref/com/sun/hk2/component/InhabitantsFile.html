43  /***
44   * Inhabitants file.
45   *
46   * <p>
47   * Inhabitants file describe inhabitants (objects) that are to be placed into the habitat.
48   * This file is generated by the APT processor, which is integrated transparently into
49   * the build process by the HK2 maven plugin.
50   *
51   * <p>
52   * The file is by convention placed into <tt>/{@value #PATH}/xyz</tt> where 'xyz'
53   * portion is {@link InhabitantAnnotation#value() the identifier of the habitat}.
54   * This allows multiple different habitats to be created over the same set of classes.
55   * For example, there can be one habitat for the whole GF, then there are smaller habitats
56   * for each JAX-WS deployment.
57   *
58   * <h2>Format of the inhabitants file</h2>
59   * <p>
60   * The file is a UTF-8 encoded text file, and processing is line-based. A line
61   * that starts with '#' is treated as a comment and ignored.
62   * Other lines are assumed to be in <tt>key=value,key=value,...</tt> format.
63   * <tt>=value</tt> portion is optional, and this can be used to design keys
64   * that are conceptually boolean. The same key can appear multiple times.s
65   *
66   * <p>
67   * The following keys are defined:
68   *
69   * <table>
70   * <tr>
71   * <td>Key</td>
72   * <td>Value</td>
73   * </tr>
74   *
75   * <tr>
76   * <td>{@value #CLASS_KEY}</td>
77   * <td>
78   * The fully qualified class name of the inhabitant.
79   * </td>
80   * </tr>
81   *
82   * <tr>
83   * <td>{@value #INDEX_KEY}</td>
84   * <td>
85   * The index under which the inhabitant is registered.
86   * Multiple values allowed. A value is of the form:
87   * {@code PRIMARYNAME[:SUBNAME]}.
88   *
89   * This is used for all kinds of indexing needs, including
90   * {@link Contract} (where PRIMARYNAME is the FQCN of the contract name
91   * and SUBNAME is the component name.)
92   * </td>
93   * </tr>
94   *
95   * @author Kohsuke Kawaguchi
96   */
97  public class InhabitantsFile {
98      public static final String PATH = "META-INF/inhabitants";
99  
100     public static final String CLASS_KEY = "class";
101     public static final String INDEX_KEY = "index";
102     /***
103      * Used as a metadata for inhabitants of {@link CompanionSeed},
104      * to indicate what is the actual companion class. 
105      */
106     public static final String COMPANION_CLASS_KEY = "companionClass";
107     /***
108      * Used to point to {@link CageBuilder} for this component.
109      */
110     public static final String CAGE_BUILDER_KEY = "cageBuilder";
111     /***
112      * Used as metadafa for inhabitants of {@link CompanionSeed} to capture
113      * metadata of the actual companioin class.
114      */
115     public static final String COMPANION_CLASS_METADATA_KEY = "companionClassMetadata";
116 
117     /***
118      * Used as metadata for indentifying the type on which a {@link InhabitantAnnotation}
119      * was annotated.
120      */
121     public static final String TARGET_TYPE = "targetType";
122 }

56  /**
57   * Marks the companion relationship with another component.
58   *
59   * <p>
60   * Implementations of this contract are generated for each component
61   * with the {@link CompanionOf} annotation.  
62   *
63   * @author Kohsuke Kawaguchi
64   */
65  @Contract
66  @InhabitantAnnotation("default")
67  @CagedBy(CompanionSeed.Registerer.class)
68  public @interface CompanionSeed {
69      /**
70       * The type of the lead class.
71       */
72      @Index
73      @InhabitantMetadata("lead")
74      Class<?> lead();
75  
76      /**
77       * Other metadata to capture the companion class.
78       */
79      @InhabitantMetadata("companionClass")
80      Class<?> companion();
81  
82      /**
83       * When {@link CompanionSeed} enters habitat, look
84       * for existing inhabitants and make sure all of them
85       * get its corresponding companions from this new seed.
86       *
87       * <p>
88       * Normal {@link CageBuilder}s register themselves as components,
89       * but this one is defined in HK2 so we have to get in by a special means.
90       */
91      public static final class Registerer<T> implements CageBuilder<T> {
92          private final Habitat habitat;
93  
94          public Registerer(Habitat habitat) {
95              this.habitat = habitat;
96          }
97  
98          public void onEntered(Inhabitant<T> i) {
99              // and if a companion seed is added to something else, make sure
100             // existing lead inhabitants will get this as a companion
101             assert i.metadata()!=null;
102             for (Inhabitant<?> lead : habitat.getInhabitantsByType(i.metadata().getOne("lead")))
103                 lead.setCompanions(cons(lead.companions(),createCompanion(habitat,lead,i)));
104         }
105 
106         /**
107          * Allocates a new read-only list by adding one more element.
108          */
109         @SuppressWarnings({ "unchecked", "hiding" })
110         private <T> List<T> cons(Collection<T> list, T oneMore) {
111             int sz = list.size();
112             Object[] a = list.toArray(new Object[sz+1]);
113             a[sz]=oneMore;
114             return (List<T>) Arrays.asList(a);
115         }
116 
117         /**
118          * Creates a companion inhabitant from the inhabitant of a {@link CompanionSeed},
119          * to be associated with a lead component.
120          */
121         public static Inhabitant<?> createCompanion(Habitat habitat, final Inhabitant<?> lead, final Inhabitant<?> seed) {
122             Holder<ClassLoader> cl = new Holder<ClassLoader>() {
123                     public ClassLoader get() {
124                         return seed.type().getClassLoader();
125                     }
126                 };
127 
128             // this is the name of the class with @CompanionOf
129             String fqcn = seed.metadata().getOne(COMPANION_CLASS_KEY);
130 
131             String metadataLine = seed.metadata().getOne(COMPANION_CLASS_METADATA_KEY);
132             MultiMap<String,String> metadata;
133             if(metadataLine==null)
134                 metadata = MultiMap.emptyMap();
135             else
136                 metadata = InhabitantFileBasedParser.buildMetadata(new KeyValuePairParser(metadataLine));
137 
138             Inhabitant<?> ci = Inhabitants.createInhabitant(habitat, cl, fqcn, metadata, lead, null);
139             habitat.add(ci);
140             return ci;
141         }
142     }

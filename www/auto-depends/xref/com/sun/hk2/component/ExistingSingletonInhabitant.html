37  package com.sun.hk2.component;
38  
39  import org.jvnet.hk2.component.ComponentException;
40  import org.jvnet.hk2.component.Inhabitant;
41  import org.jvnet.hk2.component.MultiMap;
42  
43  /***
44   * {@link Inhabitant} built around an object that already exists.
45   *
46   * @author Kohsuke Kawaguchi
47   */
48  public class ExistingSingletonInhabitant<T> extends AbstractInhabitantImpl<T> {
49      private final T object;
50      private final Class<T> type;
51      private final MultiMap<String,String> metadata;
52  
53      public ExistingSingletonInhabitant(T object) {
54          this((Class<T>)object.getClass(),object);
55      }
56  
57      public ExistingSingletonInhabitant(Class<T> type, T object) {
58          this(type,object,MultiMap.<String,String>emptyMap());
59      }
60  
61      public ExistingSingletonInhabitant(Class<T> type, T object, MultiMap<String,String> metadata) {
62          this.type = type;
63          this.object = object;
64          this.metadata = metadata;
65      }
66  
67      public String typeName() {
68          return type.getName();
69      }
70  
71      public Class<T> type() {
72          return type;
73      }
74  
75      public MultiMap<String, String> metadata() {
76          return metadata;
77      }
78  
79      public T get(Inhabitant onBehalfOf) throws ComponentException {
80          return object;
81      }
82  
83      public boolean isInstantiated() {
84          return true;
85      }
86  
87      public void release() {
88          // since we are working on the existing object,
89          // we can't release its instance.
90          // (technically it's possible to invoke PreDestroy,
91          // not clear which is better --- you can argue both ways.)
92      }
93  }

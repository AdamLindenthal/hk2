37  package com.sun.hk2.component;
38  
39  import org.jvnet.hk2.component.ComponentException;
40  import org.jvnet.hk2.component.Inhabitant;
41  import org.jvnet.hk2.component.MultiMap;
42  
43  /**
44   * {@link Inhabitant} built around an object that already exists.
45   *
46   * @author Kohsuke Kawaguchi
47   */
48  public class ExistingSingletonInhabitant<T> extends AbstractInhabitantImpl<T> {
49      private final T object;
50      private final Class<T> type;
51      private final MultiMap<String,String> metadata;
52  
53      public ExistingSingletonInhabitant(T object) {
54          this((Class<T>)object.getClass(),object);
55      }
56  
57      public ExistingSingletonInhabitant(Class<T> type, T object) {
58          this(type,object,MultiMap.<String,String>emptyMap());
59      }
60  
61      public ExistingSingletonInhabitant(Class<T> type, T object, MultiMap<String,String> metadata) {
62          this.type = type;
63          this.object = object;
64          this.metadata = (null == metadata) ? new MultiMap<String,String>() : metadata;
65      }
66      
67      @Override
68      public String toString() {
69          return getClass().getSimpleName() + "-" + System.identityHashCode(this) + 
70            "(" + get(null) + "," + metadata + ")";
71      }
72  
73      @Override
74      public String typeName() {
75          return type.getName();
76      }
77  
78      @Override
79      public Class<T> type() {
80          return type;
81      }
82  
83      @Override
84      public MultiMap<String, String> metadata() {
85          return metadata;
86      }
87  
88      public T get(Inhabitant onBehalfOf) throws ComponentException {
89          return object;
90      }
91  
92      @Override
93      public boolean isInstantiated() {
94          return true;
95      }
96  
97      @Override
98      public void release() {
99          // since we are working on the existing object,
100         // we can't release its instance.
101         // (technically it's possible to invoke PreDestroy,
102         // not clear which is better --- you can argue both ways.)
103     }
104 }

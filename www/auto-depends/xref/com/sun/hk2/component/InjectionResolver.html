44  import java.lang.annotation.Annotation;
45  import java.lang.reflect.AnnotatedElement;
46  import java.lang.reflect.Method;

48  /***
49   * Implementation of this abstract class are handling injection resolution
50   * for a particular injection annotation {@see Inject}
51   *
52   * Injection targets are identified by the generic parameter and the constructor
53   * of this class. Potential injection targets are fields and methods of the
54   * injected type.
55   *
56   * @param <U> U is the annotation used to identify the injection targets.
57   *
58   * @author Jerome Dochez
59   */
60  public abstract class InjectionResolver<U extends Annotation> {
61      
62      public final Class<U> type;
63  
64      /***
65       * Construct a resolver with a particular injection type
66       * @param type the injection annotation type
67       */
68      public InjectionResolver(Class<U> type) {
69          this.type = type;
70      }
71  
72      /***
73       * Returns the setter method responsible for setting the resource identified by the
74       * passed annotation on the passed annotated method.
75       *
76       * This is useful when the annotation is specified on the getter for instance (due
77       * to external specification requirements for instance) while the setter should be used if
78       * values must be set using this injection resolver.
79       *
80       * By default, the setter method is the annotated method.
81       *
82       * @param annotated is the annotated {@link java.lang.reflect.Method}
83       * @param annotation the annotation on the method
84       * @return the setter method to use for injecting the annotation identified resource
85       */
86      public Method getSetterMethod(Method annotated, U annotation) {
87          return annotated;
88      }
89  
90      /***
91       * Returns true if the resolution of this injection identified by the
92       * passed annotation instance is optional
93       * @param annotated is the annotated java element {@link java.lang.reflect.Method}
94       * or {@link java.lang.reflect.Field}
95       * @param annotation the injection metadata
96       * @return true if the {@see getValue()} can return null without generating a
97       * faulty injection operation
98       */
99      public boolean isOptional(AnnotatedElement annotated, U annotation) {
100         return false;
101     }
102 
103     /***
104      * Returns the value to inject in the field or method of component annotated with
105      * the annotated annotation.
106      *
107 //     * @param habitat the habitat containing component / inhabitant
108      * @param component injection target instance
109      * @param onBehalfOf inhabitant doing the injection for
110      * @param annotated is the annotated java element {@link java.lang.reflect.Method}
111      * or {@link java.lang.reflect.Field}
112      * @param type type of the expected return
113      * @return the resource to be injected
114      * @throws ComponentException if the resource cannot be located.
115      */
116     public abstract <V> V getValue(Object component,
117         Inhabitant<?> onBehalfOf,
118         AnnotatedElement annotated,
119         Class<V> type) throws ComponentException;
120 }

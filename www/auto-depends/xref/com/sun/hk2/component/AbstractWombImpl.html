37  package com.sun.hk2.component;

39  import java.util.Collection;
40  import java.util.logging.Level;
41  import java.util.logging.Logger;
42  
43  import org.jvnet.hk2.component.*;
44  import org.jvnet.hk2.tracing.TracingThreadLocal;
45  import org.jvnet.hk2.tracing.TracingUtilities;

47  /**
48   * @author Kohsuke Kawaguchi
49   */
50  @SuppressWarnings("unchecked")
51  public abstract class AbstractWombImpl<T> extends AbstractInhabitantImpl<T> implements Womb<T> {
52      private final static Logger logger = Logger.getLogger(AbstractWombImpl.class.getName());
53    
54      protected final Class<T> type;
55      protected final Habitat habitat; 
56      private final MultiMap<String,String> metadata;
57  
58      public AbstractWombImpl(Class<T> type, Habitat habitat, MultiMap<String,String> metadata) {
59          this.type = type;
60          this.habitat = habitat;
61          this.metadata = metadata;
62      }
63  
64      public final String typeName() {
65          return type.getName();
66      }
67  
68      public final Class<T> type() {
69          return type;
70      }
71  
72      public final T get(Inhabitant onBehalfOf) throws ComponentException {
73          try {
74              if (TracingUtilities.isEnabled())
75                  TracingThreadLocal.get().push(this);
76  
77              T o = create(onBehalfOf);
78              logger.log(Level.FINER, "created object {0}", o);
79              initialize(o, onBehalfOf);
80              return o;
81          } finally {
82              if (TracingUtilities.isEnabled())
83                  TracingThreadLocal.get().pop();
84          }
85      }
86  
87      public boolean isInstantiated() {
88          return true;
89      }
90  
91      @Override
92      public void initialize(T t, Inhabitant onBehalfOf) throws ComponentException {
93        // I could rely on injection, but the algorithm is slow enough for now that I
94        // need a faster scheme.
95        if (t instanceof InhabitantRequested) {
96            ((InhabitantRequested) t).setInhabitant(onBehalfOf);
97        }
98      }
99  
100     public void release() {
101         // Womb creates a new instance every time,
102         // so there's nothing to release here.
103     }
104 
105     public MultiMap<String, String> metadata() {
106         return metadata;
107     }
108 
109     /**
110      * Performs resource injection on the given instance from the given habitat.
111      *
112      * <p>
113      * This method is an utility method for subclasses for performing injection.
114      */
115     protected void inject(Habitat habitat, T t, Inhabitant<?> onBehalfOf) {
116         logger.log(Level.FINER, "injection starting on {0}", t);
117 
118         InjectionManager injectionMgr = createInjectionManager();
119         
120         Collection<InjectionResolver> targets = habitat.getAllByType(InjectionResolver.class);
121 //        Collection<InjectionResolver> otherTargets = habitat.getAllByContract(InjectionResolver.class);
122         assert(!targets.isEmpty());
123         injectionMgr.inject(t, onBehalfOf, targets.toArray(new InjectionResolver[targets.size()]));
124 
125         // postContruct call if any
126         if (t instanceof PostConstruct) {
127             logger.log(Level.FINER, "calling PostConstruct on {0}", t);
128             ((PostConstruct)t).postConstruct();
129         }
130         
131         logger.log(Level.FINER, "injection finished on {0}", t);
132     }
133 
134     protected InjectionManager createInjectionManager() {
135       return new InjectionManager();
136     }

77      /**
78       * Obtains the value to inject, based on the type and {@link Inject} annotation.
79       */
80      @Override
81      public <V> V getValue(final Object component,
82                            final Inhabitant<?> onBehalfOf,
83                            final AnnotatedElement target,
84                            final Type genericType,
85                            final Class<V> type) throws ComponentException {
86          final Inject inject = target.getAnnotation(Inject.class);
87          final Callable<V> callable = new Callable<V>() {
88              @Override
89              public V call() throws ComponentException {
90                  V result=null;
91  
92                  if (type.isArray()) {
93                      result = getArrayInjectValue(habitat, component, onBehalfOf, target, genericType, type);
94                  } else {
95                      if (Types.isSubClassOf(type, org.glassfish.hk2.Factory.class)) {
96                          result = getHolderInjectValue(habitat, component, onBehalfOf, target, genericType, type, inject);
97                      } else {
98                          if (genericType instanceof TypeVariable) {
99                              // ok this is a bit more complicated, the user wants us to inject a parameterized type
100                             // so we need to look at the class declaration to find out in the index of the
101                             // declaration of that parameterized type and reconcile it with our inhabitant metadata.
102                             TypeVariable[] typeVariables = component.getClass().getTypeParameters();
103                             for (int i=0;i<typeVariables.length;i++) {
104                                 if (typeVariables[i].getName().equals(((TypeVariable) genericType).getName())) {
105                                     // that's the parameterized type we are looking for.
106                                     String parameterizedType = onBehalfOf.metadata().get(InhabitantsFile.PARAMETERIZED_TYPE).get(i);
107                                     try {
108                                         ClassLoader classLoader = System.getSecurityManager()==null?
109                                                 component.getClass().getClassLoader():
110                                                 AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
111                                                     @Override
112                                                     public ClassLoader run() {
113                                                         return component.getClass().getClassLoader();
114                                                     }
115                                                 });
116                                         Class<?> clazz = classLoader.loadClass(parameterizedType);
117                                         ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, clazz, habitat.isContract(clazz));
118                                         populateContractLocator(contractLocator, target, inject);
119                                         result = (V) contractLocator.getByType(org.jvnet.tiger_types.Types.erasure(genericType));
120                                     } catch(ClassNotFoundException e) {
121                                         Logger.getAnonymousLogger().warning("Cannot load class " + parameterizedType);
122                                         return null;
123                                     }
124                                 }
125                             }
126                         } else {
127                             ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, genericType, habitat.isContractExt(genericType));
128                             populateContractLocator(contractLocator, target, inject);
129                             result = (V) contractLocator.getByType(org.jvnet.tiger_types.Types.erasure(genericType));
130                         }
131                     }
132                 }
133 
134                 return result==null?null:validate(component, onBehalfOf, result);
135             }
136         };
137 
138         try {
139             if (habitat.isContextualFactoriesPresent()) {
140                 InjectionPoint ip = new InjectionPointImpl(component, target, type, inject, onBehalfOf);
141                 return Hk2ThreadContext.captureIPandRun(ip, callable);
142             } else {
143                 return callable.call();
144             }
145         } catch (Exception e) {
146             if (e instanceof ComponentException) throw ComponentException.class.cast(e);
147             throw new ComponentException(e);
148         }
149     }

151     protected <V> V getArrayInjectValue(Habitat habitat,
152                                         Object component,
153                                         Inhabitant<?> onBehalfOf,
154                                         AnnotatedElement target,
155                                         Type genericType,
156                                         Class<V> type) {
157         V result;
158         Class<?> ct = type.getComponentType();
159 
160         Collection<?> instances;
161         if (habitat.isContract(ct)) {
162             instances = getAllByContract(onBehalfOf, habitat, ct);
163         } else {
164             instances = getAllByType(onBehalfOf, habitat, ct);
165         }
166 
167         result = type.cast(instances.toArray((Object[]) Array.newInstance(ct, instances.size())));
168         // TODO: validate() here too
169         return result;
170     }

172     protected <V> V getHolderInjectValue(final Habitat habitat,
173                                          final Object component,
174                                          final Inhabitant<?> onBehalfOf,
175                                          final AnnotatedElement target,
176                                          final Type genericType,
177                                          final Class<V> type,
178                                          final Inject inject) throws ComponentException {
179 
180         final Type t = Types.getTypeArgument(genericType, 0);
181 
182         if (habitat.isContractExt(t)) {
183             ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, t, true);
184             populateContractLocator(contractLocator, target, inject);
185             return type.cast(contractLocator.getProvider());
186         }
187 
188         // the receiver maybe requesting the inhabitant pointing to itself to have
189         // access to its own metadata.
190         final Class<?> finalType = Types.erasure(t);
191         try {
192             if (finalType.cast(component) != null) {
193                 return type.cast(onBehalfOf);
194             }
195         } catch (ClassCastException e) {
196             // ignore
197         }
198 
199         ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, finalType, false);
200         return type.cast(contractLocator.getProvider());
201     }

246     @SuppressWarnings("unchecked")
247     protected <V> Collection<V> manage(Inhabitant<?> onBehalfOf, Iterable<?> inhabitants) {
248         if (null == inhabitants) {
249             return null;
250         }
251 
252         final ArrayList<V> managed = new ArrayList<V>();
253         for (Object iObj : inhabitants) {
254             Inhabitant<V> i = (Inhabitant<V>) iObj;
255             managed.add((V) manage(onBehalfOf, i).get());
256         }
257 
258         return managed;
259     }

272     void populateContractLocator(ContractLocatorImpl<?> contractLocator, AnnotatedElement target, Inject inject) {
273         Named named = target.getAnnotation(Named.class);
274         String name = inject.name();
275         if (named!=null && !inject.name().isEmpty()) {
276             throw new RuntimeException("Field or method [" + target.toString() + "] is annotated with both a @Named" +
277                     " annotation as well as a @Inject.name() value, please choose");
278         }
279         if (named!=null) {
280             name = named.value();
281         }
282         contractLocator.named(name);
283 
284         // now we need to obtain all qualifiers on the injection target so we narrow the search.
285         Annotation[] targetAnnotations = target.getAnnotations();
286 
287         // if there is only one annotation, it's @Inject, we can ignore qualifiers narrowing
288         if (name.isEmpty() && targetAnnotations.length > 1) {
289             for (Annotation annotation : target.getAnnotations()) {
290                 for (Annotation annotationAnnotation : annotation.annotationType().getAnnotations()) {
291                     if (annotationAnnotation.annotationType()==Qualifier.class) {
292                         contractLocator.annotatedWith(annotation.annotationType());
293                     }
294                 }
295             }
296         }

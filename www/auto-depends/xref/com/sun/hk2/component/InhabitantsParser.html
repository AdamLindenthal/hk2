37  package com.sun.hk2.component;

39  import org.jvnet.hk2.component.Habitat;
40  import org.jvnet.hk2.component.Inhabitant;
41  
42  import java.io.IOException;
43  import java.util.Collections;
44  import java.util.HashMap;
45  import java.util.HashSet;
46  import java.util.Iterator;
47  import java.util.Map;
48  import java.util.Set;

50  /***
51   * Parses <tt>/META-INF/inhabitants</tt> and populate {@link Habitat}.
52   *
53   * <p>
54   * This class can be subclasses to customize the parsing behavior, which is useful
55   * for ignoring some components.
56   *
57   * @author Kohsuke Kawaguchi
58   * @author Jerome Dochez
59   */
60  public class InhabitantsParser {
61      public final Habitat habitat;
62  
63      /***
64       * Fully qualified class names of components to be replaced, to {@link Class} that replaces them.
65       * If the value is null, that means just drops it without a replacement.
66       *
67       * @see #drop(Class)
68       * @see #replace(Class, Class)
69       */
70      // Whether this feature should belong to this base class is arguable --- perhaps a better
71      // approach is to create a sub classs that does it?
72      private final Map<String,Class> replacements = new HashMap<String,Class>();
73  
74      public InhabitantsParser(Habitat habitat) {
75          this.habitat = habitat;
76      }
77  
78      /***
79       * Tells {@link InhabitantsParser} that if it encounters the specified component
80       * while parsing inhabitants file,
81       * simply drop it and pretend that such an inhabitant had never existed.
82       *
83       * <p>
84       * This is useful when the application that's hosting an HK2 environment
85       * wants to tweak the inhabitant population at sub-module level.
86       */
87      public void drop(Class component) {
88          drop(component.getName());
89      }
90  
91      public void drop(String fullyQualifiedClassName) {
92          replace(fullyQualifiedClassName,null);
93      }
94  
95      /***
96       * Tells {@link InhabitantsParser} that if it encounters the specified component
97       * while parsing inhabitants file,
98       * ignore the one in the inhabitants file and instead insert the specified 'new' component.
99       *
100      * <p>
101      * This is useful when the application that's hosting an HK2 environment
102      * wants to tweak the inhabitant population at sub-module level.
103      */
104     public void replace(Class oldComponent, Class newComponent) {
105         replace(oldComponent.getName(),newComponent);
106     }
107 
108     public void replace(String oldComponentFullyQualifiedClassName, Class newComponent) {
109         replacements.put(oldComponentFullyQualifiedClassName,newComponent);
110     }
111 
112     /***
113      * Parses the inhabitants file (which is represented by {@link InhabitantsScanner}.
114      *
115      * <p>
116      * All the earlier drop/replace commands will be honored during this process.
117      */
118     @SuppressWarnings("unchecked")
119     public void parse(Iterable<InhabitantParser> scanner, Holder<ClassLoader> classLoader) throws IOException {
120         if (scanner==null)
121             return;
122         for( InhabitantParser inhabitantParser : scanner) {
123             String typeName = inhabitantParser.getImplName();
124             if(replacements.containsKey(typeName)) {
125                 // create a replacement instead
126                 Class<?> target = replacements.get(typeName);
127                 if(target!=null) {
128                     inhabitantParser.setImplName(target.getName());
129                     Inhabitant i = Inhabitants.create(target,habitat,inhabitantParser.getMetaData());
130                     add(i, inhabitantParser);
131                     // add index so that the new component can be looked up by the name of the old component.
132                     habitat.addIndex(i,typeName,null);
133                 }
134             } else {
135                 Set<String> indicies = new HashSet<String>();
136                 Iterator<String> iter = inhabitantParser.getIndexes().iterator();
137                 while (iter.hasNext()) {
138                   indicies.add(iter.next());
139                 }
140                 Inhabitant<?> i = com.sun.hk2.component.Inhabitants.
141                     createInhabitant(habitat, classLoader, typeName,
142                         inhabitantParser.getMetaData(), null,
143                         Collections.unmodifiableSet(indicies));
144                 add(i, inhabitantParser);
145             }
146         }
147     }
148 
149     /***
150      * Adds the given inhabitant to the habitat, with all its indices.
151      */
152     protected void add(Inhabitant<?> i, InhabitantParser parser) {
153         habitat.add(i);
154 
155         for (String v : parser.getIndexes()) {
156             // register inhabitant to the index
157             int idx = v.indexOf(':');
158             if(idx==-1) {
159                 // no name
160                 habitat.addIndex(i,v,null);
161             } else {
162                 // v=contract:name
163                 String contract = v.substring(0, idx);
164                 String name = v.substring(idx + 1);
165                 habitat.addIndex(i, contract, name);
166             }
167         }
168     }

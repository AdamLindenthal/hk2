37  package com.sun.hk2.component;
38  
39  import org.jvnet.hk2.component.ComponentException;
40  import org.jvnet.hk2.component.Habitat;
41  import org.jvnet.hk2.component.Inhabitant;
42  import org.jvnet.hk2.component.MultiMap;
43  import org.jvnet.hk2.component.Womb;
44  import org.jvnet.hk2.component.Wombs;
45  
46  /***
47   * @author Kohsuke Kawaguchi
48   */
49  public class LazyInhabitant<T> extends EventPublishingInhabitant<T> {
50      private final String typeName;
51  
52      /***
53       * Lazy reference to {@link ClassLoader}.
54       */
55      private final Holder<ClassLoader> classLoader;
56  
57      protected final Habitat habitat;
58  
59      private final MultiMap<String,String> metadata;
60      
61      private final Inhabitant<?> lead;
62  
63      
64      public LazyInhabitant(Habitat habitat, Holder<ClassLoader> cl, String typeName, MultiMap<String,String> metadata) {
65          this(habitat, cl, typeName, metadata, null);
66      }
67  
68      public LazyInhabitant(Habitat habitat, Holder<ClassLoader> cl, String typeName, MultiMap<String,String> metadata, Inhabitant<?> lead) {
69        assert metadata!=null;
70        this.habitat = habitat;
71        this.classLoader = cl;
72        this.typeName = typeName;
73        this.metadata = metadata;
74        this.lead = lead;
75      }
76  
77      @Override
78      public Inhabitant<?> lead() {
79          return lead;
80      }
81      
82      @Override
83      public String typeName() {
84          return typeName;
85      }
86      
87      @Override
88      public Class<T> type() {
89        // fetching is too heavy of an operation because it will activate/write the class
90  //        fetch();
91  
92          Inhabitant<T> real = this.real;
93          if (null != real) {
94              return real.type();
95          } else {
96              return loadClass();
97          }
98      }
99  
100     @Override
101     public MultiMap<String,String> metadata() {
102         return metadata;
103     }
104 
105     @Override
106     protected synchronized void fetch() {
107         if (null == real) {
108           Class<T> c = loadClass();
109           real = Inhabitants.wrapByScope(c,createWomb(c),habitat);
110         }
111     }
112     
113     public final ClassLoader getClassLoader() {
114         return classLoader.get();
115     }
116     
117     @SuppressWarnings("unchecked")
118     private Class<T> loadClass() {
119         final ClassLoader cl = getClassLoader();
120         try {
121             Class<T> c = (Class<T>) cl.loadClass(typeName);
122             return c;
123         } catch (ClassNotFoundException e) {
124             throw new ComponentException("Failed to load "+typeName+" from " + cl, e);
125         }
126     }
127 
128     @Override
129     public synchronized void release() {
130         super.release();
131         real = null;
132     }
133 
134     /***
135      * Creates {@link Womb} for instantiating objects.
136      */
137     protected Womb<T> createWomb(Class<T> c) {
138         return Wombs.create(c,habitat,metadata);
139     }

141 }

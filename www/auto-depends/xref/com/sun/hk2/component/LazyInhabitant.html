37  package com.sun.hk2.component;
38  
39  import org.jvnet.hk2.component.ComponentException;
40  import org.jvnet.hk2.component.Habitat;
41  import org.jvnet.hk2.component.Inhabitant;
42  import org.jvnet.hk2.component.Inhabitants;
43  import org.jvnet.hk2.component.MultiMap;
44  import org.jvnet.hk2.component.Womb;
45  import org.jvnet.hk2.component.Wombs;
46  
47  /***
48   * @author Kohsuke Kawaguchi
49   */
50  public class LazyInhabitant<T> extends AbstractInhabitantImpl<T> {
51      private final String typeName;
52      /***
53       * Real {@link Inhabitant} object. Lazily created.
54       */
55      private volatile Inhabitant<T> real;
56      /***
57       * Lazy reference to {@link ClassLoader}.
58       */
59      private final Holder<ClassLoader> classLoader;
60  
61      protected final Habitat habitat;
62  
63      private final MultiMap<String,String> metadata;
64  
65      public LazyInhabitant(Habitat habitat, Holder<ClassLoader> cl, String typeName, MultiMap<String,String> metadata) {
66          assert metadata!=null;
67          this.habitat = habitat;
68          this.classLoader = cl;
69          this.typeName = typeName;
70          this.metadata = metadata;
71      }
72  
73      public String typeName() {
74          return typeName;
75      }
76  
77      public Class<T> type() {
78          fetch();
79          return real.type();
80      }
81  
82      public MultiMap<String,String> metadata() {
83          return metadata;
84      }
85  
86      @SuppressWarnings("unchecked")
87      private synchronized void fetch() {
88          if(real!=null)  return;
89  
90          try {
91              Class<T> c = (Class<T>) classLoader.get().loadClass(typeName);
92              real = Inhabitants.wrapByScope(c,createWomb(c),habitat);
93          } catch (ClassNotFoundException e) {
94              throw new ComponentException("Failed to load "+typeName+" from "+classLoader.get(),e);
95          }
96      }
97  
98      /***
99       * Creates {@link Womb} for instantiating objects.
100      */
101     protected Womb<T> createWomb(Class<T> c) {
102         return Wombs.create(c,habitat,metadata);
103     }
104 
105     public T get(Inhabitant onBehalfOf) throws ComponentException {
106         fetch();
107         return real.get(onBehalfOf);
108     }
109 
110     public void release() {
111         if(real!=null)
112             real.release();
113     }
114 
115     public boolean isInstantiated() {
116         return (real!=null && real.isInstantiated());
117     }
118 }

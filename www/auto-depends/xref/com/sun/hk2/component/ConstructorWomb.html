37  package com.sun.hk2.component;

39  import org.jvnet.hk2.annotations.Scoped;
40  import org.jvnet.hk2.component.ComponentException;
41  import org.jvnet.hk2.component.Habitat;
42  import org.jvnet.hk2.component.Inhabitant;
43  import org.jvnet.hk2.component.MultiMap;
44  import org.jvnet.hk2.component.PerLookup;
45  import org.jvnet.hk2.component.Scope;
46  import org.jvnet.hk2.component.Singleton;
47  
48  import java.security.AccessController;
49  import java.security.PrivilegedAction;
50  
51  /***
52   * Creates an object from its constructor.
53   * @author Kohsuke Kawaguchi
54   */
55  @SuppressWarnings("unchecked")
56  public class ConstructorWomb<T> extends AbstractWombImpl<T> {
57      private final ScopeInstance singletonScope;
58  
59      public ConstructorWomb(Class<T> type, Habitat habitat, MultiMap<String,String> metadata) {
60          super(type, habitat, metadata);
61          singletonScope = habitat.singletonScope;
62      }
63  
64      public T create(Inhabitant onBehalfOf) throws ComponentException {
65          try {
66              return type.newInstance();
67          } catch (InstantiationException e) {
68              throw new ComponentException("Failed to create "+type,e);
69          } catch (IllegalAccessException e) {
70              throw new ComponentException("Failed to create "+type,e);
71          } catch (LinkageError e) {
72              throw new ComponentException("Failed to create "+type,e);
73          } catch (RuntimeException e) {
74              throw new ComponentException("Failed to create "+type,e);
75          }
76      }
77  
78      public void initialize(final T t, final Inhabitant onBehalfOf) throws ComponentException {
79          super.initialize(t, onBehalfOf);
80          
81          Scoped scoped = t.getClass().getAnnotation(Scoped.class);
82          ScopeInstance si = (scoped == null ? singletonScope : getScope(scoped));
83          if (System.getSecurityManager() != null) {
84            AccessController.doPrivileged(new PrivilegedAction() {
85                // privileged required for running with SecurityManager ON
86                public java.lang.Object run() {
87                    inject(habitat, t, onBehalfOf);
88                    return null;
89                }
90            });
91          } else {
92            inject(habitat, t, onBehalfOf);
93          }
94      }
95  
96      /***
97       * Determines the {@link ScopeInstance} that stores the component.
98       *
99       * @return
100      *      null for prototype scope. (Note that in {@link Scope#current()}
101      *      null return value is an error.)
102      */
103     private ScopeInstance getScope(Scoped svc) throws ComponentException {
104         Class<? extends Scope> s = svc.value();
105         // for performance reason and to avoid infinite recursion,
106         // recognize these two fundamental built-in scopes and process them differently.
107         if(s==Singleton.class)
108             return singletonScope;
109         if(s==PerLookup.class)
110             return null;
111 
112         // for all the other scopes, including user-defined ones.
113         Scope scope = habitat.getByType(s);
114         ScopeInstance si = scope.current();
115         if(si==null) // scope is an extension point, so beware for broken implementations
116             throw new ComponentException(scope+" returned null from the current() method");
117         return si;
118     }

60  /***
61   * JUnit runner for hk2 enabled tests. Life-cycle of the test will be managed by
62   * this runner which will allow for dependency injection
63   *
64   * @author Jerome Dochez
65   */
66  public class Hk2Runner extends Runner {
67  
68      final Class<?> testClass;
69      final Description description;
70      final Map<Description, Method> testMethods = 
71            new LinkedHashMap<Description, Method>();
72      final List<Method> beforeMethods =
73            new ArrayList<Method>();
74      final List<Method> afterMethods =
75          new ArrayList<Method>();
76  
77      private Object instance;
78  
79      private final Hk2RunnerOptions options;
80  
81      
82      public Hk2Runner(Class<?> testClass) {
83          this.testClass = testClass;
84          this.options = testClass.getAnnotation(Hk2RunnerOptions.class);
85          this.description = Description.createSuiteDescription(testClass);
86          
87          for (Method m : testClass.getDeclaredMethods()) {
88              if (m.getAnnotation(Test.class)!=null) {
89                  Description testDescription = Description.createTestDescription(testClass, m.getName());
90                  description.addChild(testDescription);
91                  testMethods.put(testDescription, m);
92              }
93              
94              if (m.getAnnotation(Before.class) != null) {
95                  beforeMethods.add(m);
96              }
97              
98              if (m.getAnnotation(After.class) != null) {
99                  afterMethods.add(m);
100             }
101         }
102     }
103 
104     @Override
105     public Description getDescription() {
106         return description;        
107     }
108 
109     @Override
110     public void run(RunNotifier notifier) {
111         if (testClass.isAnnotationPresent(Ignore.class)) {
112             notifier.fireTestIgnored(getDescription());
113             return;
114         }
115 
116         // Run the @BeforeClass methods.
117         for (Method m : testClass.getMethods()) {
118             int mod = m.getModifiers();
119             if (Modifier.isStatic(mod) && m.getAnnotation(BeforeClass.class)!=null) {
120                 if (m.getAnnotation(Ignore.class)!=null) continue;
121                 try {
122                     m.invoke(null);
123                 } catch (IllegalAccessException e) {
124                     Failure failure = new Failure(Description.createTestDescription(testClass, m.getName()), e);
125                     notifier.fireTestFailure(failure);
126                 } catch (InvocationTargetException e) {
127                     Failure failure = new Failure(Description.createTestDescription(testClass, m.getName()), e);
128                     notifier.fireTestFailure(failure);
129                 }
130             }
131         }
132 
133         
134         boolean reinitPerTest = (null != options) ? options.reinitializePerTest() : false;
135         try {
136             wombInit();
137         } catch (ComponentException e) {
138             notifier.fireTestFailure(new Failure(getDescription(),e));
139             return;
140         }
141         
142         Iterator<Description> iter = description.getChildren().iterator();
143         while (iter.hasNext()) {
144             Description testDescription = iter.next();
145             
146             Method m = testMethods.get(testDescription);
147             if (m.isAnnotationPresent(Ignore.class)) {
148                 notifier.fireTestIgnored(testDescription);
149                 continue;
150             }
151 
152             try {
153                 runBefores(notifier);
154             } catch (Throwable e1) {
155                 throw new RuntimeException(e1);
156             }
157             
158             notifier.fireTestStarted(testDescription);
159             try {
160                 m.invoke(instance);
161             } catch (IllegalAccessException e) {
162                 Failure failure = new Failure(testDescription, e);
163                 notifier.fireTestFailure(failure);
164             } catch (InvocationTargetException e) {
165                 Failure failure = new Failure(testDescription, e.getTargetException());
166                 notifier.fireTestFailure(failure);
167             }
168             
169             notifier.fireTestFinished(testDescription);
170 
171             try {
172                 runAfters(notifier);
173             } catch (Throwable e1) {
174                 throw new RuntimeException(e1);
175             }
176             
177             if (reinitPerTest && iter.hasNext()) {
178                 try {
179                     wombInit();
180                 } catch (ComponentException e) {
181                     notifier.fireTestFailure(new Failure(getDescription(),e));
182                     return;
183                 }
184             }
185         }
186 
187         // Run the @AfterClass methods.
188         for (Method m : testClass.getMethods()) {
189             int mod = m.getModifiers();
190             if (Modifier.isStatic(mod) && m.getAnnotation(AfterClass.class)!=null) {
191                 if (m.getAnnotation(Ignore.class)!=null) continue;
192                 try {
193                     m.invoke(null);
194                 } catch (IllegalAccessException e) {
195                     Failure failure = new Failure(Description.createTestDescription(testClass, m.getName()), e);
196                     notifier.fireTestFailure(failure);
197                 } catch (InvocationTargetException e) {
198                     Failure failure = new Failure(Description.createTestDescription(testClass, m.getName()), e);
199                     notifier.fireTestFailure(failure);
200                 }
201             }
202         }
203     }
204 
205     private void runMethods(RunNotifier notifier, List<Method> methods) throws Throwable {
206         for (Method m: methods) {
207            if (m.getAnnotation(Ignore.class) != null) continue;
208            
209            try {
210             m.invoke(instance);
211            } catch (InvocationTargetException e) {
212                Throwable cause = e.getCause();
213                Failure failure = new Failure(Description.createTestDescription(testClass, m.getName()), cause);
214                notifier.fireTestFailure(failure);
215                throw cause;
216            } 
217         }
218     }
219     
220     private void runBefores(RunNotifier notifier) throws Throwable {
221         runMethods(notifier, beforeMethods);        
222     }
223     
224     private void runAfters(RunNotifier notifier) throws Throwable {
225         runMethods(notifier, afterMethods);
226     }

228     public static Habitat getHabitat() {
229         return singleton.getHabitat();
230     }

232     private void wombInit() {
233       singleton = new Hk2TestServices(null == options ? null : options.habitatFactory());
234       
235       Habitat habitat = singleton.getHabitat();
236       // so far we don't support extra meta-data on our tests.
237       Womb<?> womb = Wombs.create(testClass, habitat, new MultiMap<String, String>());
238       instance = womb.get();
239     }

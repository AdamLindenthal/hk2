233     /***
234      * Find all jars referenced directly and indirectly via a classpath
235      * specification typically drawn from java.class.path or 
236      * surefire.test.class.path System properties 
237      * @param cpSet a Set to hold classpath entries
238      * @param classPath a classpath with entries separated by {@link File.pathSeparator}
239      */
240     static void findEntriesInClasspath(Set<String> cpSet, String classPath) {
241         if (classPath != null) {
242             String[] filenames = classPath.split(File.pathSeparator);
243 
244             for (String filename : filenames) {
245 
246                 if (!filename.equals("")) {
247                     final File classpathEntry = new File(filename);
248 
249                     addTransitiveJars(cpSet, classpathEntry);
250                 }
251             }
252         }
253     }

255     /***
256      * Add provided File and all of its transitive manifest classpath entries to
257      * the provided set
258      * 
259      * @param cpSet a Set to hold classpath entries
260      * @param classpathFile File to transitively add to set
261      */
262     private static void addTransitiveJars(Set<String> cpSet, final File classpathFile) {
263         cpSet.add(classpathFile.getAbsolutePath());
264 
265         if (classpathFile.exists()) {
266 
267             try {
268                 if (classpathFile.isFile()) {
269                     JarFile jarFile = null;
270                     Manifest mf;
271                     try {
272                         jarFile = new JarFile(classpathFile);
273                         
274                         mf = jarFile.getManifest();
275                     } finally {
276                         if (jarFile != null) {
277                             jarFile.close();
278                         }
279                     }
280 
281                     // manifest may contain additional classpath
282                     if (mf != null) {
283                         String additionalClasspath = mf.getMainAttributes().getValue(
284                                 Attributes.Name.CLASS_PATH);
285 
286                         if (additionalClasspath != null) {
287 
288                             for (String classpathEntry : additionalClasspath.split(" ")) {
289 
290                                 if (!classpathEntry.equals("")) {
291                                     File mfClasspathFile = new File(classpathFile.getParent(),
292                                             classpathEntry.trim());
293 
294                                     if (mfClasspathFile.exists() && !cpSet.contains(mfClasspathFile.getAbsolutePath())) {
295                                         addTransitiveJars(cpSet, mfClasspathFile);
296                                     }
297                                 }
298                             }
299                         }
300                     }
301                 }
302             } catch (Exception ex) {
303                 ex.printStackTrace();
304             }
305 
306         }
307     }

309     private void parse(Parser parser, final File f) throws IOException {
310         Manifest manifest=null;
311         if (f.isDirectory()) {
312             File manifestFile = new File(f, JarFile.MANIFEST_NAME);
313             if (manifestFile.exists()) {
314                 InputStream is = new BufferedInputStream(new FileInputStream(manifestFile));
315                 try {
316                     manifest = new Manifest(is);
317                 } finally {
318                     is.close();
319                 }
320             }
321         } else {
322             JarFile jar = new JarFile(f);
323             manifest = jar.getManifest();
324             jar.close();
325         }
326         
327         if (manifest!=null) {
328             String imports = manifest.getMainAttributes().getValue("Import-Package");
329             if (imports==null || imports.indexOf("hk2")==-1) {
330 //                System.out.println("Ignoring service-less " + f.getName());
331                 logger.log(Level.FINE, "ignoring service-less {0}", f.getName());
332                 return;
333             }
334         }
335         
336         parseAlways(parser, f);
337     }

339     private void parseAlways(Parser parser, final File f) throws IOException {
340       parser.parse(f, new Runnable() {
341           public void run() {
342 //              System.out.println("Finished introspecting " + f.getName());
343               logger.log(Level.FINE, "Finished introspecting {0}", f.getName());
344           }
345       });
346     }

24  package org.jvnet.hk2.component;
25  
26  import java.util.Collection;
27  
28  /***
29   * A resource manager is responsible for adding and retrieving resource from a backing store.
30   *
31   * <p>
32   * ResourceManager must be thread-safe. Multiple threads may concurrently
33   * execute various methods of a single ResourceManager instance.
34   */
35  public interface ResourceManager {
36  
37      /***
38       * Adds a new resource to this manager, using the Extract instance information
39       * to qualify the resource, null if the resource is not qualified.
40       *
41       * @param resourceInfo is the locator information for that resource
42       * @param value new value to add to this resource manager
43       */
44      public abstract <T> void add(ResourceLocator<T> resourceInfo, T value);
45  
46      /***
47       * Looks up a previously stored instance of the provided type, further
48       * identifying the requested resource using the Inject meta data about it.
49       * If no qualification of the resource is provided and there is more than
50       * one instance of the provided type stored in this manager, the last entry
51       * will be returned.
52       *
53       * @param resourceInfo is the locator information for that resource
54       * @return the instance if found, null otherwise.
55       */
56      public abstract <T> T lookup(ResourceLocator<T> resourceInfo);
57  
58      /***
59       * Returns all previously stored instances of a particular type, null if none
60       * were registered.
61       *
62       * @param type of the resources looked up
63       * @return
64       *      Can be empty but never null.
65       */
66      public abstract <T> Collection<? extends T> lookupAll(Class<T> type);
67  
68      /***
69       * Removes a resource from this manager.
70       * @param value of the resource to be removed
71       */
72      // TODO: the other methods on this interface allows the same object to be listed
73      // under multiple names, so this definition is strange.
74      // remove when I work on component de-initialization.
75      public abstract void remove(Object value);
76      
77  }

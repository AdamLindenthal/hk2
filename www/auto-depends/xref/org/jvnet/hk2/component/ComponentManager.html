42  /***
43   * ComponentManager is responsible for instantiating components, as well as injecting
44   * and extracting resources from components. Components are defined with an interface they
45   * implement. Usually a component will implement the {@link Service} interface so the
46   * corresponding META-INF/services entries are created automatically.
47   *
48   * The component manager is a thread safe, it keeps no context between invocations.
49   *
50   * @author Jerome Dochez
51   * @author Kohsuke Kawaguchi
52   */
53  @Service(scope=Singleton.class)
54  public class ComponentManager {

95      /***
96       * Obtains a reference to the component inside the manager.
97       *
98       * <p>
99       * This is the "new Foo()" equivalent in the IoC world.
100      *
101      * <p> 
102      * Depending on the {@link Scope} of the component, a new instance
103      * might be created, or an existing instance might be returned.
104      *
105      * @return
106      *      non-null.
107      * @throws ComponentException
108      *      If failed to obtain a requested instance.
109      *      In practice, failure only happens when we try to create a
110      *      new instance of the component.
111      */
112     public <T> T getComponent(Class<T> clazz) throws ComponentException {
113         Service svc = clazz.getAnnotation(Service.class);
114         if(svc==null)
115 
116             throw new ComponentException(clazz+" is not a service");
117         try {
118             ScopeInstance si = getScope(svc);
119             if(si==null)    // prototype scope?
120                 return instantiate(clazz,svc,null);
121 
122             // TODO: ResourceLocator needs to get contract class, not service class.
123             ResourceLocator<T> loc = new ResourceLocator<T>(svc, clazz);
124 
125             T c = si.store.lookup(loc);
126             if(c!=null) return c; // component already exists
127 
128             // so far the execution is concurrent, so to avoid creating
129             // the same component twice, enter a lock
130             final Object lock = instanciationLock[(loc.hashCode()&0x7FFFFFFF)%instanciationLock.length];
131             synchronized(lock) {
132                 c = si.store.lookup(loc);
133                 if(c!=null)
134                     return c; // glad we locked!
135 
136                 // really create a new instance now
137                 c = instantiate(clazz,svc,si);
138                 si.store.add(loc,c);
139             }
140 
141             return c;
142         } catch (ComponentException e) {
143             throw new ComponentException("Failed to obtain "+clazz+" component",e);
144         }
145     }

147     /***
148      * Add an already instantiated component to this manager. The component has
149      * been instantiated by external code, however dependency injection, PostConstruct
150      * invocation and dependency extraction will be performed on this instance before
151      * it is store in the relevant scope's resource manager.
152      *
153      * @param name name of the component, could be default name
154      * @param component component instance
155      * @throws ComponentException if the passed object is not an HK2 component or
156      * injection/extraction failed.
157      */
158     public <T> void addComponent(String name, T component)
159         throws ComponentException {
160 
161         Service svc = component.getClass().getAnnotation(Service.class);
162         if (svc==null) {
163             throw new ComponentException(component.getClass() + " is not annotated with @Service");
164         }
165         // perform injection
166         inject(component);
167 
168         // postContruct call if any
169         if(component instanceof PostConstruct)
170             ((PostConstruct)component).postConstruct();
171 
172         ScopeInstance si = getScope(svc);
173         if(si!=null)
174             // extraction amounts to no-op if this is prototype scope. so skip that.
175             extract(component, si);
176 
177         // return component for future use.
178         si.store.add(new ResourceLocator(name, component.getClass()), component);
179     }

181     private static final Object[] instanciationLock = new Object[32];
182     static {
183         for( int i=0; i<instanciationLock.length; i++ )
184             instanciationLock[i] = new Object();

187     /***
188      * Determines the {@link ScopeInstance} that stores the component.
189      *
190      * @return
191      *      null for prototype scope. (Note that in {@link Scope#current()}
192      *      null return value is an error.)
193      */
194     private ScopeInstance getScope(Service svc) throws ComponentException {
195         Class<? extends Scope> s = svc.scope();
196         // for performance reason and to avoid infinite recursion,
197         // recognize these two fundamental built-in scopes and process them differently.
198         if(s==Singleton.class)
199             return singletonScope;
200         if(s==PerLookup.class)
201             return null;
202 
203         // for all the other scopes, including user-defined ones.
204         Scope scope = getComponent(s);
205         ScopeInstance si = scope.current();
206         if(si==null) // scope is an extension point, so beware for broken implementations
207             throw new ComponentException(scope+" returned null from the current() method");
208         return si;
209     }

211     /***
212      * Instantiate a new component instance and performs injection.
213      *
214      * @param impl
215      *      Implementation class to be instanciated.
216      * @param svc
217      *      Service annotation on the class.
218      * @param si
219      *      the instanciated component will be put into this scope.
220      *      this is used to store all {@link Extract extracted} resources.
221      * @return a new instance of a service implementing the contract interface
222      */
223     private <T> T instantiate(Class<T> impl, Service svc, ScopeInstance si) throws ComponentException {
224 
225         T component;
226         try {
227             if(svc.factory()!=Factory.class) {
228                 // create a component via a factory
229                 Factory factory = getComponent(svc.factory());
230                 Object o = factory.getObject();
231                 try {
232                     component = impl.cast(o);
233                 } catch (ClassCastException e) {
234                     throw new ComponentException(svc.factory()+" returned "+o+" but expected "+impl, e);
235                 }
236             } else {
237                 component = impl.newInstance();
238             }
239         } catch (InstantiationException e) {
240             throw new ComponentException("Exception while instantiating component of type " + impl, e);
241         } catch (IllegalAccessException e) {
242             throw new ComponentException("Exception while instantiating component of type " + impl, e);
243         }
244 
245         // perform injection
246         inject(component);
247 
248         // postContruct call if any
249         if(component instanceof PostConstruct)
250             ((PostConstruct)component).postConstruct();
251 
252         if(si!=null)
253             // extraction amounts to no-op if this is prototype scope. so skip that.
254             extract(component, si);
255 
256         // return component for future use.
257         return component;
258     }

260     /***
261      * Initializes the component by performing injection.
262      *
263      * @param component component instance to inject
264      * @throws ComponentException
265      *      if injection failed for some reason. 
266      */
267     private  void inject(Object component) throws ComponentException {
268         assert component!=null;
269 
270         // TODO: faster implementation needed.
271 
272         Class currentClass = component.getClass();
273         while (!currentClass.equals(Object.class)) {
274             // get the list of the instances variable
275             for (Field field : currentClass.getDeclaredFields()) {
276                 Inject inject = field.getAnnotation(Inject.class);
277                 if (inject == null)     continue;
278 
279                 Class<?> fieldType = field.getType();
280                 try {
281                     Object value = getValue(inject, fieldType);
282                     if (value != null) {
283                         field.setAccessible(true);
284                         field.set(component, value);
285                     } else {
286                         if(!inject.optional())
287                             throw new ComponentException("Failed to find component to be injected to "+field.toGenericString());
288                     }
289                 } catch (IllegalAccessException e) {
290                     throw new ComponentException("Injection failed on " + field.toGenericString(), e);
291                 }
292             }
293             for (Method method : currentClass.getDeclaredMethods()) {
294                 Inject inject = method.getAnnotation(Inject.class);
295                 if (inject == null)     continue;
296 
297                 if (method.getReturnType() != void.class) {
298                     throw new ComponentException("Injection failed on " + method.toGenericString()
299                             + " : setter method is not declared with a void return type");
300                 }
301 
302                 Class<?>[] paramTypes = method.getParameterTypes();
303 
304                 if (paramTypes.length > 1) {
305                     throw new ComponentException("injection failed on " + method.toGenericString() + " : setter method takes more than 1 parameter");
306                 }
307                 if (paramTypes.length == 0) {
308                     throw new ComponentException("injection failed on " + method.toGenericString() + " : setter method does not take a parameter");
309                 }
310 
311                 try {
312                     Object value = getValue(inject, paramTypes[0]);
313                     if (value != null) {
314                         method.setAccessible(true);
315                         method.invoke(component, value);
316                     } else {
317                         if (!inject.optional())
318                             throw new ComponentException("Failed to find component to be injected to " + method.toGenericString());
319                     }
320                 } catch (IllegalAccessException e) {
321                     throw new ComponentException("Injection failed on " + method.toGenericString(), e);
322                 } catch (InvocationTargetException e) {
323                     throw new ComponentException("Injection failed on " + method.toGenericString(), e);
324                 }
325             }
326             currentClass = currentClass.getSuperclass();
327         }
328     }

330     /***
331      * Extracts resources identified with {@link Extract} annotations.
332      *
333      * TODO: it's not clear what type under which such components should be stored.
334      *
335      * @param component the component we should extract resources from
336      * @throws ComponentException if the resource extract fail 
337      */
338     private void extract(Object component, ScopeInstance si) throws ComponentException {
339 
340         Class<?> currentClass = component.getClass();
341         while (!currentClass.equals(Object.class)) {
342             for (Field field : currentClass.getDeclaredFields()) {
343                 Extract extract = field.getAnnotation(Extract.class);
344                 if (extract == null)    continue;
345 
346                 try {
347                     field.setAccessible(true);
348                     Object value = field.get(component);
349                     Class<?> type = field.getType();
350 
351                     if (LOGGER.isLoggable(Level.FINER)) {
352                         LOGGER.log(Level.FINER, "Extracting resource " + value + " returned from " + field);
353                     }
354                     if (value!=null) {
355                         extractValue(value, si, type, extract);
356                     } else {
357                         if (LOGGER.isLoggable(Level.FINE)) {
358                             LOGGER.log(Level.FINE, "Resource returned from " + field + " is null");
359                         }
360                     }
361                 } catch (IllegalArgumentException ex) {
362                     throw new ComponentException("Extraction failed on " + field, ex);
363 
364                 } catch (IllegalAccessException ex) {
365                     throw new ComponentException("Extraction failed on " + field, ex);
366                 }
367             }
368             for (Method method : currentClass.getDeclaredMethods()) {
369                 Extract extract = method.getAnnotation(Extract.class);
370                 if (extract == null)    continue;
371 
372                 Class<?> type = method.getReturnType();
373                 if (type == null) {
374                     throw new ComponentException("Extraction failed : " + method + " has a void return type");
375                 }
376                 if (method.getParameterTypes().length > 0) {
377                     throw new ComponentException("Extraction failed : " + method + " takes parameters, it should not");
378                 }
379                 try {
380                     method.setAccessible(true);
381                     Object value = method.invoke(component);
382                     if (LOGGER.isLoggable(Level.FINER)) {
383                         LOGGER.log(Level.FINER, "Extracting resource " + value + " returned from " + method);
384                     }
385                     if (value!=null) {
386                         extractValue(value, si, type, extract);
387                     } else {
388                         if (LOGGER.isLoggable(Level.FINE)) {
389                             LOGGER.log(Level.FINE, "Resource returned from " + method + " is null");
390                         }
391                     }
392                 } catch (IllegalArgumentException ex) {
393                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
394                 } catch (InvocationTargetException ex) {
395                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
396                 } catch (IllegalAccessException ex) {
397                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
398                 }
399             }
400             currentClass = currentClass.getSuperclass();
401         }
402 
403     }

405     @SuppressWarnings("unchecked")
406     private void extractValue(Object value, ScopeInstance si, Class type, Extract extract) {
407         if(value instanceof Iterable) {
408             for (Object o : Iterable.class.cast(value)) {
409                 si.store.add(null, o);
410             }
411         } else
412         if (type.isArray()) {
413             Object[] values = (Object[]) value;
414             for (Object o : values) {
415                 si.store.add(null, o);
416             }
417         } else {
418             si.store.add(new ResourceLocator(extract, type), value);
419         }
420     }

423     /***
424      * Obtains all the components that implement the given contract.
425      *
426      * <p>
427      * If necessary, the components are created.
428      *
429      * @param contract type of service to load/intitialized
430      * @return
431      *      can be empty but never null.
432      */
433     public <T> Collection<T> getComponents(Class<T> contract, Logger logger) {
434 
435         Set<T> providers = new HashSet<T>();
436         for (Class<? extends T> providerClass : serviceLookup.getProvidersClass(contract)) {
437 
438             try {
439                 Service service = providerClass.getAnnotation(Service.class);
440                 assert service!=null;
441 
442                 Configurable configurable = providerClass.getAnnotation(Configurable.class);
443                 if (configurable!=null) {
444                     // get them from the scope.
445                     ScopeInstance scope = getScope(service);
446                     providers.addAll(scope.store.lookupAll(providerClass));
447                 } else {
448                     providers.add(getComponent(providerClass));
449                 }
450             } catch(ComponentException e) {
451                 logger.log(Level.SEVERE, "Cannot instantiate service " + providerClass, e);
452             }
453         }
454         return providers;
455         
456     }

459     /***
460      * Loads a component that implements the given contract.
461      *
462      * <p>
463      * If this method is used with a non-null name, then a service
464      * that implements the given contract AND have the specified name
465      * will be returned.
466      *
467      * <p>
468      * If this method is used with null name, then one service
469      * that implements the given contract will be returned, regardless of its name.
470      * If there are multiple service classes, which one gets returned is unspecified.
471      *
472      * @return
473      *      null if no such servce exists. 
474      */
475     public <T> T getComponent(ResourceLocator<T> locator) throws ComponentException {
476         // TODO: more efficient implementation needed, but for now to get things moving
477 
478 
479         Class<? extends T> cachedProvider = cachedProviders.get(locator);
480         if (cachedProvider!=null) {
481             return getComponent(cachedProvider);
482         }
483         for (Class<? extends T>  providerClass : serviceLookup.getProvidersClass(locator.getType())) {
484             // let's cache this for future usage
485             ResourceLocator cacheID = new ResourceLocator(providerClass.getAnnotation(Service.class),
486                 locator.getType());
487             cachedProviders.put(cacheID, providerClass);
488 
489             if(locator.getName()!=null
490             && !providerClass.getAnnotation(Service.class).name().equals(locator.getName()))
491                 continue;   // name doesn't match
492 
493             return getComponent(providerClass);
494         }
495 
496         return null; // not found 
497     }

499     /***
500      * Destroys the component instance explicitly.
501      * <p>
502      * Normally, a component is destroyed when the scope it's in is destroyed,
503      * but you can call this method to force the early destruction.
504      * <p>
505      * Note that as soon as someone else tries to {@link #getComponent(Class)}
506      * a component will be reinstanciated.
507      */
508     public void releaseComponent(Object component) throws ComponentException {
509         Service svc = component.getClass().getAnnotation(Service.class);
510         if(svc==null)
511             throw new ComponentException("Unable to release "+component+"; no @Service annotation");
512         ScopeInstance si = getScope(svc);
513         if(si==null)
514             // if si==null, then this component is prototype scope, so no there's store.
515             // as per the prototype scope contract, there's no @PreDestroy
516             return;
517 
518         // call the preDestroy if defined on the service
519         if(component instanceof PreDestroy)
520             ((PreDestroy)component).preDestroy();
521         
522         si.store.remove(component);
523     }

535     /***
536      * Obtains the value to inject, based on the type and {@link Inject} annotation.
537      */
538     @SuppressWarnings("unchecked")
539     protected <T> Object getValue(Inject inject, Class<T> type) throws ComponentException {
540         if (type.isArray()) {
541             Class<?> ct = type.getComponentType();
542 
543             Contract ctr = ct.getAnnotation(Contract.class);
544             if(ctr!=null) {
545                 Collection instances = getComponents(ct, LOGGER);
546                 return instances.toArray((Object[])Array.newInstance(ct, instances.size()));
547             }
548         } else {
549             Service svc = type.getAnnotation(Service.class);
550             if(svc!=null)
551                 // component injection
552                 return getComponent(type);
553 
554             Contract ctr = type.getAnnotation(Contract.class);
555             if(ctr!=null)
556                 // service lookup injection
557                 return getComponent(new ResourceLocator(inject, type));
558         }
559         throw new ComponentException(type+" cannot be injected: it's neither a contract nor a service");
560     }

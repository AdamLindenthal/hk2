37  package org.jvnet.hk2.component;
38  
39  import org.jvnet.hk2.annotations.CagedBy;
40  import org.jvnet.hk2.annotations.Contract;
41  
42  /***
43   * Receives notifications when a component with {@link CagedBy}
44   * is entered into habitat.
45   *
46   * <p>
47   * The concrete registration hook classes are instanciated as components
48   * themselves, so in that way sub-classes can obtain contextual information,
49   * such as {@link Habitat}, via injection. 
50   *
51   * @author Kohsuke Kawaguchi
52   * @see CagedBy
53   */
54  @Contract
55  @CagedBy(CageBuilder.Registerer.class)
56  public interface CageBuilder<T> {
57      /***
58       * Called when an {@link Inhabitant} is entered into habitat.
59       */
60      void onEntered(Inhabitant<T> i);
61  
62      /***
63       * When {@link CageBuilder} enters habitat, look
64       * for existing inhabitants that were supposed to be
65       * caged by this and cage them all.
66       *
67       * <p>
68       * This is because we don't generally know which one comes first
69       * into the habitat &mdash; cage builder or caged component.
70       *
71       * <p>
72       * Normal {@link CageBuilder}s register themselves as components,
73       * but this one is defined in HK2 so we have to get in by a special means.
74       */
75      public static final class Registerer<T> implements CageBuilder<T> {
76          private final Habitat habitat;
77  
78          public Registerer(Habitat habitat) {
79              this.habitat = habitat;
80          }
81  
82          public void onEntered(Inhabitant<T> i) {
83              // and if a companion seed is added to something else, make sure
84              // existing lead inhabitants will get this as a companion
85              assert i.metadata()!=null;
86              for (Inhabitant supposedToBeCaged : habitat.getInhabitants(CagedBy.class,i.typeName()))
87                  ((CageBuilder)i.get()).onEntered(supposedToBeCaged);
88          }
89      }
90  }

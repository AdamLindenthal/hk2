15    public WildcardMatcher(String pattern) {
16      this.pattern = pattern;
17      this.parts = pattern.split("//*");
18    }
19  
20    public boolean matches(String text) {
21      if (0 == parts.length) {
22        return true;
23      }
24      
25      if (!pattern.startsWith("*") && !text.startsWith(parts[0])) {
26        return false;
27      }
28      
29      for (String part : parts) {
30        int pos = text.indexOf(part);
31        if (pos == -1) {
32          return false;
33        }
34        text = text.substring(pos + part.length());
35      }
36  
37      return (text.length() == 0 || pattern.endsWith("*"));
38    }

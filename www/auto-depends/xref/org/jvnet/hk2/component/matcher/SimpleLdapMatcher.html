119   protected boolean getTheAndSetFor(Set<String> result, String propName, boolean removeIt, Node node) {
120     boolean hadMatch = false;
121     if (node.type == BASE_TYPE) {
122       if (node.leftHandSide.equals(propName)) {
123         result.add(node.rightHandSide);
124         hadMatch = true;
125         if (removeIt) {
126           node.leftHandSide="";
127         }
128       }
129     } else if (node.type == AND_TYPE) {
130       Iterator<Node> iter = node.children.iterator();
131       while (iter.hasNext()) {
132         Node andNode = iter.next();
133         boolean matched = getTheAndSetFor(result, propName, removeIt, andNode);
134         hadMatch |= matched;
135         if (matched && removeIt) {
136           iter.remove();
137         }
138       }
139     }
140     return hadMatch;
141   }

144   private Map<String, Set<String>> getMatchingMap(MultiMap<String, String> props) {
145     Map<String, Set<String>> result = new HashMap<String, Set<String>>();
146     for (Entry<String, List<String>> entry : props.entrySet()) {
147       String key = entry.getKey().toLowerCase();
148       Set<String> set = result.get(key);
149       if (null == set) {
150         set = new HashSet<String>();
151         result.put(entry.getKey().toLowerCase(), set);
152       }
153       set.addAll(entry.getValue());
154     }
155     return result;

180   /* package */static Map<String, Set<String>> getMatchingMap(
181       Map<String, Object> paramProperties) {
182     HashMap<String, Set<String>> retVal = new HashMap<String, Set<String>>();
183 
184     // Get the "fixed" map AFTER adding the system generated properties
185     for (Map.Entry<String, Object> entry : paramProperties.entrySet()) {
186       String key = entry.getKey();
187       if (key == null) {
188         throw new RuntimeException("Null key in property map");
189       }
190 
191       Object value = entry.getValue();
192       if (value == null) {
193         throw new RuntimeException("Null value in property map for " + key);
194       }
195 
196       String toLowerKey = key.toLowerCase();
197       Set<String> valueSet = new HashSet<String>();
198 
199       Class<?> valueClass = value.getClass();
200       if (valueClass.isArray()) {
201         int valueLength = Array.getLength(value);
202         for (int lcv = 0; lcv < valueLength; lcv++) {
203           Object arrayValue = Array.get(value, lcv);
204           if (arrayValue == null) {
205             throw new RuntimeException("Null value in array for " + key
206                 + " at index " + lcv);
207           }
208 
209           String arrayValueString = arrayValue.toString();
210           valueSet.add(arrayValueString);
211         }
212       } else {
213         String valueString = value.toString();
214         valueSet.add(valueString);
215       }
216 
217       retVal.put(toLowerKey, valueSet);
218     }
219 
220     return retVal;
221   }

223   private static Node parse(ActiveString workString) {
224     Node retVal = new Node();
225 
226     // I am looking for the start of an expression
227     workString.toStatementStart();
228 
229     int type = workString.getStatementType();
230     switch (type) {
231     case BASE_TYPE:
232       retVal.setType(BASE_TYPE);
233       String lhs = workString.getLeftHandSide();
234       boolean isEquals = workString.getEqualsOrNotEquals();
235       String rhs = workString.getRightHandSide();
236 
237       retVal.setLeftHandSide(lhs);
238       retVal.setIsEquals(isEquals);
239       retVal.setRightHandSide(rhs);
240       break;
241     case AND_TYPE:
242     case OR_TYPE:
243       retVal.setType(type);
244       while (workString.finishedOrNewStatement()) {
245         Node child = parse(workString);
246         retVal.addChild(child);
247       }
248 
249       if (retVal.getNumChildren() <= 1) {
250         throw new RuntimeException(
251             "There was zero or one children of an AND or OR construct");
252       }
253 
254       break;
255     case NEST_TYPE:
256       // This type was NOT incremented past the '('
257       Node lastNode = parse(workString);
258       retVal.copyNode(lastNode);
259       break;
260     case FALSE_TYPE: // getStatementType cannot return FALSE_TYPE
261     default:
262       throw new AssertionError("Unknown node type " + type);
263     }
264 
265     return retVal;
266   }

329     private boolean doesMatch(Map<String, Set<String>> toMatch) {
330       if (type == BASE_TYPE) {
331         Set<String> valueSet = toMatch.get(leftHandSide);
332         if (valueSet == null) {
333           if (equals) {
334             return (null == leftHandSide || leftHandSide.length() == 0);
335           }
336 
337           return true;
338         }
339 
340         WildcardMatcher matcher = new WildcardMatcher(rightHandSide);
341         for (String val : valueSet) {
342           if (matcher.matches(val)) {
343             return equals;
344           }
345         }
346 
347         // valueSet does *not* contain the value
348         if (equals) {
349           return false;
350         }
351 
352         return true;
353       }
354 
355       if (type == AND_TYPE) {
356         for (Node andNode : children) {
357           if (!andNode.doesMatch(toMatch)) {
358             return false;
359           }
360         }
361 
362         return true;
363       }
364 
365       if (type == OR_TYPE) {
366         for (Node orNode : children) {
367           if (orNode.doesMatch(toMatch)) {
368             return true;
369           }
370         }
371 
372         return false;
373       }
374 
375       if (type == FALSE_TYPE) {
376         return false;
377       }
378 
379       throw new AssertionError("Unknown node type " + type);
380     }

410     final String original;

416     private ActiveString(String fullString) {
417       original = fullString;

421     /***
422      * Increments dot safely
423      * 
424      * @param doThrow
425      *          Tells this method to throw if it reaches the end of the string,
426      *          the state machine knows that this should not reach the end
427      * @return true if there are more characters, false if this is the end
428      */
429     private boolean incrementDot(boolean doThrow) {
430       if (dot + 1 >= data.length) {
431         // finished
432         if (doThrow) {
433           throw new RuntimeException("Invalid end of string reached in "
434               + original + " at character " + dot);
435         }
436 
437         return false;
438       }
439 
440       dot++;
441       return true;
442     }
443 
444     /***
445      * Skips white space
446      * 
447      * @param doThrow
448      *          if true throw an exception if the end is reached
449      * @return true if there are more characters, false if this is the end
450      */
451     private boolean skipWhiteSpace(boolean doThrow) {
452       boolean retVal = false;
453 
454       while (Character.isWhitespace(data[dot])) {
455         retVal = incrementDot(doThrow);
456       }
457 
458       return retVal;
459     }
460 
461     private void toStatementStart() {
462       skipWhiteSpace(true);
463 
464       if (!(data[dot] == LEFT)) {
465         throw new RuntimeException("Statement did not start with '(', found "
466             + data[dot] + " at character " + dot);
467       }
468 
469       // Get past the left parenthesis
470       incrementDot(true);
471     }
472 
473     private int getStatementType() {
474       skipWhiteSpace(true);
475 
476       if (data[dot] == AND) {
477         incrementDot(true);
478         return AND_TYPE;
479       }
480 
481       if (data[dot] == OR) {
482         incrementDot(true);
483         return OR_TYPE;
484       }
485 
486       if (data[dot] == LEFT) {
487         // Do not increment dot
488         return NEST_TYPE;
489       }
490 
491       // Do not increment dot
492       return BASE_TYPE;
493     }
494 
495     /***
496      * 
497      * @return false if this is finished (increments past the final ')' and true
498      *         if there is another statement (does NOT increment past the '(')
499      */
500     private boolean finishedOrNewStatement() {
501       skipWhiteSpace(true);
502 
503       if (data[dot] == RIGHT) {
504         incrementDot(false);
505         return false;
506       }
507 
508       // Do not increment past the dot
509       return true;
510     }
511 
512     private String getLeftHandSide() {
513       int startIndex = dot;
514       int count = 0;
515 
516       while (data[dot] != EQ) {
517         if ((data[dot] == NOT) && ((dot + 1) < data.length)
518             && (data[dot + 1] == EQ)) {
519           // Found a not equals
520           return new String(data, startIndex, count);
521         }
522 
523         count++;
524         incrementDot(true);
525       }
526 
527       return new String(data, startIndex, count);
528     }
529 
530     private boolean getEqualsOrNotEquals() {
531       if (data[dot] == EQ) {
532         incrementDot(true);
533         return true;
534       }
535 
536       if (data[dot] == NOT) {
537         incrementDot(true);
538         if (!(data[dot] == EQ)) {
539           throw new AssertionError("Invalid not equals at character " + dot);
540         }
541 
542         incrementDot(true);
543         return false;
544       }
545 
546       throw new AssertionError(
547           "Was looking for equals or not equals but found " + data[dot]
548               + " at character " + dot);
549     }
550 
551     private String getRightHandSide() {
552       int startIndex = dot;
553       int count = 0;
554 
555       while (data[dot] != RIGHT) {
556         count++;
557         incrementDot(true);
558       }
559 
560       // This might be the end, but need to get past the )
561       incrementDot(false);
562       return new String(data, startIndex, count);
563     }

565     private void finish() {
566       if (skipWhiteSpace(false) == true) {
567         throw new RuntimeException(
568             "Found extra characters at end of LDAP string " + original
569                 + " at character " + dot);
570       }
571     }

576       return "ActiveString(" + original + "," + dot + "," + dataDot + ")";

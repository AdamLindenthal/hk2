24  package org.jvnet.hk2.component;
25  
26  import java.lang.reflect.Field;
27  import java.lang.reflect.Method;
28  import java.lang.reflect.InvocationTargetException;
29  import java.lang.reflect.AnnotatedElement;
30  import java.lang.annotation.Annotation;
31  
32  /***
33   * InjectionManager is responsible for injecting resources into a component.
34   * Resources tagged for injection are identified via an annotation which is
35   * a parameter type of this class.
36   *
37   * @author Jerome Dochez
38   */
39  public abstract class InjectionManager<T extends Annotation> {
40  
41      /***
42       * Initializes the component by performing injection.
43       *
44       * @param component component instance to inject
45       * @throws ComponentException
46       *      if injection failed for some reason.
47       */
48      public void inject(Object component, Class<T> type) throws ComponentException {
49          assert component!=null;
50  
51          // TODO: faster implementation needed.
52  
53          Class currentClass = component.getClass();
54          while (!currentClass.equals(Object.class)) {
55              // get the list of the instances variable
56              for (Field field : currentClass.getDeclaredFields()) {
57  
58                  T inject = field.getAnnotation(type);
59                  if (inject == null)     continue;
60  
61                  Class fieldType = field.getType();
62                  try {
63                      Object value = getValue(field, fieldType);
64                      if (value != null) {
65                          field.setAccessible(true);
66                          field.set(component, value);
67                      } else {
68                          if(!isOptional(inject))
69                              throw new UnsatisfiedDepedencyException(field);
70                      }
71                  } catch (ComponentException e) {
72                      if (!isOptional(inject)) {
73                          throw new UnsatisfiedDepedencyException(field,e);
74                      }
75                  } catch (IllegalAccessException e) {
76                      throw new ComponentException("Injection failed on " + field.toGenericString(), e);
77                  }
78              }
79              for (Method method : currentClass.getDeclaredMethods()) {
80                  T inject = method.getAnnotation(type);
81                  if (inject == null)     continue;
82  
83                  if (method.getReturnType() != void.class) {
84                      throw new ComponentException("Injection failed on " + method.toGenericString()
85                              + " : setter method is not declared with a void return type");
86                  }
87  
88                  Class<?>[] paramTypes = method.getParameterTypes();
89  
90                  if (paramTypes.length > 1) {
91                      throw new ComponentException("injection failed on " + method.toGenericString() + " : setter method takes more than 1 parameter");
92                  }
93                  if (paramTypes.length == 0) {
94                      throw new ComponentException("injection failed on " + method.toGenericString() + " : setter method does not take a parameter");
95                  }
96  
97                  try {
98                      Object value = getValue(method, paramTypes[0]);
99                      if (value != null) {
100                         method.setAccessible(true);
101                         method.invoke(component, value);
102                     } else {
103                         if (!isOptional(inject))
104                             throw new UnsatisfiedDepedencyException(method);
105                     }
106                 } catch (IllegalAccessException e) {
107                     throw new ComponentException("Injection failed on " + method.toGenericString(), e);
108                 } catch (InvocationTargetException e) {
109                     throw new ComponentException("Injection failed on " + method.toGenericString(), e);
110                 }
111             }
112             currentClass = currentClass.getSuperclass();
113         }
114     }
115 
116     protected abstract boolean isOptional(T annotation);
117 
118     protected abstract Object getValue(AnnotatedElement annotated, Class type) throws ComponentException;
119 }

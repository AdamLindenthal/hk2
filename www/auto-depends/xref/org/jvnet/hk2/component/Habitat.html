107     /***
108      * Removes all imhabitants for a particular type
109      *
110      * @param type of the compoment
111      */
112     public void removeAllByType(Class<?> type) {
113 
114         String name = type.getName();
115 
116         // remove all existing inhabitant.
117         synchronized(byType.get(name)) {
118             for (Inhabitant existing : byType.get(name)) {
119                 existing.release();
120             }
121             byType.get(name).clear();
122         }
123     }

125     /***
126      * Adds a new inhabitant.
127      *
128      * <p>
129      * See {@link Inhabitants} for typical ways to create {@link Inhabitant}s.
130      */
131     public void add(final Inhabitant<?> i) {
132         String name = i.typeName();
133         byType.add(name,i);
134 
135         // for each companion, create an inhabitat that goes with the lead and hook them up
136         List<Inhabitant> companions=null;
137         for(Inhabitant<?> c : getInhabitantsByAnnotation(CompanionSeed.class,name)) {
138             if(companions==null)
139                 companions = new ArrayList<Inhabitant>();
140             companions.add(createCompanion(this,i,c));
141         }
142         i.setCompanions(companions);
143 
144         String cageBuilderName = i.metadata().getOne(CAGE_BUILDER_KEY);
145         if(cageBuilderName!=null) {
146             Inhabitant cageBuilder = byType.getOne(cageBuilderName);
147             if(cageBuilder!=null)
148                 ((CageBuilder)cageBuilder.get()).onEntered(i);
149             // if cageBuilder==null, we can't cage this component now, but
150             // we'll do that when cageBuilder comes into the habitat.
151             // that happens because every CageBuilder implementations are caged by
152             // CageBuilder.Registerer.
153         }
154     }

194     /***
195      * Removes a NamedInhabitant for a specific contract
196      *
197      * @param index contract name
198      * @param name instance name
199      * @return true if the removal was successful
200      */
201     public boolean removeIndex(String index, String name) {
202         if (byContract.containsKey(index)) {
203             List<NamedInhabitant> contracted = byContract.get(index);
204             for (NamedInhabitant i : contracted) {
205                 if ((i.name == null && name == null) ||
206                         (i.name != null && i.name.equals(name))) {
207 
208                     // remember to remove the components stored under its type
209                     if (byType.get(index).contains(i.inhabitant)) {
210                         byType.get(index).remove(i.inhabitant);
211                     }
212                     return contracted.remove(i);
213                 }
214             }
215         }
216         return false;
217     }

225     public boolean removeIndex(String index, Object service) {
226         if (byContract.containsKey(index)) {
227              List<NamedInhabitant> contracted = byContract.get(index);
228              for (NamedInhabitant i : contracted) {
229                  // call to isInstantiated is necessary to avoid loading of services that are not yet loaded
230                  if (i.inhabitant.isInstantiated() && i.inhabitant.get().equals(service)) {
231                     // remember to remove the components stored under its type
232                     if (byType.get(index).contains(i.inhabitant)) {
233                         byType.get(index).remove(i.inhabitant);
234                     }                     
235                      return contracted.remove(i);
236                  }
237              }
238          }
239          return false;
240     }

344     /***
345      * Loads a component that implements the given contract and has the given name.
346      *
347      * @param name
348      *      can be null, in which case it'll only match to the unnamed component.
349      * @return
350      *      null if no such servce exists.
351      */
352     public <T> T getComponent(Class<T> contract, String name) throws ComponentException {
353 
354         if (name!=null && name.length()==0)
355             name=null;
356         Inhabitant i = getInhabitant(contract, name);
357         if(i!=null)
358             try {
359                 return contract.cast(i.get());
360             } catch (ClassCastException e) {
361                 Logger.getAnonymousLogger().severe("ClassCastException between contract " + contract + " and service " + i.get());
362                 Logger.getAnonymousLogger().severe("Contract class loader " + contract.getClassLoader());
363                 Logger.getAnonymousLogger().severe("Service class loader " + i.get().getClass().getClassLoader());
364                 throw e;
365             }
366         else
367             return null;
368     }

534     // intentionally not generified so that the getInhabitants methods can choose the right signature w/o error
535     private Iterable _getInhabitants(final Class contract, final String name) {
536         return new Iterable<Inhabitant>() {
537             private final Iterable<NamedInhabitant> base = byContract.get(contract.getName());
538 
539             public Iterator<Inhabitant> iterator() {
540                 return new Iterator<Inhabitant>() {
541                     private Inhabitant i = null;
542                     private final Iterator<NamedInhabitant> itr = base.iterator();
543 
544                     public boolean hasNext() {
545                         while(i==null && itr.hasNext()) {
546                             NamedInhabitant ni = itr.next();
547                             if(eq(ni.name,name))
548                                 i = ni.inhabitant;
549                         }
550                         return i!=null;
551                     }
552 
553                     public Inhabitant next() {
554                         if(i==null)
555                             throw new NoSuchElementException();
556                         Inhabitant r = i;
557                         i = null;
558                         return r;
559                     }
560 
561                     public void remove() {
562                         throw new UnsupportedOperationException();
563                     }
564                 };
565             }
566         };
567     }

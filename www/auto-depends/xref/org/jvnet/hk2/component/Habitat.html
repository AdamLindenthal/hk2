176     Habitat(Services parent, String name, Boolean concurrency_controls) {
177         this.parent = parent;
178         // this.name = name;
179         this.concurrencyControls = (null == concurrency_controls) ? CONCURRENCY_CONTROLS_DEFAULT
180                 : concurrency_controls;
181         this.byContract = new MultiMap<String, NamedInhabitant>(
182                 this.concurrencyControls);
183         this.byType = new MultiMap<String, Inhabitant>(this.concurrencyControls);
184         this.singletonScope = new ScopeInstanceImpl("singleton", new HashMap());
185 
186         // add the set of ExecutorServices by name
187         if (concurrencyControls) {
188             logger.fine("concurrency controls enabled");
189             // TODO: remove me (eventually)
190             // System.out.println("CONCURRENCY CONTROLS ENABLED");
191             addIndex(new ExistingSingletonInhabitant<ExecutorService>(
192                         ExecutorService.class, exec),
193                     ExecutorService.class.getName(),
194                     Constants.EXECUTOR_INHABITANT_INJECTION_MANAGER);
195             addIndex(new ExistingSingletonInhabitant<ExecutorService>(
196                         ExecutorService.class, exec),
197                     ExecutorService.class.getName(),
198                     Constants.EXECUTOR_INHABITANT_ACTIVATOR);
199         }
200         addIndex(new ExistingSingletonInhabitant<ExecutorService>(
201                     ExecutorService.class, SameThreadExecutor.instance),
202                 ExecutorService.class.getName(),
203                 Constants.EXECUTOR_HABITAT_LISTENERS_AND_TRACKERS);
204 
205         // make the listeners available very early in lifecycle
206         addHabitatListener(new SelfListener());
207 
208         // add the set of injection resolvers
209         InjectInjectionResolver injectresolver = new InjectInjectionResolver(this);
210         addIndex(new ExistingSingletonInhabitant<InjectionResolver>(
211                     InjectionResolver.class, injectresolver),
212                 InjectionResolver.class.getName(), "Inject");
213 
214         // make the habitat itself available
215         Inhabitant<Habitat> habitatInh = new ExistingSingletonInhabitant<Habitat>(Habitat.class, this);
216         add(habitatInh);
217         addIndex(habitatInh, Injector.class.getName(), null);
218         addIndex(habitatInh, Services.class.getName(), null);
219         if (parent != null && name != null) {
220             DynamicBinderFactory parentBinder = parent.bindDynamically();
221             parentBinder.bind(Services.class).named(name).toInstance(this);
222             parentBinder.commit();
223         }
224 
225         InhabitantProviderInterceptor interceptor = new RunLevelInhabitantProvider(
226                 this);
227         addIndex(
228                 new ExistingSingletonInhabitant<InhabitantProviderInterceptor>(
229                         InhabitantProviderInterceptor.class, interceptor),
230                 InhabitantProviderInterceptor.class.getName(), null);
231 
232         // the default RunLevelService
233         DefaultRunLevelService rls = new DefaultRunLevelService(this);
234         ExistingSingletonInhabitant<RunLevelService> rlsI = new ExistingSingletonInhabitant<RunLevelService>(
235                 RunLevelService.class, rls);
236         add(rlsI);
237         addIndex(rlsI, RunLevelService.class.getName(),
238                 DefaultRunLevelService.DEFAULT_NAME);
239     }

268     @Override
269     public Collection<Binding<?>> getDeclaredBindings(Descriptor descriptor) {
270         LinkedHashSet<Binding<?>> result = new LinkedHashSet<Binding<?>>();
271 
272         // try optimized approaches first
273         if (DescriptorImpl.isEmpty(descriptor)) {
274             includeBinding(result, byContract);
275             includeBinding(result, byType);
276         } else if (null != descriptor && !DescriptorImpl.isEmpty(descriptor.getTypeName())) {
277             includeBindingIfMatches(result, descriptor, byType.get(descriptor.getTypeName()));
278             // Note, that next we are treating the type as a Contract (seems appropriate to do)
279             includeBindingIfMatches(result, descriptor, byContract.get(descriptor.getTypeName()));
280         } else if (null != descriptor && !DescriptorImpl.isEmpty(descriptor.getContracts())) {
281             includeBindingIfMatches(result, descriptor, byContract.getIntersectionOfAll(descriptor.getContracts()));
282         } else {
283             // no optimized approaches, let's try them all
284             includeBindingIfMatches(result, descriptor, byType.entrySet());
285             includeBindingIfMatches(result, descriptor, byContract.entrySet());
286         }
287         
288         return result;
289     }
290 
291     private void includeBindingIfMatches(LinkedHashSet<Binding<?>> result,
292             Descriptor descriptor,
293             Set entrySet) {
294         for (Object o : entrySet) {
295             Map.Entry<String, List<Inhabitant>> entry = (Map.Entry<String, List<Inhabitant>>)o; 
296             includeBindingIfMatches(result, descriptor, entry.getValue());
297         }
298     }
299 
300     private void includeBinding(LinkedHashSet<Binding<?>> result, MultiMap<String,?> mm) {
301         for (Entry<String, ?> entry : mm.entrySet()) {
302             includeBindingIfMatches(result, null, (List<?>)entry.getValue());
303         }
304     }
305 
306     private void includeBindingIfMatches(LinkedHashSet<Binding<?>> result,
307             Descriptor descriptor,
308             List<?> list) {
309         for (Object i : list) {
310             includeBinding(result, descriptor, i);
311         }
312     }

314     void includeBinding(LinkedHashSet<Binding<?>> result, Descriptor descriptor, Object i) {
315         if (NamedInhabitant.class.isInstance(i)) {
316             i = NamedInhabitant.class.cast(i).inhabitant;
317         }
318         
319         if (AbstractInhabitantImpl.class.isInstance(i)) {
320             AbstractInhabitantImpl<?> ai = AbstractInhabitantImpl.class.cast(i);
321             if (null == descriptor || (null != descriptor && ai.matches(descriptor))) {
322                 // TODO: we need to do more here, i think, to avoid duplicates since Inhabitants can layer on top of one another
323                 result.add(ai);
324             }
325         }
326     }

395     /**
396      * Add a habitat listener with contract-level filtering.
397      * <p/>
398      * The listener will be called based on the set of contract filters
399      * provided.
400      *
401      * @param listener  The habitat Listener to be added
402      * @param typeNames The contracts to filter on; this should be non-null
403      */
404     public void addHabitatListener(HabitatListener listener,
405                                    String... typeNames) {
406         if (null == typeNames || 0 == typeNames.length) {
407             throw new IllegalArgumentException();
408         }
409         addHabitatListener(listener,
410                 new HashSet<String>(Arrays.asList(typeNames)));
411     }

413     protected void addHabitatListener(HabitatListener listener, Set<String> typeNames) {
414         if (null == listener) {
415             throw new IllegalArgumentException();
416         }
417         
418         ExistingSingletonInhabitant<HabitatListener> inhabitant = new ExistingSingletonInhabitant<HabitatListener>(
419                 HabitatListener.class, listener, metaData(typeNames));
420         add(inhabitant);
421 
422         // track listeners by type - support of filtered notification
423         for (String contract : typeNames) {
424             ListenersByTypeInhabitant sameListeners;
425             synchronized (byContract) {
426                 sameListeners = (ListenersByTypeInhabitant) getInhabitantByContract(
427                         ListenersByTypeInhabitant.class.getName(), contract);
428                 if (null == sameListeners) {
429                     sameListeners = new ListenersByTypeInhabitant(contract);
430                     addIndex(sameListeners,
431                             ListenersByTypeInhabitant.class.getName(),
432                             contract, false); // don't send notifications for
433                     // this type (no particular
434                     // reason really)
435                 }
436             }
437             sameListeners.add(listener);
438         }
439     }

441     /**
442      * Remove a habitat listener.
443      *
444      * @param listener The habitat Listener to be removed
445      * @return true; if the listener was indeed removed
446      */
447     public boolean removeHabitatListener(HabitatListener listener) {
448         List<Inhabitant> list = byType._get(HabitatListener.class.getName());
449         List<Inhabitant> releaseList = new ArrayList<Inhabitant>();
450         Iterator<Inhabitant> iter = list.iterator();
451         while (iter.hasNext()) {
452             Inhabitant existing = iter.next();
453             if (null != existing && existing.get() == listener) {
454                 releaseList.add(existing);
455                 if (concurrencyControls) {
456                     list.remove(existing);
457                 } else {
458                     iter.remove();
459                 }
460             }
461         }
462 
463         // need to release the per-listener-contract entries too
464         for (Inhabitant released : releaseList) {
465             MultiMap<String, String> metadata = released.metadata();
466             if (null != metadata) {
467                 List<String> filters = metadata.get(Constants.OBJECTCLASS);
468                 for (String contract : filters) {
469                     ListenersByTypeInhabitant sameListeners;
470                     synchronized (byContract) {
471                         sameListeners = (ListenersByTypeInhabitant) getInhabitantByContract(
472                                 ListenersByTypeInhabitant.class.getName(),
473                                 contract);
474                     }
475                     sameListeners.remove(listener);
476                 }
477             }
478 
479             released.release();
480             notify(released, EventType.INHABITANT_REMOVED, null, released);
481         }
482 
483         return !releaseList.isEmpty();
484     }

601     protected void addIndex(Inhabitant<?> i, String index, String name, boolean notify) {
602         // TODO: it would be nice NOT to have to do this in the future!
603         if (AbstractInhabitantImpl.class.isInstance(i)) {
604             DescriptorImpl descriptor = (DescriptorImpl)i.getDescriptor();
605             if (null != descriptor) {
606                 if (null != index) {
607                     descriptor.addContract(index);
608                 }
609                 if (null != name) {
610                     descriptor.addName(name);
611                 }
612             }
613         }
614         
615         byContract.add(index, new NamedInhabitant(name, i));
616 
617         if (notify) {
618             notify(i, EventType.INHABITANT_INDEX_ADDED, index, name, null, null);
619         }
620     }

637         String name = inhabitant.typeName();
638         if (byType.remove(name, inhabitant)) {

646     /**
647      * Removes a NamedInhabitant for a specific contract
648      *
649      * @param index contract name
650      * @param name  instance name
651      * @return true if the removal was successful
652      */
653     public boolean removeIndex(String index, String name) {
654         boolean removed = false;
655         final List<NamedInhabitant> contracted = byContract._get(index);
656         if (!contracted.isEmpty()) {
657             Iterator<NamedInhabitant> iter = contracted.iterator();
658             while (iter.hasNext()) {
659                 NamedInhabitant i = iter.next();
660 
661                 if ((i.name == null && name == null)
662                         || (i.name != null && i.name.equals(name))) {
663                     if (concurrencyControls) {
664                         removed = contracted.remove(i);
665                         assert (removed);
666                     } else {
667                         iter.remove();
668                     }
669                     removed = true;
670                     notify(i.inhabitant, EventType.INHABITANT_INDEX_REMOVED,
671                             index, name, null, null);
672 
673                     // remember to remove the components stored under its type
674                     remove(i.inhabitant);
675                 }
676             }
677         }
678 
679         return removed;

682     /**
683      * Removes a Contracted service
684      *
685      * @param index               the contract name
686      * @param serviceOrInhabitant the service instance, or an Inhabitant instance
687      */
688     public boolean removeIndex(String index, Object serviceOrInhabitant) {
689         boolean removed = false;
690         if (byContract.containsKey(index)) {
691             List<NamedInhabitant> contracted = byContract._get(index);
692             Iterator<NamedInhabitant> iter = contracted.iterator();
693             while (iter.hasNext()) {
694                 NamedInhabitant i = iter.next();
695                 if (matches(i.inhabitant, serviceOrInhabitant)) {
696                     if (concurrencyControls) {
697                         removed = contracted.remove(i);
698                         assert (removed);
699                     } else {
700                         iter.remove();
701                     }
702                     removed = true;
703                     notify(i.inhabitant, EventType.INHABITANT_INDEX_REMOVED,
704                             index, null, service(serviceOrInhabitant), null);
705 
706                     // remember to remove the components stored under its type
707                     remove(i.inhabitant);
708                 }
709             }
710         }
711 
712         return removed;
713     }

715     protected boolean matches(Inhabitant<?> inhabitant,
716                               Object serviceOrInhabitant) {
717         boolean matches;
718         if (serviceOrInhabitant instanceof Inhabitant) {
719             matches = (serviceOrInhabitant == inhabitant);
720         } else {
721             // call to isActive is necessary to avoid loading of services that
722             // are not yet loaded.
723             matches = inhabitant.isActive()
724                     && inhabitant.get() == serviceOrInhabitant;
725         }
726         return matches;
727     }

731                 .isActive()) ? ((Inhabitant) serviceOrInhabitant).get()
732                 : serviceOrInhabitant;

838     private void doNotify(final NotifyCall innerCall,
839                           final Inhabitant<?> inhabitant, final EventType event,
840                           final String index,
841                           final Inhabitant<HabitatListener> extraListenerToBeNotified) {
842         final ListenersByTypeInhabitant sameListeners = (ListenersByTypeInhabitant) getInhabitantByContract(
843                 ListenersByTypeInhabitant.class.getName(), index);
844         if (null != sameListeners && !sameListeners.listeners.isEmpty()) {
845             getComponent(ExecutorService.class,
846                     Constants.EXECUTOR_HABITAT_LISTENERS_AND_TRACKERS).execute(
847                     new Runnable() {
848                         public void run() {
849                             Iterator<HabitatListener> iter = sameListeners.listeners
850                                     .iterator();
851                             while (iter.hasNext()) {
852                                 Object entry = iter.next();
853                                 HabitatListener listener = (entry instanceof HabitatListener) ? (HabitatListener) entry
854                                         : (HabitatListener) ((Inhabitant) entry)
855                                         .get();
856                                 try {
857                                     boolean keepMe = innerCall
858                                             .inhabitantChanged(listener);
859                                     if (!keepMe) {
860                                         removeHabitatListener(listener);
861                                     }
862                                 } catch (Exception e) {
863                                     // don't percolate the exception since it
864                                     // may negatively impact processing
865                                     logger.log(Level.WARNING,
866                                             "exception caught from listener: ",
867                                             e);
868                                 }
869                             }
870 
871                             // we take the extraListenerToBeNotified because
872                             // (a) we want to have all notifications in the
873                             // executor, and
874                             // (b) it might trigger an exception that we want to
875                             // trap
876                             if (null != extraListenerToBeNotified) {
877                                 extraListenerToBeNotified.get()
878                                         .inhabitantChanged(event, Habitat.this,
879                                                 extraListenerToBeNotified);
880                             }
881                         }
882                     });
883         }

1255     public Inhabitant getInhabitantByContract(String fullyQualifiedName,
1256                                               String name) {
1257         if (null == name || name.isEmpty() || !name.contains(",")) {
1258             for (NamedInhabitant i : byContract.get(fullyQualifiedName)) {
1259                 if (eq(i.name, name)) {
1260                     return i.inhabitant;
1261                 }
1262             }
1263         } else {
1264             String names[];
1265             if (null == name || name.equals("")) {
1266                 names = NULL_STR_ARR;
1267             } else {
1268                 names = name.split(",");
1269             }
1270 
1271             List<NamedInhabitant> list = byContract.get(fullyQualifiedName);
1272             for (String sname : names) {
1273                 sname = (null == sname || sname.equals("*")) ? null : sname;
1274                 for (NamedInhabitant i : list) {
1275                     String iname = i.name;
1276                     if (eq(iname, sname)) {
1277                         return i.inhabitant;
1278                     }
1279                 }
1280             }
1281         }

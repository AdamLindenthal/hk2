64      @SuppressWarnings({ "unchecked", "rawtypes" })
65      public static <T> Creator<T> create(Class<T> c, Habitat habitat, MultiMap<String,String> metadata) {
66          Factory f = c.getAnnotation(Factory.class);
67          if (f != null) {
68              return new FactoryCreator<T>(c,f.value(),habitat,metadata);
69          }
70  
71          Inhabitant factory = habitat.getInhabitantByAnnotation(FactoryFor.class, c.getName());
72          if (factory != null) {
73              return new FactoryCreator<T>(c,factory,habitat,metadata);
74          }
75  
76          // A constructor invoked on an abstract class will fail,
77          // so only return constructors for concrete classes
78          boolean isAbstract = Modifier.isAbstract(c.getModifiers());
79          
80          if (!isAbstract) {
81            Reference<Constructor<T>> defaultCtorRef = new Reference();
82            Reference<Constructor<T>> noArgCtorRef = new Reference();
83            qualifyingConstructors(c, null, getInjectionAnnotations(habitat), defaultCtorRef, noArgCtorRef);
84            if (defaultCtorRef.get() != null) {
85              return new InjectableParametizedConstructorCreator<T>(c, defaultCtorRef.get(), habitat, metadata);
86            }
87            if (noArgCtorRef.get() != null) {
88              return new ConstructorCreator<T>(c,habitat,metadata);
89            }
90          }
91          
92          return null;
93      }

95      /**
96       * Returns all currently available injection annotations
97       *
98       */
99      private static Set<Class<? extends Annotation>> getInjectionAnnotations(Habitat habitat) {
100 
101         Set<Class<? extends Annotation>> annotations = new HashSet<Class<? extends Annotation>>();
102         for (Inhabitant<? extends InjectionResolver> resolver : getAllInjectionResolvers(habitat)) {
103             annotations.add(resolver.get().type);
104         }
105         return annotations;
106     }

108     /**
109      * Returns all currently available injection annotations
110      * @param habitat the service registry
111      * @return the list of injection resolvers registered in this service registry in its parent.
112      *
113      */
114     public static List<Inhabitant<? extends InjectionResolver>> getAllInjectionResolvers(Habitat habitat) {
115 
116         List<Inhabitant<? extends InjectionResolver>> resolvers = new ArrayList<Inhabitant<? extends InjectionResolver>>();
117 
118         // we treat the @Inject injection resolver specially as the other injectors might be
119         // using default HK2 injection themselves.
120         Inhabitant<InjectionResolver> specialResolver = habitat.getInhabitant(InjectionResolver.class, "Inject");
121         specialResolver.get(); // force initialization
122         resolvers.add(specialResolver);
123         for (Inhabitant<? extends InjectionResolver> injectionResolver : habitat.getInhabitants(InjectionResolver.class)) {
124             if (injectionResolver!=specialResolver)
125                 resolvers.add(injectionResolver);
126         }
127 
128         if (habitat.getDefault()!=habitat) {
129             for (Provider<? extends InjectionResolver> injectionResolver : habitat.getDefault().forContract(InjectionResolver.class).all()) {
130                 resolvers.add((Inhabitant<? extends InjectionResolver>) injectionResolver);
131             }
132         }
133         return resolvers;
134     }

136     /**
137      * Returns all "eligible" constructors for injection. Eligibility is
138      * determined by:
139      * 
140      * (a) the no-arg constructor, or (b) @Inject annotated constructor.
141      * 
142      * </p>
143      * Note, however, that this does not verify that the arguments on the
144      * constructor are indeed injectable out of the habitat.
145      * 
146      * @param c
147      *      The class to find the constructors for
148      * @param defaultCtor
149      *      (optional) The "favored" default constructor reference; meaning that all arguments can be injected based on the contents of the habitat
150      * @param noArgCtor
151      *      (optional The no-arg constructor reference
152      * @return
153      *      The list of all eligible constructors
154      */
155     @SuppressWarnings("unchecked")
156     public static <T> void qualifyingConstructors(Class<T> c,
157             List<Constructor<?>> allList,
158             Set<Class<? extends Annotation>> injectionAnnotations,
159             Reference<Constructor<T>> defaultCtor,
160             Reference<Constructor<T>> noArgCtor) {
161         Constructor<T>[] ctors;
162         try {
163             ctors = (Constructor<T>[]) c.getDeclaredConstructors();
164         } catch(Exception e) {
165             ctors = (Constructor<T>[]) c.getConstructors();
166         }
167         
168         for (Constructor<T> ctor : ctors) {
169             if (0 == ctor.getParameterTypes().length) {
170                 if (null != allList) {
171                     allList.add(ctor);
172                 }
173                 if (null != noArgCtor) {
174                     noArgCtor.set(ctor);
175                 }
176             } else if (isInjectable(injectionAnnotations, ctor)) {
177                 if (null != allList) {
178                     allList.add(ctor);
179                 }
180                 if (null != defaultCtor && null == defaultCtor.get()) {
181                     defaultCtor.set(ctor);
182                 }
183             } else {
184                 boolean allInjectable = true;
185                 for (Annotation[] a : ctor.getParameterAnnotations()) {
186                     if (!isInject(injectionAnnotations, a)) {
187                         allInjectable = false;
188                         break;
189                     }
190                 }
191                 if (allInjectable) {
192                     if (null != allList) {
193                         allList.add(ctor);
194                     }
195                     if (null != defaultCtor && null == defaultCtor.get()) {
196                         defaultCtor.set(ctor);
197                     }
198                 }
199             }
200         }
201     }

203     private static boolean isInject(Set<Class<? extends Annotation>> injectionAnnotations, Annotation... a) {
204         for (Annotation ae : a) {
205             if (injectionAnnotations.contains(ae.annotationType()) ||
206                     ae.annotationType().equals(javax.inject.Inject.class)) {
207                 return true;
208             }
209         }
210         
211         return false;
212     }
213 
214     public static boolean isInjectable(Set<Class<? extends Annotation>> injectionAnnotations,AnnotatedElement ae) {
215         for (Class<? extends Annotation> injectionAnnotation : injectionAnnotations) {
216             if (ae.getAnnotation(injectionAnnotation)!=null) return true;
217         }
218         return (null != ae.getAnnotation(Inject.class) ||
219                 null != ae.getAnnotation(javax.inject.Inject.class));
220     }

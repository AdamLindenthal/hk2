37  package org.jvnet.hk2.component;
38  
39  import java.util.ArrayList;
40  import java.util.Collections;
41  import java.util.HashMap;
42  import java.util.List;
43  import java.util.Map;
44  import java.util.Map.Entry;
45  import java.util.Set;
46  import java.util.Collection;
47  import java.io.Serializable;
48  
49  /***
50   * Map from a key to multiple values.
51   * Order is significant among values, and null values are allowed, although null keys are not.
52   * 
53   * @author Kohsuke Kawaguchi
54   */
55  public class MultiMap<K,V> implements Serializable {
56      private final Map<K,List<V>> store;
57  
58      public final String toString() {
59          final StringBuilder builder = new StringBuilder();
60          
61          final String newline = System.getProperty( "line.separator" );
62          builder.append( "{" );
63          for ( final K key : store.keySet() ) {
64              builder.append( key  + ": {" );
65              for( final V value : store.get(key) ) {
66                  builder.append( value.toString() + "," );
67              }
68              // trailing comma is OK
69              builder.append( "}" + newline );
70          }
71          builder.append( "}" );
72          return builder.toString();
73      }
74      
75      /***
76       * Creates an empty multi-map.
77       */
78      public MultiMap() {
79          store = new HashMap<K, List<V>>();
80      }
81  
82      /***
83       * Creates a multi-map backed by the given store.
84       */
85      private MultiMap(Map<K,List<V>> store) {
86          this.store = store;
87      }
88  
89      /***
90       * Copy constructor.
91       */
92      public MultiMap(MultiMap<K,V> base) {
93          this();
94          for (Entry<K, List<V>> e : base.entrySet())
95              store.put(e.getKey(),new ArrayList<V>(e.getValue()));
96      }
97  
98      /***
99       * Adds one more value.
100      */
101     public final void add(K k,V v) {
102         List<V> l = store.get(k);
103         if(l==null) {
104             l = new ArrayList<V>();
105             store.put(k,l);
106         }
107         l.add(v);
108     }
109 
110     /***
111      * Replaces all the existing values associated with the key
112      * by the given value.
113      *
114      * @param v
115      *      Can be null or empty.
116      */
117     public void set(K k, Collection<? extends V> v) {
118         store.put(k,new ArrayList<V>(v));
119     }
120 
121     /***
122      * Replaces all the existing values associated wit hthe key
123      * by the given single value.
124      *
125      * <p>
126      * This is short for <tt>set(k,Collections.singleton(v))</tt>
127      */
128     public void set(K k, V v) {
129         ArrayList<V> vlist = new ArrayList<V>(1);
130         vlist.add(v);
131         store.put(k, vlist);
132     }
133 
134     /***
135      * @return
136      *      Can be empty but never null. Read-only.
137      */
138     public final List<V> get(K k) {
139         List<V> l = store.get(k);
140         if(l==null) return Collections.emptyList();
141         return l;
142     }
143 
144     /***
145      * Checks if the map contains the given key.
146      */
147     public boolean containsKey(K key) {
148         return !get(key).isEmpty();
149     }
150 
151     /***
152      * Gets the first value if any, or null.
153      * <p>
154      * This is useful when you know the given key only has one value and you'd like
155      * to get to that value.
156      *
157      * @return
158      *      null if the key has no values or it has a value but the value is null.
159      */
160     public final V getOne(K k) {
161         List<V> lst = store.get(k);
162         if(lst ==null)  return null;
163         if(lst.isEmpty())   return null;
164         return lst.get(0);
165     }
166 
167     /***
168      * Lists up all entries.
169      */
170     public Set<Entry<K,List<V>>> entrySet() {
171         return store.entrySet();
172     }
173 
174     /***
175      * Format the map as "key=value1,key=value2,...."
176      */
177     public String toCommaSeparatedString() {
178         StringBuilder buf = new StringBuilder();
179         for (Entry<K,List<V>> e : entrySet()) {
180             for (V v : e.getValue()) {
181                 if(buf.length()>0)  buf.append(',');
182                 buf.append(e.getKey()).append('=').append(v);
183             }
184         }
185         return buf.toString();
186     }
187 
188     /***
189      * Creates a copy of the map that contains the exact same key and value set.
190      * Keys and values won't cloned.
191      */
192     public MultiMap<K,V> clone() {
193         return new MultiMap<K,V>(this);
194     }
195 
196     /***
197      * Returns the size of the map
198      * @return integer or 0 if the map is empty
199      */
200     public int size() {
201         return store.size();
202     }
203 
204     private static final MultiMap EMPTY = new MultiMap(Collections.emptyMap());
205 
206     /***
207      * Gets the singleton read-only empty multi-map.
208      *
209      * @see Collections#emptyMap()
210      */
211     public static <K,V> MultiMap<K,V> emptyMap() {
212         return EMPTY;
213     }
214 }

50  /***
51   * Map from a key to multiple values.
52   * Order is significant among values, and null values are allowed, although null keys are not.
53   * 
54   * @author Kohsuke Kawaguchi
55   * @author Jerome Dochez
56   */
57  public class MultiMap<K,V> implements Serializable {
58      private final Map<K,List<V>> store;
59      private final boolean concurrencyControls;
60  
61      public final String toString() {
62          final StringBuilder builder = new StringBuilder();
63          final String newline = System.getProperty( "line.separator" );
64          builder.append("{");
65          for ( final K key : store.keySet() ) {
66              builder.append(key).append(": ");
67              builder.append(store.get(key));
68              builder.append(newline);
69          }
70          builder.append("}");
71          return builder.toString();
72      }
73      
74      /***
75       * Creates an empty multi-map with default concurrency controls
76       */
77      public MultiMap() {
78         this(new HashMap<K, List<V>>(), Habitat.CONCURRENCY_CONTROLS_DEFAULT);
79      }
80  
81      /***
82       * Creates an empty multi-map with option to use concurrency controls.
83       * Concurrency controls applies to the inner List collection held per each key.
84       * There are currently no concurrency controls around the Map portion of the data
85       * structure.
86       */
87      MultiMap(Map<K,List<V>> store) {
88        this(store, Habitat.CONCURRENCY_CONTROLS_DEFAULT);
89      }
90  
91      /***
92       * Creates an empty multi-map with option to use concurrency controls
93       */
94      MultiMap(boolean concurrencyControls) {
95        this(new HashMap<K, List<V>>(), concurrencyControls);
96      }
97      
98      
99      /***
100      * Creates a multi-map backed by the given store.
101      * @param store map to copy
102      */
103     protected MultiMap(Map<K,List<V>> store, boolean concurrencyControls) {
104         this.store = store;
105         this.concurrencyControls = concurrencyControls;
106     }
107 
108     /***
109      * Copy constructor.
110      * @param base map to copy
111      */
112     public MultiMap(MultiMap<K,V> base) {
113         this();
114         for (Entry<K, List<V>> e : base.entrySet())
115             store.put(e.getKey(), newList(e.getValue()));
116     }
117 
118     /***
119      * Creates an optionally populated list to be used as an entry in the map.
120      * @param initialVal
121      * @return
122      */
123     protected List<V> newList(Collection<? extends V> initialVals) {
124       if (concurrencyControls) {
125         if (null == initialVals) {
126           return new CopyOnWriteArrayList<V>();
127         } else {
128           return new CopyOnWriteArrayList<V>(initialVals);
129         }
130       } else {
131         if (null == initialVals) {
132           return new ArrayList<V>(1);
133         } else {
134           return new ArrayList<V>(initialVals);
135         }
136       }
137     }
138     
139     /***
140      * Adds one more key-value pair.
141      * @param k key to store the entry under
142      * @param v value to store in the k's values.
143      */
144     public final void add(K k,V v) {
145         List<V> l = store.get(k);
146         if (l==null) {
147             l = newList(null);
148             store.put(k,l);
149         }
150         l.add(v);
151     }
152 
153     /***
154      * Replaces all the existing values associated with the key
155      * by the given value.
156      *
157      * @param k key for the values
158      * @param v
159      *      Can be null or empty.
160      */
161     public void set(K k, Collection<? extends V> v) {
162         store.put(k, newList(v));
163     }
164 
165     /***
166      * Replaces all the existing values associated with the key
167      * by the given single value.
168      *
169      * @param k key for the values
170      * @param v singleton value for k key
171      * <p>
172      * This is short for <tt>set(k,Collections.singleton(v))</tt>
173      */
174     public void set(K k, V v) {
175         List<V> vlist = newList(null);
176         vlist.add(v);
177         store.put(k, vlist);
178     }
179 
180     /***
181      * Returns the elements indexed by the provided key
182      * @param k key for the values
183      * @return
184      *      Can be empty but never null. Read-only.
185      */
186     public final List<V> get(K k) {
187         List<V> l = store.get(k);
188         if (l==null) return Collections.emptyList();
189         return l;
190     }
191 
192     /***
193      * Package private (for getting the raw map for direct manipulation by the habitat)
194      * @param k the key
195      * @return
196      */
197     final List<V> _get(K k) {
198       List<V> l = store.get(k);
199       if (l==null) return Collections.emptyList();
200       return l;
201     }

203     /***
204      * Checks if the map contains the given key.
205      * @param k key to test
206      * @return true if the map contains at least one element for this key
207      */
208     public boolean containsKey(K k) {
209         return !get(k).isEmpty();
210     }
211 
212     /***
213      * Removes an key value from the map
214      * @param key key to be removed
215      * @return the value stored under this key or null if there was none
216      */
217     public List<V> remove(K key) {
218         return store.remove(key);
219     }
220 
221     /***
222      * Removes an key value pair from the map
223      * @param key key to be removed
224      * @param entry the entry to be removed from the key'ed list
225      * @return true if there was none that was deleted
226      */
227     public boolean remove(K key, V entry) {
228       List<V> list = store.get(key);
229       return (null == list) ? false : list.remove(entry);
230     }
231 
232     /***
233      * Gets the first value if any, or null.
234      * <p>
235      * This is useful when you know the given key only has one value and you'd like
236      * to get to that value.
237      *
238      * @param k key for the values
239      * @return
240      *      null if the key has no values or it has a value but the value is null.
241      */
242     public final V getOne(K k) {
243         List<V> lst = store.get(k);
244         if(lst ==null)  return null;
245         if(lst.isEmpty())   return null;
246         return lst.get(0);
247     }
248 
249     /***
250      * Lists up all entries.
251      * @return a {@link java.util.Set} of {@link java.util.Map.Entry} of entries
252      */
253     public Set<Entry<K,List<V>>> entrySet() {
254         return store.entrySet();
255     }
256 
257     /***
258      * @return the map as "key=value1,key=value2,...."
259      */
260     public String toCommaSeparatedString() {
261         StringBuilder buf = new StringBuilder();
262         for (Entry<K,List<V>> e : entrySet()) {
263             for (V v : e.getValue()) {
264                 if(buf.length()>0)  buf.append(',');
265                 buf.append(e.getKey()).append('=').append(v);
266             }
267         }
268         return buf.toString();
269     }

275     public MultiMap<K,V> clone() {
276         return new MultiMap<K,V>(this);
277     }

284         return store.size();
285     }

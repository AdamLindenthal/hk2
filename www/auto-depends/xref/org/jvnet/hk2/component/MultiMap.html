47  /**
48   * Map from a key to multiple values.

123         if (!MultiMap.class.isInstance(another)) {
124             return false;
125         }
126         
127         @SuppressWarnings("rawtypes")
128         MultiMap other = MultiMap.class.cast(another);
129         if (size() != other.size()) {

173     /**
174      * Creates an optionally populated list to be used as an entry in the map.
175      *
176      * @param initialVal
177      * @return
178      */
179     protected List<V> newList(Collection<? extends V> initialVals) {
180         if (concurrencyControls) {
181             if (null == initialVals) {
182                 return new CopyOnWriteArrayList<V>();
183             } else {
184                 return new CopyOnWriteArrayList<V>(initialVals);
185             }
186         } else {
187             if (null == initialVals) {
188                 return new ArrayList<V>(1);
189             } else {
190                 return new ArrayList<V>(initialVals);
191             }
192         }
193     }

206     public final void add(K k, V v) {
207         if (!modifiable) {
208             throw new UnsupportedOperationException("unmodifiable collection");
209         }
210         
211         List<V> l = store.get(k);
212         if (l == null) {
213             l = newList(null);
214             store.put(k, l);
215         }
216         l.add(v);
217     }

260     /**
261      * Returns the union of all elements indexed by the provided keys.
262      * This is a disjunctive operation.
263      *  
264      * @param keys the collection of keys
265      * @return the union collection of values
266      */
267     public List<V> getUnionOfAll(Collection<K> keys) {
268         LinkedHashSet<V> set = new LinkedHashSet<V>();
269         for (K k : keys) {
270             set.addAll(get(k));
271         }
272         return Collections.unmodifiableList(new ArrayList<V>(set));
273     }

275     /**
276      * Returns the intersection of all elements indexed by the provided keys.
277      * That means that all values must appear for each and every key specified.
278      * This is a conjunctive operation.
279      *  
280      * @param keys the collection of keys
281      * @return the intersecting collection of values
282      */
283     public List<V> getIntersectionOfAll(Collection<K> keys) {
284         if (1 == keys.size()) {
285             return get(keys.iterator().next());
286         }
287         
288         LinkedHashSet<V> set = new LinkedHashSet<V>();
289         for (K k : keys) {
290             List<V> vals = get(k);
291             
292             if (vals.isEmpty()) {
293                 return Collections.emptyList();
294             }
295             
296             if (set.isEmpty()) {
297                 // 1st iteration
298                 set.addAll(vals);
299             } else {
300                 // >1st iteration
301                 if (1 == set.size()) {
302                   V item = set.iterator().next();
303                   if (vals.contains(item)) {
304                       set.clear();
305                       set.add(item);
306                   } else {
307                       return Collections.emptyList();
308                   }
309                 } else if (1 == vals.size()) {
310                     V item = vals.iterator().next();
311                     if (set.contains(item)) {
312                         set.clear();
313                         set.add(item);
314                     } else {
315                         return Collections.emptyList();
316                     }
317                 } else {
318                     Iterator<V> iter = set.iterator();
319                     while (iter.hasNext()) {
320                         V item = iter.next();
321                         if (!vals.contains(item)) {
322                             iter.remove();
323                         }
324                         if (set.isEmpty()) {
325                             return Collections.emptyList();
326                         }
327                     }
328                 }
329             }
330         }
331 
332         if (set.isEmpty()) {
333             return Collections.emptyList();
334         }
335 
336         return Collections.unmodifiableList(new ArrayList<V>(set));
337     }

339     public void mergeAll(org.glassfish.hk2.MultiMap<K, V> another) {
340         if (null != another) {
341             for (Entry<K, List<V>> entry : another.entrySet()) {
342                 List<V> ourList = store.get(entry.getKey());
343                 if (null == ourList) {
344                     ourList = newList(entry.getValue());
345                     store.put(entry.getKey(), ourList);
346                 } else {
347                     for (V v : entry.getValue()) {
348                         if (!ourList.contains(v)) {
349                             ourList.add(v);
350                         }
351                     }
352                 }
353             }
354         }
355     }

431     @Override
432     public V getFirst(K k) {
433         List<V> lst = store.get(k);
434         if (null == lst) {
435             return null;
436         }

438         if (lst.isEmpty()) {

455     /**

457      */
458     public String toCommaSeparatedString() {
459         StringBuilder buf = new StringBuilder();
460         for (Entry<K, List<V>> e : entrySet()) {
461             for (V v : e.getValue()) {
462                 if (buf.length() > 0) {
463                     buf.append(',');
464                 }
465                 buf.append(e.getKey()).append('=').append(v);
466             }
467         }

77  /**
78   * Responsible for generating the collection of inhabitants, decoupling from
79   * implementation detail for the caller.
80   * <p/>
81   * The caller is expected to continually build up the InhabitantsGenerator
82   * context by calling add*(), followed by calling getModelInhabitants() to
83   * obtain the progenitors of the inhabitants.
84   * <p/>
85   * There are two ways to close this instance, either through {@link #getContext()}
86   * or through calling {@link #close()} directly.
87   *
88   * @author Jerome Dochez, Jeff Trent
89   */
90  public abstract class InhabitantsParsingContextGenerator implements Closeable {
91  
92      private static final Logger logger = 
93        Logger.getLogger(InhabitantsParsingContextGenerator.class.getName());
94  
95      private Parser parser;
96      private final ParsingContext context;
97  
98      private final LinkedHashMap<String, InhabitantsScanner> metaInfScanners = 
99              new LinkedHashMap<String, InhabitantsScanner>();

103     private final LinkedHashSet<URI> parsed = new LinkedHashSet<URI>();

106     /**
107      * Factory for the {@link InhabitantsParsingContextGenerator}
108      *
109      * @param h habitat not currently used; reserved for future use
110      * @param es the executor to use for any async processing (e.g., parsing)
111      * @param inhabitantsClassPath the fully qualified classpath in order to resolve the class-model
112      * @param inhabitantsClassPathFilter the filter used for pruning the classpath (may also be a {@link ClassPathAdvisor})
113      * 
114      * @return an empty context InhabitantsGenerator
115      */
116     public static InhabitantsParsingContextGenerator create(Habitat h, 
117             ExecutorService es,
118             ClassPath inhabitantsClassPath,
119             FileFilter inhabitantsClassPathFilter) {
120         return new InhabitantsParsingContextGenerator(es, inhabitantsClassPath, inhabitantsClassPathFilter) {};
121     }

123     protected InhabitantsParsingContextGenerator(final ExecutorService es,
124             final ClassPath inhabitantsClassPath,
125             final FileFilter inhabitantsClassPathFilter) {
126         // setup the parser
127         ParsingContext.Builder builder = new ParsingContext.Builder();
128         final Set<String> annotations = new HashSet<String>();
129         annotations.add(Contract.class.getName());
130         annotations.add(Service.class.getName());
131         annotations.add(InhabitantMetadata.class.getName());
132         annotations.add(RunLevel.class.getName());
133         annotations.add("org.jvnet.hk2.config.Configured");
134 
135         builder.config(new ParsingConfig() {
136             final Set<String> empty = Collections.emptySet();
137 
138             public Set<String> getAnnotationsOfInterest() {
139                 return empty;
140             }
141 
142             public Set<String> getTypesOfInterest() {
143                 return annotations;
144             }
145 
146             @Override
147             public boolean modelUnAnnotatedMembers() {
148                 return false;
149             }
150         });
151 
152         // optionally provide an executor
153         builder.executorService(es);
154         
155         // Important Note:
156         // we are careful not to create a locator if we don't have to because it will
157         // mean additional burden on the parser since it will need to locate and resolve
158         // all of the dangling types.  If we are being called in unit-test mode then
159         // the chances for this is very unlikely because we have (or should have) a
160         // "full" testing classpath.
161         
162         Locator locator = null;
163         if (null == inhabitantsClassPathFilter) {
164           this.inhabitantsClassPathFilter = null;
165           locator = (null == inhabitantsClassPath) ? null : new Locator(inhabitantsClassPath);
166         } else {
167           this.inhabitantsClassPathFilter = inhabitantsClassPathFilter;
168           if (ClassPathAdvisor.class.isInstance(this.inhabitantsClassPathFilter)) {
169             ClassPathAdvisor advisor = ClassPathAdvisor.class.cast(this.inhabitantsClassPathFilter);
170             advisor.starting(inhabitantsClassPath);
171           }
172 
173           if (null != inhabitantsClassPath) {
174             LinkedHashSet<File> accepted = new LinkedHashSet<File>();
175             for (File f : inhabitantsClassPath.getFileEntries()) {
176               if (inhabitantsClassPathFilter.accept(f)) {
177                 accepted.add(f);
178               }
179             }
180             
181             locator = new Locator(ClassPath.create(null, accepted));
182           }
183         }
184         
185         builder.locator(locator);
186         
187         this.context = builder.build();
188         this.parser = new Parser(context);
189     }

191     /**
192      * Add the collection of files to the current InhabitantsGenerator context.
193      *
194      * @param files the files to parse.
195      * @throws IOException
196      */
197     public void parse(Collection<File> files) throws IOException {
198         for (File file : files) {
199             parse(file);
200         }
201     }

203     /**
204      * Retrieves the parsing context that can be used for model generation elsewhere.
205      * <b>Note that this can be called at most once and then this instance is implicitly
206      * closed.</b>
207      *
208      * @return the parsing context given the code sources provided
209      */
210     public ParsingContext getContext() {
211         if (null != parser) {
212           try {
213             parser.awaitTermination();
214           } catch (InterruptedException e) {
215             close();
216             throw new RuntimeException(e);
217           }
218         }
219         
220         if (ClassPathAdvisor.class.isInstance(this.inhabitantsClassPathFilter)) {
221           ClassPathAdvisor advisor = ClassPathAdvisor.class.cast(this.inhabitantsClassPathFilter);
222           advisor.finishing(getSignificantURIReferences(), getInsignificantURIReferences());
223           
224           // we don't want to be notified again
225           this.inhabitantsClassPathFilter = null;
226         }
227         
228         return context;
229     }

231     /**
232      * Given the set of annotations that comprise Hk2, as well as the provided classpath / files to
233      * introspect, return the set of URIs that actually reference something of "significant value"
234      * pertaining to habitat creation.
235      */
236     private Set<URI> getSignificantURIReferences() {
237       LinkedHashSet<URI> result = new LinkedHashSet<URI>();
238 
239       Types types = context.getTypes();
240       for (String annotation : context.getConfig().getTypesOfInterest()) {
241         AnnotationType atype = (AnnotationType) types.getBy(annotation);
242         if (null != atype) {
243           Collection<AnnotatedElement> coll = atype.allAnnotatedTypes();
244           for (AnnotatedElement ae : coll) {
245             Type type = types.getBy(ae.getName());
246             if (null != type) {
247               for (URI uri : type.getDefiningURIs()) {
248                 try {
249                   result.add(new File(uri).getCanonicalFile().toURI());
250                 } catch (IOException e) {
251                   throw new RuntimeException(e);
252                 }
253               }
254             }
255           }
256         }
257       }
258       
259       return Collections.unmodifiableSet(result);
260     }
261 
262     /**
263      * Given the set of annotations that comprise Hk2, as well as the provided classpath / files to
264      * introspect, return the set of URIs that did not provide any "significant value"
265      * pertaining to habitat creation.
266      */
267     private Set<URI> getInsignificantURIReferences() {
268       LinkedHashSet<URI> result = new LinkedHashSet<URI>();
269 
270       result.addAll(parsed);
271       result.removeAll(getSignificantURIReferences());
272       
273       return Collections.unmodifiableSet(result);
274     }

283     protected void addInhabitantsScanner(String name, InhabitantsScanner is) {
284         synchronized (metaInfScanners) {
285             if (!metaInfScanners.containsKey(name)) {
286                 metaInfScanners.put(name, is);
287             }
288         }
289     }

291     /**
292      * Eventually we can perform optimizations here instead of a pass-thru to parseAlways.
293      */
294     public void parse(File f) throws IOException {
295       if (null == parser) {
296         throw new IllegalStateException("parser closed");
297       }
298       
299       f = f.getCanonicalFile();
300       if (null == inhabitantsClassPathFilter || inhabitantsClassPathFilter.accept(f)) {
301         parseAlways(parser, f);
302       }
303     }
304 
305     protected void parseAlways(Parser parser, final File f) throws IOException {
306       logger.log(Level.FINE, "introspecting {0}", f);
307       parsed.add(f.toURI());
308       try {
309         parser.parse(f, new Runnable() {
310             public void run() {
311                 logger.log(Level.FINER, "Finished introspecting {0}", f.getName());
312             }
313         });
314       } catch (IOException e) {
315         logger.log(Level.WARNING, "problem during parsing - closing prematurely", e);
316         close();
317       }
318     }
319 
320     @Override
321     public void close() {
322       if (null != parser) {
323         parser.close();
324         parser = null;
325       }
326     }

329     private static class Locator implements ResourceLocator {
330       private final ClassLoader resourceLoader;
331       
332       public Locator(ClassPath inhabitantsClassPath) {
333         try {
334           resourceLoader = new URLClassLoader(inhabitantsClassPath.getRawURLs(), null);
335           
336           if (logger.isLoggable(Level.FINER)) {
337             logger.log(Level.FINER, "resourceLoader is {0}", Arrays.asList(inhabitantsClassPath.getRawURLs()).toString());
338           }
339         } catch (IOException e) {
340           throw new RuntimeException(e);
341         }
342       }
343 
344       @Override
345       public InputStream openResourceStream(String name) {
346         if (name.startsWith("java/")) {
347           logger.log(Level.FINE, "skipping {0}", name);
348           return null;  // wasteful to parse these
349         }
350         if (name.indexOf(".")==-1) {
351           return null; // intrinsic types.
352         }
353 
354         logger.log(Level.FINE, "loading resource {0}", name);
355         
356         try {
357           InputStream resource = resourceLoader.getResourceAsStream(name);
358           logger.log(Level.FINE, "resource {0} resolved to {1}", new Object[] {name, resource});
359           return resource;
360         } catch (Exception e) {
361           throw new RuntimeException("failed while getting resource: " + name);
362         }
363       }
364 
365       @Override
366       public URL getResource(String name) {
367         if (name.startsWith("java/")) {
368           logger.log(Level.FINE, "skipping {0}", name);
369           return null;  // wasteful to parse these
370         }
371         if (name.indexOf(".")==-1) {
372           return null; // intrinsic types.
373         }
374         
375         return resourceLoader.getResource(name);
376       }
377     }

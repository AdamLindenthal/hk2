111     public <T extends V> ResolvedBinder<T> toFactory(final org.glassfish.hk2.Factory<T> provider) {
112 
113         AbstractResolvedBinder<T> resolvedBinder = new AbstractResolvedBinder<T>(this) {
114             @Override
115             void registerIn(Habitat habitat) {
116                 MultiMap<String, String> metadataMap = super.populateMetadata();
117                 Class contractType=null;
118                 Iterator<Class> itrClasses = contracts.values().iterator();
119                 while(itrClasses.hasNext() && contractType==null) {
120                     contractType = itrClasses.next();
121                 }
122                 if (contractType==null) 
123                     throw new RuntimeException("You must use a bind(Class contractType) API to bind a Provider");
124                 Inhabitant<T> inh = new AbstractCreatorImpl<T>(contractType, habitat, metadataMap) {
125                     @Override
126                     public T create(Inhabitant onBehalfOf) throws ComponentException {
127                         T t = provider.get();
128                         inject(habitat, t, onBehalfOf);
129                         return t;
130                     }
131                 };
132                 super.registerIn(habitat, inh);
133             }

139     @Override
140     public <T extends V> ResolvedBinder<T> toFactory(final Class<? extends org.glassfish.hk2.Factory<? extends T>> factoryType) {
141         AbstractResolvedBinder<T> resolvedBinder = new AbstractResolvedBinder<T>(this) {
142             @Override
143             void registerIn(Habitat habitat) {
144                 MultiMap<String, String> metadataMap = super.populateMetadata();
145                 Class contractType=null;
146                 Iterator<Class> itrClasses = contracts.values().iterator();
147                 while(itrClasses.hasNext() && contractType==null) {
148                     contractType = itrClasses.next();
149                 }
150                 if (contractType==null) throw new RuntimeException("You must use a bind(Class contractType) API to bind a Provider");
151 
152                 Inhabitant<T> inh = new AbstractCreatorImpl<T>(contractType, habitat, metadataMap) {
153                     @Override
154                     public T create(Inhabitant onBehalfOf) throws ComponentException {
155                         Inhabitant<? extends org.glassfish.hk2.Factory<? extends T>> factoryInhabitant =
156                                 habitat.getInhabitantByType(factoryType);
157                         if (factoryInhabitant==null) {
158                             factoryInhabitant = new ConstructorCreator<Factory<? extends T>>(factoryType, habitat, null);
159                         }
160                         Factory<? extends T> f = factoryInhabitant.get();
161                         T t = f.get();
162                         inject(habitat, t, onBehalfOf);
163                         return t;
164                     }
165                 };
166                 super.registerIn(habitat, inh);
167             }
168         };
169         owner.add(resolvedBinder);
170         return resolvedBinder;
171     }

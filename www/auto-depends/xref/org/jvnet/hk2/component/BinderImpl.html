91          AbstractResolvedBinder<V> resolvedBinder = new ClassNameBasedBinder(this, className,

125     @Override
126     public <T extends V> ResolvedBinder<T> toFactory(final org.glassfish.hk2.Factory<T> provider) {
127 
128         AbstractResolvedBinder<T> resolvedBinder = new AbstractResolvedBinder<T>(this) {
129 
130             @Override
131             Inhabitant<?> registerIn(Habitat habitat) {
132                 MultiMap<String, String> metadataMap = super.populateMetadata();
133                 Class contractType = null;
134                 Iterator<Class> itrClasses = contracts.values().iterator();
135                 while (itrClasses.hasNext() && contractType == null) {
136                     contractType = itrClasses.next();
137                 }
138                 if (contractType == null) {
139                     throw new RuntimeException("You must use a bind(Class contractType) API to bind a Provider");
140                 }
141                 Inhabitant<T> inh = com.sun.hk2.component.Inhabitants.wrapByScope(new AbstractCreatorImpl<T>(contractType, habitat, metadataMap) {
142 
143                     @Override
144                     public T create(Inhabitant onBehalfOf) throws ComponentException {
145                         T t = provider.get();
146                         if (t != null) {
147                             inject(habitat, t, onBehalfOf);
148                         }
149                         return t;
150                     }
151                 }, habitat, metadata.scope);
152                 super.registerIn(habitat, inh);
153                 return inh;
154             }
155         };
156         owner.add(resolvedBinder);
157         return resolvedBinder;
158     }

160     @Override
161     public <T extends V> ResolvedBinder<T> toFactory(final Class<? extends Factory<? extends T>> factoryType) {
162         AbstractResolvedBinder<T> resolvedBinder = new AbstractResolvedBinder<T>(this) {
163 
164             @Override
165             Inhabitant<?> registerIn(Habitat habitat) {
166                 MultiMap<String, String> metadataMap = super.populateMetadata();
167                 Class contractType = null;
168                 Iterator<Class> itrClasses = contracts.values().iterator();
169                 while (itrClasses.hasNext() && contractType == null) {
170                     contractType = itrClasses.next();
171                 }
172                 if (contractType == null) {
173                     throw new RuntimeException("You must use a bind(Class contractType) API to bind a Provider");
174                 }
175 
176                 Inhabitant<T> inh = com.sun.hk2.component.Inhabitants.wrapByScope(new AbstractCreatorImpl<T>(contractType, habitat, metadataMap) {
177 
178                     @Override
179                     public T create(Inhabitant onBehalfOf) throws ComponentException {
180                         Inhabitant<? extends Factory<? extends T>> factoryInhabitant =
181                                 habitat.getInhabitantByType(factoryType);
182                         if (factoryInhabitant == null) {
183                             factoryInhabitant = Creators.create(factoryType, habitat, null);
184                         }
185                         Factory<? extends T> f = factoryInhabitant.get();
186                         T t = f.get();
187                         if (t != null) {
188                             inject(habitat, t, onBehalfOf);
189                         }
190                         return t;
191                     }
192                 }, habitat, metadata.scope);
193                 super.registerIn(habitat, inh);
194                 return inh;
195             }
196         };
197         owner.add(resolvedBinder);
198         return resolvedBinder;
199     }

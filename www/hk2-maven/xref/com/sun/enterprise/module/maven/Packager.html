62  /***
63   * Prepares manifest entries in {@link MavenArchiveConfiguration}.
64   *
65   * <p>
66   * This class is moved outside {@code PackageMojo} and doesn't rely
67   * on Mojo resource injection to allow other plugins to reuse
68   * this capability without hitting http://jira.codehaus.org/browse/MNG-2789.
69   *
70   * @author Kohsuke Kawaguchi
71   */
72  public class Packager {
73      /***
74       * Creates META-INF/MANIFEST.MF with all the entries needed for HK2 runtime.
75       *
76       * @param pom
77       *      The project from which we are creating manifest. 
78       */
79      public void writeManifest(MavenProject pom, File classesDirectory) throws IOException {
80          Manifest mf = new Manifest();
81          for( Map.Entry<String,String> e : configureManifest(pom,null,classesDirectory).entrySet()) {
82              mf.getMainAttributes().put(
83                  new Attributes.Name(e.getKey()),e.getValue());
84          }
85          mf.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION,"1.0");
86          
87          File manifest = new File(classesDirectory, "META-INF/MANIFEST.MF");
88          manifest.getParentFile().mkdir();
89          FileOutputStream os = new FileOutputStream(manifest);
90          try {
91              mf.write(os);
92          } finally{
93              os.close();
94          }
95      }
96      public Map<String,String> configureManifest(MavenProject pom, MavenArchiveConfiguration archive, File classesDirectory) throws IOException {
97          Map<String,String> entries;
98          if(archive!=null)
99              entries = archive.getManifestEntries();
100         else
101             entries = new HashMap<String,String>();
102 
103         entries.put(ManifestConstants.BUNDLE_NAME, pom.getGroupId()+'.'+pom.getArtifactId() );
104 
105         // check META-INF/services/xxx.ImportPolicy to fill in Import-Policy
106         configureImportPolicy(classesDirectory, entries, ImportPolicy.class, ManifestConstants.IMPORT_POLICY);
107         configureImportPolicy(classesDirectory, entries, LifecyclePolicy.class, ManifestConstants.LIFECYLE_POLICY);
108 
109 
110         // check direct dependencies to find out dependency modules.
111         // we don't need to list transitive dependencies here, so use getDependencyArtifacts().
112 
113         TokenListBuilder dependencyModuleNames = new TokenListBuilder();
114         Set<String> dependencyModules = new HashSet<String>();  // used to find transitive dependencies through other modules.
115         for (Artifact a : (Set<Artifact>)pom.getDependencyArtifacts()) {
116             if(a.getScope()!=null && a.getScope().equals("test"))
117                 continue;
118             // http://www.nabble.com/V3-gf%3Arun-throws-NPE-tf4816802.html indicates
119             // that some artifacts are not resolved at this point. Not sure when that could happen
120             // so aborting with diagnostics if we find it. We need to better understand what this
121             // means and work accordingly. - KK
122             if(a.getFile()==null) {
123                 throw new AssertionError(a.getId()+" is not resolved. a="+a);
124             }
125             Jar jar;
126             try {
127                 jar = Jar.create(a.getFile());
128             } catch (IOException ioe) {
129                 // not a jar file, so continue.
130                 continue;
131             }
132             Manifest manifest = jar.getManifest();
133             String name = null;
134             if (manifest!=null) {
135                 Attributes attributes = manifest.getMainAttributes();
136 
137                 name = attributes.getValue(ManifestConstants.BUNDLE_NAME);
138             }
139             if (name != null) {
140                 // this is a hk2 module
141                 if (!a.isOptional())
142                     dependencyModuleNames.add(name);
143 
144                 // even optional modules need to be listed here
145                 dependencyModules.add(a.getGroupId() + '.' + a.getArtifactId() + ':' + a.getVersion());
146             }
147         }
148 
149         // find jar files to be listed in Class-Path. This needs to include transitive
150         // dependencies, except when the path involves a hk2 module.
151         TokenListBuilder classPathNames = new TokenListBuilder(" ");
152         TokenListBuilder classPathIds = new TokenListBuilder(" ");
153         for (Artifact a : (Set<Artifact>)pom.getArtifacts()) {
154             // check the trail. does that include hk2 module in the path?
155             boolean throughModule = false;
156             for (String module : dependencyModules)
157                 throughModule |= a.getDependencyTrail().get(1).toString().startsWith(module);
158             if(throughModule)
159                 continue;   // yep
160 
161             if(a.getScope().equals("system") || a.getScope().equals("provided") || a.getScope().equals("test"))
162                 continue;   // ignore tools.jar and such dependencies.
163 
164             if(a.isOptional())
165                 continue;   // optional dependency
166             
167             classPathNames.add(stripVersion(a));
168             classPathIds.add(a.getId());
169         }
170         if(!classPathNames.isEmpty()) {
171             String existingClassPath = entries.get(ManifestConstants.CLASS_PATH);
172             if(existingClassPath!=null)
173                 entries.put(ManifestConstants.CLASS_PATH,existingClassPath+" "+classPathNames);
174             else
175                 entries.put(ManifestConstants.CLASS_PATH,classPathNames.toString());
176 
177             entries.put(ManifestConstants.CLASS_PATH_ID,classPathIds.toString());
178         }
179 
180         return entries;
181     }

183     private void configureImportPolicy(File classesDirectory, Map<String, String> entries, Class<?> clazz, String entryName) throws IOException {
184         File importPolicy = new File(classesDirectory, "META-INF/services/" + clazz.getName());
185         if(importPolicy.exists()) {
186             BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(importPolicy), "UTF-8"));
187             entries.put(entryName,in.readLine());
188             in.close();
189         }
190     }

192     private static String stripVersion(Artifact a) {
193         String origName = a.getFile().getName();
194         String newName = origName;
195         int idx = origName.lastIndexOf(a.getVersion());
196         if (idx != -1) {
197             // remove version in artifactId-version(-classifier).type
198             String baseFilename = origName.substring( 0, idx - 1 );
199             String extension = origName.substring( idx + a.getVersion().length());
200             newName = baseFilename + extension;
201         }
202         return newName;
203     }

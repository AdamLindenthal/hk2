87      public MavenProjectRepository(MavenProject project, ArtifactResolver artifactResolver, ArtifactRepository localRepository, ArtifactFactory artifactFactory) {
88          super(project.getName(),project.getFile().toURI());
89          this.project = project;
90          this.artifactResolver = artifactResolver;
91          this.localRepository = localRepository;
92          this.artifactFactory = artifactFactory;
93  
94          // adds the ones that we already know of.
95          Artifact ma = project.getArtifact();
96          if(ma.getFile()!=null) {
97              // if the 'ma' is the distribution module, it won't have its own output.
98              artifacts.put(ma.getId(),ma);
99          }
100 
101         for (Artifact a : (Set<Artifact>) project.getArtifacts())
102             artifacts.put(a.getId(),a);
103     }

105     /***
106      * When creating {@link MavenProjectRepository} from the current project (which is used
107      * to launch mvn), and if the compile phase has run yet, then the main artifact is
108      * still null.
109      *
110      * <p>
111      * However, it's often convenient to pick up the files that were left in the file system
112      * from the previous execution. This method checks this situation and updates {@link MavenProject}
113      * accordingly, so that it can be then passed to the constructor of {@link MavenProjectRepository}.
114      *
115      * <p>
116      * Think of this as a pre-processing phase to compensate for the lack of the compile phase
117      * invocation.
118      */
119     public static void prepareProject(MavenProject project) throws IOException {
120         Artifact ma = project.getArtifact();
121         if(!project.getPackaging().equals("pom") && ma.getFile()==null) {
122             File outdir = new File(project.getBuild().getOutputDirectory());
123             if(!outdir.exists())
124                 logger.warning("No output directory "+outdir);
125             else
126                 ma.setFile(outdir);
127         }
128 
129         if(ma.getFile()!=null) {
130             // if the 'ma' is the distribution module, it won't have its own output.
131             if(ma.getFile().isDirectory()) {
132                 // if the main artifact is from target/classes, create META-INF.MF
133                 new Packager().writeManifest(project,ma.getFile());
134             }
135         }
136     }

138     @Override
139     protected void loadModuleDefs(Map<String, ModuleDefinition> moduleDefs, List<URI> libraries) throws IOException {
140 
141         logger.info("Loading modules list from "+project.getFile());
142 
143         MavenModuleDefinition main = buildModule(project.getArtifact(), moduleDefs, libraries);
144         if(main!=null) {
145             // artifact from the main project, in case those are not compiled yet
146             main.addClasspath(new File(project.getBuild().getOutputDirectory()));
147             for (Resource res : (List<Resource>)project.getBuild().getResources())
148                 main.addClasspath(new File(res.getDirectory()));
149         }
150 
151         for (Artifact a : (List<Artifact>) project.getAttachedArtifacts()) {
152             buildModule(a, moduleDefs, libraries);
153         }
154 
155         for (Artifact a : (Set<Artifact>) project.getArtifacts()) {
156             buildModule(a, moduleDefs, libraries);
157         }
158 
159         if(moduleDefs.isEmpty()) {
160 //            throw new Error("No modules found");// should this error check be done by the caller of loadModuleDefs?
161             logger.warning("No modules found");            
162         }
163 
164     }

166     private MavenModuleDefinition buildModule(Artifact a, Map<String, ModuleDefinition> moduleDefs, List<URI> libraries)
167             throws IOException {
168 
169         File jarFile = a.getFile();
170         if(jarFile==null || (!jarFile.getName().endsWith(".jar") && !jarFile.isDirectory()))
171             // between the compile phase and the package phase, the main artifact is
172             // set to the target/classes. allow that to be used as a jar.
173             return null;
174 
175         MavenModuleDefinition moduleDef = loadJar(jarFile);
176         if(moduleDef.getManifest().getMainAttributes().getValue(ManifestConstants.BUNDLE_NAME)==null) {
177             // project.getArtifacts() pick up all the transitive dependencies,
178             // including to the normal jar files through modules.
179             libraries.add(jarFile.toURI());
180             return null;
181         }
182 
183         if(logger.isLoggable(Level.CONFIG))
184             logger.config("Adding module "+a.getId()+" trail: "+a.getDependencyTrail());
185 
186         moduleDefs.put(moduleDef.getName(), moduleDef);
187         return moduleDef;
188     }

190     /*package*/ File resolveArtifact(String id) throws IOException {
191         Artifact artifact = artifacts.get(id);
192         if(artifact!=null)  return artifact.getFile();
193         
194         try {
195             Matcher m = ID_PATTERN.matcher(id);
196             if(!m.matches())
197                 throw new IllegalArgumentException("Wrong ID: "+id);
198             Artifact a = artifactFactory.createArtifactWithClassifier(
199                 m.group(1), m.group(2), m.group(5), m.group(3), m.group(4)
200             );
201 
202             artifactResolver.resolve(a, project.getRemoteArtifactRepositories(), localRepository );
203             artifacts.put(a.getId(),a);
204             return a.getFile();
205         } catch (ArtifactResolutionException e) {
206             throw new IOException2("Failed to resolve "+id,e);
207         } catch (ArtifactNotFoundException e) {
208             throw new IOException2("Failed to resolve "+id,e);
209         }
210     }

212     // the format is 'groupId:artifactId:type(:classifier)?:version
213     private static final String TOKEN = "([^:]+)";
214     private static final Pattern ID_PATTERN = Pattern.compile(MessageFormat.format("{0}:{0}:{0}(?://:{0})?:{0}",TOKEN));

218         return new MavenModuleDefinition(this,jar);

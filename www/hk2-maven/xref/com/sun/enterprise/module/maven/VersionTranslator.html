50  /***
51   * The versioning scheme used by Maven, HK2 and OSGi are all different in
52   * subtle ways. e.g., something that you represent as 1.0-SNAPSHOT in Maven
53   * is represented as 1.0.0.SNAPSHOT in OSGi. This class encapsulate those
54   * different schemes and provides utility methods to translate from one
55   * to another.
56   *
57   * @author Sanjeeb.Sahoo@Sun.COM
58   */
59  class VersionTranslator {
60      /*
61       * Implementation Note:
62       * Maven version scheme looks like this:
63       * <major>.<minor>.<revision>([-<qualifier>]|[-<build>])
64       * It is impemented at:
65       * http://svn.apache.org/viewvc/maven/artifact/trunk/src/main/java/org/apache/maven/artifact/versioning/DefaultArtifactVersion.java?view=markup
66       * It is further described (to some extent) at:
67       * http://docs.codehaus.org/display/MAVEN/Versioning
68       * OSGi versioning scheme is defined at
69       * http://www2.osgi.org/javadoc/r4/org/osgi/framework/Version.html#Version(java.lang.String)
70       * Since they differ, we need to translate.
71       */
72  
73      private static Logger logger = Logger.getAnonymousLogger();
74  
75      /***
76       * Translates Maven Version String to OSGi Version String
77       * @param mvs
78       * @return a OSGi version string
79       */
80      public static String MavenToOSGi(String mvs) {
81          try {
82              ArtifactVersion mv = new DefaultArtifactVersion(mvs);
83              int major = mv.getMajorVersion();
84              int minor = mv.getMinorVersion();
85              int micro = mv.getIncrementalVersion();
86  
87              // 0 is never a valid build number
88              String qualifier = mv.getBuildNumber() != 0 ?
89                      Integer.toString(mv.getBuildNumber()) : mv.getQualifier();
90              logger.logp(Level.INFO, "VersionTranslator", "MavenToOSGi",
91                      "qualifier = {0}", qualifier);
92              Version ov =
93                      qualifier == null ?
94                      new Version(major, minor, micro)
95                      : new Version(major, minor, micro, qualifier);
96              logger.logp(Level.INFO, "VersionTranslator", "MavenToOSGi", "{0} -> {1}", new Object[]{mvs, ov});
97              return ov.toString();
98          } catch(RuntimeException e) {
99              logger.logp(Level.INFO, "VersionTranslator", "MavenToOSGi",
100                     "Following exception was raised " +
101                     "while translating Maven version {0} to OSGi version: {1}",
102                     new Object[]{mvs, e});
103             throw e;
104         }
105     }

120     public static void main(String[] args) throws IOException {
121         System.out.println("Enter maven version:");
122         DataInputStream in = new DataInputStream(System.in);
123         String mvs = in.readLine();

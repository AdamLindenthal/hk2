21          super(location);
22  
23          try {
24              String classpath = mainAttributes.getValue(ManifestConstants.CLASS_PATH_ID);
25              for( String id : new Tokenizer(classpath," ")) {
26                  File jar = repository.resolveArtifact(id);
27                  classPath.add(jar.toURI());
28              }
29          } catch (IOException e) {
30              throw new IOException2("Failed to process "+ManifestConstants.CLASS_PATH_ID+" for "+location,e);
31          }
32      }

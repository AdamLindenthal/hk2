83      /***
84       * Reads information from the POM and the artifact archive to configure
85       * the OSGi manifest entries. Returns a new set of entries if the archive
86       * does not already have manifest entries, else it uses the existing entries
87       * map. If any of the attribute already exists, then
88       * it skips its processing honoring user's request. It uses the following
89       * rules:
90       *
91       * Bundle-SymbolicName is assumed to be "${groupId}.${artifactId}"
92       * Bundle-Version is derived from "${pom.version}"
93       * using {@link VersionTranslator#MavenToOSGi(String)}
94       * Bundle-Description is assumed to be "${pom.description}".
95       * Bundle-Vendor is assumed to be "${pom.organization.name}".
96       * Require-Bundle is populated by values read from pom dependencies
97       * Note:
98       * There is no support for Export-Package yet.
99       * It sets Bundle-ManifestVersion as 2 which indicates OSGi r4 bundle.
100      *
101      * @param pom The Maven project object
102      * @param archive The archive that is being built
103      * @param classesDirectory output for javac
104      * @return Manifest entries
105      * @throws java.io.IOException
106      */
107     public Map<String,String> configureOSGiManifest(MavenProject pom, MavenArchiveConfiguration archive, File classesDirectory) throws IOException {
108         Map<String,String> entries;
109         if(archive!=null)
110             entries = archive.getManifestEntries();
111         else
112             entries = new HashMap<String,String>();
113 
114         if(entries.get(BUNDLE_MANIFESTVERSION) == null) {
115             // 2 indicates compliance with r4, note: there is no value called 1
116             entries.put(BUNDLE_MANIFESTVERSION, "2");
117         }
118 
119         if (entries.get(BUNDLE_NAME) == null) {
120             // Bundle-Name is a human readable localizable name that can contain spaces
121             entries.put(BUNDLE_NAME, pom.getName());
122         }
123 
124         if (entries.get(BUNDLE_SYMBOLICNAME) == null) {
125             // OSGi convention is to use reverse domain name for SymbolicName, hence use .
126             entries.put(BUNDLE_SYMBOLICNAME, pom.getGroupId()+'.'+pom.getArtifactId());
127         }
128 
129         if (entries.get(BUNDLE_VERSION) == null) {
130             entries.put(BUNDLE_VERSION, VersionTranslator.MavenToOSGi(pom.getVersion()));
131         }
132 
133         if (entries.get(BUNDLE_DESCRIPTION) == null) {
134             if (pom.getDescription()!=null)
135                 entries.put(BUNDLE_DESCRIPTION, pom.getDescription());
136         }
137 
138         if (entries.get(BUNDLE_VENDOR) == null) {
139             if (pom.getOrganization()!=null && pom.getOrganization().getName() != null)
140                 entries.put(BUNDLE_VENDOR, pom.getOrganization().getName());
141         }
142 
143         // Handle Require-Bundle.
144         if (entries.get(REQUIRE_BUNDLE) == null) {
145             String requiredBundles = generateRequireBundleHeader(discoverRequiredBundles(pom));
146             if (requiredBundles.length()>0) {
147                 entries.put(REQUIRE_BUNDLE, requiredBundles);
148             }
149         }
150 
151         // Handle Export-Package
152         if (entries.get(EXPORT_PACKAGE) == null) {
153             List<ExportedPackage> packages = discoverPackages(classesDirectory);
154 
155             // don't use version until we resolve split package issues in GF
156             String exportPackages = generateExportPackageHeader(packages, null);
157             if (exportPackages.length()> 0) {
158                 entries.put(EXPORT_PACKAGE, exportPackages);
159             }
160         }
161         return entries;
162     }

164     static class BundleDependency {
165         String bundleSymbolicName;
166         String versionRange; // version = 1.0.0.SNAPSHOT or "[1.0.0.SNAPSHOT, 1.0.0.SNAPSHOT]"
167         String resolution; // optional or mandatory
168         String visibility; // private or reexport
169     }

171     public List<BundleDependency> discoverRequiredBundles(MavenProject pom)
172             throws IOException {
173         List<BundleDependency> dependencies = new ArrayList<BundleDependency>();
174         for (Artifact a : (Set<Artifact>)pom.getDependencyArtifacts()) {
175             if("test".equals(a.getScope()) || "provided".equals(a.getScope()))
176                 continue;
177             // http://www.nabble.com/V3-gf%3Arun-throws-NPE-tf4816802.html indicates
178             // that some artifacts are not resolved at this point. Not sure when that could happen
179             // so aborting with diagnostics if we find it. We need to better understand what this
180             // means and work accordingly. - KK
181             if(a.getFile()==null) {
182                 throw new AssertionError(a.getId()+" is not resolved. a="+a);
183             }
184 
185             Attributes attributes = null;
186             String name = null;
187             Manifest manifest = Jar.create(a.getFile()).getManifest();
188             if (manifest!=null) {
189                 attributes = manifest.getMainAttributes();
190                 name = attributes.getValue(BUNDLE_SYMBOLICNAME);
191             }
192             if (name != null && attributes.getValue(FRAGMENT_HOST)==null) {
193                 // this is a OSGi host module 
194                 BundleDependency bd = new BundleDependency();
195                 bd.bundleSymbolicName = name;
196 
197                 final String version = attributes.getValue(BUNDLE_VERSION);
198                 // no need to translate, for it's already an OSGi version
199                 bd.versionRange = version; // maps to [version, infinity)
200 
201                 // resolution=optional or mandatory
202                 bd.resolution = props.getProperty("resolution", RESOLUTION_MANDATORY);
203 
204                 bd.visibility = props.getProperty("visibility", VISIBILITY_PRIVATE);
205                 dependencies.add(bd);
206             }
207         }
208         return dependencies;
209     }

211     public String generateRequireBundleHeader(
212             List<BundleDependency> dependencies) {
213         // Require-Bundle:
214         // a.b.c;version=1.0.0.b58g;resolution:=mandatory;visibility:=reexport,
215         // p.q.r;version=1.0.0.SNAPSHOT;resolution:=optional;visibility:=private
216         StringBuilder requiredBundles = new StringBuilder();
217         for (BundleDependency bd : dependencies) {
218             if(requiredBundles.length()!=0) {
219                 requiredBundles.append(",");
220             }
221             requiredBundles.append(bd.bundleSymbolicName);
222 
223             if(bd.versionRange!= null) {
224                 requiredBundles.append(';').
225                         append(BUNDLE_VERSION_ATTRIBUTE).
226                         append('=').
227                         append(bd.versionRange);
228             }
229 
230             requiredBundles.append(";").
231                     append(RESOLUTION_DIRECTIVE).
232                     append(":=").append(bd.resolution);
233 
234             requiredBundles.append(';').
235                     append(VISIBILITY_DIRECTIVE).
236                     append(":=").append(bd.visibility);
237         }
238         return requiredBundles.toString();
239     }

241     static class ExportedPackage implements Comparable<ExportedPackage>{
242         String packageName;
243         Collection<String> packagesUsed; // uses directive
244 
245         public ExportedPackage(
246                 String packageName,
247                 Collection<String> packagesUsed) {
248             this.packageName = packageName;
249             this.packagesUsed = packagesUsed;
250         }
251 
252         public ExportedPackage(String packageName) {
253             this(packageName, null);
254         }
255 
256         @Override public int hashCode() {
257             return packageName.hashCode();
258         }
259 
260         @Override public boolean equals(Object obj) {
261             if (obj instanceof ExportedPackage) {
262                 return packageName.equals(ExportedPackage.class.cast(obj).packageName);
263             }
264             return false;
265         }
266 
267         public int compareTo(ExportedPackage exportedPackage) {
268             return this.packageName.compareTo(exportedPackage.packageName);
269         }
270     }

272     public static List<ExportedPackage> discoverPackages(final File classesDirectory) {
273         final List<ExportedPackage> packages = new ArrayList<ExportedPackage>();
274         classesDirectory.listFiles(
275                 new FilenameFilter() {
276                     public boolean accept(File dir, String name) {
277                         final File file = new File(dir, name);
278                         if (file.isDirectory()) {
279                             file.listFiles(this);
280                         }
281                         final String CLASS_EXT = ".class";
282                         if (!name.endsWith(CLASS_EXT)) {
283                             return false;
284                         }
285                         final String packagePath = dir.getPath().substring(classesDirectory.getPath().length()+1);
286                         String packageName = packagePath.replace(File.separatorChar, '.');
287                         logger.fine("packageName = " + packageName);
288                         final ExportedPackage ep = new ExportedPackage(packageName);
289                         if (!packages.contains(ep)) {
290                             packages.add(ep);
291                         }
292                         return true;
293                     }
294                 }
295         );
296         Collections.sort(packages);
297         return packages;
298     }

300     public static String generateExportPackageHeader(
301             Collection<ExportedPackage> packages, String version) {
302         // Export-Packages=a.b.c;version=1.0.0.SNAPSHOT,p.q.r;x.y.z;version=2.0
303 
304         if (packages.isEmpty()) {
305             return "";
306         }
307         StringBuilder sb = new StringBuilder();
308         for(ExportedPackage p : packages) {
309             if(sb.length()>0) {
310                 sb.append(","); // exclude the first one
311             }
312             sb.append(p.packageName);
313 
314             // TODO: Make use of uses directive
315 
316             if (version!=null) {
317                 sb.append(";").append(VERSION_ATTRIBUTE).append("=").append(version);
318             }
319         }
320         return sb.toString();
321     }

51      /***
52       * For each class annotated with {@link Configured}.
53       */
54      public void visitClassDeclaration(ClassDeclaration clz) {
55          Configured c = clz.getAnnotation(Configured.class);
56  
57          String name = clz.getQualifiedName();
58  
59          Builder builder = new BuilderImpl(name+"Reader",null,null);
60          builder.setReaderBaseClass(ReaderEx.class);
61          JCodeModel cm = builder.getCodeModel();
62          ElementParserBuilder root = builder.getParserBuilder();
63  
64          ElementParserBuilder b = root.expectElement(new QName(c.name()));
65          JClass target = cm.ref(name);
66          JVar $target = b.getBody().decl(target, "result", JExpr._new(target));
67          b.getBody().add(JExpr._this().invoke("inject").arg(b.getXSR()).arg($target));
68          b.getBody()._return($target.type(), JExpr.invoke("wrapUp").arg($target));
69  
70          Set<Property> props = new LinkedHashSet<Property>();
71  
72          for (FieldDeclaration f : clz.getFields()) {
73              Property p = Property.createField(env,f);
74              if(handleAttribute(p,b,$target)|handleElement(p,b,$target))
75                  props.add(p);
76          }
77  
78          for (MethodDeclaration m : clz.getMethods()) {
79              Property p = new Property.Method(m);
80              if(handleAttribute(p,b,$target)|handleElement(p,b,$target))
81                  props.add(p);
82          }
83  
84          //for (Property p : props) {
85          //    p.pre(cm,b.getBody().getBlock());
86          //    p.post(cm,b.getTailBlock());
87          //}
88  
89          try {
90              builder.write(new FilerCodeWriter(env.getFiler()));
91          } catch (IOException e) {
92              throw new Error(e);
93          }
94      }

96      private boolean handleElement(Property p, ElementParserBuilder b, JVar $target) {
97          FromElement a = p.getAnnotation(FromElement.class);
98          if(a==null)     return false; // not an attribute
99  
100         ElementParserBuilder child = b.expectElement(new QName(p.inferName(a.value())));
101 
102         // try to handle it as a reference
103         TypeMirror t = p.type();
104         if(t instanceof ClassType) {
105             ClassDeclaration decl = ((ClassType) t).getDeclaration();
106             Configured cfg = decl.getAnnotation(Configured.class);
107             if(cfg!=null) {
108                 // load from structure
109                 JCodeModel cm = child.getCodeModel();
110                 p.assign(child.passParentVariable($target),
111                     child.getBody().getBlock(),
112                     // the read1 method is always the one that loads the object
113                     JExpr.cast(cm.ref(decl.getQualifiedName()),
114                         JExpr._new(cm.ref(decl.getQualifiedName()+"Reader")).
115                             arg(JExpr.ref("context")).invoke("read1").arg(child.getXSR()).arg(JExpr.ref("properties"))
116                         ));
117                 return true;
118             }
119         }
120 
121         handleSimpleValue(child, $target, p);
122         return true;
123     }

125     /***
126      * Generates the handler that reads an attribute and sets the value.
127      */
128     private boolean handleAttribute(Property p, ElementParserBuilder b, JVar $target) {
129         FromAttribute a = p.getAnnotation(FromAttribute.class);
130         if(a==null)     return false; // not an attribute
131 
132         String name = p.inferName(a.value());
133         handleSimpleValue(b.expectAttribute(new QName(name)), $target, p);
134         return true;
135     }

137     /***
138      * Completes the handler that reads a simple text value.
139      */
140     private void handleSimpleValue(ParserBuilder pb, JVar $target, Property p) {
141         $target = pb.passParentVariable($target);
142         p.assign($target, pb.getBody().getBlock(), pb.as(toSimpleClass(p.type(),p.decl().getPosition())));
143     }

146     /***
147      * Determines the Class object from the given type
148      */
149     private Class<?> toSimpleClass(TypeMirror type, final SourcePosition position) {
150         final Class[] r = new Class[1];
151         env.getTypeUtils().getErasure(type).accept(new SimpleTypeVisitor() {
152             public void visitPrimitiveType(PrimitiveType p) {
153                 switch (p.getKind()) {
154                 case BOOLEAN:       r[0]=boolean.class;break;
155                 case BYTE:          r[0]=byte.class;break;
156                 case CHAR:          r[0]=char.class;break;
157                 case DOUBLE:        r[0]=double.class;break;
158                 case FLOAT:         r[0]=float.class;break;
159                 case INT:           r[0]=int.class;break;
160                 case LONG:          r[0]=long.class;break;
161                 case SHORT:         r[0]=short.class;break;
162                 }
163             }
164 
165             public void visitClassType(ClassType classType) {
166                 String name = classType.getDeclaration().getQualifiedName();
167                 try {
168                     r[0] = Class.forName(name);
169                 } catch (ClassNotFoundException e) {
170                     env.getMessager().printError(position,"Not a valid type for attribute "+name);
171                 }
172             }
173         });
174 
175         // TODO: improve error handling when r[0]==null
176         return r[0];
177     }

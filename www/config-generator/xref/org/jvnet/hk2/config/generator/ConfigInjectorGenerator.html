115     public void process() {
116         cm = new JCodeModel();
117 
118         AnnotationTypeDeclaration ann = (AnnotationTypeDeclaration) env.getTypeDeclaration(Configured.class.getName());
119         for(Declaration d : env.getDeclarationsAnnotatedWith(ann))
120             d.accept(this);
121 
122         try {
123             cm.build(new FilerCodeWriter(env.getFiler()));
124         } catch (IOException e) {
125             throw new Error(e);
126         }
127         cm = null;
128     }

141     /***
142      * For each {@link ConfigBeanProxy} annotated with {@link Configured}.
143      */
144     public void visitInterfaceDeclaration(InterfaceDeclaration clz) {
145         try {
146             if(!isSubType(clz,configBeanProxy)) {
147                 env.getMessager().printError(clz.getPosition(),
148                     clz.getQualifiedName()+" has @Configured but doesn't extend ConfigBeanProxy");
149             } else
150                 new ClassGenerator(clz,true).generate();
151         } catch (JClassAlreadyExistsException e) {
152             env.getMessager().printError(clz.getPosition(),e.toString());
153         }
154     }

173         /***
174          * If true, generate a ConfigInjector that extends from {@link NoopConfigInjector}.
175          * This is used as the metadata place holder for {@link ConfigBeanProxy}s.
176          * <p>
177          * If false, generate a real {@link ConfigInjector}.
178          *
179          * <p>
180          * See the test-config test module for this difference in action. 
181          */
182         private final boolean generateNoopConfigInjector;

185         public ClassGenerator(TypeDeclaration clz, boolean generateNoopConfigInjector) throws JClassAlreadyExistsException {
186             this.clz = clz;
187             this.generateNoopConfigInjector = generateNoopConfigInjector;
188             Configured c = clz.getAnnotation(Configured.class);
189 
190             String name = clz.getQualifiedName();
191             targetType = cm.ref(name);
192 
193             // [RESULT]
194             // @Service(name='...')
195             // @InjectionTarget(target)
196             // public class XYZInjector extends ConfigInjector<XYZ>
197             injector = cm._class(name+"Injector");
198             String elementName = c.name();
199             if(c.local()) {
200                 if(elementName.length()>0) {
201                     env.getMessager().printError(clz.getPosition(),"@Configured.local and @Configured.name is mutually exclusive");
202                     elementName = ""; // error recovery
203                 }
204             } else {
205                 if(elementName.length()==0) // infer default
206                     elementName = Dom.convertName(clz.getSimpleName());
207             }
208 
209             service = injector.annotate(Service.class).param("name",elementName);
210             injector.annotate(InjectionTarget.class).param("value",targetType);
211             if(generateNoopConfigInjector) {
212                 injector._extends(cm.ref(NoopConfigInjector.class));
213                 injectAttributeMethod = null;
214                 injectMethod = null;
215                 injectElementMethod = null;
216             } else {
217                 injector._extends(cm.ref(ConfigInjector.class).narrow(targetType));
218 
219                 // [RESULT]
220                 // public void inject(Dom dom, Property target) { ... }
221                 injectMethod = injector.method(JMod.PUBLIC, void.class, "inject");
222                 injectMethod.param(Dom.class, "dom");
223                 injectMethod.param(targetType, "target");
224                 injectMethod.body();
225 
226                 injectAttributeMethod = injector.method(JMod.PUBLIC,void.class,"injectAttribute");
227                 addReinjectionParam(injectAttributeMethod);
228 
229                 injectElementMethod = injector.method(JMod.PUBLIC,void.class,"injectElement");
230                 addReinjectionParam(injectElementMethod);
231             }
232 
233             metadata.add(ConfigMetadata.TARGET,name);
234 
235             // locate additional contracts for the target.
236             for (TypeDeclaration t : ContractFinder.find(clz))
237                 metadata.add(ConfigMetadata.TARGET_CONTRACTS,t.getQualifiedName());
238         }
239 
240         private void addReinjectionParam(JMethod method) {
241             method.param(Dom.class,"dom");
242             method.param(String.class,"name");
243             method.param(targetType, "target");
244         }

246         /***
247          * Visits all annotated fields/methods and
248          * generates the body of the {@link ConfigInjector#inject(Dom, Object)} code.
249          */
250         public void generate() {
251             Stack<TypeDeclaration> q = new Stack<TypeDeclaration>();
252             Set<TypeDeclaration> visited = new HashSet<TypeDeclaration>();
253             q.push(clz);
254 
255             while(!q.isEmpty()) {
256                 TypeDeclaration t = q.pop();
257                 if(!visited.add(t)) continue;   // been here already
258 
259                 for (FieldDeclaration f : t.getFields())
260                     generate(new Property.Field(f));
261 
262                 for (MethodDeclaration m : t.getMethods())
263                     generate(new Property.Method(m));
264 
265                 for (InterfaceType it : clz.getSuperinterfaces())
266                     q.add(it.getDeclaration());
267 
268                 if (t instanceof ClassDeclaration) {
269                     ClassDeclaration cd = (ClassDeclaration) t;
270                     if(cd.getSuperclass()!=null)
271                         q.add(cd.getSuperclass().getDeclaration());
272                 }
273             }
274 
275             service.param("metadata", metadata.toCommaSeparatedString());
276         }

278         private void generate(Property p) {
279             Attribute a = p.getAnnotation(Attribute.class);
280             Element e = p.getAnnotation(Element.class);
281 
282             if(a!=null) {
283                 new AttributeMethodGenerator(p,a).generate();
284                 if(e!=null)
285                     env.getMessager().printError(p.decl().getPosition(),"Cannot have both @Element and @Attribute at the same time");
286             } else {
287                 if(e!=null)
288                     new ElementMethodGenerator(p,e).generate();
289             }
290 
291             // Updates #key with error check.
292             if(p.isKey()) {
293                 if(key!=null) {
294                     env.getMessager().printError(p.decl().getPosition(),"Multiple key properties");
295                     env.getMessager().printError(key.decl().getPosition(),"Another one is at here");
296                 }
297                 key = p;
298             }
299         }

301         /***
302          * Generates a single injection method, which injects
303          * value(s) of a particular element/attribute name.
304          */
305         private abstract class MethodGenerator {
306             final JBlock body;
307             final JVar $dom;
308             final JVar $target;
309             /***
310              * Element name or attribute name.
311              * A special case is "*" for elements that indicate substitute-by-type.
312              */
313             final String xmlName;
314             final Property p;
315             private int iota=1;
316             /***
317              * Erasure of {@code p.type()}
318              */
319             final TypeMirror erasure;
320             /***
321              * If this is a multi-value property, the packer knows how to create a collection value.
322              */
323             final Packer packer;
324             /***
325              * The type of individual item. If this is a multi-value property, this is a type of the collection
326              * item, otherwise the same as {@link #erasure}.
327              */
328             final TypeMirror itemType;
329             /***
330              * Converter for {@link #itemType}.
331              */
332             /*semi-final*/ Converter conv;
333 
334             MethodGenerator(String methodNamePrefix, JMethod reinjectionMethod, Property p, String xmlName) {
335                 this.xmlName = p.inferName(xmlName);
336                 this.p = p;
337 
338                 if(generateNoopConfigInjector) {
339                     body = null;
340                     $dom = null;
341                     $target = null;
342                 } else {
343                     JMethod m = injector.method(JMod.PUBLIC,void.class, methodNamePrefix+p.seedName());
344                     $dom = m.param(Dom.class,"dom");
345                     $target = m.param(targetType,"target");
346                     body = m.body();
347 
348                     injectMethod.body().invoke(m).arg($dom).arg($target);
349 
350                     reinjectionMethod.body()._if(JExpr.lit(this.xmlName).invoke("equals").arg(JExpr.ref("name")))
351                         ._then().invoke(m).arg($dom).arg($target);
352                 }
353 
354                 erasure = erasure(p.type());
355                 packer = createPacker(p.type(),erasure);
356 
357                 itemType = packer==null ? erasure : erasure(packer.itemType());
358             }
359 
360             private void assign(JExpression rhs) {
361                 p.assign($target,body,rhs);
362             }
363 
364             /***
365              * Returns '@xmlName' for attributes and '&lt;xmlName>' for elements.
366              */
367             protected abstract String xmlTokenName();
368 
369             protected void generate() {
370                 conv = createConverter(itemType);
371                 conv.addMetadata(xmlTokenName(),itemType);
372 
373                 if(!isVariableExpansion() && TO_JTYPE.apply(itemType,null)!=cm.ref(String.class))
374                     env.getMessager().printError(p.decl().getPosition(),"variableExpansion=false is only allowed on String");
375 
376                 if(!generateNoopConfigInjector) {
377                     JVar value = var(
378                         packer!=null ? cm.ref(List.class).narrow(conv.sourceType()) : conv.sourceType(),getXmlValue());
379 
380                     if(!isRequired())
381                         body._if(value.eq(JExpr._null()))._then()._return();
382 
383                     if(packer!=null)
384                         handleMultiValue(value);
385                     else
386                         assign(conv.as(value,itemType));
387                 }
388 
389                 if(p.isKey())
390                     addKey();
391             }

425              * Invokes a method on DOM by adjusting the name for plural.

427             final JInvocation invokeDom(String methodName) {
428                 if(packer!=null)    methodName+='s';

432             private void handleMultiValue(JVar values) {
433                 // [RESULT]
434                 // List<S> values = dom.leafElements("...");
435                 // <packer init>
436                 // for( S v : values ) {
437                 //   <packer set>(<as>(v));
438                 // }
439                 //  ... assign ...
440                 packer.start(values.invoke("size"));
441                 JForEach forEach = body.forEach(conv.sourceType(), id(), values);
442                 packer.pack(forEach.body(),conv.as(forEach.var(),packer.itemType()), forEach.var());
443                 assign(packer.end());
444             }

464             private Packer createPacker(TypeMirror type, TypeMirror erasure) {
465                 if(erasure instanceof ArrayType) {
466                     // T=X[]
467                     return new ArrayPacker((ArrayType)erasure);
468                 }
469 
470                 TypeMirror itemType = math.isCollection(type);
471                 if(itemType!=null) {
472                     // T=Collection[]
473                     return new ListPacker(type,itemType);
474                 }
475 
476                 TypeMirror mapType = TypeMath.baseClassFinder.apply(type, env.getTypeDeclaration(Map.class.getName()));
477                 if(mapType!=null) {
478                     // T=Map<...>
479                     DeclaredType d = (DeclaredType)mapType;
480                     Iterator<TypeMirror> itr = d.getActualTypeArguments().iterator();
481                     itr.next();
482                     return new MapPacker(itr.next());
483                 }
484 
485                 return null;
486             }

488             abstract class Packer {
489                 abstract TypeMirror itemType();
490                 /***
491                  * Starts packing.
492                  */
493                 abstract void start(JExpression $valueSize);
494                 /***
495                  * Adds one more item to the pack.
496                  */
497                 abstract void pack(JBlock block, JExpression item, JExpression sourceValue);
498 
499                 /***
500                  * Returns the packed value to be set.
501                  */
502                 abstract JExpression end();
503             }

505             final class ArrayPacker extends Packer {
506                 private JVar $array,$index;
507                 private final JType arrayT;
508                 private final JType componentT;
509                 private final ArrayType at;
510 
511                 public ArrayPacker(ArrayType t) {
512                     this.at = t;
513                     this.componentT = TO_JTYPE.apply(itemType(),null);
514                     this.arrayT = componentT.array();
515                 }
516 
517                 TypeMirror itemType() {
518                     return at.getComponentType();
519                 }
520 
521                 void start(JExpression $valueSize) {
522                     // [RESULT]
523                     // T[] x = new T[values.size()];
524                     $array = var(arrayT, JExpr.newArray(componentT, $valueSize));
525                     $index = var(int.class,JExpr.lit(0));
526                 }
527 
528                 void pack(JBlock block, JExpression item, JExpression sourceValue) {
529                     // [RESULT]
530                     // x[i++] = <rhs>;
531                     block.assign($array.component($index.incr()),item);
532                 }
533 
534                 JExpression end() {
535                     return $array;
536                 }
537             }

539             final class ListPacker extends Packer {
540                 private JVar $list;
541                 private final JClass collectionType,itemType;
542                 private final TypeMirror itemT;
543 
544                 public ListPacker(TypeMirror collectionType, TypeMirror itemType) {
545                     this.collectionType = TO_JTYPE.apply(collectionType,null).boxify();
546                     this.itemType       = TO_JTYPE.apply(itemType,null).boxify();
547                     this.itemT = itemType;
548                 }
549 
550                 TypeMirror itemType() {
551                     return itemT;
552                 }
553 
554                 void start(JExpression $valueSize) {
555                     // [RESULT]
556                     // T x = new ArrayList<T>(values.size());
557                     $list = var(collectionType,JExpr._new(implType()).arg($valueSize));
558                 }
559 
560                 /***
561                  * Figure out the concrete implementation class to be used.
562                  */
563                 JType implType() {
564                     if(cm.ref(Set.class).isAssignableFrom(collectionType))
565                         return cm.ref(HashSet.class).narrow(itemType);
566                     return cm.ref(ArrayList.class).narrow(itemType);
567                 }
568 
569                 void pack(JBlock block, JExpression item, JExpression sourceValue) {
570                     // [RESULT]
571                     // x.add(<rhs>);
572                     block.invoke($list,"add").arg(item);
573                 }
574 
575                 JExpression end() {
576                     return $list;
577                 }
578             }

584                 public MapPacker(TypeMirror itemType) {
585                     this.itemT = itemType;
586                 }

609             private Converter createConverter(TypeMirror itemType) {
610                 try {
611                     // is this a leaf value?
612                     math.SIMPLE_VALUE_CONVERTER.apply(itemType, JExpr._null());
613                     return new LeafConverter();
614                 } catch (UnsupportedOperationException e) {
615                     // nope
616                 }
617 
618                 // try to handle it as a reference
619                 if (itemType instanceof DeclaredType) {
620                     if (!(itemType instanceof ClassType || itemType instanceof InterfaceType)) {
621                         env.getMessager().printError(p.decl().getPosition(),
622                         "I only inject interfaces or classes,  "+itemType+" is not one of them");
623                         return new NodeConverter();
624                     }
625                     TypeDeclaration decl = ((DeclaredType)itemType).getDeclaration();
626                     Configured cfg = decl.getAnnotation(Configured.class);
627                     if(cfg!=null) {
628                         // node value
629                         if(isReference())
630                             return new ReferenceConverter();
631                         else
632                             return new NodeConverter();
633                     }
634                 }
635 
636                 if(isAllElementMatch()) {
637                     return new NodeByTypeConverter(itemType);
638                 }
639 
640                 env.getMessager().printError(p.decl().getPosition(),
641                     "I don't know how to inject "+itemType+" from configuration");
642                 return new NodeConverter(); // error recovery
643             }

645             /***
646              * Encapsulates the source value representation in {@link Dom}.
647              */
648             abstract class Converter {
649                 /***
650                  * Generates an expression that converts 'rhs'.
651                  *
652                  * @param targetType
653                  *      The expected type of the expression, so that the generated expression
654                  *      can contain cast operation if necessary.
655                  */
656                 abstract JExpression as(JExpression rhs, TypeMirror targetType);
657                 /***
658                  * Source value type as returned by {@link Dom}.
659                  */
660                 abstract JClass sourceType();
661 
662                 /***
663                  * True if the XML representation of the source value is a leaf (string value)
664                  * as opposed to node (an XML fragment.)
665                  */
666                 abstract boolean isLeaf();
667                 
668                 abstract void addMetadata(String key,TypeMirror itemType);
669 
670                 protected final String makeCollectionIfNecessary(String s) {
671                     if(packer!=null)    return "collection:"+s;
672                     else                return s;
673                 }
674             }
675 
676             class LeafConverter extends Converter {
677                 JExpression as(JExpression rhs, TypeMirror targetType) {
678                     return math.SIMPLE_VALUE_CONVERTER.apply(targetType, rhs);
679                 }
680                 JClass sourceType() {
681                     return cm.ref(String.class);
682                 }
683 
684                 boolean isLeaf() {
685                     return true;
686                 }
687 
688                 void addMetadata(String key,TypeMirror itemType) {
689                     metadata.add(key,makeCollectionIfNecessary("leaf"));
690                 }
691             }

693             class NodeConverter extends Converter {
694                 JExpression as(JExpression rhs, TypeMirror targetType) {
695                     return JExpr.cast(TO_JTYPE.apply(targetType,null),rhs.invoke("get"));
696                 }
697                 JClass sourceType() {
698                     return cm.ref(Dom.class);
699                 }
700 
701                 boolean isLeaf() {
702                     return false;

121     public void process() {
122         cm = new JCodeModel();
123 
124         AnnotationTypeDeclaration ann = (AnnotationTypeDeclaration) env.getTypeDeclaration(Configured.class.getName());
125         for(Declaration d : env.getDeclarationsAnnotatedWith(ann))
126             d.accept(this);
127 
128         try {
129             cm.build(new FilerCodeWriter(env.getFiler()));
130         } catch (IOException e) {
131             throw new Error(e);
132         }
133         cm = null;
134     }

147     /***
148      * For each {@link ConfigBeanProxy} annotated with {@link Configured}.
149      */
150     public void visitInterfaceDeclaration(InterfaceDeclaration clz) {
151         try {
152             if(!isSubType(clz,configBeanProxy)) {
153                 env.getMessager().printError(clz.getPosition(),
154                     clz.getQualifiedName()+" has @Configured but doesn't extend ConfigBeanProxy");
155             } else
156                 new ClassGenerator(clz,true).generate();
157         } catch (JClassAlreadyExistsException e) {
158             env.getMessager().printError(clz.getPosition(),e.toString());
159         }
160     }

163         Types u = env.getTypeUtils();
164         return u.isSubtype(u.getDeclaredType(subType),u.getDeclaredType(baseType));

179         /***
180          * If true, generate a ConfigInjector that extends from {@link NoopConfigInjector}.
181          * This is used as the metadata place holder for {@link ConfigBeanProxy}s.
182          * <p>
183          * If false, generate a real {@link ConfigInjector}.
184          *
185          * <p>
186          * See the test-config test module for this difference in action. 
187          */
188         private final boolean generateNoopConfigInjector;

191         public ClassGenerator(TypeDeclaration clz, boolean generateNoopConfigInjector) throws JClassAlreadyExistsException {
192             this.clz = clz;
193             this.generateNoopConfigInjector = generateNoopConfigInjector;
194             Configured c = clz.getAnnotation(Configured.class);
195 
196             String name = clz.getQualifiedName();
197             targetType = cm.ref(name);
198 
199             // [RESULT]
200             // @Service(name='...')
201             // @InjectionTarget(target)
202             // public class XYZInjector extends ConfigInjector<XYZ>
203             injector = cm._class(name+"Injector");
204             String elementName = c.name();
205             if(c.local()) {
206                 if(elementName.length()>0) {
207                     env.getMessager().printError(clz.getPosition(),"@Configured.local and @Configured.name is mutually exclusive");
208                     elementName = ""; // error recovery
209                 }
210             } else {
211                 if(elementName.length()==0) // infer default
212                     elementName = Dom.convertName(clz.getSimpleName());
213             }
214 
215             service = injector.annotate(Service.class).param("name",elementName);
216             injector.annotate(InjectionTarget.class).param("value",targetType);
217             if(generateNoopConfigInjector) {
218                 injector._extends(cm.ref(NoopConfigInjector.class));
219                 injectAttributeMethod = null;
220                 injectMethod = null;
221                 injectElementMethod = null;
222             } else {
223                 injector._extends(cm.ref(ConfigInjector.class).narrow(targetType));
224 
225                 // [RESULT]
226                 // public void inject(Dom dom, Property target) { ... }
227                 injectMethod = injector.method(JMod.PUBLIC, void.class, "inject");
228                 injectMethod.param(Dom.class, "dom");
229                 injectMethod.param(targetType, "target");
230                 injectMethod.body();
231 
232                 injectAttributeMethod = injector.method(JMod.PUBLIC,void.class,"injectAttribute");
233                 addReinjectionParam(injectAttributeMethod);
234 
235                 injectElementMethod = injector.method(JMod.PUBLIC,void.class,"injectElement");
236                 addReinjectionParam(injectElementMethod);
237             }
238 
239             metadata.add(ConfigMetadata.TARGET,name);
240 
241             // locate additional contracts for the target.
242             for (TypeDeclaration t : ContractFinder.find(clz))
243                 metadata.add(ConfigMetadata.TARGET_CONTRACTS,t.getQualifiedName());
244         }
245 
246         private void addReinjectionParam(JMethod method) {
247             method.param(Dom.class,"dom");
248             method.param(String.class,"name");
249             method.param(targetType, "target");
250         }

252         /***
253          * Visits all annotated fields/methods and
254          * generates the body of the {@link ConfigInjector#inject(Dom, Object)} code.
255          */
256         public void generate() {
257             Stack<TypeDeclaration> q = new Stack<TypeDeclaration>();
258             Set<TypeDeclaration> visited = new HashSet<TypeDeclaration>();
259             q.push(clz);
260 
261             while(!q.isEmpty()) {
262                 TypeDeclaration t = q.pop();
263                 if(!visited.add(t)) continue;   // been here already
264 
265                 for (FieldDeclaration f : t.getFields())
266                     generate(new Property.Field(f));
267 
268                 for (MethodDeclaration m : t.getMethods())
269                     generate(new Property.Method(m));
270 
271                 for (InterfaceType it : clz.getSuperinterfaces())
272                     q.add(it.getDeclaration());
273 
274                 if (t instanceof ClassDeclaration) {
275                     ClassDeclaration cd = (ClassDeclaration) t;
276                     if(cd.getSuperclass()!=null)
277                         q.add(cd.getSuperclass().getDeclaration());
278                 }
279             }
280 
281             service.param("metadata", metadata.toCommaSeparatedString());
282         }

284         private void generate(Property p) {
285             Attribute a = p.getAnnotation(Attribute.class);
286             Element e = p.getAnnotation(Element.class);
287 
288             if(a!=null) {
289                 new AttributeMethodGenerator(p,a).generate();
290                 if(e!=null)
291                     env.getMessager().printError(p.decl().getPosition(),"Cannot have both @Element and @Attribute at the same time");
292             } else {
293                 if(e!=null)
294                     new ElementMethodGenerator(p,e).generate();
295             }
296 
297             // Updates #key with error check.
298             if(p.isKey()) {
299                 if(key!=null) {
300                     env.getMessager().printError(p.decl().getPosition(),"Multiple key properties");
301                     env.getMessager().printError(key.decl().getPosition(),"Another one is at here");
302                 }
303                 key = p;
304             }
305         }

307         /***
308          * Generates a single injection method, which injects
309          * value(s) of a particular element/attribute name.
310          */
311         private abstract class MethodGenerator {
312             final JBlock body;
313             final JVar $dom;
314             final JVar $target;
315             /***
316              * Element name or attribute name.
317              * A special case is "*" for elements that indicate substitute-by-type.
318              */
319             final String xmlName;
320             final Property p;
321             private int iota=1;
322             /***
323              * Erasure of {@code p.type()}
324              */
325             final TypeMirror erasure;
326             /***
327              * If this is a multi-value property, the packer knows how to create a collection value.
328              */
329             final Packer packer;
330             /***
331              * The type of individual item. If this is a multi-value property, this is a type of the collection
332              * item, otherwise the same as {@link #erasure}.
333              */
334             final TypeMirror itemType;
335             /***
336              * Converter for {@link #itemType}.
337              */
338             /*semi-final*/ Converter conv;
339 
340             MethodGenerator(String methodNamePrefix, JMethod reinjectionMethod, Property p, String xmlName) {
341                 this.xmlName = p.inferName(xmlName);
342                 this.p = p;
343 
344                 if(generateNoopConfigInjector) {
345                     body = null;
346                     $dom = null;
347                     $target = null;
348                 } else {
349                     JMethod m = injector.method(JMod.PUBLIC,void.class, methodNamePrefix+p.seedName());
350                     $dom = m.param(Dom.class,"dom");
351                     $target = m.param(targetType,"target");
352                     body = m.body();
353 
354                     injectMethod.body().invoke(m).arg($dom).arg($target);
355 
356                     reinjectionMethod.body()._if(JExpr.lit(this.xmlName).invoke("equals").arg(JExpr.ref("name")))
357                         ._then().invoke(m).arg($dom).arg($target);
358                 }
359 
360                 erasure = erasure(p.type());
361                 packer = createPacker(p.type(),erasure);
362 
363                 itemType = packer==null ? erasure : erasure(packer.itemType());
364             }
365 
366             private void assign(JExpression rhs) {
367                 p.assign($target,body,rhs);
368             }
369 
370             /***
371              * Returns '@xmlName' for attributes and '&lt;xmlName>' for elements.
372              */
373             protected abstract String xmlTokenName();
374 
375             protected void generate() {
376                 conv = createConverter(itemType);
377                 conv.addMetadata(xmlTokenName(),itemType);
378 
379                 if(!isVariableExpansion() && TO_JTYPE.apply(itemType,null)!=cm.ref(String.class))
380                     env.getMessager().printError(p.decl().getPosition(),"variableExpansion=false is only allowed on String");
381 
382                 if(!generateNoopConfigInjector) {
383                     JVar value = var(
384                         packer!=null ? cm.ref(List.class).narrow(conv.sourceType()) : conv.sourceType(),getXmlValue());
385 
386                     if(!isRequired())
387                         body._if(value.eq(JExpr._null()))._then()._return();
388 
389                     if(packer!=null)
390                         handleMultiValue(value);
391                     else
392                         assign(conv.as(value,itemType));
393                 }
394 
395                 if(p.isKey())
396                     addKey();
397             }

438             private void handleMultiValue(JVar values) {
439                 // [RESULT]
440                 // List<S> values = dom.leafElements("...");
441                 // <packer init>
442                 // for( S v : values ) {
443                 //   <packer set>(<as>(v));
444                 // }
445                 //  ... assign ...
446                 packer.start(values.invoke("size"));
447                 JForEach forEach = body.forEach(conv.sourceType(), id(), values);
448                 packer.pack(forEach.body(),conv.as(forEach.var(),packer.itemType()), forEach.var());
449                 assign(packer.end());
450             }

470             private Packer createPacker(TypeMirror type, TypeMirror erasure) {
471                 if(erasure instanceof ArrayType) {
472                     // T=X[]
473                     return new ArrayPacker((ArrayType)erasure);
474                 }
475 
476                 TypeMirror itemType = math.isCollection(type);
477                 if(itemType!=null) {
478                     // T=Collection[]
479                     return new ListPacker(type,itemType);
480                 }
481 
482                 TypeMirror mapType = TypeMath.baseClassFinder.apply(type, env.getTypeDeclaration(Map.class.getName()));
483                 if(mapType!=null) {
484                     // T=Map<...>
485                     DeclaredType d = (DeclaredType)mapType;
486                     Iterator<TypeMirror> itr = d.getActualTypeArguments().iterator();
487                     itr.next();
488                     return new MapPacker(itr.next());
489                 }
490 
491                 return null;
492             }

494             abstract class Packer {
495                 abstract TypeMirror itemType();
496                 /***
497                  * Starts packing.
498                  */
499                 abstract void start(JExpression $valueSize);
500                 /***
501                  * Adds one more item to the pack.
502                  */
503                 abstract void pack(JBlock block, JExpression item, JExpression sourceValue);
504 
505                 /***
506                  * Returns the packed value to be set.
507                  */
508                 abstract JExpression end();
509             }

511             final class ArrayPacker extends Packer {
512                 private JVar $array,$index;
513                 private final JType arrayT;
514                 private final JType componentT;
515                 private final ArrayType at;
516 
517                 public ArrayPacker(ArrayType t) {
518                     this.at = t;
519                     this.componentT = TO_JTYPE.apply(itemType(),null);
520                     this.arrayT = componentT.array();
521                 }
522 
523                 TypeMirror itemType() {
524                     return at.getComponentType();
525                 }
526 
527                 void start(JExpression $valueSize) {
528                     // [RESULT]
529                     // T[] x = new T[values.size()];
530                     $array = var(arrayT, JExpr.newArray(componentT, $valueSize));
531                     $index = var(int.class,JExpr.lit(0));
532                 }
533 
534                 void pack(JBlock block, JExpression item, JExpression sourceValue) {
535                     // [RESULT]
536                     // x[i++] = <rhs>;
537                     block.assign($array.component($index.incr()),item);
538                 }
539 
540                 JExpression end() {
541                     return $array;
542                 }
543             }

545             final class ListPacker extends Packer {
546                 private JVar $list;
547                 private final JClass collectionType,itemType;
548                 private final TypeMirror itemT;
549 
550                 public ListPacker(TypeMirror collectionType, TypeMirror itemType) {
551                     this.collectionType = TO_JTYPE.apply(collectionType,null).boxify();
552                     this.itemType       = TO_JTYPE.apply(itemType,null).boxify();
553                     this.itemT = itemType;
554                 }
555 
556                 TypeMirror itemType() {
557                     return itemT;
558                 }
559 
560                 void start(JExpression $valueSize) {
561                     // [RESULT]
562                     // T x = new ArrayList<T>(values.size());
563                     $list = var(collectionType,JExpr._new(implType()).arg($valueSize));
564                 }
565 
566                 /***
567                  * Figure out the concrete implementation class to be used.
568                  */
569                 JType implType() {
570                     if(cm.ref(Set.class).isAssignableFrom(collectionType))
571                         return cm.ref(HashSet.class).narrow(itemType);
572                     return cm.ref(ArrayList.class).narrow(itemType);
573                 }
574 
575                 void pack(JBlock block, JExpression item, JExpression sourceValue) {
576                     // [RESULT]
577                     // x.add(<rhs>);
578                     block.invoke($list,"add").arg(item);
579                 }
580 
581                 JExpression end() {
582                     return $list;
583                 }
584             }

586             final class MapPacker extends Packer {
587                 private JVar $map;
588                 private final TypeMirror itemT;
589 
590                 public MapPacker(TypeMirror itemType) {
591                     this.itemT = itemType;
592                 }
593 
594                 TypeMirror itemType() {
595                     return itemT;

598                 void start(JExpression $valueSize) {
599                     // [RESULT]

601                     $map = var(Map.class,JExpr._new(cm.ref(HashMap.class)).arg($valueSize));

608                 }
609

611                     return $map;
612                 }

615             private Converter createConverter(TypeMirror itemType) {
616                 try {
617                     // is this a leaf value?
618                     math.SIMPLE_VALUE_CONVERTER.apply(itemType, JExpr._null());
619                     return new LeafConverter();
620                 } catch (UnsupportedOperationException e) {
621                     // nope
622                 }
623 
624                 // try to handle it as a reference
625                 if (itemType instanceof DeclaredType) {
626                     if (!(itemType instanceof ClassType || itemType instanceof InterfaceType)) {
627                         env.getMessager().printError(p.decl().getPosition(),
628                         "I only inject interfaces or classes,  "+itemType+" is not one of them");
629                         return new NodeConverter();
630                     }
631                     TypeDeclaration decl = ((DeclaredType)itemType).getDeclaration();
632                     Configured cfg = decl.getAnnotation(Configured.class);
633                     if(cfg!=null) {
634                         // node value
635                         if(isReference())
636                             return new ReferenceConverter();
637                         else
638                             return new NodeConverter();
639                     }
640                 }
641 
642                 if(isAllElementMatch()) {
643                     return new NodeByTypeConverter(itemType);
644                 }
645 
646                 env.getMessager().printError(p.decl().getPosition(),
647                     "I don't know how to inject "+itemType+" from configuration");
648                 return new NodeConverter(); // error recovery
649             }

651             /***
652              * Encapsulates the source value representation in {@link Dom}.
653              */
654             abstract class Converter {
655                 /***
656                  * Generates an expression that converts 'rhs'.
657                  *
658                  * @param targetType
659                  *      The expected type of the expression, so that the generated expression
660                  *      can contain cast operation if necessary.
661                  */
662                 abstract JExpression as(JExpression rhs, TypeMirror targetType);
663                 /***
664                  * Source value type as returned by {@link Dom}.
665                  */
666                 abstract JClass sourceType();
667 
668                 /***
669                  * True if the XML representation of the source value is a leaf (string value)
670                  * as opposed to node (an XML fragment.)
671                  */
672                 abstract boolean isLeaf();
673                 
674                 abstract void addMetadata(String key,TypeMirror itemType);
675 
676                 protected final String makeCollectionIfNecessary(String s) {
677                     if(packer!=null)    return "collection:"+s;
678                     else                return s;
679                 }
680             }
681 
682             class LeafConverter extends Converter {
683                 JExpression as(JExpression rhs, TypeMirror targetType) {
684                     return math.SIMPLE_VALUE_CONVERTER.apply(targetType, rhs);
685                 }
686                 JClass sourceType() {
687                     return cm.ref(String.class);
688                 }
689 
690                 boolean isLeaf() {
691                     return true;
692                 }
693 
694                 void addMetadata(String key,TypeMirror itemType) {
695                     metadata.add(key,makeCollectionIfNecessary("leaf"));
696                 }
697             }

787                 return (!noDefaultValue);

789             /***
790              * Generates the injector that reads an attribute and sets the value.
791              */
792             @Override
793             protected void generate() {
794                 metadata.add(xmlTokenName(),isRequired()?"required":"optional");
795                 if (this.hasDefault()) {
796                     if (a.defaultValue().indexOf(',')!=-1) {
797                         metadata.add(xmlTokenName(), '"' + "default:" + a.defaultValue() + '"');
798                     } else {
799                         metadata.add(xmlTokenName(), "default:" + a.defaultValue());                        
800                     }
801                 }
802                 String ant = "";
803                 try {
804                     Class c = a.dataType();
805                 } catch(MirroredTypeException me) { //hack?
806                      ant = getCanonicalTypeFrom(me);

808                 if (ant.length() == 0) { //take it from the return type of method
809                     Property.Method m = (Property.Method)p; // Method needn't be Property's inner class
810                     String typeReturnedByMethodDecl = m.decl.getReturnType().toString();
811                     metadata.add(xmlTokenName(), "datatype:" + typeReturnedByMethodDecl);
812                 } else {
813                     metadata.add(xmlTokenName(), "datatype:" + ant);
814                 }
815                 super.generate();
816             }

894             case BYTE:      return cm.BYTE;

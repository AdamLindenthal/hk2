154 		correctSet.clear();

156 		correctSet.add(Named.class.getName());

230         Assert.assertEquals(PerLookup.class, cDesc.getScopeAnnotation());

234         Assert.assertTrue(cDesc.getQualifiers().contains(Green.class.getName()));
235         Assert.assertTrue(cDesc.getQualifiers().contains(Blue.class.getName()));
236

238         boolean red = false;

240         boolean blue = false;
241         
242         for (Annotation anno : cDesc.getQualifierAnnotations()) {
243             if (Red.class.equals(anno.annotationType())) red = true;
244             if (Green.class.equals(anno.annotationType())) green = true;
245             if (Blue.class.equals(anno.annotationType())) blue = true;
246         }
247         
248         Assert.assertTrue(red);
249         Assert.assertTrue(green);
250         Assert.assertTrue(blue);
251         
252         Assert.assertEquals(DescriptorType.CLASS, cDesc.getDescriptorType());
253         Assert.assertTrue(cDesc.getMetadata().isEmpty());
254         Assert.assertNull(cDesc.getLoader());
255         Assert.assertEquals(0, cDesc.getRanking());
256         Assert.assertNull(cDesc.getBaseDescriptor());
257         Assert.assertNull(cDesc.getServiceId());
258         Assert.assertNull(cDesc.getLocatorId());
259         Assert.assertTrue(cDesc.isReified());
260         Assert.assertTrue(cDesc.getInjectees().isEmpty());
261         
262         Assert.assertEquals(c, cDesc.create(null));
263         
264         // Call the destroy, though it should do nothing
265         cDesc.dispose(c);
266         
267         // Check the cache
268         Assert.assertEquals(c, cDesc.getCache());
269         Assert.assertTrue(cDesc.isCacheSet());
270         
271         String asString = cDesc.toString();
272         Assert.assertTrue(asString.contains("implementation=org.glassfish.hk2.tests.api.FullDescriptorImpl"));

352     /**
353      * Tests that an object with a complex hierarchy is analyzed properly
354      */
355     @Test
356     public void testComplexHierarchyObject() {
357         AbstractActiveDescriptor<ComplexHierarchy> d = BuilderHelper.createConstantDescriptor(new ComplexHierarchy());
358         
359         Set<String> contracts = d.getAdvertisedContracts();
360         
361         Assert.assertTrue(contracts.contains(ComplexHierarchy.class.getName()));
362         Assert.assertTrue(contracts.contains(MarkerInterfaceImpl.class.getName()));
363         Assert.assertTrue(contracts.contains(MarkerInterface2.class.getName()));
364         Assert.assertTrue(contracts.contains(ParameterizedInterface.class.getName()));
365         
366         Assert.assertEquals(4, contracts.size());
367         
368         Set<Type> contractsAsTypes = d.getContractTypes();
369         
370         int lcv = 0;
371         for (Type contractAsType : contractsAsTypes) {
372             // This tests that this is an ordered iterator
373             switch(lcv) {
374             case 0:
375                 Assert.assertEquals(ComplexHierarchy.class, contractAsType);
376                 break;
377             case 1:
378                 Assert.assertEquals(MarkerInterfaceImpl.class, contractAsType);
379                 break;
380             case 2:
381                 Assert.assertEquals(MarkerInterface2.class, contractAsType);
382                 break;
383             case 3:
384                 ParameterizedType pt = (ParameterizedType) contractAsType;
385                 
386                 Assert.assertEquals(ParameterizedInterface.class, pt.getRawType());
387                 Assert.assertEquals(String.class, pt.getActualTypeArguments()[0]);
388                 break;
389             default:
390                 Assert.fail("Too many types: " + contractAsType);
391             }
392             
393             lcv++;
394         }
395         
396         Assert.assertNotNull(d.isProxiable());
397         Assert.assertEquals(false, d.isProxiable().booleanValue());
398     }

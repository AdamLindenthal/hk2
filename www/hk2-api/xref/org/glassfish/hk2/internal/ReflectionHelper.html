65      /**
66       * Given the type parameter gets the raw type represented
67       * by the type, or null if this has no associated raw class
68       * @param type The type to find the raw class on
69       * @return The raw class associated with this type
70       */
71      public static Class<?> getRawClass(Type type) {
72          if (type == null) return null;
73          
74          if (type instanceof Class) {
75              return (Class<?>) type;
76          }
77          
78          if (type instanceof ParameterizedType) {
79              Type rawType = ((ParameterizedType) type).getRawType();
80              if (rawType instanceof Class) {
81                  return (Class<?>) rawType;
82              }
83          }
84          
85          return null;
86      }

88      private static String getNamedName(Named named, Class<?> implClass) {
89          String name = named.value();
90          if (name != null && !name.equals("")) return name;
91          
92          String cn = implClass.getName();
93              
94          int index = cn.lastIndexOf(".");
95          if (index < 0) return cn;
96          
97          return cn.substring(index + 1);
98      }

102      * 
103      * @param implClass The class to evaluate
104      * @return The name this class should have

106     public static String getName(Class<?> implClass) {
107         Named named = implClass.getAnnotation(Named.class);
108         
109         String namedName = (named != null) ? getNamedName(named, implClass) : null ;

116     /**
117      * Returns the set of types this class advertises
118      * @param t the object we are analyzing
119      * @return The type itself and the contracts it implements
120      */
121     public static Set<Type> getAdvertisedTypesFromObject(Object t) {
122         Set<Type> retVal = new HashSet<Type>();
123         if (t == null) return retVal;
124         
125         retVal.add(t.getClass());
126         
127         Type genericInterfaces[] = t.getClass().getGenericInterfaces();
128         for (Type genericInterface : genericInterfaces) {
129             Class<?> rawClass = getRawClass(genericInterface);
130             if (rawClass == null) continue;
131             
132             if (rawClass.isAnnotationPresent(Contract.class)) {
133                 retVal.add(genericInterface);
134             }
135         }
136         
137         return retVal;
138     }

140     /**
141      * Gets the scope annotation from the object
142      * @param t The object to analyze
143      * @return The class of the scope annotation
144      */
145     public static Class<? extends Annotation> getScopeFromObject(Object t) {
146         if (t == null) return PerLookup.class;
147         
148         Class<?> oClass = t.getClass();
149         for (Annotation annotation : oClass.getAnnotations()) {
150             Class<? extends Annotation> annoClass = annotation.annotationType();
151             
152             if (annoClass.isAnnotationPresent(Scope.class)) {
153                 return annoClass;
154             }
155             
156         }
157         
158         return PerLookup.class;
159     }

164      * @param t The object to analyze
165      * @return The set of qualifiers.  Will not return null but may return an empty set

167     public static Set<Annotation> getQualifiersFromObject(Object t) {
168         Set<Annotation> retVal = new HashSet<Annotation>();
169         if (t == null) return retVal;
170         
171         Class<?> oClass = t.getClass();
172         for (Annotation annotation : oClass.getAnnotations()) {
173             Class<? extends Annotation> annoClass = annotation.annotationType();
174             
175             if (annoClass.isAnnotationPresent(Qualifier.class)) {
176                 retVal.add(annotation);
177             }
178             
179         }
180         
181         return retVal;
182     }

184     private static String writeSet(Set<?> set) {
185         StringBuffer sb = new StringBuffer("{");
186         
187         boolean first = true;
188         for (Object writeMe : set) {
189             if (first) {
190                 first = false;
191                 sb.append(writeMe.toString());
192             }
193             else {
194                 sb.append("," + writeMe.toString());
195             }
196         }
197         
198         sb.append("}");
199         
200         return sb.toString();

223         StringBuffer sb = new StringBuffer("{");
224         
225         boolean first = true;
226         for (Map.Entry<String, List<String>> entry : metadata.entrySet()) {
227             if (first) {
228                 first = false;
229                 sb.append(entry.getKey() + "=");
230             }
231             else {
232                 sb.append("," + entry.getKey() + "=");
233             }
234             
235             sb.append(writeList(entry.getValue()));
236         }
237         
238         sb.append("}");

244      * Pretty prints this descriptor
245      * 
246      * @param d The descriptor to write out nicely
247      * @return The descriptor to print
248      */
249     public static String prettyPrintDescriptor(Descriptor d) {
250         StringBuffer sb = new StringBuffer("Descriptor(");
251         
252         sb.append("\n\timplementation=" + d.getImplementation());
253         
254         if (d.getName() != null) {
255             sb.append("\n\tname=" + d.getName());
256         }
257         
258         sb.append("\n\tcontracts=");
259         sb.append(writeSet(d.getAdvertisedContracts()));
260         
261         sb.append("\n\tscope=" + d.getScope());
262         
263         sb.append("\n\tqualifiers=");
264         sb.append(writeSet(d.getQualifiers()));
265         
266         sb.append("\n\tdescriptorType=" + d.getDescriptorType());
267         
268         sb.append("\n\tmetadata=");
269         sb.append(writeMetadata(d.getMetadata()));
270         
271         sb.append("\n\tloader=" + d.getLoader());
272         
273         sb.append("\n\tid=" + d.getServiceId());
274         
275         sb.append("\n\tlocatorId=" + d.getServiceId());
276         
277         sb.append("\n\tidentityHashCode=" + System.identityHashCode(d));
278         
279         sb.append(")");
280         
281         return sb.toString();

284     /**
285      * Adds a value to the list of values associated with this key
286      * 
287      * @param metadatas The base metadata object
288      * @param key The key to which to add the value.  May not be null
289      * @param value The value to add.  May not be null
290      */
291     public static void addMetadata(Map<String, List<String>> metadatas, String key, String value) {
292         if (key == null || value == null) return;
293         List<String> inner = metadatas.get(key);
294         if (inner == null) {
295             inner = new LinkedList<String>();
296             metadatas.put(key, inner);
297         }
298         
299         inner.add(value);
300     }

302     /**
303      * Removes the given value from the given key
304      * 
305      * @param metadatas The base metadata object
306      * @param key The key of the value to remove.  May not be null
307      * @param value The value to remove.  May not be null
308      * @return true if the value was removed
309      */
310     public static boolean removeMetadata(Map<String, List<String>> metadatas, String key, String value) {
311         if (key == null || value == null) return false;
312         
313         List<String> inner = metadatas.get(key);
314         if (inner == null) return false;
315         
316         boolean retVal = inner.remove(value);
317         if (inner.size() <= 0) metadatas.remove(key);
318         
319         return retVal;
320     }

322     /**
323      * Removes all the metadata values associated with key
324      * 
325      * @param metadatas The base metadata object
326      * @param key The key of the metadata values to remove
327      * @return true if any value was removed
328      */
329     public static boolean removeAllMetadata(Map<String, List<String>> metadatas, String key) {
330         List<String> values = metadatas.remove(key);
331         return (values != null && values.size() > 0);
332     }

334     /**
335      * This method does a deep copy of the incoming meta-data, (which basically means we will
336      * also make copies of the value list)
337      * 
338      * @param copyMe The guy to copy (if null, null will be returned)
339      * @return A deep copy of the metadata
340      */
341     public static Map<String, List<String>> deepCopyMetadata(Map<String, List<String>> copyMe) {
342         if (copyMe == null) return null;
343         
344         Map<String, List<String>> retVal = new LinkedHashMap<String, List<String>>();
345         
346         for (Map.Entry<String, List<String>> entry : copyMe.entrySet()) {
347             String key = entry.getKey();
348             List<String> value = entry.getValue();
349             
350             LinkedList<String> valueCopy = new LinkedList<String>(value);
351             
352             retVal.put(key, valueCopy);
353         }
354         
355         return retVal;
356     }

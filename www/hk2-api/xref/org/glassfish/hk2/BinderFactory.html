40  package org.glassfish.hk2;
41  
42  /**
43   * Factory for adding services instances to a {@link Services}. Services
44   * definition can vary per service, some services implement well defined
45   * interfaces usually annotated with {@link org.jvnet.hk2.annotations.Contract}
46   * which are called contracts. Some services do not implement such
47   * interfaces and are registered under their implementation type.
48   *
49   * <p/>
50   * When registering services using this programmatic API, it is not
51   * necessary to use the HK2 annotations. Optional presence of such
52   * annotations will be ignored when registering servics.
53   * 
54   * <p/>
55   * Important Note: A new binding is completed only after calling
56   * one of the "to*(...)" methods of {@link NamedBinder}.
57   *
58   * @author Jerome Dochez, Jeff Trent
59   */
60  public interface BinderFactory {
61  
62      /**
63       * Return the parent binder factory instance which can be used to
64       * bind services for wider visibility outside of this module
65       * definition.
66       *
67       * @return the parent binder factory or null if there is no parent.
68       */
69      BinderFactory inParent();
70  
71      /**
72       * Binds a service using a {@link String} interface name.
73       *
74       * @param contractName the interface fqcn.
75       * @return a {@link Binder} instance that can be used to further
76       * qualify the binding request.
77       */
78      Binder<Object> bind(String contractName);
79  
80      /**
81       * Starts the binding process of a service using {@link String}
82       * interfaces names that can be used to lookup the service.
83       *
84       * @param contractNames the interfaces fully qualified class names.
85       * @return a {@link Binder} instance that can be used to further
86       * qualify the binding request.
87       */
88      Binder<Object> bind(String... contractNames);
89  
90      /**
91       * Starts a binding process of a service using at least one
92       * interface or abstract class {@link Class} reference.
93       *
94       * Supplemental interfaces can be used to register the service
95       * under different contracts. Each contract can be used to look
96       * up the service.
97       *
98       * The service must implements all the contracts passed in
99       * this method.
100      *
101      * @param contract the main contract {@link Class} reference
102      * @param contracts supplemental contracts references
103      * @param <T> the main contract type
104      * @return a {@link Binder} instance for the main contract type type to further
105      * qualify the binding request.
106      */
107     <T> Binder<T> bind(Class<T> contract, Class<?>... contracts);
108 
109     /**
110      * Binds a parameterized type by forcing users to create a subclass of
111      * {@link TypeLiteral} which will allow HK2 to retrieve the parameterized
112      * type at runtime.
113      *
114      * @param typeLiteral a {@link TypeLiteral} subclass instance
115      * @param <T> the parameterized type to use as a contract definition. This
116      * will allow users to inject instances of the parameterized type. HK2
117      * cannot bind a generic type like Set<E> although it can injects it as
118      * long as E is specified at the injection point and the parameterized
119      * type can be looked up.
120      *
121      * @return a {@link Binder} instance for that parameterized type to further
122      * qualify the binding request.
123      */
124     <T> Binder<T> bind(TypeLiteral<T> typeLiteral);
125 
126     /**
127      * Binds a service which does not implement a contract or
128      * interface that can be used to look it up. The bound service
129      * can only be looked up using the service type directly.
130      *
131      * A service registered by its type cannot be named.
132      *
133      * @return a {@link NamedBinder} to register the service.
134      */
135     NamedBinder<Object> bind();
136 }

72      /**
73       * This method will add the ability to use the {@link PerThread} scope to
74       * the given locator.  If the locator already has a {@link Context} implementation
75       * that handles the {@link PerThread} scope this method does nothing.
76       *
77       * @param locator The non-null locator to enable the PerThread scope on
78       * @throws MultiException if there were errors when committing the service
79       */
80      public static void enablePerThreadScope(ServiceLocator locator) {
81          Context<PerThread> perThreadContext = locator.getService((new TypeLiteral<Context<PerThread>>() {}).getType());
82          if (perThreadContext != null) return;
83  
84          DynamicConfigurationService dcs = locator.getService(DynamicConfigurationService.class);
85          DynamicConfiguration config = dcs.createDynamicConfiguration();
86          final DescriptorImpl descriptor = BuilderHelper.link(PerThreadContext.class).
87                  to(Context.class).
88                  in(Singleton.class.getName()).
89                  build();
90  
91          ClassLoader loader = ServiceLocatorUtilities.class.getClassLoader();
92          final ClassLoader binderClassLoader = loader == null ? ClassLoader.getSystemClassLoader() : loader;
93          descriptor.setLoader(new HK2Loader() {
94              @Override
95              public Class<?> loadClass(String className) throws MultiException {
96                  try {
97                      return binderClassLoader.loadClass(className);
98                  } catch (ClassNotFoundException e) {
99                      throw new MultiException(e);
100                 }
101 
102             }
103         });
104 
105         config.bind(descriptor);
106         config.commit();
107     }

111      * single config transaction.
112      *
113      * @param locator The non-null locator to use for the configuration action
114      * @param binders The non-null list of binders to be added to the locator
115      * @throws MultiException if any error was encountered while binding services

117     public static void bind(ServiceLocator locator, Binder... binders) {
118         DynamicConfigurationService dcs = locator.getService(DynamicConfigurationService.class);
119         DynamicConfiguration config = dcs.createDynamicConfiguration();
120 
121         for (Binder binder : binders) {
122             binder.bind(config);
123         }
124 
125         config.commit();
126     }

196     /**
197      * It is very often the case that one wishes to add a single descriptor to
198      * a service locator.  This method adds that one descriptor.  If the descriptor
199      * is an {@link ActiveDescriptor} and is reified, it will be added as an
200      * {@link ActiveDescriptor}.  Otherwise it will be bound as a {@link Descriptor}.
201      *
202      * @param locator The non-null locator to add this descriptor to
203      * @param descriptor The non-null descriptor to add to this locator
204      * @throws MultiException On a commit failure
205      */
206     @SuppressWarnings("unchecked")
207     public static <T> ActiveDescriptor<T> addOneDescriptor(ServiceLocator locator, Descriptor descriptor) {
208         DynamicConfigurationService dcs = locator.getService(DynamicConfigurationService.class);
209         DynamicConfiguration config = dcs.createDynamicConfiguration();
210 
211         ActiveDescriptor<T> retVal;
212         if (descriptor instanceof ActiveDescriptor) {
213             ActiveDescriptor<T> active = (ActiveDescriptor<T>) descriptor;
214 
215             if (active.isReified()) {
216                 retVal = config.addActiveDescriptor(active);
217             }
218             else {
219                 retVal = config.bind(descriptor);
220             }
221 
222         }
223         else {
224             retVal = config.bind(descriptor);
225         }
226 
227         config.commit();
228 
229         return retVal;
230     }

232     /**
233      * Finds a descriptor in the given service locator.  If the descriptor has the serviceId and
234      * locatorId set then it will first attempt to use those values to get the exact descriptor
235      * described by the input descriptor.  Failing that (or if the input descriptor does not have
236      * those values set) then it will use the equals algorithm of {@DescriptorImpl} to determine
237      * the equality of the descriptor.
238      * 
239      * @param locator The non-null locator in which to find the descriptor
240      * @param descriptor The non-null descriptor to search for
241      * @return The descriptor as found in the locator, or null if no such descriptor could be found
242      */
243     @SuppressWarnings("unchecked")
244     public static <T> ActiveDescriptor<T> findOneDescriptor(ServiceLocator locator, Descriptor descriptor) {
245         if (locator == null || descriptor == null) throw new IllegalArgumentException();
246         
247         if (descriptor.getServiceId() != null && descriptor.getLocatorId() != null) {
248             ActiveDescriptor<T> retVal = (ActiveDescriptor<T>) locator.getBestDescriptor(
249                     BuilderHelper.createSpecificDescriptorFilter(descriptor));
250             
251             if (retVal != null) return retVal;
252             
253             // Fall back to DescriptorImpl.equals
254         }
255         
256         final DescriptorImpl di;
257         if (descriptor instanceof DescriptorImpl) {
258             di = (DescriptorImpl) descriptor;
259         }
260         else {
261             di = new DescriptorImpl(descriptor);
262         }
263         
264         ActiveDescriptor<T> retVal = (ActiveDescriptor<T>) locator.getBestDescriptor(new Filter() {
265 
266             @Override
267             public boolean matches(Descriptor d) {
268                 return di.equals(d);
269             }
270             
271         });
272         
273         return retVal;
274     }

276     /**
277      * This method will attempt to remove descriptors matching the passed in descriptor from
278      * the given locator.  If the descriptor has its locatorId and serviceId values set then
279      * only a descriptor matching those exact locatorId and serviceId will be removed.  Otherwise
280      * any descriptor that returns true from the {@link DescriptorImpl#equals(Object)} method
281      * will be removed from the locator.  Note that if more than one descriptor matches they
282      * will all be removed.  Hence more than one descriptor may be removed by this method.
283      * 
284      * @param locator The non-null locator to remove the descriptor from
285      * @param descriptor The non-null descriptor to remove from the locator
286      */
287     public static void removeOneDescriptor(ServiceLocator locator, Descriptor descriptor) {
288         if (locator == null || descriptor == null) throw new IllegalArgumentException();
289         
290         DynamicConfigurationService dcs = locator.getService(DynamicConfigurationService.class);
291         DynamicConfiguration config = dcs.createDynamicConfiguration();
292         
293         if (descriptor.getLocatorId() != null && descriptor.getServiceId() != null) {
294             Filter destructionFilter = BuilderHelper.createSpecificDescriptorFilter(descriptor);
295             
296             config.addUnbindFilter(destructionFilter);
297             
298             config.commit();
299             
300             return;
301         }
302         
303         // Must use second algorithm, which is not as precise, but which still mainly works
304         final DescriptorImpl di;
305         if (descriptor instanceof DescriptorImpl) {
306             di = (DescriptorImpl) descriptor;
307         }
308         else {
309             di = new DescriptorImpl(descriptor);
310         }
311         
312         Filter destructionFilter = new Filter() {
313 
314             @Override
315             public boolean matches(Descriptor d) {
316                 return di.equals(d);
317             }
318             
319         };
320         
321         config.addUnbindFilter(destructionFilter);
322         
323         config.commit();
324     }

344     /**
345      * Returns the best service matching the passed in fully qualified
346      * class name of the service
347      * 
348      * @param locator The locator to find the service in
349      * @param className The fully qualified class name of the service
350      * @return The found service, or null if there is no service with this class name
351      */
352     @SuppressWarnings("unchecked")
353     public static <T> T getService(ServiceLocator locator, String className) {
354         if (locator == null || className == null) throw new IllegalArgumentException();
355         
356         ActiveDescriptor<T> ad = (ActiveDescriptor<T>) locator.getBestDescriptor(BuilderHelper.createContractFilter(className));
357         if (ad == null) return null;
358         
359         return locator.getServiceHandle(ad).getService();
360     }

362     /**
363      * Returns the service in this service locator given the current descriptor.
364      * 
365      * @param locator The non-null locator in which to get the service associated with
366      * this descriptor
367      * @param descriptor The non-null descriptor to find the corresponding service for
368      * @return
369      */
370     @SuppressWarnings("unchecked")
371     public static <T> T getService(ServiceLocator locator, Descriptor descriptor) {
372         if (locator == null || descriptor == null) throw new IllegalArgumentException();
373         
374         Long locatorId = descriptor.getLocatorId();
375         if (locatorId != null &&
376                 (locatorId.longValue() == locator.getLocatorId()) &&
377                 (descriptor instanceof ActiveDescriptor)) {
378             return locator.getServiceHandle((ActiveDescriptor<T>) descriptor).getService();
379             
380         }
381         
382         ActiveDescriptor<T> found = findOneDescriptor(locator, descriptor);
383         if (found == null) return null;
384         
385         return locator.getServiceHandle(found).getService();
386     }

408     /**
409      * This method will first attempt to find a service corresponding to the type and qualifiers
410      * passed in to the method, and if one is found simply returns it.  If no service is found
411      * in the locator this method will call {@link ServiceLocator#createAndInitialize(Class)} on
412      * the class (ignoring the qualifiers) in order to create an instance of the service.
413      * 
414      * @param locator The service locator to search for the service with
415      * @param type The non-null type of object to either find or create
416      * @param qualifiers The set of qualifiers that should be associated with the service
417      * @return An instance of type as either found in the locator or automatically created,
418      * injected and post-constructed.  Note that the return value CAN be null IF the service
419      * was found in the service locator and the context in which the service is in allows for
420      * null services.
421      * @throws MultiException On a failure from any of the underlying operations
422      */
423     public static <T> T findOrCreateService(ServiceLocator locator, Class<T> type, Annotation... qualifiers) throws MultiException {
424         if (locator == null || type == null) throw new IllegalArgumentException();
425         
426         ServiceHandle<T> retVal = locator.getServiceHandle(type, qualifiers);
427         if (retVal == null) {
428             return locator.createAndInitialize(type);
429         }
430         
431         return retVal.getService();
432     }

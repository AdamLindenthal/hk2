154     /**
155      * Build the binding descriptor and bind it in the {@link DynamicConfiguration
156      * dynamic configuration}.
157      *
158      * @param configuration dynamic binding configuration.
159      * @param defaultLoader default HK2 loader that should be used in case a custom loader
160      *                      was not set.
161      */
162     abstract void complete(DynamicConfiguration configuration, HK2Loader defaultLoader);

164     private static class ClassBasedBindingBuilder<T> extends AbstractBindingBuilder<T> {
165         private final Class<T> service;
166 
167         public ClassBasedBindingBuilder(Class<T> service, Type serviceContractType) {
168             this.service = service;
169             if (serviceContractType != null) {
170                 super.contracts.add(serviceContractType);
171             }
172         }
173 
174         @Override
175         void complete(final DynamicConfiguration configuration, final HK2Loader defaultLoader) {
176             if (this.loader == null) {
177                 this.loader = defaultLoader;
178             }
179 
180             ActiveDescriptorBuilder builder = BuilderHelper.activeLink(service)
181                     .named(name)
182                     .andLoadWith(this.loader);
183 
184             if (scope != null) {
185                 builder.in(scope);
186             }
187 
188             if (ranked != null) {
189                 builder.ofRank(ranked);
190             }
191 
192             for (String key : metadata.keySet()) {
193                 for (String value : metadata.get(key)) {
194                     builder.has(key, value);
195                 }
196             }
197 
198             for (Annotation annotation : qualifiers) {
199                 builder.qualifiedBy(annotation);
200             }
201 
202             for (Type contract : contracts) {
203                 builder.to(contract);
204             }
205 
206             configuration.bind(builder.build());
207         }
208     }

210     private static class InstanceBasedBindingBuilder<T> extends AbstractBindingBuilder<T> {
211         private final T service;
212 
213         public InstanceBasedBindingBuilder(T service) {
214             this.service = service;
215         }
216 
217         @Override
218         void complete(DynamicConfiguration configuration, HK2Loader defaultLoader) {
219             if (this.loader == null) {
220                 this.loader = defaultLoader;
221             }
222             AbstractActiveDescriptor descriptor = BuilderHelper.createConstantDescriptor(service);
223             descriptor.setName(name);
224             descriptor.setLoader(this.loader);
225 
226             if (scope != null) {
227                 descriptor.setScope(scope.getName());
228             }
229 
230             if (ranked != null) {
231                 descriptor.setRanking(ranked);
232             }
233 
234             for (String key : metadata.keySet()) {
235                 for (String value : metadata.get(key)) {
236                     descriptor.addMetadata(key, value);
237                 }
238             }
239 
240             for (Annotation annotation : qualifiers) {
241                 descriptor.addQualifierAnnotation(annotation);
242             }
243 
244             for (Type contract : contracts) {
245                 descriptor.addContractType(contract);
246             }
247 
248             configuration.bind(descriptor);
249         }
250     }

252     private static class FactoryInstanceBasedBindingBuilder<T> extends AbstractBindingBuilder<T> {
253         private final Factory<T> factory;
254 
255         public FactoryInstanceBasedBindingBuilder(Factory<T> factory) {
256             this.factory = factory;
257         }
258 
259         @Override
260         void complete(DynamicConfiguration configuration, HK2Loader defaultLoader) {
261             if (this.loader == null) {
262                 this.loader = defaultLoader;
263             }
264 
265             AbstractActiveDescriptor factoryContractDescriptor = BuilderHelper.createConstantDescriptor(factory);
266             factoryContractDescriptor.addContractType(factory.getClass());
267             factoryContractDescriptor.setName(name);
268             factoryContractDescriptor.setLoader(this.loader);
269 
270             ActiveDescriptorBuilder descriptorBuilder = BuilderHelper.activeLink(factory.getClass())
271                     .named(name)
272                     .andLoadWith(this.loader);
273             if (scope != null) {
274                 descriptorBuilder.in(scope);
275             }
276 
277             if (ranked != null) {
278                 descriptorBuilder.ofRank(ranked);
279             }
280 
281             for (Annotation qualifier : qualifiers) {
282                 factoryContractDescriptor.addQualifierAnnotation(qualifier);
283                 descriptorBuilder.qualifiedBy(qualifier);
284             }
285 
286             for (Type contract : contracts) {
287                 factoryContractDescriptor.addContractType(new ParameterizedTypeImpl(Factory.class, contract));
288                 descriptorBuilder.to(contract);
289             }
290 
291             configuration.bind(new FactoryDescriptorsImpl(
292                     factoryContractDescriptor,
293                     descriptorBuilder.buildFactory()));
294         }
295     }

297     private static class FactoryTypeBasedBindingBuilder<T> extends AbstractBindingBuilder<T> {
298         private final Class<? extends Factory<T>> factoryClass;
299         private final Class<? extends Annotation> factoryScope;
300 
301         public FactoryTypeBasedBindingBuilder(Class<? extends Factory<T>> factoryClass, Class<? extends Annotation> factoryScope) {
302             this.factoryClass = factoryClass;
303             this.factoryScope = factoryScope;
304         }
305 
306         @Override
307         void complete(DynamicConfiguration configuration, HK2Loader defaultLoader) {
308             if (this.loader == null) {
309                 this.loader = defaultLoader;
310             }
311 
312             ActiveDescriptorBuilder factoryDescriptorBuilder = BuilderHelper.activeLink(factoryClass)
313                     .named(name)
314                     .andLoadWith(this.loader);
315             if (factoryScope != null) {
316                 factoryDescriptorBuilder.in(factoryScope);
317             }
318 
319             ActiveDescriptorBuilder descriptorBuilder = BuilderHelper.activeLink(factoryClass)
320                     .named(name)
321                     .andLoadWith(this.loader);
322             if (scope != null) {
323                 descriptorBuilder.in(scope);
324             }
325 
326             if (ranked != null) {
327 //                factoryContractDescriptor.ofRank(factoryRank);
328                 descriptorBuilder.ofRank(ranked);
329             }
330 
331             for (Annotation qualifier : qualifiers) {
332                 factoryDescriptorBuilder.qualifiedBy(qualifier);
333                 descriptorBuilder.qualifiedBy(qualifier);
334             }
335 
336             for (Type contract : contracts) {
337                 factoryDescriptorBuilder.to(new ParameterizedTypeImpl(Factory.class, contract));
338                 descriptorBuilder.to(contract);
339             }
340 
341             configuration.bind(new FactoryDescriptorsImpl(
342                     factoryDescriptorBuilder.build(),
343                     descriptorBuilder.buildFactory()));
344         }
345     }

347     private static class FactoryDescriptorsImpl implements FactoryDescriptors {
348         private final Descriptor serviceDescriptor;
349         private final Descriptor factoryDescriptor;
350 
351         public FactoryDescriptorsImpl(Descriptor serviceDescriptor, Descriptor factoryDescriptor) {
352             this.serviceDescriptor = serviceDescriptor;
353             this.factoryDescriptor = factoryDescriptor;
354         }
355 
356         @Override
357         public Descriptor getFactoryAsAService() {
358             return serviceDescriptor;
359         }
360 
361         @Override
362         public Descriptor getFactoryAsAFactory() {
363             return factoryDescriptor;
364         }
365 
366         @Override
367         public String toString() {
368             return "FactoryDescriptorsImpl(\n" +
369                     serviceDescriptor + ",\n" + factoryDescriptor + ",\n\t" + System.identityHashCode(this) + ")";

49  /**
50   * Entry point to the HK2 services runtime. Users should start with {@link org.glassfish.hk2.HK2#get()}
51   * to get an implementation of the HK2 runtime.
52   *
53   * @author Jerome Dochez
54   */
55  public class HK2 {
56  
57  
58      final HK2Provider provider;
59  
60      /**
61       * Provides an {@link HK2} APIs implementation using the provided {@link HK2Provider}
62       * instance as the implementation backend.
63       *
64       * This is a very advanced usage pattern when {@link HK2Provider} implementations
65       * are customized to provide specific domain features.
66       *
67       * @param provider the HK2 public APIs implementation backend.
68       */
69      public HK2(HK2Provider provider) {
70          this.provider = provider;
71      }
72  
73      /**
74       * Entry point to the HK2 public APIs, will initialize the implementation and return
75       * a valid instance that can be used to configure modules bindings.
76       *
77       * Implementation will be searched using the META-INF/services service location feature
78       * {@see java.util.ServiceLoader} for further reference.
79       *
80       * @return a HK2 instance to configure {@link Module} types or instances with or null if
81       * an implementation cannot be found.
82       *
83       * @throws RuntimeException if an implementation cannot be instantiated.
84       */
85      public static HK2 get() {
86          try {
87              HK2Provider provider=null;
88              Iterable<? extends HK2Provider> loader = ServiceLoader.lookupProviderInstances(HK2Provider.class);
89              if (loader!=null) {
90                  Iterator<? extends HK2Provider> iterator = loader.iterator();
91                  if (iterator!=null && iterator.hasNext()) {
92                      provider=iterator.next();
93                  }
94              }
95              // todo : we need to revisit when ServiceLoader can take care of META-INF/Services in non OSGi env.
96              if (provider==null) {
97                  Iterator<HK2Provider> providers = java.util.ServiceLoader.load(HK2Provider.class).iterator();
98                  if (providers!=null && providers.hasNext()) {
99                      provider = providers.next();
100                 } else {
101                     Logger.getLogger(HK2.class.getName()).severe("Cannot find an implementation of the HK2 public API.");
102                     return null;
103                 }
104             }
105             return new HK2(provider);
106         } catch (Exception e) {
107             // todo : better handling
108             throw new RuntimeException(e);
109         }
110     }

112     /**
113      * Creates a new {@link Services} instances to register and lookup services to and from.
114      *
115      * {@link Module} will be instantiated in the order of the their declaration unless some
116      * modules have explicit dependencies on other modules through a {@link org.jvnet.hk2.annotations.Inject}
117      * annotation.
118      *
119      * {@link Module} are components and will follow normal HK2 activation and injection
120      * procedure prior to the {@link Module#configure(BinderFactory)} call.
121      *
122      * {@link Module} that are named using the {@link org.jvnet.hk2.annotations.Service#name()}
123      * annotation attribute will be added to the returned {@link Services} instance under
124      * the name + {@link Module} contract identity. The returned {@link Services} instance will
125      * also be registered using the optional {@link Module} name if present.
126      *
127      * Returned instance of {@link Services} is not automatically added to the parent (if not null)
128      * as an indexed {@link Services} instances. Users should use the parent's
129      * {@link org.glassfish.hk2.Services#bindDynamically()} method to register children
130      * services.
131      *
132      * @param parent the parent {@link Services} instances to delegate lookup to.
133      * @param moduleTypes array of {@link Module} types that will be configured with in the
134      * returned {@link Services} instance.
135      * @return a {@link Services} instance configured with all the {@link Module} services
136      */
137     public Services create(Services parent, Class<? extends Module>... moduleTypes) {
138         return provider.create(parent, moduleTypes);
139     }

150      *
151      * @param parent the parent {@link Services} instances to delegate lookup to.
152      * @param modules array of {@link Module} that will be configured with in the
153      * returned {@link Services} instance.
154      * @return a {@link Services} instance configured with all the {@link Module} services
155      */
156     public Services create(Services parent, Module... modules) {
157         return provider.create(parent, modules);
